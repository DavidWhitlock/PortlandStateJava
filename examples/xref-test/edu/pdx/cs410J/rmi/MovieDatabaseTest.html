19    private MovieDatabase getMovieDatabase() throws RemoteException {
20      return new MovieDatabaseImpl();
21    }
22  
23    @Test
24    public void testCreateMovie() throws RemoteException {
25      MovieDatabase db = getMovieDatabase();
26      String title = "Movie 1";
27      int year = 2008;
28      long id = db.createMovie(title, year);
29      Movie movie = db.getMovie(id);
30      assertNotNull(movie);
31      assertEquals(title, movie.getTitle());
32      assertEquals(year, movie.getYear());
33      assertEquals(0, db.getFilmography(99L).size());
34    }
35  
36    @Test
37    public void testFilmography() throws RemoteException {
38      long bobId = 56l;
39      long billId = 57l;
40  
41      String title1 = "Bob's Movie 1";
42      int year1 = 2007;
43      String title2 = "Bob's Movie 2";
44      int year2 = 2008;
45  
46      MovieDatabase db = getMovieDatabase();
47      long id1 = db.createMovie(title1, year1);
48      long id2 = db.createMovie(title2, year2);
49      db.noteCharacter(id1, "Joe", bobId);
50      db.noteCharacter(id2, "Frank", bobId);
51      db.noteCharacter(id2, "Henry", billId);
52  
53      assertEquals(0, db.getFilmography(78L).size());
54  
55      SortedSet<Movie> bobMovies = db.getFilmography(bobId);
56      assertEquals(2, bobMovies.size());
57      Iterator<Movie> bobIter = bobMovies.iterator();
58      Movie bobMovie1 = bobIter.next();
59      assertEquals(title1, bobMovie1.getTitle());
60      assertEquals(year1, bobMovie1.getYear());
61      Movie bobMovie2 = bobIter.next();
62      assertEquals(title2, bobMovie2.getTitle());
63      assertEquals(year2, bobMovie2.getYear());
64  
65      SortedSet<Movie> billMovies = db.getFilmography(billId);
66      assertEquals(1, billMovies.size());
67      Movie billMovie = billMovies.iterator().next();
68      assertEquals(title2, billMovie.getTitle());
69      assertEquals(year2, billMovie.getYear());
70    }
71  
72    @Test(expected = IllegalArgumentException.class)
73    public void deletingAnNonExistentMovieThrowsIllegalArgumentException() throws RemoteException {
74      MovieDatabase db = getMovieDatabase();

76    }

78    @Test
79    public void testDeleteMovie() throws RemoteException {
80      MovieDatabase db = getMovieDatabase();
81      long movieId = db.createMovie("Movie 1", 2015);
82      db.deleteMovie(movieId);
83      assertThat(db.getMovie(movieId), nullValue());
84    }
