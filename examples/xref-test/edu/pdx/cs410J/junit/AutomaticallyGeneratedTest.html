16  /**
17   * The class demonstrates how JUnit 5's {@link DynamicTest} class can be used to generate test cases
18   * dynamically.
19   */
20  public class AutomaticallyGeneratedTest
21  {
22      @BeforeAll
23      public static void setUp() {
24          System.out.println("BeforeClass");
25      }
26  
27      @AfterAll
28      public static void tearDown() {
29          System.out.println("AfterClass");
30      }
31  
32      @TestFactory
33      Stream<DynamicTest> generateDynamicTests() {
34          return getTestNames().stream().map(name -> dynamicTest(name, () -> new GeneratedTest(name)));
35      }
36  
37      public static List<String> getTestNames() {
38          return Arrays.asList( "One", "Two", "Three" );
39      }
40  
41      public static class GeneratedTest
42      {
43          private final String name;
44  
45          public GeneratedTest(String name) {
46              this.name = name;
47          }
48  
49          public String getName() {
50              return name;
51          }
52  
53          public void run()
54          {
55              assertNotNull(getName());
56              // assertEquals( 3, getName().length() );
57          }
58      }
