11  /**
12   * A main program that launches a book store using the "production" {@link edu.pdx.cs410J.di.BookInventory} and
13   * {@link edu.pdx.cs410J.di.CreditCardService}
14   */
15  public class BookStoreApp
16  {
17     public static void main(String... args) throws JAXBException, IOException
18     {
19         String tmpdir = System.getProperty( "java.io.tmpdir" );
20         File directory = new File( tmpdir );
21         BookInventory inventory = new BookDatabase( directory );
22         addBooks(inventory);
23         CreditCardService cardService = new FirstBankOfPSU( "localhost", 8080 );
24  
25         Logger logger = Logger.getLogger( "edu.pdx.cs410J.Logger" );
26         logger.setLevel( Level.INFO );
27  
28         CheckoutPanel panel = new CheckoutPanel( inventory, cardService, logger );
29         BookStoreGUI gui = new BookStoreGUI( panel );
30         gui.pack();
31         gui.setVisible( true );
32     }
33  
34      protected static void addBooks( BookInventory inventory )
35      {
36          addBook( inventory, new Book( "The Pragmatic Programmer", "Andrew Hunt", 29.95 ), 4 );
37          addBook( inventory, new Book("Winning", "Jack Welch", 15.95), 3 );
38          addBook( inventory, new Book( "Agile Estimating and Planning", "Mike Cohn", 35.99), 2);
39      }
40  
41      private static void addBook( BookInventory inventory, Book book, int copies )
42      {
43          inventory.add( Collections.nCopies( copies, book ).toArray(new Book[copies]) );
44      }
45  }
