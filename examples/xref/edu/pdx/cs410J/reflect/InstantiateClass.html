3   /**
4    * This program reads the name of a class with a zero-argument
5    * constructor and uses Java reflection to load and instantiate the
6    * class.
7    */
8   public class InstantiateClass {
9   
10    public static void main(String[] args) {
11      ClassLoader cl = ClassLoader.getSystemClassLoader();
12      try {
13        Class c = cl.loadClass(args[0]); 
14        Object o = c.newInstance();
15        System.out.println(o);
16  
17      } catch (InstantiationException ex) {
18        String s = "Could not instantiate " + args[0];
19        System.err.println(s);
20  
21      } catch (IllegalAccessException ex) {
22        String s = "No public constructor for " + args[0];
23        System.err.println(s);
24  
25      } catch (ClassNotFoundException ex) {
26        String s = "Could not find " + args[0];
27        System.err.println(s);
28      }
29    }
30  
31  }
