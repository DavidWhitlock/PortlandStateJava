3   import java.io.*;
4   import java.security.*;
5   import java.security.spec.*;
6   
7   /**
8    * This program verifies that a given message was signed by the person
9    * with a given public key.
10   */
11  public class VerifyMessage {
12    public static void main(String[] args) {
13      String fileName = args[0];
14      String digestName = args[1];
15      String message = args[2];
16  
17      try {
18        // Read in key from file
19        FileInputStream fis = new FileInputStream(fileName);
20        byte[] encodedKey = new byte[fis.available()];
21        fis.read(encodedKey);
22        fis.close();
23  
24        X509EncodedKeySpec spec =
25          new X509EncodedKeySpec(encodedKey);
26        KeyFactory factory =
27          KeyFactory.getInstance("DSA", "SUN");
28        PublicKey publicKey = factory.generatePublic(spec);
29  
30        // Read the digest
31        fis = new FileInputStream(digestName);
32        byte[] digest = new byte[fis.available()];
33        fis.read(digest);
34        fis.close();
35  
36        // Compute DSA signature
37        Signature sig = Signature.getInstance("DSA");
38  
39        sig.initVerify(publicKey);
40        sig.update(message.getBytes());
41        if (sig.verify(digest)) {
42  	System.out.println("Success");
43        } else {
44  	System.out.println("Failure");
45        }
46        return;
47  
48      } catch (IOException ex) {
49        ex.printStackTrace(System.err);
50  
51      } catch (NoSuchProviderException ex) {
52        ex.printStackTrace(System.err);
53  
54      } catch (NoSuchAlgorithmException ex) {
55        ex.printStackTrace(System.err);
56  
57      } catch (InvalidKeySpecException ex) {
58        ex.printStackTrace(System.err);
59  
60      } catch (SignatureException ex) {
61        ex.printStackTrace(System.err);
62  
63      } catch (InvalidKeyException ex) {
64        ex.printStackTrace(System.err);
65      }
66  
67      System.exit(1);
68    }
69  }
