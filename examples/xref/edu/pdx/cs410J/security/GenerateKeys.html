3   import java.io.*;
4   import java.security.*;
5   
6   /**
7    * This program generates a pair of DSA keys in files named
8    * <code>public.key</code> and <code>private.key</code>
9    */
10  public class GenerateKeys {
11    public static void main(String[] args) {
12      KeyPairGenerator gen = null;
13  
14      try {
15        gen = KeyPairGenerator.getInstance("DSA");
16      } catch (NoSuchAlgorithmException ex) {
17        ex.printStackTrace(System.err);
18      }
19  
20      gen.initialize(1024);
21      KeyPair keys = gen.generateKeyPair();
22  
23      PrivateKey privateKey = keys.getPrivate();
24      PublicKey publicKey = keys.getPublic();
25  
26      try {
27        FileOutputStream fos = new FileOutputStream ("private.key");
28        fos.write(privateKey.getEncoded());
29  
30        fos = new FileOutputStream ("public.key");
31        fos.write(publicKey.getEncoded());
32  
33      } catch (IOException ex) {
34        ex.printStackTrace(System.err);
35        System.exit(1);
36      }
37    }
38  }
