3   import java.util.*;
4   
5   /**
6    * This class represents a student that may enroll in a section of a
7    * course.
8    */
9   public class Student {
10  
11    /** The student's id */
12    private String id;
13  
14    /** The grades that the student received in the sections of courses
15     * that he was enrolled in */
16    private Map<Section, Grade> grades;
17  
18    /////////////////////////  Constructors  //////////////////////////
19  
20    /**
21     * Creates a new <code>Student</code> with a given id
22     */
23    public Student(String id) {
24      this.id = id;
25      this.grades = new HashMap<Section, Grade>();
26    }
27  
28    //////////////////////  Accessor Methods  ////////////////////////
29  
30    /**
31     * Sets the grade this student got in a given section of a course
32     */
33    public void setGrade(Section section, Grade grade) {
34      this.grades.put(section, grade);
35    }
36  
37    /**
38     * Returns the grade this student received in a given sesion of a
39     * course.  Returns <code>null</code> if the student was never
40     * enrolled in the course or is no grade was assigned.
41     */
42    public Grade getGrade(Section section) {
43      return (Grade) this.grades.get(section);
44    }
45  
46    /**
47     * Returns the grades that this student received 
48     */
49    public Map<Section, Grade> getGrades() {
50      return grades;
51    }
52  
53    /**
54     * Returns the id of this student
55     */
56    public String getId() {
57      return id;
58    }
59  }
