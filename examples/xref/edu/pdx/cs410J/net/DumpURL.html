3   import java.io.*;
4   import java.net.*;
5   
6   /**
7    * This program dumps the contents of a URL to standard out.
8    */
9   public class DumpURL {
10    private static PrintStream err = System.err;
11  
12    /**
13     * Read the URL from the command line.
14     */
15    public static void main(String[] args) {
16      URL url = null;
17      try {
18        url = new URL(args[0]);
19  
20      } catch (MalformedURLException ex) {
21        err.println("** Bad URL: " + args[0]);
22        System.exit(1);
23      }
24  
25      try {
26        InputStream urlStream = url.openStream();
27        InputStreamReader isr = new InputStreamReader(urlStream);
28        BufferedReader br = new BufferedReader(isr);
29  
30        while (br.ready()) {
31  	String line = br.readLine();
32  	System.out.println(line);
33        }
34  
35        br.close();
36  
37      } catch (IOException ex) {
38        err.println("** IOException: " + ex);
39        System.exit(1);
40      }
41    }
42  
43  }
