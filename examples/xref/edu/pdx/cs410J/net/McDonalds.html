3   /**
4    * This class models a <code>McDonalds</code>.  There are a bunch of
5    * <code>McCustomer</code>s who all want a BigMac(tm).  There are a
6    * bunch of liberal arts majors, er, <code>McEmployee</code>s who cook
7    * the BigMacs(tm).  Each <code>McCustomer</code> and
8    * <code>McEmployee</code> runs in his or her own thread.
9    */
10  public class McDonalds {
11    private static java.io.PrintStream err = System.err;
12    private int nBigMacs;
13  
14    /**
15     * Creates a new <code>McDonalds</code> with a given number of
16     * BigMacs to cook.
17     */
18    public McDonalds(int nBigMacs) {
19      this.nBigMacs = nBigMacs;
20    }
21  
22    /**
23     * Returns <code>true</code> if there are more BigMacs to cook.
24     */
25    public synchronized boolean moreBigMacs() {
26      if (this.nBigMacs <= 0) {
27        return false;
28  
29      } else {
30        this.nBigMacs--;
31        return true;
32      }
33    }
34  
35    /**
36     * Read the number of <code>McCustomer</code>s and the number of
37     * <code>McEmployee</code>s from the command line.  Spin off threads
38     * for each one and what minimum wage at work.
39     */
40    public static void main(String[] args) {
41      int nCustomers = 0;
42      int nEmployees = 0;
43  
44      try {
45        nCustomers = Integer.parseInt(args[0]);
46        nEmployees = Integer.parseInt(args[1]);
47  
48      } catch (NumberFormatException ex) {
49        err.println("** NumberFormatException");
50        System.exit(1);
51      }
52  
53      // Each customer wants a BigMac(tm)
54      McDonalds mcDonalds = new McDonalds(nCustomers);
55  
56      // The customers enter...
57      for (int i = 0; i < nCustomers; i++) {
58        McCustomer customer = new McCustomer(i, mcDonalds);
59        (new Thread(customer)).start();
60      }
61  
62      // The employees start cooking...
63      for (int i = 0; i < nEmployees; i++) {
64        McEmployee employee = new McEmployee(i, mcDonalds);
65        (new Thread(employee)).start();
66      }
67  
68      // Our work here is done.
69    }
70  }
