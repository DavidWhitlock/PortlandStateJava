3   import java.io.*;
4   import java.net.*;

6   /**
7    * A client that connects to a {@link DateServer} and reads the {@link
8    * java.util.Date} sent to it as a <code>String</code>.  The important
9    * things to note are the ports that are used to communicate.
10   *
11   * @author David Whitlock
12   * @since Fall 2005
13   */
14  public class DateClient {
15  
16    private static final PrintStream out = System.out;
17    private static final PrintStream err = System.err;
18  
19    /**
20     * Connects to a server and receives the date.
21     */
22    public static void main(String[] args) {
23      String host = args[0];
24      int port = Integer.parseInt(args[1]);
25  
26      try {
27        out.println("Client connecting to " + host + ":" + port);
28        Socket socket = new Socket(host, port);
29  
30        out.println("Client running on " + socket.getLocalAddress() +
31                    ":" + socket.getLocalPort());
32        out.println("Client communicating with " + socket.getInetAddress() 
33                    + ":" + socket.getPort());
34  
35        InputStream is = socket.getInputStream();
36        InputStreamReader isr = new InputStreamReader(is);
37        BufferedReader br = new BufferedReader(isr);
38  
39        out.println("Client read " + br.readLine());
40        out.close();
41  
42      } catch (IOException ex) {
43        err.println("** IOException: " + ex);
44        System.exit(1);
45      }
46    }
47  
48  }
