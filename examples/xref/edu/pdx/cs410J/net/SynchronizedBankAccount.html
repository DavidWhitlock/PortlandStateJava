3   /**
4    * Synchronized methods ensure that the data in the balance is
5    * accessed correctly.
6    */
7   public class SynchronizedBankAccount extends BankAccount {
8     private static int nextId = 1;
9     int id = nextId++;
10  
11    public synchronized int getBalance() {
12      return super.getBalance();
13    }
14  
15    public synchronized void setBalance(int balance) {
16      super.setBalance(balance);
17    } 
18  
19    public synchronized void doTransaction(int trans) {
20      // Will not attempt to re-obtain lock
21      int balance = this.getBalance();
22      balance += trans;
23      this.setBalance(balance);
24    }
25  }
