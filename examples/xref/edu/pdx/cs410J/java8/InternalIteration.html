9     public void evolutionOfIteration(String[] arrayOfStrings, List<String> listOfStrings) {
10      for (int i = 0; i < arrayOfStrings.length; i++) {
11        String string = arrayOfStrings[i];
12        System.out.println(string);
13      }
14  
15      Iterator<String> iter = listOfStrings.iterator();
16      while(iter.hasNext()) {
17        String string = iter.next();
18        System.out.println(string);
19      }
20  
21      for(String string : listOfStrings) {
22        System.out.println(string);
23      }
24  
25      listOfStrings.forEach(new Consumer<String>() {
26        @Override
27        public void accept(String string) {
28          System.out.print(string);
29        }
30      });
31
