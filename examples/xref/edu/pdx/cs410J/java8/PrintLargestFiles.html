3   import java.io.IOException;
4   import java.io.UncheckedIOException;
5   import java.nio.file.Files;
6   import java.nio.file.Path;
7   import java.nio.file.Paths;
8   import java.util.Comparator;
9   import java.util.stream.Stream;
10  
11  public class PrintLargestFiles {
12  
13    public static void main(String[] args) throws IOException {
14      Path path = Paths.get(args[0]);
15  
16      printLargestFiles(path);
17    }
18  
19    private static void printLargestFiles(Path root) throws IOException {
20      Stream<Path> stream = Files.walk(root);
21      stream
22        .sorted(new FileSizeComparator().reversed())
23        .limit(10)
24        .forEach(PrintLargestFiles::printFileAndSize);
25    }
26  
27    private static void printFileAndSize(Path path) {
28      try {
29        System.out.println(Files.size(path) + " " + path);
30      } catch (IOException e) {
31        throw new UncheckedIOException(e);
32      }
33  
34    }
35  
36    private static class FileSizeComparator implements Comparator<Path> {
37      @Override
38      public int compare(Path o1, Path o2) {
39        try {
40          long size1 = Files.size(o1);
41          long size2 = Files.size(o2);
42          return size1 > size2 ? 1 : (size2 > size1 ? -1 : 0);
43        } catch (IOException ex) {
44          throw new UncheckedIOException(ex);
45        }
46      }
47    }
48  }
