3   import java.io.*;
4   import java.text.*;
5   import java.util.*;
6   
7   /**
8    * This class shows off some of Java's internationalization (i18n)
9    * capabilities using the <code>Locale</code> class.
10   */
11  public class AroundTheWorld {
12  
13    private static PrintWriter out = new PrintWriter(System.out, true);
14    private static PrintWriter err = new PrintWriter(System.err, true);
15  
16    private static void usage() {
17      err.println("usage: java AroundTheWorld [options]");
18      err.println("  -country code      Which country?");
19      err.println("  -language code     Which language?");
20      err.println("  -timeZone code     Which time zone?");
21      err.println("If any code is ??, then all possible values for " +
22  		"that code are printed");
23      System.exit(1);
24    }
25  
26    /**
27     * Prints the available locales formatted using a given locale.
28     */
29    private static void printAvailableLocales(Locale locale) {
30      Locale[] locales = Locale.getAvailableLocales();
31      out.println("Available locales:");
32      for (int i = 0; i < locales.length; i++) {
33        Locale l = locales[i];
34        out.println("Locale " + i + " of " + locales.length);
35        out.println("  Name: " + l.getDisplayName(locale));
36        out.println("  Country: " + l.getDisplayCountry(locale) + " (" +
37  		  l.getCountry() + ")");
38        out.println("  Language: " + l.getDisplayLanguage(locale) + 
39  		  " (" + l.getLanguage() + ")");
40        out.println("  Variant: " + l.getDisplayVariant(locale) + " (" +
41  		  l.getVariant() + ")");
42      }
43    }
44  
45    /**
46     * Prints the codes for all available time zones formatted for a
47     * given locale.
48     */
49    private static void printTimeZones(Locale locale) {
50      String[] codes = TimeZone.getAvailableIDs();
51      out.println("Time zones");
52      for (int i = 0; i < codes.length; i++) {
53        String code = codes[i];
54        TimeZone tz = TimeZone.getTimeZone(code);
55        out.println("  " + code + ": " + tz.getDisplayName(locale));
56      }
57    }
58  
59    /**
60     * Prints out today's date and time in a specific locale.
61     */
62    private static void printToday(Locale locale, TimeZone tz) {
63      DateFormat df = 
64        DateFormat.getDateTimeInstance(DateFormat.FULL, DateFormat.FULL,
65  				     locale);
66      df.setTimeZone(tz);
67      out.println("Right now: " + df.format(new Date()));
68    }
69  
70    /**
71     * Prints out a number in various locale-specific formats
72     */
73    private static void printNumber(double number, Locale locale) {
74      NumberFormat nf;
75  
76      nf = NumberFormat.getNumberInstance(locale);
77      out.println("A number: " + nf.format(number));
78  
79      nf = NumberFormat.getCurrencyInstance(locale);
80      out.println("Currency: " + nf.format(number));
81  
82      nf = NumberFormat.getPercentInstance(locale);
83      out.println("Percent: " + nf.format(number));
84    }
85  
86    /**
87     * Parses the command line to determine which locale to display
88     * information in.  Prints out dates, times, and currencies using
89     * various locales.
90     */
91    public static void main(String[] args) {
92      String countryCode = null;
93      String languageCode = null;
94      String timeZoneCode = null;
95  
96      // Parse the command line
97      for (int i = 0; i < args.length; i++) {
98        if (args[i].equals("-country")) {
99  	if(++i >= args.length) {
100 	  err.println("** Missing country code");
101 	  usage();
102 	}
103 
104 	countryCode = args[i];
105 
106       } else if (args[i].equals("-language")) {
107 	if(++i >= args.length) {
108 	  err.println("** Missing language code");
109 	  usage();
110 	}
111 
112 	languageCode = args[i];
113 
114       } else if (args[i].equals("-timeZone")) {
115 	if(++i >= args.length) {
116 	  err.println("** Missing time zone code");
117 	  usage();
118 	}
119 
120 	timeZoneCode = args[i];
121 
122       } else {
123 	err.println("** Unknown option: " + args[i]);
124 	usage();
125       }
126     }
127 
128     Locale locale;
129 
130     if (countryCode == null) {
131       // Use default
132       locale = Locale.getDefault();
133 
134     } else if (countryCode.equals("??")) {
135       locale = Locale.getDefault();
136       printAvailableLocales(locale);
137 
138     } else if (languageCode == null) {
139       // Use default
140       locale = Locale.getDefault();
141 
142     } else if (languageCode.equals("??")) {
143       locale = Locale.getDefault();
144       printAvailableLocales(locale);
145 
146     } else {
147       locale = new Locale(languageCode, countryCode);
148     }
149 
150     TimeZone timeZone;
151     if (timeZoneCode == null) {
152       // Use default
153       timeZone = TimeZone.getDefault();
154 
155     } else if (timeZoneCode.equals("??")) {
156       timeZone = TimeZone.getDefault();
157       printTimeZones(locale);
158 
159     } else {
160       timeZone = TimeZone.getTimeZone(timeZoneCode);
161     }
162 
163     // Print out some interesting info
164     out.println("");
165     printToday(locale, timeZone);
166     printNumber(1234.56, locale);
167     out.println("");
168 
169   }
170 
171 }
