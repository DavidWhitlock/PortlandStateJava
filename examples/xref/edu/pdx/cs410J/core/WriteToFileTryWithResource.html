3   import java.io.FileWriter;
4   import java.io.IOException;
5   import java.io.PrintWriter;
6   import java.io.Writer;
7   
8   /**
9    * This program does the same thing as <code>WriteToFile</code> but uses the new
10   * "try with resource" language feature of Java 7.
11   */
12  public class WriteToFileTryWithResource {
13  
14    /**
15     * The first argument is the file to write to.
16     */
17    public static void main(String[] args) {
18      // Wrap a PrintWriter around System.err
19      PrintWriter err = new PrintWriter(System.err, true);
20  
21      try (Writer writer = new FileWriter(args[0])) {
22        // Write the command line arguments to the file
23        for (int i = 1; i < args.length; i++) {
24        	writer.write(args[i]);
25        	writer.write('\n');
26        }
27      
28        // All done
29        writer.flush();
30        writer.close();
31  
32      } catch (IOException ex) {
33        err.println("** " + ex);
34      }
35    }
36  }
