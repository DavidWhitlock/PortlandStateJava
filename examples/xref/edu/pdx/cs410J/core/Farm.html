12    /**
13     * Prints the contents of a <code>Map</code>.
14     */
15    private static void print(Map<String, Animal> map) {
16      for (String key : map.keySet()) {
17        Object value = map.get(key);
18  
19        String s = key + " -> " + value;
20        System.out.println(s);
21      }
22    }
23  
24    /**
25     * Create a <code>Map</code> and print it.
26     */
27    public static void main(String[] args) {
28      Map<String, Animal> farm = new HashMap<String, Animal>();
29      farm.put("Old MacDonald", new Human("Old MacDonald"));
30      farm.put("Bossie", new Cow("Bossie"));
31      farm.put("Clyde", new Sheep("Clyde"));
32      farm.put("Louise", new Duck("Louise"));
33  
34      print(farm);
35    }
