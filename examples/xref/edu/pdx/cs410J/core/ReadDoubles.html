3   import java.io.*;
4   
5   /**
6    * This program takes a file whose name is given on the command line
7    * and reads <code>double</code>s from it.  It demonstrates the
8    * <code>FileInputStream</code> and <code>DataInputStream</code>
9    * classes.
10   */
11  public class ReadDoubles {
12    static PrintStream out = System.out;
13    static PrintStream err = System.err;
14  
15    /**
16     * The one command line argument is the name of the file to read.
17     */
18    public static void main(String args[]) {
19      FileInputStream fis = null;
20      try {
21        fis = new FileInputStream(args[0]);
22  
23      } catch (FileNotFoundException ex) {
24        err.println("** No such file: " + args[0]);
25      }
26  
27      DataInputStream dis = new DataInputStream(fis);
28      while (true) {
29        try {
30  	double d = dis.readDouble();
31  	out.print(d + " ");
32  	out.flush();
33  
34        } catch (EOFException ex) {
35  	// All done reading
36  	break;
37  
38        } catch (IOException ex) {
39  	err.println("** " + ex);
40  	System.exit(1);
41        }
42      }
43      out.println("");
44    }
45  }
