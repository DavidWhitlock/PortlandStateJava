6   /**
7    * This class is a <code>Comparator</code> that compares
8    * <code>Person</code>s based on their shoe size.
9    */
10  public class PersonComparator implements Comparator<Person> {
11  
12    public int compare(Person o1, Person o2) {
13      double size1 = o1.shoeSize();
14      double size2 = o2.shoeSize();
15  
16      if (size1 > size2) {
17        return 1;
18  
19      } else if (size1 < size2) {
20        return -1;
21  
22      } else {
23        return 0;
24      }
25    }
26  
27    /**
28     * Creates a set of <code>Person</code>s and prints out the
29     * contents.
30     */
31    public static void main(String[] args) {
32      Set<Person> set = new TreeSet<Person>(new PersonComparator());
33      set.add(new Person("Quan", 10.5));
34      set.add(new Person("Jerome", 11.0));
35      set.add(new Person("Dave", 10.5));
36      set.add(new Person("Nandini", 8.0));
37  
38      // Print out the people
39      Iterator iter = set.iterator();
40      while (iter.hasNext()) {
41        Person p = (Person) iter.next();
42        System.out.println(p);
43      }
44    }
45  
46  }
