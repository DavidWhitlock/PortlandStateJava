7   /**
8    * This program reads in {@link Date}s from standard in using {@link
9    * DateFormat#SHORT}.  Then it adds them to a {@link SortedSet} to see
10   * how their sorted.  One of my students claimed that {@link
11   * DateFormat#SHORT} didn't parse 4-digit dates correct.  This test
12   * will see if she was right.
13   *
14   * @author David Whitlock
15   * @version $Revision: 1.2 $
16   */
17  public class SortDateFormatSHORT {
18  
19    public static void main(String[] args) throws IOException {
20      int f = DateFormat.SHORT;
21      DateFormat df = DateFormat.getDateTimeInstance(f, f);
22      df.setLenient(false);
23  
24      System.out.println("An example of DateFormat.SHORT: " +
25                         df.format(new Date()));
26      System.out.println("\nEnter some dates to sort:");
27  
28      SortedSet<Date> sorted = new TreeSet<Date>();
29      BufferedReader br =
30        new BufferedReader(new InputStreamReader(System.in));
31      while (true) {
32        String line = br.readLine();
33        try {
34          sorted.add(df.parse(line));
35  
36        } catch (ParseException ex) {
37          break;
38        }
39      }
40  
41      System.out.println(sorted.size() + " sorted dates:");
42      for (Iterator iter = sorted.iterator(); iter.hasNext(); ) {
43        System.out.print("  ");
44        System.out.println(iter.next());
45      }
46    }
47  
48  }
