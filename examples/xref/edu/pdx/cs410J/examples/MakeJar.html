3   import java.io.*;
4   import java.util.*;
5   import java.util.jar.*;
6   import java.util.zip.*;
7   
8   /**
9    * This class demonstrates the file compression utilities in the
10   * <code>java.util.zip</code> and <code>java.util.jar</code> packages
11   * by creating a jar file whose contents are specified on the command
12   * line.
13   *
14   * @author David Whitlock
15   */
16  public class MakeJar {
17  
18    private static PrintWriter err = new PrintWriter(System.err, true);
19  
20    /**
21     * Prints out information about how this program is used.
22     */
23    private static void usage() {
24      err.println("usage: MakeJar [options] jarFile [file]+");
25      err.println("  Where [options] are:");
26      err.println("  -nocompress    Don't compress Jar file");
27      err.println("  -author name   Author of Jar file (default: CS410J)");
28      err.println("  -version n     Version of Jar file " +
29  		"(default: 1.0)");
30      System.exit(1);
31    }
32  
33    /**
34     * Reads the name of the Jar file followed by the names of the files
35     * to be added to the jar file from the command line.
36     */
37    public static void main(String[] args) {
38      String jarFileName = null;
39      Set<String> fileNames = new HashSet<String>();
40      boolean compress = true;
41      String author = "CS410J";   // Author of Jar file
42      String version = "1.0";     // Version of Jar file
43  
44      // Parse the command line
45      for (int i = 0; i < args.length; i++) {
46        if (args[i].equals("-nocompress")) {
47  	compress = false;
48  
49        } else if (args[i].equals("-author")) {
50  	if(++i >= args.length) {
51  	  err.println("** Missing author name");
52  	  usage();
53  	}
54  
55  	author = args[i];
56  
57        } else if (args[i].equals("-version")) {
58  	if(++i >= args.length) {
59  	  err.println("** Missing version");
60  	  usage();
61  	}
62  
63  	version = args[i];
64  
65        } else if (jarFileName == null) {
66  	jarFileName = args[i];
67  
68        } else {
69  	// Add this file to the Jar
70  	fileNames.add(args[i]);
71        }
72      }
73  
74      if (jarFileName == null) {
75        err.println("** No Jar file specified");
76        usage();
77      }
78  
79      if (fileNames.isEmpty()) {
80        err.println("** No files specified");
81        usage();
82      }
83  
84      // Make note of the author and version in the Manifest file in the
85      // Jar
86      Manifest manifest = new Manifest();
87      Attributes global = manifest.getMainAttributes();
88      global.put(Attributes.Name.MANIFEST_VERSION, version);
89      global.put(new Attributes.Name("Created-By"), author);
90  
91      // Create a JarOutputStream around the jar file
92      JarOutputStream jos = null;
93      try {
94        File jarFile = new File(jarFileName);
95        OutputStream os = new FileOutputStream(jarFile);
96        jos = new JarOutputStream(os, manifest);
97   
98      } catch (IOException ex) {
99        err.println("** IOException: " + ex.getMessage());
100       System.exit(1);
101     }
102 
103     if (compress) {
104       jos.setMethod(JarOutputStream.DEFLATED);
105     } else {
106       jos.setMethod(JarOutputStream.STORED);
107     }
108 
109     // Now open all the files and add them to the JAR file
110     Iterator names = fileNames.iterator();
111     while (names.hasNext()) {
112       String fileName = (String) names.next();
113       try {
114 	File file = new File(fileName);
115 
116 	JarEntry entry = new JarEntry(fileName);
117 	entry.setTime((new Date()).getTime());
118 	entry.setSize(file.length());
119 
120 	InputStream is = 
121 	  new BufferedInputStream(new FileInputStream(file));
122 	byte[] buffer = new byte[1024];
123 	int read = 0;
124 
125 	if(compress) {
126 	  entry.setMethod(JarEntry.DEFLATED);
127 
128 	} else {
129 	  entry.setMethod(JarEntry.STORED);
130 
131 	  // Compute the checksum of the file using CRC32
132 	  CRC32 checksum = new CRC32();
133 	  checksum.reset();
134 	  long total = 0;
135 	  while ((read = is.read(buffer)) != -1) {
136 	    checksum.update(buffer, 0, read);
137 	    total += read;
138 	  }
139 	  if (total != file.length()) {
140 	    throw new JarException("File length problems during " +
141 				   file.getPath() + " (" + total + 
142 				   " out of " + file.length() + ")");
143 	  }
144 	  entry.setCrc(checksum.getValue());
145 	}
146 
147 	// Add the entry to the JAR file
148 	jos.putNextEntry(entry);
149 	is = new BufferedInputStream(new FileInputStream(file));
150 	while((read = is.read(buffer, 0, buffer.length)) != -1) {
151 	  jos.write(buffer, 0, read);
152 	}
153 	is.close();
154 	jos.closeEntry();
155 
156       } catch (IOException ex) {
157 	err.println("** IOException: " + ex.getMessage());
158 	System.exit(1);
159       }
160     }
161 
162   }
