4   import java.util.*;

6   /**
7    * This class is a <code>Comparator</code> that compares
8    * <code>Object</code>s based on the name of their classes.
9    */
10  public class ClassComparator implements Comparator<Object> {
11    public int compare(Object o1, Object o2) {
12      String name1 = o1.getClass().getName();
13      String name2 = o2.getClass().getName();
14      
15      // Take advantage of String's compartTo method
16      return name1.compareTo(name2);
17    }
18  
19    /**
20     * Create a bunch of object and store them in a set using a
21     * <code>ClassComparator</code>. 
22     */
23    public static void main(String[] args) {
24      Set<Object> set = new TreeSet<Object>(new ClassComparator());
25      set.add("Hello");
26      set.add(new Cow("Betty"));
27      set.add(new Vector());
28      set.add(new Ant("Richard"));
29  
30      // Print out the set
31      Iterator iter = set.iterator();
32      while (iter.hasNext()) {
33        Object o = iter.next();
34        String s = o + " (" + o.getClass().getName() + ")";
35        System.out.println(s);
36      }
37    }
38  }
