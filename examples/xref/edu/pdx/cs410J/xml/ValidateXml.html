1   package edu.pdx.cs410J.xml;
2   
3   import java.io.*;
4   import javax.xml.parsers.*;
5   import org.xml.sax.*;
6   import org.xml.sax.helpers.DefaultHandler;
7   
8   /**
9    * This program parses and XML file and validates it for correctness
10   * against its DTD.
11   */
12  public class ValidateXml extends DefaultHandler {
13  
14    private static PrintStream err = System.err;
15  
16    public void error(SAXParseException ex) {
17      err.println("ERROR at (" + ex.getLineNumber() + ":" +
18                  ex.getColumnNumber() + "): " + ex.getMessage());
19    }
20  
21    public void fatalError(SAXParseException ex) {
22      err.println("FATAL at (" + ex.getLineNumber() + ":" +
23                  ex.getColumnNumber() + "): " + ex.getMessage());
24    }
25  
26    public void warning(SAXParseException ex) {
27      err.println("WARNING at (" + ex.getLineNumber() + ":" +
28                  ex.getColumnNumber() + "): " + ex.getMessage());
29    }
30  
31    public static void main(String[] args) {
32      try {
33        SAXParserFactory factory = SAXParserFactory.newInstance();
34        factory.setValidating(true);
35  
36        SAXParser parser = factory.newSAXParser();
37        parser.parse(new File(args[0]), new ValidateXml());
38  
39      } catch (ParserConfigurationException ex) {
40        ex.printStackTrace(System.err);
41  
42      } catch (SAXException ex) {
43        err.println("Parsing exception: " + ex);
44        
45      } catch (IOException ex) {
46        err.println("IOException: " + ex);
47      }
48    }
49  }
