001package edu.pdx.cs410J.di;
002
003import javax.xml.bind.JAXBContext;
004import javax.xml.bind.JAXBException;
005import javax.xml.bind.Marshaller;
006import javax.xml.bind.Unmarshaller;
007import java.io.File;
008import java.io.IOException;
009
010/**
011 * The superclass of classes that store objects serialized to a file using JAXB
012 */
013public abstract class JaxbDatabase
014{
015    private final File directory;
016    private final String fileName;
017    private final JAXBContext xmlContext;
018    private final File file;
019
020    protected JaxbDatabase( File directory, String fileName, Class<?>... jaxbClasses )
021        throws IOException, JAXBException
022    {
023        this.directory = directory;
024        this.fileName = fileName;
025        this.xmlContext = JAXBContext.newInstance( jaxbClasses );
026
027        this.directory.mkdirs();
028        if (!this.directory.exists()) {
029          throw new IOException( "Could not create data directory: " + this.directory);
030        }
031
032
033        this.file = new File(this.directory, this.fileName);
034    }
035
036    /**
037     * Writes an object as XML to the data file
038     * @param xml The object to marshal
039     */
040    protected void writeXml( Object xml )
041    {
042        try
043        {
044            Marshaller marshaller = this.xmlContext.createMarshaller();
045            marshaller.marshal( xml, new File(this.directory, this.fileName) );
046        }
047        catch ( JAXBException ex )
048        {
049            throw new IllegalStateException( "Could not save inventory", ex);
050        }
051    }
052
053    public File getDatabaseFile() {
054        return file;
055    }
056
057    /**
058     * Read the XML data file and returns the unmarshalled object
059     * @return The object in the XML file or null if the file doesn't exist
060     * @throws JAXBException if we can't read the file
061     */
062    protected Object readFile()
063        throws JAXBException
064    {
065        System.out.println("Reading xml data from " + file );
066
067        if ( this.file.exists()) {
068          Unmarshaller unmarshaller = this.xmlContext.createUnmarshaller();
069          return unmarshaller.unmarshal( this.file );
070        } else {
071          return null;
072        }
073    }
074}
