001package edu.pdx.cs410J.xml;
002
003import java.io.*;
004import java.net.*;
005import javax.xml.parsers.*;
006import javax.xml.transform.*;
007import javax.xml.transform.dom.DOMSource;
008import javax.xml.transform.stream.StreamResult;
009import org.w3c.dom.*;
010
011/**
012 * This program builds a DOM tree that represents a phone book and
013 * writes it to a file.
014 */
015public class BuildPhonebook {
016  private static PrintStream err = System.err;
017
018  public static void main(String[] args) {
019    Document doc = null;
020
021    String publicID = null;  // Who cares?
022    String systemID = null;
023    try {
024      File dtd = new File("phonebook.dtd");
025      systemID = dtd.toURL().toString();
026
027    } catch (MalformedURLException ex) {
028      err.println("** Bad URL: " + ex);
029      System.exit(1);
030    }
031
032    // Create an empty Document
033    try {
034      DocumentBuilderFactory factory =
035        DocumentBuilderFactory.newInstance();
036      factory.setValidating(true);
037
038      DocumentBuilder builder = factory.newDocumentBuilder();
039
040      DOMImplementation dom =
041        builder.getDOMImplementation();
042      DocumentType docType = 
043        dom.createDocumentType("phonebook", publicID, systemID);
044      doc = dom.createDocument(null, "phonebook", docType);
045
046    } catch (ParserConfigurationException ex) {
047      ex.printStackTrace(System.err);
048      System.exit(1);
049
050    } catch (DOMException ex) {
051      // Eep, this is bad
052      ex.printStackTrace(System.err);
053      System.exit(1);
054    }
055    
056    // Construct the DOM tree
057    try {
058      Element root = doc.getDocumentElement();
059
060      Element biz = doc.createElement("business");
061      root.appendChild(biz);
062
063      Element name = doc.createElement("name");
064      biz.appendChild(name);
065      String br = "CollegeNET, Inc.";
066      name.appendChild(doc.createTextNode(br));
067      
068      Element address = doc.createElement("address");
069      biz.appendChild(address);
070
071      Element street1 = doc.createElement("street");
072      address.appendChild(street1);
073      String st1 = "805 SW Broadway";
074      street1.appendChild(doc.createTextNode(st1));
075      
076      Element street2 = doc.createElement("street");
077      address.appendChild(street2);
078      String st2 = "Suite 1600";
079      street2.appendChild(doc.createTextNode(st2));
080
081      Element city = doc.createElement("city");
082      address.appendChild(city);
083      city.appendChild(doc.createTextNode("Portland"));
084
085      Element state = doc.createElement("state");
086      address.appendChild(state);
087      state.appendChild(doc.createTextNode("OR"));
088
089      Element zip = doc.createElement("zip");
090      address.appendChild(zip);
091      zip.appendChild(doc.createTextNode("97205"));
092
093      Element phone = doc.createElement("phone");
094      biz.appendChild(phone);
095      phone.setAttribute("areacode", "503");
096      phone.setAttribute("number", "973-5200");
097
098    } catch (DOMException ex) {
099      ex.printStackTrace(err);
100      System.exit(1);
101    }
102
103    // Write the XML document to the console
104    try {
105      Source src = new DOMSource(doc);
106      Result res = new StreamResult(System.out);
107
108      TransformerFactory xFactory = TransformerFactory.newInstance();
109      Transformer xform = xFactory.newTransformer();
110      xform.setOutputProperty(OutputKeys.INDENT, "yes");
111      xform.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, systemID);
112      xform.transform(src, res);
113
114    } catch (TransformerException ex) {
115      ex.printStackTrace(System.err);
116      System.exit(1);
117    }
118  }
119
120}
