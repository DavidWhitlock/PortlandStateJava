001package edu.pdx.cs410J.xml;
002
003import org.w3c.dom.*;
004
005/**
006 * This class represents a resident (person) whose first and last
007 * name, as well as whose middle initial, is listing in a phone book.
008 * A <code>Resident</code> is constructed from an XML DOM tree.  If we
009 * were doing this for real, we'd want a way of constructing an empty
010 * <code>Resident</code> and filling in its fields.
011 */
012public class Resident extends PhoneBookEntry {
013  protected String firstName;
014  protected String middleInitial;
015  protected String lastName;
016  protected boolean unlisted = false;
017
018  /**
019   * Create a new <code>Resident</code> from a <code>Element</code> in
020   * a DOM tree.
021   */
022  public Resident(Element root) {
023    NodeList elements = root.getChildNodes();
024    for (int i = 0; i < elements.getLength(); i++) {
025      Node node = elements.item(i);
026
027      if (!(node instanceof Element)) {
028        continue;
029      }
030
031      Element element = (Element) node;
032
033      if (element.getNodeName().equals("first-name")) {
034        Node name = element.getFirstChild();
035        Node text = name.getFirstChild();
036        this.firstName = text.getNodeValue();
037
038      } else if (element.getNodeName().equals("initial")) {
039        Node text = element.getFirstChild();
040        this.middleInitial = text.getNodeValue();
041
042      } else if (element.getNodeName().equals("last-name")) {
043        Node name = element.getFirstChild();
044        Node text = name.getFirstChild();
045        this.lastName = text.getNodeValue();
046
047      } else if (element.getNodeName().equals("address")) {
048        fillInAddress(element);
049
050      } else if (element.getNodeName().equals("phone")) {
051        fillInPhone(element);
052      }
053    }
054
055    // Check for the "unlisted" attribute
056    if (root.getAttribute("unlisted").equals("true")) {
057      this.unlisted = true;
058    }
059  }
060
061  public String toString() {
062    StringBuffer sb = new StringBuffer();
063    sb.append(this.firstName + " " + this.middleInitial + " " +
064              this.lastName + "\n");
065    sb.append(super.toString());
066
067    return sb.toString();
068  }
069}
