001package edu.pdx.cs410J.j2se15;

003import java.util.*;
004
005/**
006 * Demonstrates J2SE primitive data "autoboxing" features.
007 *
008 * @author David Whitlock
009 * @version $Revision: 1.1 $
010 * @since Summer 2004
011 */
012public class Autoboxing {
013
014  /**
015   * Performs a bunch of operations that demonstrate autoboxing
016   */
017  public static void main(String[] args) {
018    // Recall that Integer.valueOf returns an Integer
019    int i = Integer.valueOf("123");
020
021    List list = new ArrayList();
022    list.add(i);
023    
024    int j = (Integer) list.get(0);
025  }
026
027}
