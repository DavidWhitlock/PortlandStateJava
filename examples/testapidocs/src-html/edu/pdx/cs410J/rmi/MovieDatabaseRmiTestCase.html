001package edu.pdx.cs410J.rmi;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.AfterClass;
005import org.junit.BeforeClass;
006
007import java.net.MalformedURLException;
008import java.rmi.AlreadyBoundException;
009import java.rmi.NotBoundException;
010import java.rmi.RemoteException;
011import java.rmi.registry.LocateRegistry;
012import java.rmi.registry.Registry;
013import java.rmi.server.UnicastRemoteObject;
014
015public class MovieDatabaseRmiTestCase extends InvokeMainTestCase {
016  static final int RMI_PORT = 7777;
017  static final String RMI_HOST = "localhost";
018  private static MovieDatabaseImpl movieDatabaseImpl;
019  private static Registry rmiRegistry;
020
021  @BeforeClass
022  public static void bindMovieDatabaseIntoRmiRegistry() throws RemoteException, AlreadyBoundException, MalformedURLException {
023    movieDatabaseImpl = new MovieDatabaseImpl();
024    MovieDatabase database = (MovieDatabase) UnicastRemoteObject.exportObject(movieDatabaseImpl, RMI_PORT);
025    rmiRegistry = LocateRegistry.createRegistry(RMI_PORT);
026    rmiRegistry.bind(MovieDatabase.RMI_OBJECT_NAME, database);
027  }
028
029  @AfterClass
030  public static void unbindMovieDatabaseFromRmiRegistry() throws RemoteException, MalformedURLException, AlreadyBoundException, NotBoundException {
031    Registry registry = rmiRegistry;
032    UnicastRemoteObject.unexportObject(movieDatabaseImpl, true);
033    registry.unbind(MovieDatabase.RMI_OBJECT_NAME);
034    UnicastRemoteObject.unexportObject(rmiRegistry, true);
035    movieDatabaseImpl = null;
036  }
037
038  private static Registry getRmiRegistry() throws RemoteException {
039    return LocateRegistry.getRegistry(RMI_HOST, RMI_PORT);
040  }
041
042  static MovieDatabase getMovieDatabase() throws RemoteException, NotBoundException {
043    return (MovieDatabase) getRmiRegistry().lookup(MovieDatabase.RMI_OBJECT_NAME);
044  }
045}
