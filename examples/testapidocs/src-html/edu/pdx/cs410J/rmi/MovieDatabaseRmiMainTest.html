001package edu.pdx.cs410J.rmi;
002
003import com.google.common.collect.Iterables;
004import org.junit.FixMethodOrder;
005import org.junit.Test;
006import org.junit.runners.MethodSorters;
007
008import java.rmi.NotBoundException;
009import java.rmi.RemoteException;
010import java.util.Collection;
011
012import static org.hamcrest.CoreMatchers.containsString;
013import static org.hamcrest.CoreMatchers.equalTo;
014import static org.hamcrest.MatcherAssert.assertThat;
015
016@FixMethodOrder(MethodSorters.NAME_ASCENDING)
017public class MovieDatabaseRmiMainTest extends MovieDatabaseRmiTestCase {
018
019  @Test
020  public void test1CreateMovie() throws RemoteException, NotBoundException {
021    String title = "Black Panther";
022    int year = 2018;
023
024    MainMethodResult result = invokeMain(CreateMovie.class, RMI_HOST, String.valueOf(RMI_PORT), title, String.valueOf(year));
025    assertThat(result.getExitCode(), equalTo(0));
026    assertThat(result.getTextWrittenToStandardOut(), containsString("Created movie"));
027    assertThat(result.getTextWrittenToStandardError(), equalTo(""));
028
029    MovieDatabase database = getMovieDatabase();
030    Collection<Movie> movies = database.getMovies();
031    assertThat(movies.size(), equalTo(1));
032    Movie movie = Iterables.getOnlyElement(movies);
033    assertThat(movie.getTitle(), equalTo(title));
034    assertThat(movie.getYear(), equalTo(year));
035  }
036
037}
