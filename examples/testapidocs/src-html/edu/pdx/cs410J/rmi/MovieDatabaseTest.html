001package edu.pdx.cs410J.rmi;
002
003import org.junit.Test;
004
005import java.rmi.RemoteException;
006import java.util.Iterator;
007import java.util.SortedSet;
008
009import static org.hamcrest.CoreMatchers.nullValue;
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNotNull;
013
014/**
015 * Tests the behavior of the {@link MovieDatabase} class.
016 */
017public class MovieDatabaseTest {
018
019  private MovieDatabase getMovieDatabase() throws RemoteException {
020    return new MovieDatabaseImpl();
021  }
022
023  @Test
024  public void testCreateMovie() throws RemoteException {
025    MovieDatabase db = getMovieDatabase();
026    String title = "Movie 1";
027    int year = 2008;
028    long id = db.createMovie(title, year);
029    Movie movie = db.getMovie(id);
030    assertNotNull(movie);
031    assertEquals(title, movie.getTitle());
032    assertEquals(year, movie.getYear());
033    assertEquals(0, db.getFilmography(99L).size());
034  }
035
036  @Test
037  public void testFilmography() throws RemoteException {
038    long bobId = 56l;
039    long billId = 57l;
040
041    String title1 = "Bob's Movie 1";
042    int year1 = 2007;
043    String title2 = "Bob's Movie 2";
044    int year2 = 2008;
045
046    MovieDatabase db = getMovieDatabase();
047    long id1 = db.createMovie(title1, year1);
048    long id2 = db.createMovie(title2, year2);
049    db.noteCharacter(id1, "Joe", bobId);
050    db.noteCharacter(id2, "Frank", bobId);
051    db.noteCharacter(id2, "Henry", billId);
052
053    assertEquals(0, db.getFilmography(78L).size());
054
055    SortedSet<Movie> bobMovies = db.getFilmography(bobId);
056    assertEquals(2, bobMovies.size());
057    Iterator<Movie> bobIter = bobMovies.iterator();
058    Movie bobMovie1 = bobIter.next();
059    assertEquals(title1, bobMovie1.getTitle());
060    assertEquals(year1, bobMovie1.getYear());
061    Movie bobMovie2 = bobIter.next();
062    assertEquals(title2, bobMovie2.getTitle());
063    assertEquals(year2, bobMovie2.getYear());
064
065    SortedSet<Movie> billMovies = db.getFilmography(billId);
066    assertEquals(1, billMovies.size());
067    Movie billMovie = billMovies.iterator().next();
068    assertEquals(title2, billMovie.getTitle());
069    assertEquals(year2, billMovie.getYear());
070  }
071
072  @Test(expected = IllegalArgumentException.class)
073  public void deletingAnNonExistentMovieThrowsIllegalArgumentException() throws RemoteException {
074    MovieDatabase db = getMovieDatabase();
075    db.deleteMovie(-1);
076  }
077
078  @Test
079  public void testDeleteMovie() throws RemoteException {
080    MovieDatabase db = getMovieDatabase();
081    long movieId = db.createMovie("Movie 1", 2015);
082    db.deleteMovie(movieId);
083    assertThat(db.getMovie(movieId), nullValue());
084  }
085}
