001package edu.pdx.cs410J.junit;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007/**
008 * This class tests the functionality of the <code>Section</code> class
009 */
010public class SectionTest {
011
012  @Test
013  public void testAddStudent() {
014    Student student = new Student("123-45-6789");
015    Course course = new Course("CS", 410, 4);
016    Section section = 
017      new Section(course, Section.SPRING, 2001);
018    section.addStudent(student);
019    assertEquals(1, section.getClassSize());
020  }
021
022  @Test(expected = IllegalArgumentException.class)
023  public void testDropStudentNotEnrolled() {
024    Student student = new Student("123-45-6789");
025    Course course = new Course("CS", 410, 4);
026    Section section = 
027      new Section(course, Section.SPRING, 2001);
028    section.dropStudent(student);
029  }
030
031}
