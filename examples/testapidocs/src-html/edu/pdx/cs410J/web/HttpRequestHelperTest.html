001package edu.pdx.cs410J.web;
002
003import org.junit.Test;
004
005import java.io.IOException;
006import java.net.HttpURLConnection;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertTrue;
010
011/**
012 * Unit tests for the <code>HttpRequestHelper</code> class
013 */
014public class HttpRequestHelperTest {
015
016  private HttpRequestHelper helper = new HttpRequestHelper();
017
018  @Test
019  public void testGet() throws IOException {
020    HttpRequestHelper.Response response = helper.get("http://www.google.com");
021    assertEquals(HttpURLConnection.HTTP_OK, response.getCode());
022    assertTrue(response.getContent().contains("Google"));
023  }
024
025  @Test
026  public void testGetWithParameters() throws IOException {
027    HttpRequestHelper.Response response = helper.get("https://www.google.com/search", "p", "Java");
028    assertEquals(HttpURLConnection.HTTP_OK, response.getCode());
029    assertTrue(response.getContent().contains("Java"));
030
031  }
032}
