46    private long createMovie(String title, String year) throws IOException {
47      Response response = post(getResourceURL(MOVIES), "title", title, "year", year);
48      assertEquals(response.getContent(), HTTP_OK, response.getCode());
49      assertNotNull(response.getContent());
50      return Long.parseLong(response.getContent().trim());
51    }

53    @Test
54    public void testGetAllMovies() throws IOException {
55      String title1 = "testGetAllMovies1";
56      createMovie(title1, "2007");
57      String title2 = "testGetAllMovies2";
58      createMovie(title2, "2013");
59      Response response = get(getResourceURL(MOVIES));
60      assertNotNull(response);
61      String content = response.getContent();
62      assertTrue(content, content.contains(title1));
63      assertTrue(content, content.contains(title2));
64      int lines = response.getContentLines();
65      assertTrue("Expected more than 1 line, only got " + lines + "\n" +
66        "In content: " + content, lines > 1);
67    }

143   @Test
144   public void testUpdateMovie() throws IOException {
145     long id = createMovie("Old Title", "1998");
146 
147     String newTitle = "New Title";
148     String newYear = "2008";
149     Response response = put(getResourceURL(MOVIES), "id", String.valueOf(id), "title", newTitle, "year", newYear);
150     assertEquals(response.getContent(), HTTP_OK, response.getCode());
151     assertEquals(1, response.getContentLines());
152     assertEquals(String.valueOf(id), response.getContent());
153 
154     response = get(getResourceURL(MOVIES), "id", String.valueOf(id));
155     assertEquals(HTTP_OK, response.getCode());
156     assertEquals(1, response.getContentLines());
157     assertTrue(response.getContent().contains(newTitle));
158     assertTrue(response.getContent().contains(newYear));
159   }

169     assertThat(response.getContent(), response.getCode(), equalTo(HTTP_NOT_FOUND));
