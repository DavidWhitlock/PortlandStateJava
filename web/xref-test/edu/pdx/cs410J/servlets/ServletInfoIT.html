16    @Test
17    public void testServletInfo() throws IOException {
18      URL url = new URL("http://localhost:8080/web/info");
19      BufferedReader br = new BufferedReader(new InputStreamReader(url.openStream()));
20      StringBuilder sb = new StringBuilder();
21      int lines = 0;
22      while (br.ready()) {
23        String line = br.readLine();
24        sb.append(line);
25        sb.append("\n");
26        if (lines < 10) {
27          lines++;
28          System.out.println(line);
29        }
30      }
31  
32      String s = sb.toString();
33      assertTrue(s.indexOf("Servlet Information") >= 0);
34      
35    }
