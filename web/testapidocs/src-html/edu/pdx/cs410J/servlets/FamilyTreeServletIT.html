021/**

024public class FamilyTreeServletIT extends HttpRequestHelper {
025
026  private static final String PERSON_URL = "http://localhost:8080/web/family/person";
027
028  /**
029   * Creates a new <code>Person</code> using REST
030   *
031   * @return the id of the newly created person
032   */
033  private Person createPerson(Person person) throws IOException {
034    if (person.getId() != 0) {
035      String s = "Person " + person.getId() + " has already been created";
036      throw new IllegalArgumentException(s);
037    }
038
039    List<String> params = new ArrayList<String>();
040
041    params.add("Gender");
042    params.add(person.getGender() == Person.MALE ? "Male" : "Female");
043
044    params.add("FirstName");
045    params.add(person.getFirstName());
046    params.add("MiddleName");
047    params.add(person.getMiddleName());
048    params.add("LastName");
049    params.add(person.getLastName());
050
051    params.add("DateOfBirth");
052    params.add(String.valueOf(person.getDateOfBirth().getTime()));
053    params.add("DateOfDeath");
054    params.add(String.valueOf(person.getDateOfBirth().getTime()));
055
056    Response response = post(PERSON_URL, params.toArray(new String[params.size()]));
057    if (response.getCode() != HTTP_OK) {
058      String s = "Could not create Person (error code " + response.getCode() + "): " + response.getContent();
059      throw new IllegalArgumentException(s);
060    }
061
062    XmlParser parser = new XmlParser(new StringReader(response.getContent()));
063    FamilyTree tree = parser.parse();
064    assertEquals(1, tree.getPeople().size());
065    return tree.getPeople().iterator().next();
066  }
067
068  /**
069   * Returns the person with the given id
070   */
071  private Person getPerson(int id) throws IOException {
072    Response response = get(PERSON_URL + "/" + id);
073
074    if (response.getCode() == HTTP_NOT_FOUND) {
075      return null;
076
077    } else if (response.getCode() != HTTP_OK) {
078      String s = "Could not get Person (error code " + response.getCode() + ")";
079      throw new IllegalArgumentException(s);
080    }
081
082    XmlParser parser = new XmlParser(new StringReader(response.getContent()));
083    FamilyTree tree = parser.parse();
084    assertEquals(1, tree.getPeople().size());
085    return tree.getPeople().iterator().next();
086  }
087
088  @Test
089  public void testGetNonExistentPerson() throws IOException {
090     assertNull(getPerson(-42));
091  }
092
093  @Test
094  public void testCreateAndGetPerson() throws IOException {
095    Person person = new Person(7, Person.Gender.MALE) {
096      public int getId() {
097        return 0;
098      }
099    };
100    person.setDateOfBirth(new Date());
101    person.setDateOfDeath(new Date());
102    person.setFirstName("Test First Name");
103    person.setMiddleName("Test Middle Name");
104    person.setLastName("Test Last Name");
105
106    Person person2 = createPerson(person);
107    checkPerson(person, person2);
108
109    Person person3 = getPerson(person2.getId());
110    checkPerson(person2, person3);
111  }
112
113  /**
114   * Asserts that two <code>Person</code>s have the same content.
115   */
116  public void checkPerson(Person p1, Person p2) {
117    checkDate(p1.getDateOfBirth(), p2.getDateOfBirth());
118    checkDate(p1.getDateOfDeath(), p2.getDateOfDeath());
119    assertEquals(p1.getFirstName(), p2.getFirstName());
120    assertEquals(p1.getMiddleName(), p2.getMiddleName());
121    assertEquals(p1.getLastName(), p2.getLastName());
122
123    if (p1.getFatherId() > 0) {
124      assertEquals(p1.getFatherId(), p2.getFatherId());
125    }
126
127    if (p1.getMotherId() > 0) {
128      assertEquals(p1.getFatherId(), p2.getFatherId());
129    }
130
131  }
132
133  private void checkDate(Date d1, Date d2) {
134    if (d1 != null) {
135      assertNotNull(d2);
136
137    } else {
138      assertNull(d2);
139    }
140
141    Calendar c1 = Calendar.getInstance();
142    c1.setTime(d1);
143    Calendar c2 = Calendar.getInstance();
144    c2.setTime(d2);
145
146    assertEquals(c1.get(Calendar.YEAR), c2.get(Calendar.YEAR));
147    assertEquals(c1.get(Calendar.MONTH), c2.get(Calendar.MONTH));
148    assertEquals(c1.get(Calendar.DATE), c2.get(Calendar.DATE));    
149  }
