003import edu.pdx.cs410J.web.HttpRequestHelper;

006import java.io.IOException;

008import static java.net.HttpURLConnection.HTTP_BAD_REQUEST;
009import static java.net.HttpURLConnection.HTTP_NOT_FOUND;
010import static java.net.HttpURLConnection.HTTP_OK;
011import static org.hamcrest.CoreMatchers.equalTo;
012import static org.junit.Assert.*;
013
014/**
015 * Tests the functionality of the <code>MovieDatabaseRestServlet</code>
016 */
017public class MovieDatabaseServletIT extends HttpRequestHelper {
018
019  /**
020   * The URL for accessing the Movie database application
021   */
022  private static final String DATABASE_URL = "http://localhost:8080/web/movies";
023
024  private final String MOVIES = "";
025
026  @Test
027  public void testMissingTitle() throws IOException {
028    assertEquals(HTTP_BAD_REQUEST, post(getResourceURL(MOVIES)).getCode());
029  }
030
031  @Test
032  public void testMissingYear() throws IOException {
033    assertEquals(HTTP_BAD_REQUEST, post(getResourceURL(MOVIES), "title", "Title").getCode());
034  }
035
036  @Test
037  public void testMalformedYear() throws IOException {
038    assertEquals(HTTP_BAD_REQUEST, post(getResourceURL(MOVIES), "title", "Title", "year", "asdfa").getCode());
039  }
040
041  @Test
042  public void testCreateMovie() throws IOException {
043    createMovie("Title", "2008");
044  }
045
046  private long createMovie(String title, String year) throws IOException {
047    Response response = post(getResourceURL(MOVIES), "title", title, "year", year);
048    assertEquals(response.getContent(), HTTP_OK, response.getCode());
049    assertNotNull(response.getContent());
050    return Long.parseLong(response.getContent().trim());
051  }
052
053  @Test
054  public void testGetAllMovies() throws IOException {
055    String title1 = "testGetAllMovies1";
056    createMovie(title1, "2007");
057    String title2 = "testGetAllMovies2";
058    createMovie(title2, "2013");
059    Response response = get(getResourceURL(MOVIES));
060    assertNotNull(response);
061    String content = response.getContent();
062    assertTrue(content, content.contains(title1));
063    assertTrue(content, content.contains(title2));
064    int lines = response.getContentLines();
065    assertTrue("Expected more than 1 line, only got " + lines + "\n" +
066      "In content: " + content, lines > 1);
067  }
068
069  @Test
070  public void testGetOneMovie() throws IOException {
071    String title = "testGetOneMovie" + "-title";
072    String year = "1997";
073    long id = createMovie(title, year);
074    Response response = get(getResourceURL(MOVIES), "id", String.valueOf(id));
075    assertNotNull(response);
076    assertTrue(response.getContent(), response.getContent().contains(title));
077    assertTrue(response.getContent(), response.getContent().contains(year));
078    assertEquals(1, response.getContentLines());
079  }
080
081  @Test
082  public void testGetMoviesByTitle() throws IOException {
083    String query = "testGetMoviesByTitle";
084    String title1 = query + 1;
085    String year1 = "2001";
086    String title2 = "NotGetMoviesByTitle" + 2;
087    String year2 = "2002";
088    String title3 = query + 3;
089    String year3 = "2003";
090
091    createMovie(title1, year1);
092    createMovie(title2, year2);
093    createMovie(title3, year3);
094
095    Response response = get(getResourceURL(MOVIES), "title", query);
096    assertTrue(response.getContent(), response.getContent().contains(title1));
097    assertFalse(response.getContent(), response.getContent().contains(title2));
098    assertTrue(response.getContent(), response.getContent().contains(title3));
099  }

121  @Test
122  public void testGetMoviesByTitleAndYear() throws IOException {
123    String titleQuery = "testGetMoviesByTitleAndYear";
124    String yearQuery = "2001";
125
126    String title1 = titleQuery + 1;
127    String year1 = yearQuery;
128    String title2 = "NotGetMoviesByTitleAndYear" + 2;
129    String year2 = "2002";
130    String title3 = titleQuery + 3;
131    String year3 = yearQuery;
132
133    createMovie(title1, year1);
134    createMovie(title2, year2);
135    createMovie(title3, year3);
136
137    Response response = get(getResourceURL(MOVIES), "title", titleQuery);
138    assertTrue(response.getContent(), response.getContent().contains(title1));
139    assertFalse(response.getContent(), response.getContent().contains(title2));

142
143  @Test
144  public void testUpdateMovie() throws IOException {
145    long id = createMovie("Old Title", "1998");
146
147    String newTitle = "New Title";
148    String newYear = "2008";
149    Response response = put(getResourceURL(MOVIES), "id", String.valueOf(id), "title", newTitle, "year", newYear);
150    assertEquals(response.getContent(), HTTP_OK, response.getCode());
151    assertEquals(1, response.getContentLines());
152    assertEquals(String.valueOf(id), response.getContent());
153
154    response = get(getResourceURL(MOVIES), "id", String.valueOf(id));
155    assertEquals(HTTP_OK, response.getCode());
156    assertEquals(1, response.getContentLines());
157    assertTrue(response.getContent().contains(newTitle));
158    assertTrue(response.getContent().contains(newYear));
159  }
160
161  @Test
162  public void canDeleteMovieFromServer() throws IOException {
163    long id = createMovie("Movie to be deleted", "2015");
164
165    Response response = delete(getResourceURL(MOVIES), "id", String.valueOf(id));
166    assertThat(response.getContent(), response.getCode(), equalTo(HTTP_OK));
167
168    response = get(getResourceURL(MOVIES), "id", String.valueOf(id));
169    assertThat(response.getContent(), response.getCode(), equalTo(HTTP_NOT_FOUND));
170  }
171
172  private String getResourceURL(String resource) {
173    return DATABASE_URL + resource;
174  }
