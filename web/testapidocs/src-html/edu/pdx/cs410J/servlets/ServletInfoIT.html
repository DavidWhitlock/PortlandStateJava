001package edu.pdx.cs410J.servlets;
002
003import org.junit.Test;
004import static org.junit.Assert.assertTrue;
005
006import java.net.URL;
007import java.io.BufferedReader;
008import java.io.InputStreamReader;
009import java.io.IOException;
010
011/**
012 * Tests that we can get information about the servlet container
013 */
014public class ServletInfoIT {
015
016  @Test
017  public void testServletInfo() throws IOException {
018    URL url = new URL("http://localhost:8080/web/info");
019    BufferedReader br = new BufferedReader(new InputStreamReader(url.openStream()));
020    StringBuilder sb = new StringBuilder();
021    int lines = 0;
022    while (br.ready()) {
023      String line = br.readLine();
024      sb.append(line);
025      sb.append("\n");
026      if (lines < 10) {
027        lines++;
028        System.out.println(line);
029      }
030    }
031
032    String s = sb.toString();
033    assertTrue(s.indexOf("Servlet Information") >= 0);
034    
035  }
036}
