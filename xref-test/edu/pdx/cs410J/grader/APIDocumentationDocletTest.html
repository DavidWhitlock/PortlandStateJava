5   import java.io.PrintWriter;

8   import static org.hamcrest.CoreMatchers.equalTo;
9   import static org.hamcrest.MatcherAssert.assertThat;

11  public class APIDocumentationDocletTest {
12  
13    @Test
14    public void removePackageNameFromJavaLangString() {
15      assertThat(APIDocumentationDoclet.removePackageNames("java.lang.String"), equalTo("String"));
16    }
17  
18    @Test
19    public void removePackageNameFromArrayOfJavaLangString() {
20      assertThat(APIDocumentationDoclet.removePackageNames("java.lang.String[]"), equalTo("String[]"));
21    }
22  
23    @Test
24    public void removePackageNameFromMapOfStringToString() {
25      String fullName = "java.util.Map<java.lang.String,java.lang.String>";
26      String shortName = "Map<String,String>";
27      assertThat(APIDocumentationDoclet.removePackageNames(fullName), equalTo(shortName));
28    }
29  
30    @Test
31    public void leadingWhitespaceIsRemovedWhenIndenting() {
32      String rawText = "classes - The names of the classes the student is taking.  A student              \n" +
33        "        may take zero or more classes.";
34      String expected = "    classes - The names of the classes the student is taking. A student\n" +
35                        "    may take zero or more classes.\n";
36      assertThat(indentString(rawText, 4), equalTo(expected));
37    }
38  
39    private String indentString(String rawText, int indent) {
40      StringWriter sw = new StringWriter();
41      APIDocumentationDoclet.indent(rawText, indent, new PrintWriter(sw, true));
42      return sw.toString();
43    }
44  
45    @Test
46    public void noSpaceAfterDotInMiddleOfWord() {
47      assertThat(indentString("1.2.3", 2), equalTo("  1.2.3\n"));
48    }
49  
50    @Test
51    public void spacesAfterPeriodAreCollapsedIntoOneSpace() {
52      String rawText = "One sentence.  Two sentences.";
53      String expected = "  One sentence. Two sentences.\n";
54      assertThat(indentString(rawText, 2), equalTo(expected));
55    }
