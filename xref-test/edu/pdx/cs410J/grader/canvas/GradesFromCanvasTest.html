276   @Test
277   void ambiguousCanvasAssignmentNameThrowsIllegalStateException() {
278     GradeBook gradebook = new GradeBook("test");
279     String quizName = "restquiz";
280     Assignment quiz = new Assignment(quizName, 3.0).setType(Assignment.AssignmentType.QUIZ).setDescription("Web and REST");
281     gradebook.addAssignment(quiz);
282 
283     String projectName = "Project5";
284     Assignment project = new Assignment(projectName, 10.0).setType(Assignment.AssignmentType.PROJECT).setDescription("REST");
285     gradebook.addAssignment(project);
286 
287     String assignmentInCanvas = "Quiz 5: Web and REST";
288     CanvasAssignment canvasAssignment = new CanvasAssignment(assignmentInCanvas, 12345);
289 
290     GradesFromCanvas canvas = new GradesFromCanvas();
291     IllegalStateException ex =
292       assertThrows(IllegalStateException.class, () -> canvas.findAssignmentInGradebookForCanvasQuiz(canvasAssignment, gradebook));
293     String message = ex.getMessage();
294     assertThat(message, containsString(quizName));
295     assertThat(message, containsString(projectName));
296   }
