21    @Test
22    public void letterGradeRangesArePersistedToXml() throws IOException, TransformerException, ParserException {
23      GradeBook book = new GradeBook("test");
24  
25      LetterGradeRanges undergradRanges = book.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
26      morphLetterGradeRanges(undergradRanges, 1);
27  
28      LetterGradeRanges gradRanges = book.getLetterGradeRanges(Student.Section.GRADUATE);
29      morphLetterGradeRanges(gradRanges, 2);
30  
31      GradeBook book2 = writeAndReadGradeBookAsXml(book);
32  
33      LetterGradeRanges undergradRanges2 = book2.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
34      assertRangesAreEqual(undergradRanges, undergradRanges2);
35  
36      LetterGradeRanges gradRanges2 = book2.getLetterGradeRanges(Student.Section.GRADUATE);
37      assertRangesAreEqual(gradRanges, gradRanges2);
38    }
39  
40    private void assertRangesAreEqual(LetterGradeRanges ranges, LetterGradeRanges ranges2) {
41      for (LetterGrade letterGrade : LetterGrade.values()) {
42        LetterGradeRange range = ranges.getRange(letterGrade);
43        if (range == null) {
44          continue;
45        }
46        LetterGradeRange range2 = ranges2.getRange(letterGrade);
47        assertThat("Range for " + letterGrade, range2.minimum(), equalTo(range.minimum()));
48      }
49    }
50  
51    private void morphLetterGradeRanges(LetterGradeRanges ranges, int morphValue) {
52      for (LetterGrade letterGrade : LetterGrade.values()) {
53        LetterGradeRange range = ranges.getRange(letterGrade);
54        if (range == null) {
55          continue;
56        }
57        range.setRange(range.minimum() + morphValue, range.maximum() + morphValue);
58      }
59      LetterGradeRange fRange = ranges.getRange(LetterGrade.F);
60      fRange.setRange(0, fRange.maximum());
61    }

63    private GradeBook writeAndReadGradeBookAsXml(GradeBook book) throws IOException, TransformerException, ParserException {
64      Document doc = XmlDumper.dumpGradeBook(book, new XmlHelper());
65      byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
66  
67  //    System.out.println(new String(bytes));
68  
69      XmlGradeBookParser parser = new XmlGradeBookParser(new ByteArrayInputStream(bytes));
70      return parser.parse();
71    }

101   @Test
102   public void xmlHasNewlinesButNotIndentation() throws TransformerException, IOException {
103     GradeBook book = new GradeBook("test");
104     Document doc = XmlDumper.dumpGradeBook(book, new XmlHelper());;
105     byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
106     String string = new String(bytes);
107     assertThat(string, containsString("\n"));
108     assertThat(string.chars().filter(c -> '\n' == c).count(), greaterThan(4L));
109     assertThat(string, not(containsString(" <")));
110   }
