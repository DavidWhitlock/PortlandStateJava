25  public class MovieListPresenterTest extends MovieTestCase {

27    @Captor
28    private ArgumentCaptor<AsyncCallback<ArrayList<Movie>>> allMoviesCaptor;
29  
30    @Before
31    public void initializeMockito() {
32        MockitoAnnotations.initMocks(this);
33    }
34  
35    /**
36     * Tests that all movie titles are displayed when the presenter is created
37     */
38    @Test
39    public void testAllMovieTitlesDisplayed() {
40      MovieListPresenter.Display view = mock(MovieListPresenter.Display.class);
41      MovieServiceAsync service = mock(MovieServiceAsync.class);
42      HandlerManager eventBus = mock(HandlerManager.class);
43  
44      new MovieListPresenter(view, service, eventBus);
45      verify(service).getAllMovies(allMoviesCaptor.capture());
46  
47      List<String> titles = Lists.newArrayList("One", "Two", "Three", "Four");
48  
49      List<Movie> movies = Lists.transform(titles, new Function<String, Movie>() {
50        @Override
51        public Movie apply(String title) {
52          return newMovie(title, 2010);
53  
54        }
55      });
56      allMoviesCaptor.getValue().onSuccess(Lists.newArrayList(movies));
57  
58      verify(view).setTitles(titles);
59    }
60  
61    /**
62     * Tests that a {@link DisplayMovieEvent} is fired when a movie is selected
63     */
64    @Test
65    public void testDisplayMovieEventFiredOnSelection() {
66      Display view = mock(Display.class);
67      MovieServiceAsync service = mock(MovieServiceAsync.class);
68      HandlerManager eventBus = new HandlerManager(null);
69  
70      DisplayMovieEvent.Handler handler = mock(DisplayMovieEvent.Handler.class);
71      eventBus.addHandler(DisplayMovieEvent.TYPE, handler);
72  
73      Movie movie0 = newMovie("Zero", 2010);
74      Movie movie1 = newMovie("One", 2010);
75      newPresenter(view, service, eventBus, movie0, movie1);
76  
77      ArgumentCaptor<MovieSelectedHandler> captor = ArgumentCaptor.forClass(MovieSelectedHandler.class);
78      verify(view).addMovieSelectedHandler(captor.capture());
79  
80      captor.getValue().onMovieSelected(0);
81      verify(handler).onMovieDisplayed(movie0);
82  
83      captor.getValue().onMovieSelected(1);
84      verify(handler).onMovieDisplayed(movie1);
85    }
86  
87    /**
88     * Creates a new {@link MovieListPresenter} that displays the given movies
89     * @param view The presenter's view
90     * @param service The remote service
91     * @param eventBus The event bus
92     * @param movies The movies to show
93     * @return The presenter
94     */
95    private MovieListPresenter newPresenter(Display view, MovieServiceAsync service, HandlerManager eventBus, Movie... movies) {
96      MovieListPresenter presenter = new MovieListPresenter(view, service, eventBus);
97      verify(service).getAllMovies(allMoviesCaptor.capture());
98      allMoviesCaptor.getValue().onSuccess(Lists.newArrayList(movies));
99      return presenter;
100   }
