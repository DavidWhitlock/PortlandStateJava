3   import java.util.List;

5   /**
6    * Throws when the <code>main</code> method of a class with mutable <code>static</code>
7    * fields is invoked with {@link InvokeMainTestCase#invokeMain(Class, String...)}.
8    *
9    * Main classes with mutable static fields have proven to be problematic because the
10   * values of the mutable fields are not reset in between tests.  As a result, the values
11   * of these fields from previous tests might stick around.  This has proven to be very
12   * confusing.
13   *
14   * @see InvokeMainTestCase#invokeMainAllowingMutableStaticFields(Class, String...)
15   */
16  public class MainClassContainsMutableStaticFields extends RuntimeException {
17  
18    private final Iterable<String> fieldNames;
19  
20    public MainClassContainsMutableStaticFields(String className, List<String> fieldNames) {
21      super(getMessage(className, fieldNames));
22      this.fieldNames = fieldNames;
23    }
24  
25    private static String getMessage(String className, List<String> fieldNames) {
26      return "Main class \"" +
27        className +
28        "\" contains " +
29        fieldNames.size() +
30        " non-final static fields: " +
31        String.join(", ", fieldNames) +
32        ".  You probably don't want to store data in mutable static fields.  " +
33        "It might cause confusing behavior with your integration tests.";
34    }
35  
36    public Iterable<String> getNamesOfMutableStaticFields() {
37      return this.fieldNames;
38    }
39  }
