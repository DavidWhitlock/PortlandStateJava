11  /**
12   * This is the abstract superclass of classes that test family tree
13   * classes.  It contains a number of helper methods.
14   */
15  public abstract class FamilyTestCase {
16  
17  
18    /**
19     * Asserts that two <code>Person</code>s have the same contents
20     */
21    void assertEquals(Person p1, Person p2) {
22      if (p1 == null) {
23        assertNull(p2);
24        return;
25  
26      } else if (p2 == null) {
27        assertNull(p1);
28        return;
29      }
30  
31      assertNotNull(p1);
32      assertNotNull(p2);
33  
34      Assertions.assertEquals(p1.getId(), p2.getId());
35      Assertions.assertEquals(p1.getGender(), p2.getGender());
36      Assertions.assertEquals(p1.getFirstName(), p2.getFirstName());
37      Assertions.assertEquals(p1.getMiddleName(), p2.getMiddleName());
38      Assertions.assertEquals(p1.getLastName(), p2.getLastName());
39      assertEquals(p1.getMother(), p2.getMother());
40      assertEquals(p1.getFather(), p2.getFather());
41      assertEquals(p1.getDateOfBirth(), p2.getDateOfBirth());
42      assertEquals(p1.getDateOfDeath(), p2.getDateOfDeath());
43  
44      Assertions.assertEquals(p1.getMarriages().size(), p2.getMarriages().size());
45  
46      Iterator<Marriage> iter1 = p1.getMarriages().iterator();
47      Iterator<Marriage> iter2 = p2.getMarriages().iterator();
48      while (iter1.hasNext() && iter2.hasNext()) {
49        assertEquals( iter1.next(), iter2.next() );
50      }
51  
52    }
53  
54    /**
55     * Asserts that two <code>Date</code> represent the same day
56     * (ignores times)
57     */
58    void assertEquals(Date d1, Date d2) {
59      if (d1 == null) {
60        assertNull(d2);
61        return;
62  
63      } else if (d2 == null) {
64        assertNull(d1);
65        return;
66      }
67  
68      Calendar c1 = Calendar.getInstance();
69      c1.setTime(d1);
70      Calendar c2 = Calendar.getInstance();
71      c2.setTime(d2);
72  
73      Assertions.assertEquals(c1.get(Calendar.DAY_OF_YEAR),
74                   c2.get(Calendar.DAY_OF_YEAR));
75      Assertions.assertEquals(c1.get(Calendar.YEAR), c2.get(Calendar.YEAR));
76    }
77  
78    /**
79     * Asserts that two <code>Marriage</code>s are the same
80     */
81    void assertEquals(Marriage m1, Marriage m2) {
82      if (m1 == null) {
83        assertNotNull(m2);
84        return;
85  
86      } else if (m2 == null) {
87        assertNull(m1);
88        return;
89      }
90  
91      // To avoid infinite recursion, we compare people using their ids
92      Assertions.assertEquals(m1.getHusband().getId(), m2.getHusband().getId());
93      Assertions.assertEquals(m1.getWife().getId(), m2.getWife().getId());
94      assertEquals(m1.getDate(), m2.getDate());
95      Assertions.assertEquals(m1.getLocation(), m2.getLocation());
96    }
97  
98    /**
99     * Asserts that two <code>FamilyTree</code>s has the same contents
100    */
101   void assertEquals(FamilyTree tree1, FamilyTree tree2) {
102     Assertions.assertEquals(tree1.getPeople().size(), tree2.getPeople().size());
103 
104       for ( Person person : tree1.getPeople() )
105       {
106           Person p2 = tree2.getPerson( person.getId() );
107           assertNotNull( p2 );
108           assertEquals( person, p2 );
109       }
110   }
111 
112   /**
113    * Asserts that one String contains another
114    */
115   void assertContains(String container, String containee) {
116     assertTrue(container.indexOf(containee) != -1);
117   }
118 
119   /**
120    * Asserts that one String contains another
121    */
122   void assertContains(String message, String container,
123                       String containee) {
124     assertTrue(container.indexOf(containee) != -1, message);
125   }
126 
127 }
