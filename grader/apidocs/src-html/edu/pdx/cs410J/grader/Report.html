001package edu.pdx.cs410J.grader;
002
003import java.io.*;
004
005/**
006 * This class represents a report that is generated as a result of
007 * running a student's program.
008 *
009 * @author David Whitlock
010 */
011public class Report {
012
013  /** The original <code>System.err</code> */
014  PrintStream consoleErr = System.err;
015
016  /** The original <code>System.out</code> */
017  PrintStream consoleOut = System.out;
018
019  /** The number of character columns in the log file. */
020  private static final int WIDTH = 72;
021
022  /** Write output of tests and other stuff to here. */
023  private PrintWriter log;
024
025  /** Are we finished? */
026  private boolean done = false;
027
028  private PrintStream ps;
029
030  /**
031   * Creates a new <code>Report</code> that sends its output to a
032   * given <code>PrintWriter</code>.  This is useful when you want to
033   * dump to the console.  The "student's" id is
034   * <code>"writer"</code>.
035   */
036  public Report(PrintWriter pw) {
037    this.log = pw;
038  }
039  
040
041  /**
042   * Creates a new <code>Report</code> for a given student.  It will
043   * create a log file in given directory.
044   *
045   * @param id
046   *        The student's login id
047   * @param logDir
048   *        Directory into which the log file is placed.
049   */
050  public Report(final String id, File logDir) {
051    if (!logDir.isDirectory()) {
052      throw new IllegalArgumentException(logDir + 
053                                         " is not a directory");
054    }
055
056    if (!logDir.exists()) {
057      logDir.mkdirs();
058    }
059
060    String logFileName = id + ".log";
061    File logFile = new File(logDir, logFileName);
062    
063
064    // Create a PrintStream around the log file.  Set System.out and
065    // System.err to refer to this guy.
066    FileOutputStream fos = null;
067    try {
068      fos = new FileOutputStream(logFile);
069
070    } catch (IOException ex) {
071      throw new IllegalArgumentException("Could not open " +
072                                         logFileName);
073    }
074
075    /**
076     * Inner class used to prevent file stream from being closed until
077     * I say so.  Other code may attempt to close System.out which
078     * result in this file stream being closed. 
079     */
080    class CheckedPrintStream extends PrintStream {
081      public CheckedPrintStream(OutputStream stream) {
082        super(stream);
083      }
084
085      public void close() {
086        if(Report.this.done) {
087//        Report.this.consoleOut.println("Report " + id + 
088//                                       " is closing stream");
089          super.close();
090        }
091        
092//      Report.this.consoleOut.println("Report " + id + 
093//                                     " is not done yet");
094      }
095    }
096
097
098//      this.ps = System.out;
099    this.ps = new CheckedPrintStream(fos);
100    System.setOut(ps);
101    System.setErr(ps);
102
103    // Create a PrintWriter around the PrintStream
104    this.log = new PrintWriter(ps, true);
105  }
106
107  /**
108   * Makes note of an exception 
109   */
110  public void logException(String msg, Exception ex) {
111    println("** EXCEPTION " + msg);
112    ex.printStackTrace(this.log);
113  }
114  
115  /**
116   * Flushes the stream that is being written to
117   */
118  public void flush() {
119    this.log.flush();
120  }
121
122  /**
123   * Prints a left justified line of text to the log file.
124   */
125  public void println(Object o) {
126    this.log.println(o);
127  }
128
129  /**
130   * Prints a left justified line of text to the log file.
131   */
132  public void println(String text) {
133    this.log.println(text);
134  }
135
136  /**
137   * Prints a line of text to the log file and centers it.
138   */
139  public void printlnCentered(String text) {
140    printBanner(text, ' ');
141  }
142
143  /**
144   * Prints a line of text to the log file, centers it, around
145   * surrounds it with a given character.
146   */
147  public void printBanner(String text, char c) {
148    int length = text.length();
149
150    // Don't bother centering if the text is wider than the log.
151    if (length < WIDTH) {
152      int indent = (WIDTH - length) / 2;
153
154      for (int i = 0; i < indent - 1; i++) {
155        this.log.print(c);
156      }
157      this.log.print(' ');
158      this.log.flush();
159    }
160
161    this.log.print(text);
162
163    if (length < WIDTH) {
164      int indent = (WIDTH - length) / 2;
165
166      this.log.print(' ');
167      for (int i = 0; i < indent - 1; i++) {
168        this.log.print(c);
169      }
170      this.log.flush();
171    }
172
173    this.log.println("");
174  }
175
176  /**
177   * Signifies that this <code>Report</code> is done
178   */
179  void done() {
180    this.done = true;
181  }
182
183  /**
184   * Make sure we're done when this <code>Report</code> gets garbage
185   * collected.  This way, the stream will always get closed.
186   */
187  public void finalize() {
188    this.done = true;
189  }
190
191}
