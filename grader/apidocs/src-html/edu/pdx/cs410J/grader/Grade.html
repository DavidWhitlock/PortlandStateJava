001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004
005import java.io.File;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.PrintWriter;
009import java.time.LocalDateTime;
010import java.util.ArrayList;
011import java.util.List;
012
013/**
014 * This class represent the grade a student got on an assignment.
015 */
016public class Grade extends NotableImpl {
017  /**
018   * The assignment is not complete
019   */
020  public static final double INCOMPLETE = -1.0;
021
022  /**
023   * Some work has been submitted, but no grade has been assigned
024   */
025  public static final double NO_GRADE = -2.0;
026
027  private String assignmentName;
028  private double score;         // Score student received
029  private final List<LocalDateTime> submissionTimes = new ArrayList<>();
030
031  /**
032   * Creates a <code>Grade</code> for a given assignment
033   */
034  public Grade(String assignmentName, double score) {
035    this.assignmentName = assignmentName;
036    this.score = score;
037    this.setDirty(true);  // Initially dirty
038  }
039
040  public Grade(Assignment project, double score) {
041    this(project.getName(), score);
042  }
043
044  /**
045   * Returns the name of the assignment that this <code>Grade</code>
046   * is for.
047   */
048  public String getAssignmentName() {
049    return this.assignmentName;
050  }
051
052  /**
053   * Returns the score the student received on the assignment.
054   */
055  public double getScore() {
056    return this.score;
057  }
058
059  public boolean isIncomplete() {
060    return this.getScore() == INCOMPLETE;
061  }
062
063  /**
064   * Sets the score the student received on the assignment.
065   */
066  public void setScore(double score) {
067    this.score = score;
068    this.setDirty(true);
069  }
070
071  private static PrintWriter err = new PrintWriter(System.err, true);
072
073  /**
074   * Returns a brief textual description of this <code>Grade</code>
075   */
076  public String toString() {
077    return this.assignmentName + ": " + this.score;
078  }
079
080  /**
081   * Prints usage information about the main program.
082   */
083  private static void usage() {
084    err.println("\njava Grade [options] -xmlFile xmlFile " +
085                "-id id -assignment name");
086    err.println("  where [options] ares:");
087    err.println("  -score points          How many points student got");
088    err.println("  -incomplete            Notes that the assignment "
089                + "was INCOMPLETE");
090    err.println("  -no-grade              No grade was given for the "
091                + "assignment");
092    err.println("  -note note             A note about the grade");
093    err.println("\n");
094    System.exit(1);
095  }
096
097  /**
098   * A main program that creates/edits a student's grade.
099   */
100  public static void main(String[] args) {
101    String xmlFile = null;
102    String id = null;
103    String assignment = null;
104    String score = null;
105    boolean incomplete = false;
106    boolean noGrade = false;
107    String note = null;
108
109    // Parse the command line
110    for (int i = 0; i < args.length; i++) {
111      if (args[i].equals("-xmlFile")) {
112        if (++i >= args.length) {
113          err.println("** Missing XML file name");
114          usage();
115        }
116
117        xmlFile = args[i];
118
119      } else if (args[i].equals("-id")) {
120        if (++i >= args.length) {
121          err.println("** Missing student id");
122          usage();
123        }
124
125        id = args[i];
126
127      } else if (args[i].equals("-assignment")) {
128        if (++i >= args.length) {
129          err.println("** Missing assignment name");
130          usage();
131        }
132
133        assignment = args[i];
134
135      } else if (args[i].equals("-score")) {
136        if (++i >= args.length) {
137          err.println("** Missing score");
138          usage();
139        }
140
141        score = args[i];
142
143      } else if (args[i].equals("-incomplete")) {
144        incomplete = true;
145
146      } else if (args[i].equals("-no-grade")) {
147        noGrade = true;
148
149      } else if (args[i].equals("-note")) {
150        if (++i >= args.length) {
151          err.println("** Missing note text");
152          usage();
153        }
154
155        note = args[i];
156
157      } else if (args[i].startsWith("-")) {
158        err.println("** Unknown command line option: " + args[i]);
159        usage();
160
161      } else {
162        err.println("** Spurious command line: " + args[i]);
163        usage();
164      }
165    }
166
167    // Verify command line
168    if (xmlFile == null) {
169      err.println("** No XML file specified");
170      usage();
171    }
172
173    if (id == null) {
174      err.println("** No student id specified");
175      usage();
176      return;
177    }
178
179    if (assignment == null) {
180      err.println("** No assignment name specified");
181      usage();
182    }
183
184    File file = new File(xmlFile);
185    if (!file.exists()) {
186      err.println("** Grade book file " + xmlFile + 
187                  " does not exist");
188      System.exit(1);
189    }
190
191    // Parse XML file
192    GradeBook book = null;
193    try {
194      XmlGradeBookParser parser = new XmlGradeBookParser(file);
195      book = parser.parse();
196
197    } catch (FileNotFoundException ex) {
198      err.println("** Could not find file: " + ex.getMessage());
199      System.exit(1);
200
201    } catch (IOException ex) {
202      err.println("** IOException during parsing: " + ex.getMessage());
203      System.exit(1);
204
205    } catch (ParserException ex) {
206      err.println("** Exception while parsing " + file + ": " + ex);
207      System.exit(1);
208    }
209
210    if (!book.containsStudent(id)) {
211      err.println("** No student with id: " + id);
212      System.exit(1);
213    }
214
215    // Get the student
216    Student student = book.getStudent(id).get();
217
218    // Get the grade
219    Grade grade = student.getGrade(assignment);
220    if (grade == null) {
221      // Create a new Grade
222      double s;
223
224      if (incomplete) {
225        s = Grade.INCOMPLETE;
226
227      } else if (noGrade) {
228        s = Grade.NO_GRADE;
229
230      } else if (score == null) {
231        err.println("** No score for " + assignment);
232        System.exit(1);
233        s = -4.2;
234
235      } else {
236        try {
237          s = Double.parseDouble(score);
238
239        } catch (NumberFormatException ex) {
240          err.println("** Score is not a double: " + score);
241          System.exit(1);
242          s = -4.2;
243        }
244      }
245
246      grade = new Grade(assignment, s);
247      student.setGrade(assignment, grade);
248
249    } else {
250      // Set the grade
251      if (incomplete) {
252        grade.setScore(Grade.INCOMPLETE);
253
254      } else if (noGrade) {
255        grade.setScore(Grade.NO_GRADE);
256
257      } else if (score != null) {
258        try {
259          grade.setScore(Double.parseDouble(score));
260
261        } catch (NumberFormatException ex) {
262          err.println("** Score is not a double: " + score);
263        }
264      }
265    }
266
267    if (note != null) {
268      grade.addNote(note);
269    }
270
271    // Write the changes back out to the XML file
272    try {
273      XmlDumper dumper = new XmlDumper(file);
274      dumper.dump(book);
275
276    } catch (IOException ex) {
277      err.println("** While dumping to " + file + ": " + ex);
278      System.exit(1);
279    }
280  }
281
282  public boolean isNotGraded() {
283    return this.getScore() == Grade.NO_GRADE;
284  }
285
286  public List<LocalDateTime> getSubmissionTimes() {
287    return submissionTimes;
288  }
289
290  public void addSubmissionTime(LocalDateTime submissionTime) {
291    this.setDirty(true);
292    this.submissionTimes.add(submissionTime);
293  }
294}
