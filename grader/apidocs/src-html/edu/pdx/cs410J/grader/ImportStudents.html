001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004import java.io.*;
005import java.util.*;
006
007/**
008 * This program imports a bunch of students into a grade book.  It is
009 * used at the beginning of the term when students submit their XML
010 * files using the {@link Survey} program.
011 *
012 * @author David Whitlock
013 * @version $Revision: 1.2 $
014 */
015public class ImportStudents {
016  private static final PrintStream out = System.out;
017  private static final PrintStream err = System.err;
018
019  /**
020   * Prints usage information for this program
021   */
022  private static void usage(String s) {
023    err.println("\n** " + s + "\n");
024    err.println("usage: java ImportStudents xmlFile (studentXml)+");
025    err.println("  xmlFile     The grade book's XML file");
026    err.println("  studentXml  A student's XML file");
027    err.println("");
028    err.println("This program imports student XML files into a ");
029    err.println("grade book.");
030    err.println("\n");
031
032    System.exit(1);
033  }
034
035  public static void main(String[] args) throws Throwable {
036    String xmlFileName = null;
037    List<String> studentXmlNames = new ArrayList<String>();
038
039    for (int i = 0; i < args.length; i++) {
040      if (xmlFileName == null) {
041        xmlFileName = args[i];
042
043      } else {
044        studentXmlNames.add(args[i]);
045      }
046    }
047
048    if (xmlFileName == null) {
049      usage("Missing grade book XML file");
050
051    } else if (studentXmlNames.isEmpty()) {
052      usage("Missing student XML file");
053    }
054
055    File xmlFile = new File(xmlFileName);
056    if (!xmlFile.exists()) {
057      usage("Grade book file \"" + xmlFile + "\" does not exist");
058    }
059
060    XmlGradeBookParser parser = new XmlGradeBookParser(xmlFile);
061    GradeBook book = parser.parse();
062    
063    Iterator iter = studentXmlNames.iterator();
064    while (iter.hasNext()) {
065      File studentFile = new File((String) iter.next());
066      if (!studentFile.exists()) {
067        err.println("** Student file \"" + studentFile +
068                    "\" does not exist");
069      }
070
071      XmlStudentParser sp = new XmlStudentParser(studentFile);
072      Student student;
073      try {
074        student = sp.parseStudent();
075
076      } catch (ParserException ex) {
077        err.println("** Could not parse student file \"" + studentFile
078                    + "\"");
079        continue;
080      }
081
082      String id = student.getId();
083      if (book.containsStudent(id)) {
084        err.println("** Grade book already contains \"" + id + "\"");
085
086      } else {
087        out.println("Importing " + student);
088        book.addStudent(student);
089      }
090    }
091
092    if (book.isDirty()) {
093      XmlDumper dumper = new XmlDumper(xmlFile);
094      dumper.dump(book);
095    }
096  }
097
098}
