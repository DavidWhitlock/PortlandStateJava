001package edu.pdx.cs410J.grader;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006import javax.mail.MessagingException;
007import java.io.File;
008
009public abstract class StudentEmailAttachmentProcessor implements EmailAttachmentProcessor {
010  protected final Logger logger = LoggerFactory.getLogger(this.getClass().getPackage().getName());
011  protected final File directory;
012  protected final GradeBook gradeBook;
013
014  public StudentEmailAttachmentProcessor(File directory, GradeBook gradeBook) {
015    this.directory = directory;
016    this.gradeBook = gradeBook;
017  }
018
019  protected void logException(String message, Exception ex) {
020    this.logger.error(message, ex);
021  }
022
023  public abstract String getEmailFolder();
024
025  protected void warn(String message) {
026    this.logger.warn(message);
027  }
028
029  protected void info(String message) {
030    this.logger.info(message);
031  }
032
033  protected void debug(String message) {
034    this.logger.debug(message);
035  }
036
037  protected class SubmissionException extends Exception {
038    public SubmissionException(String message) {
039      super(message);
040    }
041
042    public SubmissionException(String message, MessagingException cause) {
043      super(message, cause);
044    }
045  }
046}
