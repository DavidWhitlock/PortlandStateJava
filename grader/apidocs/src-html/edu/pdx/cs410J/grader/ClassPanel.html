018/**
019 * This panel displays and edits information about a class stored in a
020 * <code>GradeBook</code>.
021 */
022@SuppressWarnings("serial")
023public class ClassPanel extends JPanel {
024  private GradeBook book;
025
026  private JFrame frame;
027
028  // GUI components we care about
029  private JLabel classNameLabel;
030  private JList<String> assignmentsList;
031  private AssignmentPanel assignmentPanel;
032
033  private JButton newAssignmentButton;
034  private JButton newStudentButton;
035  private LetterGradeRangesPanel undergradLetterGradeRangesPanel;
036  private LetterGradeRangesPanel gradLetterGradeRangesPanel;
037
038  /**
039   * Creats a <code>ClassPanel</code> and initializes its components.
040   */
041  public ClassPanel(JFrame frame) {
042    this.frame = frame;
043    this.setLayout(new BorderLayout());
044    this.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 2));
045
046    addClassNamePanel();
047    addAssignmentsPanel();
048    addLetterGradeRangesPanel();
049    addNewStudentPanel();
050  }
051
052  private void addLetterGradeRangesPanel() {
053    JTabbedPane letterGradeRangesPanel = new JTabbedPane();
054    Border letterGradeRangesBorder = BorderFactory.createTitledBorder("Letter Grades");
055    letterGradeRangesPanel.setBorder(letterGradeRangesBorder);
056
057    this.undergradLetterGradeRangesPanel = new LetterGradeRangesPanel(Student.Section.UNDERGRADUATE);
058    letterGradeRangesPanel.add("Undergraduate", this.undergradLetterGradeRangesPanel);
059
060    this.gradLetterGradeRangesPanel = new LetterGradeRangesPanel(Student.Section.GRADUATE);
061    letterGradeRangesPanel.add("Graduate", this.gradLetterGradeRangesPanel);
062
063    this.add(letterGradeRangesPanel, BorderLayout.EAST);
064  }
065
066  private void addAssignmentsPanel() {
067    // Set up the assignments panel
068    JPanel assignmentsPanel = new JPanel();
069    Border assignmentBorder =
070      BorderFactory.createTitledBorder("Assignments");
071    assignmentsPanel.setBorder(assignmentBorder);
072    assignmentsPanel.setLayout(new BorderLayout());
073
074    JPanel listPanel = addAssignmentsListPanel();
075    addAddAssignmentButton(listPanel);
076    assignmentsPanel.add(listPanel, BorderLayout.WEST);
077    addInfoPanel(assignmentsPanel);
078    this.add(assignmentsPanel, BorderLayout.CENTER);
079  }
080
081  private JPanel addAssignmentsListPanel() {
082    JPanel listPanel = new JPanel();
083    listPanel.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 2));
084    listPanel.setLayout(new BorderLayout());
085
086    this.assignmentsList = new JList<>();
087    assignmentsList.addListSelectionListener(new
088      ListSelectionListener() {
089        @Override
090        public void valueChanged(ListSelectionEvent e) {
091          String name = assignmentsList.getSelectedValue();
092          if (name != null) {
093            Assignment assign = book.getAssignment(name);
094            if (assign == null) {
095              String s = "No assignment named: " + name;
096              throw new IllegalArgumentException(s);
097            }
098            displayAssignment(assign);
099          }
100        }
101      });
102    listPanel.add(new JScrollPane(this.assignmentsList),
103                  BorderLayout.CENTER);
104    return listPanel;
105  }
106
107  private void addNewStudentPanel() {
108    // Add a panel for adding a new student
109    JPanel newStudentPanel = new JPanel();
110    newStudentPanel.setLayout(new FlowLayout());
111    newStudentPanel.add(new JLabel("New Student with id:"));
112    final JTextField newStudentField = new JTextField(12);
113    newStudentPanel.add(newStudentField);
114    this.newStudentButton = new JButton("Add");
115    this.newStudentButton.setEnabled(false);
116    this.newStudentButton.addActionListener(e -> {
117      String id = newStudentField.getText();
118      if (id == null || id.equals("")) {
119        showErrorMessageDialog("No student id specified");
120        return;
121      }
122
123      if (ClassPanel.this.book == null) {
124        showErrorMessageDialog("No grade book opened");
125        return;
126      }
127
128      Student student = new Student(id);
129      ClassPanel.this.book.addStudent(student);
130      newStudentField.setText("");
131    });
132    newStudentPanel.add(this.newStudentButton);
133    this.add(newStudentPanel, BorderLayout.SOUTH);
134  }
135
136  private void addAddAssignmentButton(JPanel listPanel) {
137    this.newAssignmentButton = new JButton("Add Assignment");
138    this.newAssignmentButton.setEnabled(false);
139    this.newAssignmentButton.addActionListener(e -> {
140      createAssignment();
141    });
142    listPanel.add(this.newAssignmentButton, BorderLayout.SOUTH);
143  }
144
145  private void addInfoPanel(JPanel assignmentsPanel) {
146    JPanel infoPanel = new JPanel();
147    infoPanel.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 2));
148    infoPanel.setLayout(new BorderLayout());
149    this.assignmentPanel = new AssignmentPanel(false);
150    infoPanel.add(assignmentPanel, BorderLayout.CENTER);
151
152    addUpdateButton(infoPanel);
153
154    assignmentsPanel.add(infoPanel, BorderLayout.CENTER);
155  }
156
157  private void addUpdateButton(JPanel infoPanel) {
158    JPanel p = new JPanel();
159    p.setLayout(new FlowLayout());
160    JButton updateButton = new JButton("Update");
161    updateButton.addActionListener(new ActionListener() {
162        @Override
163        public void actionPerformed(ActionEvent e) {
164          // Get the selected assignment and update it
165          String name = assignmentsList.getSelectedValue();
166          Assignment assign = book.getAssignment(name);
167
168          // Update currently selected assignment
169          updateAssignment(assign);
170        }
171      });
172    p.add(updateButton);
173    infoPanel.add(p, BorderLayout.SOUTH);
174  }
175
176  private void addClassNamePanel() {
177    JPanel classNamePanel = new JPanel();
178    classNamePanel.setLayout(new BoxLayout(classNamePanel,
179                                           BoxLayout.X_AXIS));
180    classNamePanel.add(Box.createHorizontalGlue());
181    this.classNameLabel = new JLabel();
182    classNamePanel.add(this.classNameLabel);
183    classNamePanel.add(Box.createHorizontalGlue());
184    this.add(classNamePanel, BorderLayout.NORTH);
185  }
186
187  private void showErrorMessageDialog(String message) {
188    JOptionPane.showMessageDialog(ClassPanel.this,
189      new String[]{message},
190      "Error",
191      JOptionPane.ERROR_MESSAGE);
192  }
193
194  /**
195   * Creates a new assignment based on the contents of the "name" and
196   * "max points" fields.  While we're at it, we add it to the
197   * assignment book and display it in the assignment list.
198   */
199  private Assignment createAssignment() {
200    NewAssignmentDialog dialog = new NewAssignmentDialog(this.frame);
201    dialog.pack();
202    dialog.setLocationRelativeTo(frame);
203    dialog.setVisible(true);
204
205    Assignment newAssign = dialog.getAssignment();
206
207    if (newAssign != null) {
208      book.addAssignment(newAssign);
209      this.displayAssignments(book);
210      this.assignmentsList.setSelectedValue(newAssign.getName(),
211                                            true);
212    }
213    return newAssign;
214  }
215
216  /**
217   * Fills in the contents of an <code>Assignment</code> based on the
218   * contents of the GUI.
219   */
220  private void updateAssignment(Assignment assign) {
221    this.assignmentPanel.updateAssignment(assign);
222  }
223
224  /**
225   * Sets the <code>GradeBook</code> that is displayed and edited in
226   * this <code>ClassPanel</code>.
227   */
228  void displayGradeBook(GradeBook book) {
229    this.book = book;
230    this.newAssignmentButton.setEnabled(true);
231    this.newStudentButton.setEnabled(true);
232
233    // Display name of the class
234    double points = 0.0d;
235    for (String name : book.getAssignmentNames()) {
236      Assignment assign = book.getAssignment( name );
237      points += assign.getPoints();
238    }
239
240    String className =
241      String.format( "%s (%d students, %.2f points)", book.getClassName(), book.getStudentIds().size(), points );
242    this.classNameLabel.setText(className);
243
244    this.displayAssignments(book);
245    this.assignmentsList.clearSelection();  // No initial selection
246
247    displayLetterGradeRanges(book);
248  }
249
250  private void displayLetterGradeRanges(GradeBook book) {
251    this.undergradLetterGradeRangesPanel.displayLetterGradeRanges(book.getLetterGradeRanges(Student.Section.UNDERGRADUATE));
252    this.gradLetterGradeRangesPanel.displayLetterGradeRanges(book.getLetterGradeRanges(Student.Section.GRADUATE));
253  }
254
255  /**
256   * Displays the names of all of the assignments in a given grade
257   * book in the assignmentsList.
258   */
259  void displayAssignments(GradeBook book) {
260    // Display all of the assignments
261    Vector<String> assignmentNames = new Vector<>();
262    assignmentNames.addAll(book.getAssignmentNames());
263    this.assignmentsList.setListData(assignmentNames);
264  }
265  
266  /**
267   * Displays an assignment in the appropriate fields
268   */
269  private void displayAssignment(Assignment assign) {
270    this.assignmentPanel.displayAssignment(assign);
271  }

273  /**
274   * Test program that displays and edits the information in a given
275   * grade book stored in an XML file.
276   */
277  public static void main(String[] args) {
278    if (args.length < 1) {
279      System.err.println("** usage: java ClassPanel xmlFile");
280      System.exit(1);
281    }
282
283    final String fileName = args[0];
284
285    GradeBook book = null;
286    try {
287      XmlGradeBookParser parser = new XmlGradeBookParser(fileName);
288      book = parser.parse();
289
290    } catch (FileNotFoundException ex) {
291      System.err.println("** Could not find file: " + ex.getMessage());
292      System.exit(1);
293      
294    } catch (IOException ex) {
295      System.err.println("** IOException during parsing: " + ex.getMessage());
296      System.exit(1);
297
298    } catch (ParserException ex) {
299      System.err.println("** Error during parsing: " + ex);
300      System.exit(1);
301    }
302
303    JFrame frame = new JFrame("ClassPanel test");
304    ClassPanel classPanel = new ClassPanel(frame);
305    classPanel.displayGradeBook(book);
306
307    final GradeBook theBook = book;
308    frame.addWindowListener(new WindowAdapter() {
309        @Override
310        public void windowClosing(WindowEvent e) {
311          // Write changes to grade book back to file
312          try {
313            XmlDumper dumper = new XmlDumper(fileName);
314            dumper.dump(theBook);
315
316          } catch (IOException ex) {
317            System.err.println("** Error while writing XML file: " + ex);
318          }
319
320          System.exit(1);
321        }
322      });
323
324    frame.getContentPane().add(classPanel);
325    
326    frame.pack();
327    frame.setVisible(true);
328  }
