001package edu.pdx.cs410J.grader;
002
003import javax.mail.Message;
004import javax.mail.MessagingException;
005import javax.mail.Session;
006import javax.mail.internet.InternetAddress;
007import javax.mail.internet.MimeMessage;
008import java.util.Properties;
009
010public class EmailSender {
011  /**
012   * The name of the SMTP server that is used to send email
013   */
014  protected static String serverName = "mailhost.cs.pdx.edu";
015
016  private static int emailServerPort = 25;
017
018  protected static MimeMessage newEmailTo(Session session, String recipient, String subject) throws MessagingException {
019    // Make a new email message
020    MimeMessage message = new MimeMessage(session);
021
022    InternetAddress[] to = {new InternetAddress(recipient)};
023    message.setRecipients(Message.RecipientType.TO, to);
024    message.setSubject(subject);
025    return message;
026  }
027
028  protected static Session newEmailSession(boolean debug) {
029    // Obtain a Session for sending email
030    Properties props = new Properties();
031    props.put("mail.smtp.host", serverName);
032    props.put("mail.smtp.port", emailServerPort);
033    props.put("mail.smtp.localhost", "127.0.0.1");
034    Session session = Session.getDefaultInstance(props, null);
035    session.setDebug(debug);
036    return session;
037  }
038
039  public void setEmailServerPort(int port) {
040    emailServerPort = port;
041  }
042}
