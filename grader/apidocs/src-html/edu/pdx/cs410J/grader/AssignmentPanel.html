016/**
017 * This panel is used to display and edit <code>Assignment</code>s.
018 */
019@SuppressWarnings("serial")
020public class AssignmentPanel extends JPanel {
021  private static final String QUIZ = "Quiz";
022  private static final String PROJECT = "Project";
023  private static final String OTHER = "Other";
024  private static final String OPTIONAL = "Optional";
025  private static final String POA = "POA";
026
027  static final String DATE_TIME_FORMAT_PATTERN = "M/d/yyyy h:mm a";
028  static final DateTimeFormatter DATE_TIME_FORMAT = DateTimeFormatter.ofPattern(DATE_TIME_FORMAT_PATTERN);
029
030  // GUI components we care about
031  private JTextField nameField;
032  private JComboBox<String> typeBox;
033  private JTextField pointsField;
034  private JTextField descriptionField;
035  private JTextField dueDateField;
036  private NotesPanel notes;
037
038  /**
039   * Creates and adds GUI components to a new
040   * <code>AssignmentPanel</code>.
041   */
042  public AssignmentPanel(boolean canCreate) {
043    this.setLayout(new BorderLayout());
044
045    // Panel containing information about an assignment
046    JPanel infoPanel = new JPanel();
047    infoPanel.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 2));
048    infoPanel.setLayout(new BorderLayout());
049
050    addAssignmentInformationWidgets(infoPanel);
051
052    this.add(infoPanel, BorderLayout.NORTH);
053
054    if (!canCreate) {
055      this.nameField.setEditable(false);
056    }
057
058    // Add a NotePanel
059    this.notes = new NotesPanel();
060    this.notes.setNotable(new Notable() {
061        private ArrayList<String> notes = new ArrayList<>();
062
063        @Override
064        public java.util.List<String> getNotes() {
065          return notes;
066        }
067
068        @Override
069        public void addNote(String note) {
070          notes.add(note);
071        }
072
073        @Override
074        public void removeNote(String note) {
075          notes.remove(note);
076        }
077      });
078    this.add(notes, BorderLayout.CENTER);
079  }
080
081  private void addAssignmentInformationWidgets(JPanel infoPanel) {
082    JPanel labels = new JPanel();
083    labels.setLayout(new GridLayout(0, 1));
084    labels.setBorder(BorderFactory.createEmptyBorder(0, 2, 0, 2));
085    labels.add(new JLabel("Name:"));
086    labels.add(new JLabel("Max points:"));
087    labels.add(new JLabel("Type:"));
088    labels.add(new JLabel("Description:"));
089    labels.add(new JLabel("Due date:"));
090
091    JPanel fields = new JPanel();
092    fields.setLayout(new GridLayout(0, 1));
093    this.nameField = new JTextField(8);
094    fields.add(this.nameField);
095    this.pointsField = new JTextField(5);
096    fields.add(this.pointsField);
097
098    addTypeBox(fields);
099
100    this.descriptionField = new JTextField(20);
101    fields.add(this.descriptionField);
102    this.dueDateField = new JTextField(22);
103    fields.add(this.dueDateField);
104
105    infoPanel.add(labels, BorderLayout.WEST);
106    infoPanel.add(fields, BorderLayout.CENTER);
107  }
108
109  private void addTypeBox(JPanel fields) {
110    this.typeBox = new JComboBox<>();
111    this.typeBox.addItem(QUIZ);
112    this.typeBox.addItem(PROJECT);
113    this.typeBox.addItem(POA);
114    this.typeBox.addItem(OTHER);
115    this.typeBox.addItem(OPTIONAL);
116    fields.add(this.typeBox);
117  }
118
119  /**
120   * Creates a new <code>Assignment</code> based on the contents of
121   * this <code>AssignmentPanel</code>.
122   */
123  public Assignment createAssignment() {
124    // Get the name and max points of the assignment
125    String name = nameField.getText();
126    if (isNullOrEmpty(name)) {
127      return error("No assignment name specified");
128    }
129
130    String points = pointsField.getText();
131    if (isNullOrEmpty(points)) {
132      return error("No points value specified");
133    }
134
135    // Create a new Assignment object
136    try {
137      double d = Double.parseDouble(points);
138      Assignment newAssign = new Assignment(name, d);
139      this.notes.getNotable().getNotes().forEach(newAssign::addNote);
140
141      updateAssignment(newAssign);
142      this.notes.setNotable(newAssign);
143      return newAssign;
144      
145    } catch (NumberFormatException ex) {
146      return error(points + " is not a number");
147    }    
148  }
149
150  private boolean isNullOrEmpty(String name) {
151    return name == null || name.equals("");
152  }
153
154  private Assignment error(String message) {
155    JOptionPane.showMessageDialog(AssignmentPanel.this,
156      new String[]{message},
157      "Error",
158      JOptionPane.ERROR_MESSAGE);
159    return null;
160  }
161
162  /**
163   * Displays the contents of a given <code>Assignment</code> in this
164   * <code>AssignmentPanel</code>.
165   */
166  public void displayAssignment(Assignment assign) {
167    this.nameField.setText(assign.getName());
168    this.pointsField.setText(String.valueOf(assign.getPoints()));
169    Assignment.AssignmentType type = assign.getType();
170    if (type == Assignment.AssignmentType.QUIZ) {
171      this.typeBox.setSelectedItem(QUIZ);
172
173    } else if (type == Assignment.AssignmentType.PROJECT) {
174      this.typeBox.setSelectedItem(PROJECT);
175
176    } else if (type == Assignment.AssignmentType.OTHER) {
177      this.typeBox.setSelectedItem(OTHER);
178
179    } else if (type == Assignment.AssignmentType.OPTIONAL) {
180      this.typeBox.setSelectedItem(OPTIONAL);
181
182    } else if (type == Assignment.AssignmentType.POA) {
183      this.typeBox.setSelectedItem(POA);
184
185    } else {
186      String s = "Invalid assignment type: " + type;
187      throw new IllegalArgumentException(s);
188    }
189
190    String description = assign.getDescription();
191    if (isNotEmpty(description)) {
192      this.descriptionField.setText(description);
193
194    } else {
195      this.descriptionField.setText("");
196    }
197
198    LocalDateTime dueDate = assign.getDueDate();
199    if (dueDate != null) {
200      this.dueDateField.setText(DATE_TIME_FORMAT.format(dueDate));
201
202    } else {
203      this.dueDateField.setText("");
204    }
205
206    this.notes.setNotable(assign);
207  }
208
209  /**
210   * Fills in the contents of an <code>Assignment</code> based on the
211   * contents of this <code>AssignmentPanel</code>.
212   */
213  public void updateAssignment(Assignment assign) {
214    String points = pointsField.getText();
215    if (isNullOrEmpty(points)) {
216      error("No points value specified");
217      return;
218    }
219
220    try {
221      double d = Double.parseDouble(points);
222      assign.setPoints(d);
223      
224    } catch (NumberFormatException ex) {
225      error(points + " is not a number");
226      return;
227    }
228
229    setAssignmentType(assign);
230
231    String description = this.descriptionField.getText();
232
233    if (isNotEmpty(description)) {
234      assign.setDescription(description);
235    }
236
237    String dueDate = this.dueDateField.getText();
238    if (isNotEmpty(dueDate)) {
239      try {
240        assign.setDueDate(LocalDateTime.parse(dueDate.trim(), DATE_TIME_FORMAT));
241
242      } catch (DateTimeParseException ex) {
243        error(dueDate + " is not a validate date (" + DATE_TIME_FORMAT_PATTERN + ")");
244      }
245
246    } else {
247      assign.setDueDate(null);
248    }
249
250    // Adding notes is taken care of by the NotesPanel
251  }
252
253  private void setAssignmentType(Assignment assign) {
254    String type = (String) this.typeBox.getSelectedItem();
255
256    switch (type) {
257      case QUIZ:
258        assign.setType(Assignment.AssignmentType.QUIZ);
259        break;
260      case PROJECT:
261        assign.setType(Assignment.AssignmentType.PROJECT);
262        break;
263      case OTHER:
264        assign.setType(Assignment.AssignmentType.OTHER);
265        break;
266      case OPTIONAL:
267        assign.setType(Assignment.AssignmentType.OPTIONAL);
268        break;
269      case POA:
270        assign.setType(Assignment.AssignmentType.POA);
271        break;
272      default:
273        String s = "Unknown assignment type: " + type;
274        throw new IllegalArgumentException(s);
275    }
276  }
277
278  private boolean isNotEmpty(String description) {
279    return description != null && !description.equals("");
280  }

285  public static void main(String[] args) {
286    String fileName = args[0];
287    String assignName = args[1];
288
289    GradeBook book = null;
290    try {
291      XmlGradeBookParser parser = new XmlGradeBookParser(fileName);
292      book = parser.parse();
293
294    } catch (FileNotFoundException ex) {
295      System.err.println("** Could not find file: " + ex.getMessage());
296      System.exit(1);
297      
298    } catch (IOException ex) {
299      System.err.println("** IOException during parsing: " + ex.getMessage());
300      System.exit(1);
301
302    } catch (ParserException ex) {
303      System.err.println("** Error during parsing: " + ex);
304      System.exit(1);
305    }
306
307    Assignment assign = book.getAssignment(assignName);
308    if (assign == null) {
309      System.err.println("Not such assignment: " + assignName);
310      System.exit(1);
311    }
312
313    AssignmentPanel assignPanel = new AssignmentPanel(false);
314    assignPanel.displayAssignment(assign);
315
316    JFrame frame = new JFrame("AssignmentPanel test");
317    frame.addWindowListener(new WindowAdapter() {
318        @Override
319        public void windowClosing(WindowEvent e) {
320          System.exit(1);
321        }
322      });
323
324    frame.getContentPane().add(assignPanel);
325    
326    frame.pack();
327    frame.setVisible(true);
328  }
