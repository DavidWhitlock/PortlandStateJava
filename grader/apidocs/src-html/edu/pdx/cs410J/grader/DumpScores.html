001package edu.pdx.cs410J.grader;
002
003import java.io.*;
004import java.util.*;
005
006/**
007 * This program sorts the scores for a given assignment and dumps them
008 * to standard out.
009 */
010public class DumpScores {
011
012  /**
013   * Inner class that represents a score/Student tuple
014   */
015  private static class Tuple implements Comparable {
016    private double score;
017    private Student student;
018
019    Tuple(Student student, double score) {
020      this.student = student;
021      this.score = score;
022    }
023
024    public int compareTo(Object o) {
025      Tuple other = (Tuple) o;
026      if (this.score == other.score) {
027        return this.student.getId().compareTo(other.student.getId());
028
029      } else if (this.score > other.score) {
030        return -1;
031
032      } else if (this.score < other.score) {
033        return 1;
034
035      } else {
036//          assert this.score == other.score;
037        return 0;
038      }
039    }
040
041    public String toString() {
042      return this.student + ": " + this.score;
043    }
044    
045  }
046
047  //////////////////////  Main Program  ///////////////////////
048
049  private static PrintStream out = System.out;
050  private static PrintStream err = System.err;
051
052  /**
053   * Prints usage information about this program
054   */
055  private static void usage(String s) {
056    err.println("\n** " + s + "\n");
057    err.println("usage: java DumpScores xmlFile assignment");
058    err.println("");
059    err.println("Dumps the grades for a given assignment to " +
060                "standard out");
061    err.println("");
062    System.exit(1);
063  }
064
065  public static void main(String[] args) throws Throwable {
066    String xmlFileName = null;
067    String assignmentName = null;
068
069    for (int i = 0; i < args.length; i++) {
070      if (xmlFileName == null) {
071        xmlFileName = args[i];
072
073      } else if (assignmentName == null) {
074        assignmentName = args[i];
075
076      } else {
077        usage("Extraneous command line: " + args[i]);
078      }
079    }
080
081    if (xmlFileName == null) {
082      usage("Missing XML file");
083
084    } else if (assignmentName == null) {
085      usage("Missing assignment");
086    }
087
088    File xmlFile = new File(xmlFileName);
089    if (!xmlFile.exists()) {
090      usage("File \"" + xmlFileName + "\" does not exist");
091    }
092
093    GradeBook book = (new XmlGradeBookParser(xmlFile)).parse();
094    Assignment assign = book.getAssignment(assignmentName);
095    if (assign == null) {
096      usage("No such assignment \"" + assignmentName + "\"");
097    }
098
099    // Maps score to the student
100    SortedSet<Tuple> scores = new TreeSet<Tuple>();
101
102    book.forEachStudent(student -> {
103      Grade grade = student.getGrade(assign.getName());
104      if (grade != null) {
105        double score = grade.getScore();
106        scores.add(new Tuple(student, score));
107      }
108    });
109
110    out.println("\nGrades for " + assign + ": " +
111                assign.getDescription());
112
113    Iterator iter = scores.iterator();
114    while (iter.hasNext()) {
115      out.println("  " + iter.next());
116    }
117
118    out.println("");
119  }
120
121}
