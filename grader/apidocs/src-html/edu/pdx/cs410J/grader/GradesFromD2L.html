001package edu.pdx.cs410J.grader;
002
003import java.util.*;
004
005public class GradesFromD2L {
006  private List<D2LStudent> students = new ArrayList<>();
007
008  public static D2LStudentBuilder newStudent() {
009    return new D2LStudentBuilder();
010  }
011
012  public void addStudent(D2LStudent student) {
013    this.students.add(student);
014  }
015
016  public Optional<Student> findStudentInGradebookForD2LStudent(D2LStudent d2lStudent, GradeBook book) {
017    return book.studentsStream().filter((Student student) -> {
018      if (haveSameD2LId(d2lStudent, student)) {
019        return true;
020
021      } else if (studentIdIsSameAsD2LId(d2lStudent, student)) {
022        student.setD2LId(d2lStudent.getD2lId());
023        return true;
024
025      } else if (haveSameFirstAndLastNameIgnoringCase(d2lStudent, student)) {
026        student.setD2LId(d2lStudent.getD2lId());
027        return true;
028
029      } else {
030        return false;
031      }
032    }).findAny();
033  }
034
035  private boolean studentIdIsSameAsD2LId(D2LStudent d2lStudent, Student student) {
036    return d2lStudent.getD2lId().equals(student.getId());
037  }
038
039  private boolean haveSameFirstAndLastNameIgnoringCase(D2LStudent d2lStudent, Student student) {
040    return d2lStudent.getFirstName().equalsIgnoreCase(student.getFirstName()) &&
041      d2lStudent.getLastName().equalsIgnoreCase(student.getLastName());
042  }
043
044  private boolean haveSameD2LId(D2LStudent d2lStudent, Student student) {
045    return d2lStudent.getD2lId().equals(student.getD2LId());
046  }
047
048  public List<D2LStudent> getStudents() {
049    return this.students;
050  }
051
052  public Optional<Assignment> findAssignmentInGradebookForD2lQuiz(String quizName, GradeBook gradebook) {
053    Assignment assignment = gradebook.getAssignment(quizName);
054    if (assignment != null) {
055      return Optional.ofNullable(assignment);
056    }
057
058    return findAssignmentInGradebookLike(quizName, gradebook);
059  }
060
061  private Optional<Assignment> findAssignmentInGradebookLike(String quizName, GradeBook gradebook) {
062    for (String assignmentName : gradebook.getAssignmentNames()) {
063      Assignment assignment = gradebook.getAssignment(assignmentName);
064      if (quizName.startsWith(assignmentName)) {
065        return Optional.ofNullable(assignment);
066      }
067
068      if (quizName.startsWith(assignment.getDescription())) {
069        return Optional.of(assignment);
070      }
071    }
072    return Optional.empty();
073  }
074
075  static class D2LStudent {
076    private final String firstName;
077    private final String lastName;
078    private final String d2lId;
079    private Map<String, Double> scores = new HashMap<>();
080
081    public D2LStudent(String firstName, String lastName, String d2lId) {
082      this.firstName = firstName;
083      this.lastName = lastName;
084      this.d2lId = d2lId;
085    }
086
087    public String getD2lId() {
088      return d2lId;
089    }
090
091    public String getFirstName() {
092      return firstName;
093    }
094
095    public String getLastName() {
096      return lastName;
097    }
098
099    public void setScore(String quizName, double score) {
100      this.scores.put(quizName, score);
101    }
102
103    public Double getScore(String quizName) {
104      return this.scores.get(quizName);
105    }
106
107    public Iterable<String> getQuizNames() {
108      return this.scores.keySet();
109    }
110
111    @Override
112    public String toString() {
113      return "D2L student \"" + getFirstName() + " " + getLastName() + "\" with id " + getD2lId();
114    }
115  }
116
117  static class D2LStudentBuilder {
118    private String firstName;
119    private String lastName;
120    private String d2lId;
121
122    private D2LStudentBuilder() {
123
124    }
125
126    public D2LStudentBuilder setFirstName(String firstName) {
127      this.firstName = firstName;
128      return this;
129    }
130
131    public D2LStudentBuilder setLastName(String lastName) {
132      this.lastName = lastName;
133      return this;
134    }
135
136    public D2LStudentBuilder setD2LId(String d2lId) {
137      this.d2lId = d2lId;
138      return this;
139    }
140
141    public D2LStudent create() {
142      if (firstName == null) {
143        throw new IllegalStateException("Missing first name");
144      }
145
146      if (lastName == null) {
147        throw new IllegalStateException("Missing last name");
148      }
149
150      if (d2lId == null) {
151        throw new IllegalStateException("Missing d2l Id");
152      }
153
154      return new D2LStudent(firstName, lastName, d2lId);
155    }
156  }
157}
