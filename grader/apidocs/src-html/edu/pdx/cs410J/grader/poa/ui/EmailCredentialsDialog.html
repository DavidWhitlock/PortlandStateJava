001package edu.pdx.cs410J.grader.poa.ui;
002
003import com.google.inject.Inject;
004import com.google.inject.Singleton;
005import edu.pdx.cs410J.grader.poa.EmailCredentialsView;
006
007import javax.swing.*;
008import java.awt.*;
009import java.awt.event.FocusEvent;
010import java.awt.event.FocusListener;
011import java.util.ArrayList;
012import java.util.List;
013
014@Singleton
015public class EmailCredentialsDialog extends JDialog implements EmailCredentialsView {
016
017  private final JTextField emailAddressField = new JTextField(30);
018  private final JPasswordField passwordField = new JPasswordField(30);
019  private final JButton okButton = new JButton("OK");
020  private final JButton cancelButton = new JButton("Cancel");
021  private final List<SubmitCredentialsListener> submitCredentialsListeners = new ArrayList<>();
022
023  @Inject
024  public EmailCredentialsDialog(TopLevelJFrame parent) {
025    super(parent, "Enter Email Credentials", true);
026    this.setVisible(false);
027
028    Container rootPane1 = this.getContentPane();
029    rootPane1.setLayout(new BorderLayout());
030    rootPane1.add(createCredentialsPanel(), BorderLayout.CENTER);
031    rootPane1.add(createOkCancelPanel(), BorderLayout.SOUTH);
032
033    this.cancelButton.addActionListener(e -> hideDialog());
034    this.okButton.addActionListener(e -> submitCredentials());
035  }
036
037  private JPanel createOkCancelPanel() {
038    JPanel panel = new JPanel();
039    BoxLayout layout = new BoxLayout(panel, BoxLayout.LINE_AXIS);
040    panel.setLayout(layout);
041
042    panel.add(Box.createHorizontalGlue());
043    panel.add(okButton);
044    panel.add(cancelButton);
045
046    return panel;
047  }
048
049  private JPanel createCredentialsPanel() {
050    JPanel credentialsPanel = new JPanel(new GridBagLayout());
051    addComponentOnGrid(new JLabel("Email Address:"), credentialsPanel, 0, 0);
052    addComponentOnGrid(this.emailAddressField, credentialsPanel, 0, 1);
053    addComponentOnGrid(new JLabel("Password:"), credentialsPanel, 1, 0);
054    addComponentOnGrid(this.passwordField, credentialsPanel, 1, 1);
055
056    return credentialsPanel;
057  }
058
059  private void addComponentOnGrid(JComponent component, JPanel container, int row, int column) {
060    GridBagConstraints constraints = new GridBagConstraints();
061    constraints.fill = GridBagConstraints.HORIZONTAL;
062    constraints.gridx = column;
063    constraints.gridy = row;
064
065    container.add(component, constraints);
066  }
067
068
069  @Override
070  public void setIsVisible(boolean isVisible) {
071    this.pack();
072    this.setVisible(isVisible);
073  }
074
075  @Override
076  public void addEmailAddressValueListener(EmailAddressValueListener listener) {
077    this.emailAddressField.addFocusListener(new FocusListener() {
078      @Override
079      public void focusGained(FocusEvent e) {
080
081      }
082
083      @Override
084      public void focusLost(FocusEvent e) {
085        listener.setEmailAddress(emailAddressField.getText());
086      }
087    });
088
089    this.emailAddressField.addActionListener(e -> {
090      listener.setEmailAddress(emailAddressField.getText());
091      submitCredentials();
092    });
093  }
094
095  @Override
096  public void addPasswordValueListener(PasswordValueListener listener) {
097    this.passwordField.addFocusListener(new FocusListener() {
098      @Override
099      public void focusGained(FocusEvent e) {
100
101      }
102
103      @Override
104      public void focusLost(FocusEvent e) {
105        listener.setPassword(getPasswordFromWidget());
106      }
107    });
108
109    this.passwordField.addActionListener(e -> {
110      listener.setPassword(getPasswordFromWidget());
111      submitCredentials();
112    });
113  }
114
115  private void submitCredentials() {
116    this.submitCredentialsListeners.forEach(this::submitCredentials);
117  }
118
119  private String getPasswordFromWidget() {
120    return new String(passwordField.getPassword());
121  }
122
123  @Override
124  public void addSubmitCredentialsListener(SubmitCredentialsListener listener) {
125    this.submitCredentialsListeners.add(listener);
126  }
127
128  private void submitCredentials(SubmitCredentialsListener listener) {
129    listener.submitCredentials();
130    hideDialog();
131  }
132
133  private void hideDialog() {
134    this.setVisible(false);
135  }
136
137}
