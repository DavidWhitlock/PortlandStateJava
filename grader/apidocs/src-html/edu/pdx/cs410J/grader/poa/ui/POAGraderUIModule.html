001package edu.pdx.cs410J.grader.poa.ui;
002
003import com.google.common.eventbus.EventBus;
004import com.google.inject.AbstractModule;
005import com.google.inject.Provides;
006import com.google.inject.Singleton;
007import edu.pdx.cs410J.grader.poa.*;
008
009public class POAGraderUIModule extends AbstractModule {
010  @Override
011  protected void configure() {
012
013    bind(POASubmissionsView.class).to(POASubmissionsPanel.class);
014    bind(POASubmissionView.class).to(POASubmissionInformationWidgets.class);
015    bind(GradeBookView.class).to(GradeBookWidget.class);
016    bind(UnhandledExceptionView.class).to(UnhandledExceptionDialog.class).asEagerSingleton();
017    bind(StudentsView.class).to(StudentsWidget.class);
018    bind(POAAssignmentsView.class).to(POAAssignmentsWidget.class);
019    bind(POAGradeView.class).to(POAGradeWidgets.class);
020    bind(EmailCredentialsView.class).to(EmailCredentialsDialog.class).asEagerSingleton();
021
022    bind(POASubmissionsPresenter.class).asEagerSingleton();
023    bind(POASubmissionPresenter.class).asEagerSingleton();
024    bind(GradeBookPresenter.class).asEagerSingleton();
025    bind(UnhandledExceptionPresenter.class).asEagerSingleton();
026    bind(StudentsPresenter.class).asEagerSingleton();
027    bind(POAAssignmentsPresenter.class).asEagerSingleton();
028    bind(POAGradePresenter.class).asEagerSingleton();
029    bind(EmailCredentialsPresenter.class).asEagerSingleton();
030
031    bind(GradeBookFileManager.class).asEagerSingleton();
032    bind(POASubmissionsDownloader.class).asEagerSingleton();
033  }
034
035  @Provides
036  @Singleton
037  public EventBus provideEventBus() {
038    return new EventBusThatPublishesUnhandledExceptionEvents();
039  }
040
041}
