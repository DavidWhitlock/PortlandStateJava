001package edu.pdx.cs410J.grader.poa.ui;
002
003import com.google.inject.Inject;
004import com.google.inject.Singleton;
005import edu.pdx.cs410J.grader.poa.POASubmissionView;
006
007import javax.swing.*;
008
009@Singleton
010public class POASubmissionInformationWidgets implements POASubmissionView {
011
012  private final JLabel subjectLabel;
013  private final JLabel submitterLabel;
014  private final JLabel submissionTimeLabel;
015  private final JTextArea submissionContent;
016  private final POAGradeWidgets gradesWidgets;
017
018  @Inject
019  public POASubmissionInformationWidgets(POAGradeWidgets gradesWidgets) {
020    this.gradesWidgets = gradesWidgets;
021
022    this.subjectLabel = new JLabel();
023    this.submitterLabel = new JLabel();
024    this.submissionTimeLabel = new JLabel();
025    this.submissionContent = new JTextArea(20, 80);
026    this.submissionContent.setEditable(false);
027  }
028
029  public JComponent getSubjectWidget() {
030    return createLabeledWidget("Subject:", this.subjectLabel);
031  }
032
033  private JPanel createLabeledWidget(String label, JLabel widget) {
034    JPanel panel = new JPanel();
035    panel.setLayout(new BoxLayout(panel, BoxLayout.LINE_AXIS));
036    panel.add(new JLabel(label));
037    panel.add(Box.createHorizontalStrut(3));
038    panel.add(widget);
039    return panel;
040  }
041
042  public JComponent getSubmitterWidget() {
043    return createLabeledWidget("Submitted by:", this.submitterLabel);
044  }
045
046  public JComponent getSubmissionTimeWidget() {
047    JPanel panel = createLabeledWidget("Submitted on:", this.submissionTimeLabel);
048    panel.add(Box.createHorizontalGlue());
049    panel.add(gradesWidgets.getIsLateCheckbox());
050    return panel;
051  }
052
053  @Override
054  public void setSubmissionSubject(String subject) {
055    this.subjectLabel.setText(subject);
056  }
057
058  @Override
059  public void setSubmissionSubmitter(String submitter) {
060    this.submitterLabel.setText(submitter);
061  }
062
063  @Override
064  public void setSubmissionTime(String time) {
065    this.submissionTimeLabel.setText(time);
066  }
067
068  @Override
069  public void setContent(String content) {
070    this.submissionContent.setText(content);
071    scrollPaneToTop();
072  }
073
074  private void scrollPaneToTop() {
075    this.submissionContent.setCaretPosition(0);
076  }
077
078  public JComponent getSubmissionContentWidget() {
079    return new JScrollPane(this.submissionContent);
080  }
081}
