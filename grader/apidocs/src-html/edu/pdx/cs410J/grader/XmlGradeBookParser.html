020/**
021 * This class creates a <code>GradeBook</code> from the contents of an
022 * XML file.
023 */
024public class XmlGradeBookParser extends XmlHelper {
025  private GradeBook book;    // The grade book we're creating
026  private InputStream in;    // Input source of grade book
027  private File studentDir;   // Where the student XML file live

042    this(new FileInputStream(file));

055  /**
056   * Sets the directory in which the XML files for students are
057   * generated.
058   */
059  public void setStudentDir(File dir) {
060    if (!dir.exists()) {
061      throw new IllegalArgumentException(dir + " does not exist");
062    }
063
064    if (!dir.isDirectory()) {
065      throw new IllegalArgumentException(dir + " is not a directory");
066    }
067
068    this.studentDir = dir;
069  }

071  /**
072   * Extracts an <code>Assignment</code> from an <code>Element</code>
073   */
074  private static Assignment extractAssignmentFrom(Element element) 
075    throws ParserException {
076    Assignment assign = null;
077
078    String name = null;
079    String description = null;
080
081    NodeList children = element.getChildNodes();
082    for (int i = 0; i < children.getLength(); i++) {
083      Node node = children.item(i);
084      if (!(node instanceof Element)) {
085        continue;
086      }
087
088      Element child = (Element) node;
089      if (child.getTagName().equals("name")) {
090        name = extractTextFrom(child);
091
092      } else if (child.getTagName().equals("description")) {
093        description = extractTextFrom(child);
094
095      } else if (child.getTagName().equals("points")) {
096        String points = extractTextFrom(child);
097        try {
098          if (name == null) {
099            throw new ParserException("No name for assignment with " +
100              "points " + points);
101          }
102          assign = new Assignment(name, Double.parseDouble(points));
103          if (description != null) {
104            assign.setDescription(description);
105          }
106
107        } catch (NumberFormatException ex) {
108          throw new ParserException("Invalid points value: " +
109            points);
110        }
111
112      } else if (child.getTagName().equals("due-date")) {
113        String dueDate = extractTextFrom(child);
114        try {
115          assign.setDueDate(LocalDateTime.parse(dueDate, DATE_TIME_FORMAT));
116
117        } catch(DateTimeParseException ex) {
118          throw new ParserException("Invalidate LocalDateTime: " + dueDate, ex);
119        }
120
121      } else if (child.getTagName().equals("notes")) {
122        for (String note : extractNotesFrom(child)) {
123          assign.addNote(note);
124        }
125      }
126    }
127
128    if (assign == null ) {
129      throw new ParserException("No assignment found!");
130    }
131
132    setAssignmentTypeFromXml(element, assign);
133
134    return assign;
135  }
136
137  private static void setAssignmentTypeFromXml(Element assignmentElement, Assignment assign) throws ParserException {
138    String type = assignmentElement.getAttribute("type");
139    switch (type) {
140      case "PROJECT":
141        assign.setType(Assignment.AssignmentType.PROJECT);
142        break;
143      case "QUIZ":
144        assign.setType(Assignment.AssignmentType.QUIZ);
145        break;
146      case "OTHER":
147        assign.setType(Assignment.AssignmentType.OTHER);
148        break;
149      case "OPTIONAL":
150        assign.setType(Assignment.AssignmentType.OPTIONAL);
151        break;
152      case "POA":
153        assign.setType(Assignment.AssignmentType.POA);
154        break;
155      default:
156        throw new ParserException("Unknown assignment type: " + type);
157    }
158  }

160  /**
161   * Parses the source and from it creates a <code>GradeBook</code>.
162   */
163  public GradeBook parse() throws ParserException {
164    Document doc = parseDocumentFromInputStream();
165
166
167    Element root = null;
168    if (doc != null) {
169      root = doc.getDocumentElement();
170    }
171    if (doc == null || root == null) {
172      throw new ParserException("Document parsing failed");
173    }
174
175    NodeList children = root.getChildNodes();
176    for (int i = 0; i < children.getLength(); i++) {
177      Node node = children.item(i);
178      if (!(node instanceof Element)) {
179        continue;
180      }
181
182      Element child = (Element) node;
183      if (child.getTagName().equals("name")) {
184        this.book = new GradeBook(extractTextFrom(child));
185              this.book.setDirty(false);
186
187      } else if (this.book == null) {
188        throw new ParserException("name element is not first");
189
190      } else if (child.getTagName().equals("assignments")) {
191        extractAssignmentsFrom(child);
192
193      } else if (child.getTagName().equals("letter-grade-ranges")) {
194        extractLetterGradeRangesFrom(child);
195
196      } else if (child.getTagName().equals("students")) {
197        extractStudentsFrom(child);
198
199      } else if (child.getTagName().equals("lateDays")) {
200        // Fill in later, maybe.
201      }
202    }
203
204    if (this.book != null) {
205      // The book is initially clean
206      this.book.makeClean();
207    }
208
209    return this.book;
210  }
211
212  private void extractLetterGradeRangesFrom(Element parent) {
213    Student.Section section;
214    String attributeName = "for-section";
215    if (parent.hasAttribute(attributeName)) {
216      String value = parent.getAttribute(attributeName);
217      section = Student.Section.fromString(value);
218
219    } else {
220      section = Student.Section.UNDERGRADUATE;
221    }
222
223    NodeList ranges = parent.getChildNodes();
224    for (int j = 0; j < ranges.getLength(); j++) {
225      Node range = ranges.item(j);
226
227      if (!(range instanceof Element)) {
228        continue;
229      }
230
231      extractLetterGradeRangeFrom((Element) range, section);
232    }
233  }
234
235  private void extractLetterGradeRangeFrom(Element element, Student.Section section) {
236    String letterGradeString = element.getAttribute("letter-grade");
237    LetterGrade letterGrade = LetterGrade.fromString(letterGradeString);
238    LetterGradeRange range = this.book.getLetterGradeRanges(section).getRange(letterGrade);
239    range.setRange(toInt(element.getAttribute("minimum-score")), toInt(element.getAttribute("maximum-score")));
240
241  }
242
243  private int toInt(String value) {
244    return Integer.parseInt(value);
245  }
246
247  private void extractStudentsFrom(Element child) throws ParserException {
248    NodeList students = child.getChildNodes();
249    for (int j = 0; j < students.getLength(); j++) {
250      Node student = students.item(j);
251
252      if (!(student instanceof Element)) {
253        continue;
254      }
255
256      Element idElement = (Element) student;
257      if (idElement.getTagName().equals("id")) {
258        String id = extractTextFrom(idElement);
259            // Locate the XML file for the Student
260        File file =
261          new File(this.studentDir, id + ".xml");
262        if (!file.exists()) {
263          throw new IllegalArgumentException("No XML file for " +
264                                             id);
265        }
266
267        try {
268          XmlStudentParser sp = new XmlStudentParser(file);
269          Student stu = sp.parseStudent();
270          this.book.addStudent(stu);
271
272        } catch (Exception ex) {
273          String s = "While parsing " + file + ": " + ex;
274          throw new ParserException(s);
275        }
276      }
277    }
278  }
279
280  private void extractAssignmentsFrom(Element child) throws ParserException {
281    NodeList assignments = child.getChildNodes();
282    for (int j = 0; j < assignments.getLength(); j++) {
283      Node assignment = assignments.item(j);
284      if (assignment instanceof Element) {
285        Assignment assign =
286          extractAssignmentFrom((Element) assignment);
287        this.book.addAssignment(assign);
288      }
289    }
290  }
291
292  private Document parseDocumentFromInputStream() throws ParserException {
293    // Parse the source
294    Document doc;
295
296    // Create a DOM tree from the XML source
297    try {
298      DocumentBuilderFactory factory =
299        DocumentBuilderFactory.newInstance();
300      factory.setValidating(true);
301
302      DocumentBuilder builder =
303        factory.newDocumentBuilder();
304      builder.setErrorHandler(this);
305      builder.setEntityResolver(this);
306
307      doc = builder.parse(new InputSource(this.in));
308
309    } catch (ParserConfigurationException | SAXException | IOException ex) {
310      throw new ParserException("While parsing XML source: " + ex);
311
312    }
313    return doc;
314  }
