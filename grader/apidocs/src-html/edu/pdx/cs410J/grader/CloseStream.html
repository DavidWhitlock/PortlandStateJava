001package edu.pdx.cs410J.grader;
002
003import java.io.*;
004
005/**
006 * Can we still write to a stream after we close it?
007 */
008public class CloseStream {
009
010  public static void main(String[] args) {
011    FileOutputStream fos = null;
012
013    try {
014      fos = new FileOutputStream("TEST");
015
016    } catch (FileNotFoundException ex) {
017      ex.printStackTrace();
018      System.exit(1);
019    }
020
021    PrintWriter pw1 = new PrintWriter(fos, true);
022
023    pw1.println("First line");
024
025    PrintWriter pw2 = new PrintWriter(fos, true);
026
027    try {
028      fos.close();
029
030    } catch (IOException ex) {
031      ex.printStackTrace();
032      System.exit(1);
033    }
034
035    pw2.println("Second line");
036
037    System.out.println("error = " + pw2.checkError());
038  }
039
040}
