004/**
005 * <P>This class allows us to call a <code>main</code> method multiple
006 * times without the JVM exiting.  Basically, every time the JVM goes
007 * to exit, we throw a <code>TesterExitException</code> which we then
008 * catch and ignore.</P>
009 *
010 * <P>Note that if you install this security manager, you will have to
011 * write a policy file that grants permissions to the code base from
012 * which the user class was loaded.</P>
013 *
014 * @author My good buddy, Nate Nystrom
015 */
016public class TesterSecurityManager extends SecurityManager {
017
018  /**
019   * Controls whether or not we allow the JVM to exit.
020   */
021  private boolean allowExit = false;
022
023  /**
024   * Sets whether or not this security manager will allow the VM to
025   * exit.
026   */
027  public void setAllowExit(boolean allowExit) {
028    this.allowExit = allowExit;
029  }
030
031  /** 
032   * A <code>SecurityException</code> is thrown if we do not allow the
033   * virtual machine to exit.  
034   */
035  public void checkExit(int status) {
036//      System.err.println("exit " + status);
037    if (! allowExit) {
038      throw new TesterExitException("Tried to exit (status=" +
039                                    status + ")");
040    }
041  }
