001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004
005import java.io.File;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.PrintWriter;
009import java.time.LocalDateTime;
010
011/**
012 * This class represents an assignment give to students in CS410J.
013 *
014 * @author David Whitlock
015 */
016public class Assignment extends NotableImpl {
017
018  public enum AssignmentType {
019    PROJECT,
020    QUIZ,
021    OTHER,
022    POA,
023    OPTIONAL
024  }
025
026  private String name;
027  private String description;
028  private double points;
029  private AssignmentType type;
030  private LocalDateTime dueDate;
031
032  /**
033   * Creates a new <code>Assignment</code> with the given name and
034   * point value.
035   */
036  public Assignment(String name, double points) {
037    this.name = name;
038    this.points = points;
039    this.type = AssignmentType.PROJECT;
040    this.setDirty(false);
041  }
042
043  /**
044   * Returns the name of this <code>Assignment</code>
045   */
046  public String getName() {
047    return this.name;
048  }
049
050  /**
051   * Returns the number of points this <code>Assignment</code> is
052   * worth
053   */
054  public double getPoints() {
055    return this.points;
056  }
057
058  /**
059   * Sets the number of points that this <code>Assignment</code> is
060   * worth.
061   */
062  public Assignment setPoints(double points) {
063    this.setDirty(true);
064    this.points = points;
065    return this;
066  }
067
068  /**
069   * Returns a description of this <code>Assignment</code>
070   */
071  public String getDescription() {
072    return this.description;
073  }
074
075  /**
076   * Sets the description of this <code>Assignment</code>
077   */
078  public void setDescription(String description) {
079    this.setDirty(true);
080    this.description = description;
081  }
082
083  /**
084   * Returns the type of this <code>Assignment</code>
085   */
086  public AssignmentType getType() {
087    return this.type;
088  }
089
090  /**
091   * Sets the type of this <code>Assignment</code>
092   */
093  public Assignment setType(AssignmentType type) {
094    this.setDirty(true);
095    this.type = type;
096    return this;
097  }
098
099  /**
100   * Returns a brief textual description of this
101   * <code>Assignment</code>
102   */
103  public String toString() {
104    return "Assignment " + this.getName() + " worth " +
105           this.getPoints();
106  }
107  
108
109  private static PrintWriter err = new PrintWriter(System.err, true);
110
111  /**
112   * Prints usage information about the main program.
113   */
114  private static void usage() {
115    err.println("\njava Assignment [options] -xmlFile xmlFile " +
116                "-name name");
117    err.println("  where [options] are:");
118    err.println("  -points points        Points assignment is worth");
119    err.println("  -description desc     Description of assignment");
120    err.println("  -type type            Type of assignment " +
121                "(PROJECT, QUIZ, or OTHER)");
122    err.println("  -note note            Note about assignment");
123    err.println("\n");
124    System.exit(1);
125  }
126
127  /**
128   * Main program that creates/edits an <code>Assignment</code> in a
129   * grade book stored in an XML file.
130   */
131  public static void main(String[] args) {
132    String fileName = null;
133    String name = null;
134    String points = null;
135    String desc = null;
136    AssignmentType type = null;
137    String note = null;
138
139    // Parse the command line
140    for (int i = 0; i < args.length; i++) {
141      if (args[i].equals("-xmlFile")) {
142        if (++i >= args.length) {
143          err.println("** Missing XML file name");
144          usage();
145        }
146
147        fileName = args[i];
148
149      } else if (args[i].equals("-name")) {
150        if (++i >= args.length) {
151          err.println("** Missing assignment name");
152          usage();
153        }
154
155        name = args[i];
156
157      } else if (args[i].equals("-points")) {
158        if (++i >= args.length) {
159          err.println("** Missing points value");
160          usage();
161        }
162
163        points = args[i];
164
165      } else if (args[i].equals("-description")) {
166        if (++i >= args.length) {
167          err.println("** Missing description");
168          usage();
169        }
170        
171        desc = args[i];
172
173      } else if (args[i].equals("-type")) {
174        if (++i >= args.length) {
175          err.println("** Missing type");
176          usage();
177        }
178
179        // Make sure type is valid
180        switch (args[i]) {
181          case "PROJECT":
182            type = AssignmentType.PROJECT;
183            break;
184          case "QUIZ":
185            type = AssignmentType.QUIZ;
186            break;
187          case "POA":
188            type = AssignmentType.POA;
189            break;
190          case "OTHER":
191            type = AssignmentType.OTHER;
192            break;
193          case "OPTIONAL":
194            type = AssignmentType.OPTIONAL;
195            break;
196          default:
197            err.println("** Invalid type: " + args[i]);
198            usage();
199            break;
200        }
201
202      } else if (args[i].equals("-note")) {
203        if (++i >= args.length) {
204          err.println("** Missing note");
205          usage();
206        }
207
208        note = args[i];
209
210      } else {
211        err.println("** Unknown option: " + args[i]);
212      }
213    }
214
215    // Verify command line
216    if (fileName == null) {
217      err.println("** No file name specified");
218      usage();
219    }
220
221    if (name == null) {
222      err.println("** No assignment name specified");
223      usage();
224    }
225
226    File file = new File(fileName);
227    if (!file.exists()) {
228      err.println("** Grade book file " + fileName + 
229                  " does not exist");
230      System.exit(1);
231    }
232
233    if (name == null) {
234      err.println("** No assignment name specified");
235      usage();
236    }
237
238    GradeBook book = null;
239    try {
240      XmlGradeBookParser parser = new XmlGradeBookParser(file);
241      book = parser.parse();
242
243    } catch (FileNotFoundException ex) {
244      err.println("** Could not find file: " + ex.getMessage());
245      System.exit(1);
246
247    } catch (IOException ex) {
248      err.println("** IOException during parsing: " + ex.getMessage());
249      System.exit(1);
250
251    } catch (ParserException ex) {
252      err.println("** Exception while parsing " + file + ": " + ex);
253      System.exit(1);
254    }
255
256    // Get the assignment
257    Assignment assign = book.getAssignment(name);
258    if (assign == null) {
259      // Did we specify a points value?
260      if (points == null) {
261        err.println("** No points specified");
262        usage();
263      }
264
265      double value = -1.0;
266      try {
267        value = Double.parseDouble(points);
268
269      } catch (NumberFormatException ex) {
270        err.println("** Not a valid point value: " + points);
271        System.exit(1);
272      }
273
274      if (value < 0.0) {
275        err.println("** Not a valid point value: " + value);
276        System.exit(1);
277      }
278
279      // Create a new Assignment
280      assign = new Assignment(name, value);
281      book.addAssignment(assign);
282    }
283
284    // Set the properties of the assignment
285    if (desc != null) {
286      assign.setDescription(desc);
287    }
288
289    if (type != null) {
290      assign.setType(type);
291    }
292
293    if (note != null) {
294      assign.addNote(note);
295    }
296
297    // Write the grade book back out to the XML file
298    try {
299      XmlDumper dumper = new XmlDumper(file);
300      dumper.dump(book);
301
302    } catch (IOException ex) {
303      err.println("** While dumping to " + file + ": " + ex);
304      System.exit(1);
305    }
306  }
307
308  public void setDueDate(LocalDateTime dueDate) {
309    this.dueDate = dueDate;
310  }
311
312  public LocalDateTime getDueDate() {
313    return dueDate;
314  }
315
316  public boolean isSubmissionLate(LocalDateTime submissionTime) {
317    if (this.dueDate == null) {
318      return false;
319
320    } else {
321      return submissionTime.isAfter(this.dueDate);
322    }
323  }
