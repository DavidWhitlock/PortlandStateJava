001package edu.pdx.cs410J.grader;
002
003import javax.swing.*;
004import java.awt.*;
005import java.util.HashMap;
006import java.util.Map;
007
008import static edu.pdx.cs410J.grader.GradeBook.LetterGradeRanges.LetterGradeRange;
009
010public class LetterGradeRangesPanel extends JPanel {
011
012  private final Map<LetterGrade, MinMaxValueFields> minMaxValueFields = new HashMap<>();
013  private GradeBook.LetterGradeRanges letterGradeRanges;
014
015  LetterGradeRangesPanel(Student.Section undergraduate) {
016    addLetterGradeRanges();
017  }
018
019  private void addLetterGradeRanges() {
020    this.setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
021
022    this.add(createRangesPanel());
023
024    this.add(Box.createVerticalGlue());
025
026    JButton update = new JButton("Update");
027    update.addActionListener(e -> updateLetterGradeRanges());
028    this.add(update);
029
030  }
031
032  private JPanel createRangesPanel() {
033    JPanel rangesPanel = new JPanel(new GridBagLayout());
034    int row = 0;
035
036    for (LetterGrade letter : LetterGrade.values()) {
037      if (letter.hasNumericRange()) {
038
039        addLetterLabel(letter, rangesPanel, row);
040
041        MinMaxValueFields fields = createMinMaxValueFieldsFor(letter);
042
043        addMaxValueField(fields, rangesPanel, row);
044        addToLabel(rangesPanel, row);
045        addMinValueField(fields, rangesPanel, row);
046
047        row++;
048      }
049    }
050
051    return rangesPanel;
052  }
053
054  private void addMinValueField(MinMaxValueFields fields, JPanel rangesPanel, int row) {
055    addComponentToRow(fields.getMinValueField(), rangesPanel, row, 3);
056  }
057
058  private void addToLabel(JPanel rangesPanel, int row) {
059    addComponentToRow(new JLabel("to"), rangesPanel, row, 2);
060  }
061
062  private void addMaxValueField(MinMaxValueFields fields, JPanel rangesPanel, int row) {
063    addComponentToRow(fields.getMaxValueField(), rangesPanel, row, 1);
064  }
065
066  private void addLetterLabel(LetterGrade letter, JPanel rangesPanel, int row) {
067    JLabel letterLabel = new JLabel(letter.asString());
068    addComponentToRow(letterLabel, rangesPanel, row, 0);
069  }
070
071  private void addComponentToRow(JComponent component, JPanel rangesPanel, int row, int gridx) {
072    GridBagConstraints constraints = new GridBagConstraints();
073    constraints.fill = GridBagConstraints.HORIZONTAL;
074    constraints.gridx = gridx;
075    constraints.gridy = row;
076    rangesPanel.add(component, constraints);
077  }
078
079  private void updateLetterGradeRanges() {
080    this.minMaxValueFields.forEach((LetterGrade grade, MinMaxValueFields fields) -> {
081      int minValue = getIntValue(fields.getMinValueField());
082      int maxValue = getIntValue(fields.getMaxValueField());
083
084      LetterGradeRange range = this.letterGradeRanges.getRange(grade);
085      range.setRange(minValue, maxValue);
086    });
087
088    this.letterGradeRanges.validate();
089  }
090
091  private int getIntValue(JTextField field) {
092    return Integer.parseInt(field.getText());
093  }
094
095  private MinMaxValueFields createMinMaxValueFieldsFor(LetterGrade letter) {
096    MinMaxValueFields fields = new MinMaxValueFields(new JTextField(3), new JTextField(3));
097    this.minMaxValueFields.put(letter, fields);
098    return fields;
099  }
100
101  public void displayLetterGradeRanges(GradeBook.LetterGradeRanges letterGradeRanges) {
102    clearAllMinMaxValueFields();
103
104    this.letterGradeRanges = letterGradeRanges;
105    letterGradeRanges.forEach(this::displayLetterGradeRange);
106  }
107
108  private void clearAllMinMaxValueFields() {
109    this.minMaxValueFields.values().forEach(MinMaxValueFields::clearFields);
110  }
111
112  private void displayLetterGradeRange(LetterGradeRange range) {
113    setTextFieldValue(getMinValueField(range.letterGrade()), range.minimum());
114    setTextFieldValue(getMaxValueField(range.letterGrade()), range.maximum());
115  }
116
117  private void setTextFieldValue(JTextField field, int value) {
118    field.setText(String.valueOf(value));
119  }
120
121  private JTextField getMinValueField(LetterGrade letterGrade) {
122    MinMaxValueFields fields = this.minMaxValueFields.get(letterGrade);
123    return fields.getMinValueField();
124  }
125
126  private JTextField getMaxValueField(LetterGrade letterGrade) {
127    MinMaxValueFields fields = this.minMaxValueFields.get(letterGrade);
128    return fields.getMaxValueField();
129  }
130
131  private class MinMaxValueFields {
132    private final JTextField minValueField;
133    private final JTextField maxValueField;
134
135    private MinMaxValueFields(JTextField minValueField, JTextField maxValueField) {
136      this.minValueField = minValueField;
137      this.maxValueField = maxValueField;
138    }
139
140    public JTextField getMinValueField() {
141      return minValueField;
142    }
143
144    public JTextField getMaxValueField() {
145      return maxValueField;
146    }
147
148    public void clearFields() {
149      this.maxValueField.setText("");
150      this.minValueField.setText("");
151    }
152  }
153}
