027  /**
028   * Computes the student's final average and makes a pretty report.
029   */
030  @VisibleForTesting
031  static void dumpReportTo(GradeBook book, Student student,
032                                   PrintWriter pw, boolean assignLetterGrades) {
033    NumberFormat format = NumberFormat.getNumberInstance();
034    format.setMinimumFractionDigits(1);
035    format.setMaximumFractionDigits(1);
036
037    Assignment lowestQuiz = null;
038    double best = 0.0;
039    double total = 0.0;
040
041    pw.println("Grade summary for: " + student.getFullName());
042    SimpleDateFormat df = 
043      new SimpleDateFormat("EEEE MMMM d, yyyy 'at' h:mm a");
044    pw.println("Generated on: " + df.format(new Date()));
045    pw.println("");
046
047    for (String assignmentName : getSortedAssignmentNames(book)) {
048      Assignment assignment = book.getAssignment(assignmentName);
049      if (noStudentHasGradeFor(assignment, book)) {
050        continue;
051      }
052
053      Grade grade = student.getGrade(assignmentName);
054      double score = getScore(grade);
055
056
057//       System.out.println("Examining " + assign + ", score: " + score);
058
059      if (assignment.getType() == Assignment.AssignmentType.QUIZ) {
060        if (lowestQuiz == null) {
061          lowestQuiz = assignment;
062//           System.out.println("Lowest quiz: " + lowestQuiz + 
063//                              ", score: " + score);
064
065        } else {
066          Grade lowestGrade = student.getGrade(lowestQuiz.getName());
067          if (lowestGrade != null && score < lowestGrade.getScore()) {
068            lowestQuiz = assignment;
069//             System.out.println("Lowest quiz: " + lowestQuiz + ", score: "
070//                                + score + ", lowest grade: " +
071//                                student.getGrade(lowestQuiz.getName()));
072          }
073        }
074      }
075
076      StringBuffer line = new StringBuffer();
077      line.append("  ");
078      line.append(assignment.getName());
079      line.append(" (");
080      line.append(assignment.getDescription());
081      line.append(")");
082      if (assignment.getType() == Assignment.AssignmentType.OPTIONAL) {
083        line.append(" (OPTIONAL)");
084      }
085      line.append(": ");
086      line.append(format.format(score));
087      line.append("/");
088      line.append(format.format(assignment.getPoints()));
089
090
091      // Skip incompletes and no grades
092      if (grade == null) {
093        if (dueDateHasPassed(assignment)) {
094          line.append(" (MISSING GRADE)");
095
096        } else {
097          line.append(String.format(" (due %s)", formatDueDate(assignment)));
098        }
099
100      } else if (grade.isIncomplete()) {
101        line.append(" (INCOMPLETE)");
102
103      } else if (grade.isNotGraded()) {
104        line.append( " (NOT GRADED)");
105      }
106
107      pw.println(line);
108
109      // Don't count optional assignments toward the maximum point
110      // total
111      if (assignment.getType() != Assignment.AssignmentType.OPTIONAL) {
112        best += assignment.getPoints();
113      }
114
115      total += score;
116    }
117
118    if (lowestQuiz != null) {
119      pw.println("");
120      pw.println("Lowest Quiz grade dropped: " +
121                  lowestQuiz.getName());
122      pw.println("");
123
124      // Subtract lowest quiz grade
125      Grade lowestGrade = student.getGrade(lowestQuiz.getName());
126      total -= (lowestGrade != null ? lowestGrade.getScore() : 0);
127      best -= lowestQuiz.getPoints();
128    }
129
130    // Print out late and resubmitted assignments
131    pw.println("Late assignments:");
132    for (String late : student.getLate()) {
133      pw.println("  " + late);
134    }
135    pw.println();
136
137    pw.println("Resubmitted assignments:");
138    for (String resubmitted : student.getResubmitted()) {
139      pw.println("  " + resubmitted);
140    }
141    pw.println("");
142
143    pw.println("Total grade: " + format.format(total)  + "/" +
144               format.format(best));
145
146    double overallScore = total / best;
147
148    if (assignLetterGrades) {
149      LetterGrade letterGrade = book.getLetterGradeForScore(student.getEnrolledSection(), overallScore * 100.0);
150      student.setLetterGrade(letterGrade);
151    }
152
153    if (student.getLetterGrade() != null) {
154      pw.println("Letter Grade: " + student.getLetterGrade());
155    }
156
157    allTotals.put(student, overallScore);
158  }
159
160  private static String formatDueDate(Assignment assignment) {
161    LocalDateTime dueDate = assignment.getDueDate();
162    return dueDate.format(DateTimeFormatter.ofLocalizedDateTime(FormatStyle.SHORT));
163  }
164
165  @VisibleForTesting
166  static boolean dueDateHasPassed(Assignment assignment) {
167    LocalDateTime dueDate = assignment.getDueDate();
168    LocalDateTime now = LocalDateTime.now();
169    return now.isAfter(dueDate);
170  }
171
172  static boolean noStudentHasGradeFor(Assignment assignment, GradeBook book) {
173    return book.studentsStream().noneMatch(student -> studentHasGradeFor(student, assignment));
174  }
175
176  private static boolean studentHasGradeFor(Student student, Assignment assignment) {
177    Grade grade = student.getGrade(assignment.getName());
178    return grade != null && !grade.isNotGraded();
179  }
180
181  private static double getScore(Grade grade) {
182    // Average non-existent scores as zero
183    double score;
184    if (grade == null) {
185      score = 0.0;
186
187    } else {
188      score = grade.getScore();
189    }
190    return score;
191  }

193  private static SortedSet<String> getSortedAssignmentNames(GradeBook book) {
194    return new TreeSet<>(book.getAssignmentNames());
195  }

214  /**
215   * Main program that creates summary reports for every student in a
216   * grade book located in a given XML file.
217   */
218  public static void main(String[] args) {
219    boolean assignLetterGrades = false;
220    String xmlFileName = null;
221    String outputDirName = null;
222    Collection<String> students = new ArrayList<>();
223
224    for (String arg : args) {
225      if (arg.equals("-assignLetterGrades")) {
226        assignLetterGrades = true;
227
228      } else if (xmlFileName == null) {
229        xmlFileName = arg;
230
231      } else if (outputDirName == null) {
232        outputDirName = arg;
233
234      } else {
235        students.add(arg);
236      }
237    }
238
239    if (xmlFileName == null) {
240      usage("Missing XML file name");
241    }
242
243    if (outputDirName == null) {
244      usage("Missing output dir name");
245      return;
246    }
247
248    String xmlFile = xmlFileName;
249    File outDir = new File(outputDirName);
250    if (!outDir.exists()) {
251      outDir.mkdirs();
252
253    } else if (!outDir.isDirectory()) {
254      usage(outDir + " is not a directory");
255    }
256
257    File file = new File(xmlFile);
258    if (!file.exists()) {
259      usage("Grade book file " + xmlFile + " does not exist");
260    }
261
262    GradeBook book = parseGradeBook(file);
263
264    // Create a SummaryReport for every student
265    Iterable<String> studentIds;
266
267    if (!students.isEmpty()) {
268      studentIds = students;
269
270    } else {
271      studentIds = book.getStudentIds();
272    }
273
274    dumpReports(studentIds, book, outDir, assignLetterGrades);
275
276    // Sort students by totals and print out results:
277    Set<Student> students1 = allTotals.keySet();
278    printOutStudentTotals(students1, out);
279
280    saveGradeBookIfDirty(xmlFileName, book);
281
282  }

284  @VisibleForTesting
285  static void printOutStudentTotals(Set<Student> allStudents, PrintWriter out) {
286    SortedSet<Student> sorted = getStudentSortedByTotalPoints(allStudents);
287
288    out.println("Undergraduates:");
289    Stream<Student> undergrads = sorted.stream().filter(student -> student.getEnrolledSection() == UNDERGRADUATE);
290    printOutStudentTotals(out, undergrads);
291
292    out.println("Graduate Students:");
293    Stream<Student> grads = sorted.stream().filter(student -> student.getEnrolledSection() == GRADUATE);
294    printOutStudentTotals(out, grads);
295
296  }
297
298  private static void printOutStudentTotals(PrintWriter out, Stream<Student> students) {
299    NumberFormat format = NumberFormat.getPercentInstance();
300
301    students.forEach(student -> {
302      Double d = allTotals.get(student);
303      out.print("  " + student + ": " + format.format(d.doubleValue()));
304
305      if (student.getLetterGrade() != null) {
306        out.print(" " + student.getLetterGrade());
307      }
308
309      out.println();
310    });
311  }

313  private static void saveGradeBookIfDirty(String xmlFileName, GradeBook book) {
314    if (book.isDirty()) {
315      try {
316        XmlDumper dumper = new XmlDumper(xmlFileName);
317        dumper.dump(book);
318
319      } catch (IOException ex) {
320        printErrorMessageAndExit("While saving gradebook to " + xmlFileName, ex);
321      }
322    }
323  }

325  private static SortedSet<Student> getStudentSortedByTotalPoints(Set<Student> students) {
326    SortedSet<Student> sorted = new TreeSet<>(new Comparator<Student>() {
327        @Override
328        public int compare(Student s1, Student s2) {
329          Double d1 = allTotals.get(s1);
330          Double d2 = allTotals.get(s2);
331          if ( d2.compareTo( d1 ) == 0 ) {
332            return s1.getId().compareTo( s2.getId() );
333          } else {
334            return d2.compareTo(d1);
335          }
336        }
337
338        @Override
339        public boolean equals(Object o) {
340          return true;
341        }
342      });
343
344    sorted.addAll(students);
345    return sorted;
346  }

348  @VisibleForTesting
349  static void dumpReports(Iterable<String> studentIds, GradeBook book, File outDir, boolean assignLetterGrades) {
350    for (String id : studentIds) {
351      err.println(id);
352
353      Student student = book.getStudent(id).orElseThrow(noStudentWithId(id));
354
355      File outFile = new File(getDirectoryForReportFileForStudent(outDir, student), getReportFileName(id));
356      try {
357        PrintWriter pw =
358          new PrintWriter(new FileWriter(outFile), true);
359        dumpReportTo(book, student, pw, assignLetterGrades);
360
361//         dumpReportTo(book, student, out);
362      } catch (IOException ex) {
363        printErrorMessageAndExit("While writing report to " + outFile, ex);
364      }
365    }
366  }

368  private static File getDirectoryForReportFileForStudent(File parentDirectory, Student student) {
369    String directoryName;
370    Student.Section enrolledSection = student.getEnrolledSection();
371    switch (enrolledSection) {
372      case UNDERGRADUATE:
373        directoryName = UNDERGRADUATE_DIRECTORY_NAME;
374        break;
375      case GRADUATE:
376        directoryName = GRADUATE_DIRECTORY_NAME;
377        break;
378      default:
379        throw new IllegalStateException("Don't know directory name for " + enrolledSection);
380    }
381    File directory = new File(parentDirectory, directoryName);
382    directory.mkdirs();
383    return directory;
384  }

386  @VisibleForTesting
387  static String getReportFileName(String studentId) {
388    return studentId + ".report";
389  }

391  private static GradeBook parseGradeBook(File gradeBookFile) {
392    GradeBook book = null;
393    try {
394      err.println("Parsing " + gradeBookFile);
395      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
396      book = parser.parse();
397
398    } catch (FileNotFoundException ex) {
399      printErrorMessageAndExit("** Could not find grade book file: " + gradeBookFile, ex);
400
401    } catch (ParserException | IOException ex) {
402      printErrorMessageAndExit("** Exception while parsing " + gradeBookFile, ex);
403    }
404    return book;
405  }
406
407  private static void printErrorMessageAndExit(String message, Throwable ex) {
408    err.println(message);
409    System.exit(1);
410  }

412  @SuppressWarnings("ThrowableInstanceNeverThrown")
413  private static Supplier<? extends IllegalStateException> noStudentWithId(String id) {
414    return () -> new IllegalStateException("No student with id " + id);
415  }
