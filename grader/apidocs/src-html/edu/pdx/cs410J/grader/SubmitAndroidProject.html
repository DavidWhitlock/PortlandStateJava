001package edu.pdx.cs410J.grader;
002
003import javax.mail.MessagingException;
004import java.io.File;
005import java.io.IOException;
006import java.util.Set;
007
008public class SubmitAndroidProject extends Submit {
009
010  private SubmitAndroidProject() {
011    this.setProjectName("Project5");
012  }
013
014  @Override
015  protected boolean canBeSubmitted(File file) {
016    return fileExists(file) && getZipEntryNameFor(file) != null;
017  }
018
019  @Override
020  protected void warnIfMainProjectClassIsNotSubmitted(Set<File> sourceFiles) {
021
022  }
023
024  @Override
025  protected String getZipEntryNameFor(File file) {
026    return AndroidZipFixer.getFixedEntryName(file.getAbsolutePath());
027  }
028
029  public static void main(String[] args) throws IOException, MessagingException {
030    SubmitAndroidProject submit = new SubmitAndroidProject();
031    submit.parseCommandLineAndSubmit(args);
032  }
033
034}
