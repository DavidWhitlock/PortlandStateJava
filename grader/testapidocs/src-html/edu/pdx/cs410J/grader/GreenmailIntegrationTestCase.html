001package edu.pdx.cs410J.grader;
002
003import com.icegreen.greenmail.imap.AuthorizationException;
004import com.icegreen.greenmail.store.FolderException;
005import com.icegreen.greenmail.user.GreenMailUser;
006import com.icegreen.greenmail.util.GreenMail;
007import com.icegreen.greenmail.util.ServerSetup;
008import com.sun.mail.util.MailSSLSocketFactory;
009import org.junit.After;
010import org.junit.Before;
011
012import javax.mail.*;
013import javax.mail.internet.InternetAddress;
014import javax.mail.internet.MimeMessage;
015import java.security.GeneralSecurityException;
016import java.util.Properties;
017
018public class GreenmailIntegrationTestCase {
019  protected final String emailAddress = "test@email.com";
020  protected final String emailServerHost = "127.0.0.1";
021  protected final int smtpPort = 2525;
022  protected final String imapUserName = "emailUser";
023  protected final String imapPassword = "emailPassword";
024  protected final int imapsPort = 9933;
025  protected GreenMail emailServer;
026
027  @Before
028  public void startEmailServer() throws FolderException, AuthorizationException {
029    ServerSetup smtp = new ServerSetup(smtpPort, emailServerHost, ServerSetup.PROTOCOL_SMTP);
030    ServerSetup imaps = new ServerSetup(imapsPort, emailServerHost, ServerSetup.PROTOCOL_IMAPS);
031    emailServer = new GreenMail(new ServerSetup[]{ smtp, imaps });
032
033    GreenMailUser user = emailServer.setUser(emailAddress, imapUserName, imapPassword);
034    doSomethingWithUser(user);
035
036    emailServer.start();
037  }
038
039  protected void doSomethingWithUser(GreenMailUser user) throws FolderException, AuthorizationException {
040
041  }
042
043  @Before
044  public void enableDebugLogging() {
045    System.setProperty("mail.imap.parse.debug", Boolean.TRUE.toString());
046  }
047
048  @After
049  public void stopEmailServer() {
050    emailServer.stop();
051  }
052
053  protected MimeMessage newEmailTo(Session session, String recipient, String subject) throws MessagingException {
054    MimeMessage message = new MimeMessage(session);
055
056    InternetAddress[] to = {new InternetAddress(recipient)};
057    message.setRecipients(Message.RecipientType.TO, to);
058    message.setSubject(subject);
059    return message;
060  }
061
062  protected Session newEmailSession(boolean debug) {
063    Properties props = new Properties();
064    props.put("mail.smtp.host", emailServerHost);
065    props.put("mail.smtp.port", smtpPort);
066    Session session = Session.getDefaultInstance(props, null);
067    session.setDebug(debug);
068    return session;
069  }
070
071  protected Folder openFolder(Store store, String folderName) throws MessagingException {
072    Folder folder = store.getDefaultFolder();
073    folder = folder.getFolder(folderName);
074    folder.open(Folder.READ_WRITE);
075    return folder;
076  }
077
078  protected Store connectToIMAPServer() throws GeneralSecurityException, MessagingException {
079    Properties props = new Properties();
080
081    MailSSLSocketFactory socketFactory = new MailSSLSocketFactory();
082    socketFactory.setTrustedHosts(new String[]{"127.0.0.1", "localhost"});
083    props.put("mail.imaps.ssl.socketFactory", socketFactory);
084
085    Session session = Session.getInstance(props, null);
086    Store store = session.getStore("imaps");
087    store.connect(emailServerHost, imapsPort, imapUserName, imapPassword);
088
089    return store;
090  }
091}
