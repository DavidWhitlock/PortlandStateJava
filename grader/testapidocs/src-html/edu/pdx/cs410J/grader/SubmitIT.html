028public class SubmitIT {
029
030  private final String studentEmail = "student@email.com";
031  private final String studentName = "Student Name";
032  private final Collection<File> filesToSubmit = new ArrayList<>();
033  private final String studentLoginId = "student";
034  private final String projectName = "Project";
035  private GreenMail emailServer;
036  private final String emailServerHost = "127.0.0.1";
037  private final int smtpPort = 2525;
038  private final File tempDirectory = new File(System.getProperty("java.io.tmpdir"));
039  private final String imapUserName = "emailUser";
040  private final String imapPassword = "emailPassword";
041  private final int imapsPort = 9933;
042  private String graderEmail = EmailSender.TA_EMAIL;
043
044  @Before
045  public void createFilesToSubmit() throws IOException {
046    File dir = createDirectoriesWithFilesToSubmit("edu", "pdx", "cs410J", studentLoginId);
047    List<String> fileNames = Arrays.asList(projectName + ".java", "File1.java", "File2.java");
048    for (String fileName : fileNames) {
049      this.filesToSubmit.add(createEmptyFile(dir, fileName));
050    }
051  }
052
053  @Before
054  public void startEmailServer() throws FolderException, AuthorizationException {
055    ServerSetup smtp = new ServerSetup(smtpPort, emailServerHost, ServerSetup.PROTOCOL_SMTP);
056    ServerSetup imaps = new ServerSetup(imapsPort, emailServerHost, ServerSetup.PROTOCOL_IMAPS);
057    emailServer = new GreenMail(new ServerSetup[]{ smtp, imaps });
058
059    GreenMailUser graderUser = emailServer.setUser(graderEmail, imapUserName, imapPassword);
060
061    moveEmailsFromInboxToProjectSubmissions(graderUser);
062
063    emailServer.start();
064  }
065
066  private void moveEmailsFromInboxToProjectSubmissions(GreenMailUser user) throws AuthorizationException, FolderException {
067    ImapHostManager manager = emailServer.getManagers().getImapHostManager();
068    MailFolder submissions = manager.createMailbox(user, ProjectSubmissionsProcessor.EMAIL_FOLDER_NAME);
069    MailFolder inbox = manager.getInbox(user);
070    inbox.addListener(new FolderListener() {
071      @Override
072      public void expunged(int msn) {
073
074      }
075
076      @Override
077      public void added(int msn) {
078        try {
079          inbox.copyMessage(msn, submissions);
080
081        } catch (FolderException ex) {
082          throw new IllegalStateException("Can't copy message to submissions folder", ex);
083        }
084      }
085
086      @Override
087      public void flagsUpdated(int msn, Flags flags, Long uid) {
088
089      }
090
091      @Override
092      public void mailboxDeleted() {
093
094      }
095    });
096  }
097
098  @Before
099  public void enableDebugLogging() {
100    GraderTools.setLoggingLevelToDebug();
101  }
102
103  @After
104  public void deleteFilesToSubmit() {
105    filesToSubmit.forEach(File::delete);
106  }
107
108  @After
109  public void stopEmailServer() {
110    emailServer.stop();
111  }
112
113  private File createEmptyFile(File dir, String fileName) throws IOException {
114    File file = new File(dir, fileName);
115    FileWriter writer = new FileWriter(file);
116    writer.write("");
117    return file;
118  }
119
120  private File createDirectoriesWithFilesToSubmit(String... directoryNames) {
121    File dir = tempDirectory;
122    for (String dirName : directoryNames) {
123      dir = new File(dir, dirName);
124      dir.mkdir();
125    }
126
127    return dir;
128  }
129
130  @Test
131  public void submitFilesAndDownloadSubmission() throws IOException, MessagingException {
132    Submit submit = new Submit();
133    submit.setProjectName(projectName);
134    submit.setUserEmail(studentEmail);
135    submit.setUserId(studentLoginId);
136    submit.setUserName(studentName);
137
138    for (File file : filesToSubmit) {
139      submit.addFile(file.getAbsolutePath());
140    }
141
142    submit.setEmailServerHostName(emailServerHost);
143    submit.setEmailServerPort(smtpPort);
144    submit.setDebug(true);
145    submit.setSendReceipt(false);
146    submit.submit(false);
147
148    GradeBook gradeBook = new GradeBook("SubmitIT");
149    gradeBook.addStudent(new Student(studentLoginId));
150    gradeBook.addAssignment(new Assignment(projectName, 3.5));
151
152    GraderEmailAccount account = new GraderEmailAccount(emailServerHost, imapsPort, imapUserName, imapPassword, true);
153    FetchAndProcessGraderEmail.fetchAndProcessGraderEmails("projects", account, this.tempDirectory, gradeBook);
154
155    Grade grade = gradeBook.getStudent(studentLoginId).get().getGrade(projectName);
156    assertThat(grade, is(notNullValue()));
157    assertThat(grade.getScore(), equalTo(Grade.NO_GRADE));
158    assertThat(grade.getSubmissionTimes().size(), equalTo(1));
159  }
