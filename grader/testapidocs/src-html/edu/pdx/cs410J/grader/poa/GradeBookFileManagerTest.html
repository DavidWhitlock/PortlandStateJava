001package edu.pdx.cs410J.grader.poa;
002
003import com.google.common.eventbus.Subscribe;
004import edu.pdx.cs410J.ParserException;
005import edu.pdx.cs410J.grader.*;
006import org.junit.Test;
007import org.mockito.ArgumentCaptor;
008
009import java.io.File;
010import java.io.IOException;
011
012import static org.hamcrest.MatcherAssert.assertThat;
013import static org.hamcrest.Matchers.any;
014import static org.hamcrest.Matchers.contains;
015import static org.hamcrest.Matchers.equalTo;
016import static org.mockito.Mockito.mock;
017import static org.mockito.Mockito.verify;
018
019public class GradeBookFileManagerTest extends EventBusTestCase {
020
021  @Override
022  public void setUp() {
023    super.setUp();
024    new GradeBookFileManager(this.bus);
025  }
026
027  @Test
028  public void gradeBookLoadedOnLoadGradeBookEvent() throws IOException {
029    String gradeBookName = "Test Grade Book";
030    File file = writeGradeBookToFile(gradeBookName);
031
032    GradeBookLoadedHandler handler = mock(GradeBookLoadedHandler.class);
033    this.bus.register(handler);
034
035    this.bus.post(new LoadGradeBook(file));
036
037    ArgumentCaptor<GradeBookLoaded> event = ArgumentCaptor.forClass(GradeBookLoaded.class);
038    verify(handler).handle(event.capture());
039
040    assertThat(event.getValue().getGradeBook().getClassName(), equalTo(gradeBookName));
041  }
042
043  protected File writeGradeBookToFile(String gradeBookName) throws IOException {
044    File file = File.createTempFile("testGradeBook", "xml");
045
046    GradeBook book = new GradeBook(gradeBookName);
047    XmlDumper dumper = new XmlDumper(file);
048    dumper.dump(book);
049
050    return file;
051  }
052
053  private interface GradeBookLoadedHandler {
054    @Subscribe
055    void handle(GradeBookLoaded event);
056  }
057
058  @Test
059  public void unhandledExceptionEventPublishedWhenLoadingBadGradeBook() throws IOException {
060    File badFile = File.createTempFile("badGradeBook", "xml");
061
062    UnhandledExceptionEventHandler handler = mock(UnhandledExceptionEventHandler.class);
063    this.bus.register(handler);
064
065    this.unhandledExceptionHandler = this::doNotFailTestWhenUnhandledExceptionEncountered;
066
067    this.bus.post(new LoadGradeBook(badFile));
068
069    ArgumentCaptor<UnhandledExceptionEvent> event = ArgumentCaptor.forClass(UnhandledExceptionEvent.class);
070    verify(handler).handle(event.capture());
071
072    assertThat(event.getValue().getUnhandledException(), any(Throwable.class));
073  }
074
075  private interface UnhandledExceptionEventHandler {
076    @Subscribe
077    void handle(UnhandledExceptionEvent event);
078  }
079
080  @Test
081  public void gradeBookSavedToFileOnSaveGradeBookEvent() throws IOException, ParserException {
082    String gradeBookName = "Test Grade Book";
083    File file = writeGradeBookToFile(gradeBookName);
084
085    GradeBookLoadedHandler handler = mock(GradeBookLoadedHandler.class);
086    this.bus.register(handler);
087
088    this.bus.post(new LoadGradeBook(file));
089
090    ArgumentCaptor<GradeBookLoaded> event = ArgumentCaptor.forClass(GradeBookLoaded.class);
091    verify(handler).handle(event.capture());
092
093    GradeBook book = event.getValue().getGradeBook();
094    Assignment poa = new Assignment("poa", 1.0);
095    book.addAssignment(poa);
096
097    this.bus.post(new SaveGradeBook(book));
098
099    book = readGradeBookFromFile(file);
100    assertThat(book.getAssignmentNames(), contains(poa.getName()));
101  }
102
103  private GradeBook readGradeBookFromFile(File file) throws IOException, ParserException {
104    XmlGradeBookParser parser = new XmlGradeBookParser(file);
105    return parser.parse();
106  }
107
108}
