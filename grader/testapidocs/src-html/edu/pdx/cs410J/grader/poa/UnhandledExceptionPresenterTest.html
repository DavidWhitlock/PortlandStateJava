001package edu.pdx.cs410J.grader.poa;
002
003import org.junit.Test;
004
005import java.io.IOException;
006import java.io.PrintWriter;
007import java.io.StringWriter;
008
009import static org.mockito.Mockito.mock;
010import static org.mockito.Mockito.verify;
011
012public class UnhandledExceptionPresenterTest extends EventBusTestCase {
013
014  private UnhandledExceptionView view;
015
016  @Override
017  public void setUp() {
018    super.setUp();
019    this.unhandledExceptionHandler = this::doNotFailTestWhenUnhandledExceptionEncountered;
020
021    this.view = mock(UnhandledExceptionView.class);
022    new UnhandledExceptionPresenter(this.bus, this.view);
023  }
024
025  @Test
026  public void viewUpdatedOnUnhandledExceptionEvent() throws IOException {
027    String message = "This is an exception message";
028    Throwable exception = new IllegalStateException(message).fillInStackTrace();
029
030    this.bus.post(new UnhandledExceptionEvent(exception));
031
032    verify(this.view).setExceptionMessage(message);
033    verify(this.view).setExceptionDetails(getStackTrace(exception));
034    verify(this.view).displayView();
035  }
036
037  private String getStackTrace(Throwable exception) throws IOException {
038    try (StringWriter sw = new StringWriter()) {
039      exception.printStackTrace(new PrintWriter(sw));
040      sw.flush();
041      return sw.toString();
042    }
043  }
044}
