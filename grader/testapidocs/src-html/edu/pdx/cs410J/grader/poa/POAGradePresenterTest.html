352  @Test
353  public void recordingGradeInViewPublishesRecordGradeMessage() {
354    ArgumentCaptor<ScoreValueHandler> scoreHandler = ArgumentCaptor.forClass(ScoreValueHandler.class);
355    verify(this.view).addScoreValueHandler(scoreHandler.capture());
356
357    ArgumentCaptor<RecordGradeHandler> recordGrade = ArgumentCaptor.forClass(RecordGradeHandler.class);
358    verify(this.view).addRecordGradeHandler(recordGrade.capture());
359
360    RecordGradeEventHandler eventHandler = mock(RecordGradeEventHandler.class);
361    this.bus.register(eventHandler);
362
363    this.assignment.setDueDate(LocalDateTime.now().minusDays(5));
364    postEventsToBus();
365
366    double score = 0.75;
367    scoreHandler.getValue().scoreValue(formatTotalPoints(score));
368    recordGrade.getValue().recordGrade();
369
370    ArgumentCaptor<RecordGradeEvent> eventCaptor = ArgumentCaptor.forClass(RecordGradeEvent.class);
371    verify(eventHandler).handleRecordGradeEvent(eventCaptor.capture());
372
373    RecordGradeEvent event = eventCaptor.getValue();
374    assertThat(event.getScore(), equalTo(score));
375    assertThat(event.getStudent(), equalTo(this.student));
376    assertThat(event.getAssignment(), equalTo(this.assignment));
377    assertThat(event.isLate(), equalTo(true));
378
379    verify(this.view).setScoreHasBeenRecorded(true);
380  }
