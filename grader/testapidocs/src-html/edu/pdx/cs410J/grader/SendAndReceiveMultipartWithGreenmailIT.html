038  private void moveEmailsFromInboxToProjectSubmissions(GreenMailUser user) throws AuthorizationException, FolderException {
039    ImapHostManager manager = emailServer.getManagers().getImapHostManager();
040    MailFolder submissions = manager.createMailbox(user, emailFolderName);
041    MailFolder inbox = manager.getInbox(user);
042    inbox.addListener(new FolderListener() {
043      @Override
044      public void expunged(int msn) {
045
046      }
047
048      @Override
049      public void added(int msn) {
050        try {
051          inbox.copyMessage(msn, submissions);
052
053        } catch (FolderException ex) {
054          throw new IllegalStateException("Can't copy message to submissions folder", ex);
055        }
056      }
057
058      @Override
059      public void flagsUpdated(int msn, Flags flags, Long uid) {
060
061      }
062
063      @Override
064      public void mailboxDeleted() {
065
066      }
067    });
068  }

071  @Test
072  public void sendAndFetchMailMessageWithMultipleAttachments() throws IOException, MessagingException, GeneralSecurityException {
073    sendMailWithAttachedZipFile();
074    fetchAttachmentsFromUnreadMessagesInFolder();
075  }
076
077  public void fetchAttachmentsFromUnreadMessagesInFolder() throws GeneralSecurityException, MessagingException {
078    Store store = connectToIMAPServer();
079    Folder folder = openFolder(store, emailFolderName);
080
081    fetchAttachmentsFromUnreadMessagesInFolder(folder);
082
083    try {
084      folder.close(false);
085      store.close();
086
087    } catch (MessagingException ex) {
088      throw new IllegalStateException("While closing folder and store", ex);
089    }
090
091  }
092
093  private void fetchAttachmentsFromUnreadMessagesInFolder(Folder folder) throws MessagingException {
094    Message[] messages = folder.getMessages();
095
096    FetchProfile profile = new FetchProfile();
097    profile.add(FetchProfile.Item.ENVELOPE);
098    profile.add(FetchProfile.Item.FLAGS);
099
100    folder.fetch(messages, profile);
101
102    for (Message message : messages) {
103      System.out.println(message.getContentType());
104      assertThat(message.getContentType() + " is not multi-part", isMultipartMessage(message), equalTo(true));
105    }
106  }
107
108  private boolean isMultipartMessage(Message message) throws MessagingException {
109    return message.isMimeType("multipart/*");
110  }
111
112  private void sendMailWithAttachedZipFile() throws IOException, MessagingException {
113    File tempDirectory = new File(System.getProperty("java.io.tmpdir"));
114    File tempFile = File.createTempFile("TempFile", ".java", tempDirectory);
115    tempFile.createNewFile();
116    File[] filesToSubmit = new File[]{tempFile};
117    File zipFile = File.createTempFile("TempFile", ".zip", tempDirectory);
118    mailTA(makeZipFile(zipFile, filesToSubmit));
119  }
120
121  private File makeZipFile(File zipFile, File[] files) throws IOException {
122    Map<File, String> sourceFilesAndNames =
123      Arrays.stream(files).collect(Collectors.toMap(file -> file, File::getName));
124
125    ZipFileOfFilesMaker maker = new ZipFileOfFilesMaker(sourceFilesAndNames, zipFile, new HashMap<>());
126    maker.makeZipFile();
127
128    return zipFile;
129  }
130
131  private void mailTA(File zipFile) throws MessagingException {
132    MimeMessage message = newEmailTo(newEmailSession(true), emailAddress, "Email subject");
133
134    MimeBodyPart textPart = createTextPartOfTAEmail();
135    MimeBodyPart filePart = createZipAttachment(zipFile);
136
137    Multipart mp = new MimeMultipart();
138    mp.addBodyPart(textPart);
139    mp.addBodyPart(filePart);
140
141    message.setContent(mp);
142
143    Transport.send(message);
144  }
145
146  private MimeBodyPart createTextPartOfTAEmail() throws MessagingException {
147    MimeBodyPart textPart = new MimeBodyPart();
148    textPart.setContent("This is some text to be displayed inline", "text/plain");
149
150    // Try not to display text as separate attachment
151    textPart.setDisposition("inline");
152    return textPart;
153  }
154
155  private MimeBodyPart createZipAttachment(File zipFile) throws MessagingException {
156    // Now attach the Zip file
157    DataSource ds = new FileDataSource(zipFile) {
158      @Override
159      public String getContentType() {
160        return "application/zip";
161      }
162    };
163    DataHandler dh = new DataHandler(ds);
164    MimeBodyPart filePart = new MimeBodyPart();
165
166    String zipFileTitle = "ZipFile.zip";
167
168    filePart.setDataHandler(dh);
169    filePart.setFileName(zipFileTitle);
170    filePart.setDescription("Zip File");
171
172    return filePart;
173  }
