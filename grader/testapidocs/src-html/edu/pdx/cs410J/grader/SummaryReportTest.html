028  @Before
029  public void createTempDirectoryForTest() {
030    String tempDirName = "SummaryReportTestTempDirectory-" + System.currentTimeMillis();

038  @After
039  public void deleteTempDirectoryForTest() throws IOException {
040    Files.walkFileTree(tempDirectory.toPath(), new SimpleFileVisitor<Path>() {
041      @Override
042      public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
043        return deleteFile(file);
044      }
045
046      @Override
047      public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
048        return deleteFile(dir);
049      }
050
051      private FileVisitResult deleteFile(Path file) throws IOException {
052        Files.delete(file);
053        return FileVisitResult.CONTINUE;
054      }
055    });
056
057    assertThat(tempDirectory.exists(), equalTo(false));
058  }

075  @Test
076  public void reportsAreGeneratedInCorrectDirectories() throws IOException {
077    GradeBook gradeBook = new GradeBook("test");
078    Assignment assignment = new Assignment("assignment", 4.0);
079    gradeBook.addAssignment(assignment);
080
081    double undergradScore = 3.0;
082    Student undergrad = addStudentInSectionWithScore(gradeBook, assignment, "undergrad", UNDERGRADUATE, undergradScore);
083
084    double gradScore = 3.5;
085    Student grad = addStudentInSectionWithScore(gradeBook, assignment, "grad", GRADUATE, gradScore);
086
087    SummaryReport.dumpReports(gradeBook.getStudentIds(), gradeBook, tempDirectory, false);
088
089    assertThatStudentHasReportInDirectoryWithScore(undergrad, UNDERGRADUATE_DIRECTORY_NAME, undergradScore);
090    assertThatStudentHasReportInDirectoryWithScore(grad, GRADUATE_DIRECTORY_NAME, gradScore);
091  }
092
093  private Student addStudentInSectionWithScore(GradeBook gradeBook, Assignment assignment, String studentId, Student.Section enrolledSection, double undergradScore) {
094    Student undergrad = new Student(studentId);
095    undergrad.setEnrolledSection(enrolledSection);
096    gradeBook.addStudent(undergrad);
097    undergrad.setGrade(assignment, undergradScore);
098    return undergrad;
099  }
100
101  private void assertThatStudentHasReportInDirectoryWithScore(Student student, String dirFileName, double studentScore) throws IOException {
102    File directory = new File(tempDirectory, dirFileName);
103    assertThat(directory.exists(), equalTo(true));
104
105    File reportFile = new File(directory, SummaryReport.getReportFileName(student.getId()));
106    assertThat(reportFile.exists(), equalTo(true));
107
108    String reportContents = readTextFromFile(reportFile);
109    assertThat(reportContents, containsString(String.valueOf(studentScore)));
110  }
111
112  private String readTextFromFile(File file) throws IOException {
113    StringWriter writer = new StringWriter();
114    CharStreams.copy(new FileReader(file), writer);
115    return writer.getBuffer().toString();
116  }

146  private void calculateTotalGradesForStudents(GradeBook gradeBook, boolean assignLetterGrades) {
147    PrintWriter pw = new PrintWriter(new Writer() {
148      @Override
149      public void write(char[] cbuf, int off, int len) throws IOException {
150
151      }
152
153      @Override
154      public void flush() throws IOException {
155
156      }
157
158      @Override
159      public void close() throws IOException {
160
161      }
162    });
163
164    gradeBook.studentsStream().forEach(student -> {
165      SummaryReport.dumpReportTo(gradeBook, student, pw, assignLetterGrades);
166    });
167  }

169  private class CapturingPrintWriter extends PrintWriter {
170    public CapturingPrintWriter() {
171      super(new StringWriter());
172    }
173
174    public String getTextWrittenToWriter() {
175      return ((StringWriter) this.out).getBuffer().toString();
176    }
177  }

179  @Test
180  public void assignLetterGradesAccordingToGradeBookRangesForSection() {
181    GradeBook gradeBook = new GradeBook("test");
182    Assignment assignment = new Assignment("assignment", 10.0);
183    gradeBook.addAssignment(assignment);
184
185    GradeBook.LetterGradeRanges undergradLetterGradeRanges = gradeBook.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
186    undergradLetterGradeRanges.getRange(LetterGrade.A).setRange(95, 100);
187    undergradLetterGradeRanges.getRange(LetterGrade.A_MINUS).setRange(90, 94);
188    undergradLetterGradeRanges.validate();
189
190    GradeBook.LetterGradeRanges gradLetterGradeRanges = gradeBook.getLetterGradeRanges(Student.Section.GRADUATE);
191    gradLetterGradeRanges.getRange(LetterGrade.A).setRange(92, 100);
192    gradLetterGradeRanges.getRange(LetterGrade.A_MINUS).setRange(90, 91);
193    gradLetterGradeRanges.validate();
194
195    String betterUndergradStudentName = "betterUndergrad";
196    Student betterUndergrad = addStudentInSectionWithScore(gradeBook, assignment, betterUndergradStudentName, UNDERGRADUATE, 9.7);
197    assertThat(betterUndergrad.getLetterGrade(), nullValue());
198
199    String worseUndergradStudentName = "worseUndergrad";
200    Student worseUndergrad = addStudentInSectionWithScore(gradeBook, assignment, worseUndergradStudentName, UNDERGRADUATE, 9.2);
201    assertThat(worseUndergrad.getLetterGrade(), nullValue());
202
203    String gradStudentName = "grad";
204    Student grad = addStudentInSectionWithScore(gradeBook, assignment, gradStudentName, GRADUATE, 9.2);
205    assertThat(grad.getLetterGrade(), nullValue());
206
207    calculateTotalGradesForStudents(gradeBook, true);
208
209    assertThat(betterUndergrad.getLetterGrade(), equalTo(LetterGrade.A));
210    assertThat(worseUndergrad.getLetterGrade(), equalTo(LetterGrade.A_MINUS));
211    assertThat(grad.getLetterGrade(), equalTo(LetterGrade.A));
212  }
