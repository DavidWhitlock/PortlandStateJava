1   package edu.pdx.cs410J.grader;

3   import com.google.common.annotations.VisibleForTesting;
4   
5   import javax.mail.Message;
6   import javax.mail.MessagingException;
7   import javax.mail.Session;
8   import javax.mail.internet.InternetAddress;
9   import javax.mail.internet.MimeMessage;
10  import java.util.Properties;
11  
12  public class EmailSender {
13    /**
14     * The grader's email address
15     */
16    @VisibleForTesting
17    static final String TA_EMAIL = "sjavata@gmail.com";
18    /**
19     * The name of the SMTP server that is used to send email
20     */
21    protected static String serverName = "mailhost.cs.pdx.edu";
22  
23    private static int emailServerPort = 25;
24  
25    protected static MimeMessage newEmailTo(Session session, String recipient, String subject) throws MessagingException {
26      return newEmailTo(session, TA_EMAIL, recipient, subject);
27    }
28  
29    protected static MimeMessage newEmailTo(Session session, String sender, String recipient, String subject) throws MessagingException {
30      // Make a new email message
31      MimeMessage message = new MimeMessage(session);
32  
33      InternetAddress[] to = {new InternetAddress(recipient)};
34      message.setRecipients(Message.RecipientType.TO, to);
35      message.setFrom(sender);
36      message.setSubject(subject);
37      return message;
38    }
39  
40    protected static Session newEmailSession(boolean debug) {
41      // Obtain a Session for sending email
42      Properties props = new Properties();
43      props.put("mail.smtp.host", serverName);
44      props.put("mail.smtp.port", emailServerPort);
45      props.put("mail.smtp.localhost", "127.0.0.1");
46      Session session = Session.getDefaultInstance(props, null);
47      session.setDebug(debug);
48      return session;
49    }
50  
51    public void setEmailServerPort(int port) {
52      emailServerPort = port;
53    }
54  }
