19    public static void importGradesFromCanvas(GradesFromCanvas canvasGrades, GradeBook gradebook) {
20      List<GradesFromCanvas.CanvasStudent> students = canvasGrades.getStudents();
21      logger.debug("Importing grades for " + students.size() + " students");
22  
23      for (GradesFromCanvas.CanvasStudent canvasStudent : students) {
24        Optional<Student> student = canvasGrades.findStudentInGradebookForCanvasStudent(canvasStudent, gradebook);
25        if (student.isEmpty()) {
26          String message = "Could not find student " + canvasStudent + " in gradebook";
27          System.err.println(message);
28  
29        } else {
30          for (GradesFromCanvas.CanvasAssignment canvasAssignment : canvasStudent.getAssignments()) {
31            Optional<Assignment> optional = canvasGrades.findAssignmentInGradebookForCanvasQuiz(canvasAssignment, gradebook);
32            Assignment assignment = optional.orElseThrow(() -> new IllegalStateException("No assignment named \"" + canvasAssignment.getName() + "\" in gradebook"));
33  
34            Double score = canvasStudent.getScore(canvasAssignment);
35            logger.debug("Recording grade of " + score + " for " + student.get() + " for " + assignment.getName());
36            student.get().setGrade(assignment.getName(), new Grade(assignment, score));
37          }
38        }
39      }
40    }
41  
42    public static void main(String[] args) throws IOException, ParserException {
43      String canvasCsvFileName = null;
44      String gradeBookFileName = null;
45  
46      for (String arg : args) {
47        if (canvasCsvFileName == null) {
48          canvasCsvFileName = arg;
49  
50        } else if (gradeBookFileName == null) {
51          gradeBookFileName = arg;
52  
53        } else {
54          usage("Extraneous command line argument: " + arg);
55        }
56      }
57  
58      if (canvasCsvFileName == null) {
59        usage("Missing Canvas CSV file name");
60      }
61  
62      if (gradeBookFileName == null) {
63        usage("Missing grade book file name");
64      }
65  
66      GradesFromCanvas canvasGrades = parseCanvasCsvFile(canvasCsvFileName);
67      GradeBook gradebook = parseGradeBook(gradeBookFileName);
68      importGradesFromCanvas(canvasGrades, gradebook);
69  
70      saveGradeBookIfModified(gradebook, gradeBookFileName);
71    }
72  
73    private static void saveGradeBookIfModified(GradeBook gradeBook, String gradeBookFileName) {
74      if (gradeBook.isDirty()) {
75        File file = new File(gradeBookFileName);
76        try {
77          XmlDumper dumper = new XmlDumper(file);
78          dumper.dump(gradeBook);
79  
80        } catch (IOException e) {
81          usage("Can't write grade book in \"" + gradeBookFileName + "\"");
82        }
83      }
84    }
85  
86    private static GradeBook parseGradeBook(String gradeBookFileName) throws IOException, ParserException {
87      File gradeBookFile = new File(gradeBookFileName);
88      if (!gradeBookFile.exists()) {
89        usage("Grade Book file \"" + gradeBookFile + "\" does not exist");
90      }
91  
92      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
93      return parser.parse();
94    }
95  
96    private static GradesFromCanvas parseCanvasCsvFile(String canvasCsvFileName) throws IOException {
97      File canvasCsvFile = new File(canvasCsvFileName);
98      if (!canvasCsvFile.exists()) {
99        usage("Canvas CSV file \"" + canvasCsvFile + "\" does not exist");
100     }
101 
102     CanvasGradesCSVParser parser = new CanvasGradesCSVParser(new FileReader(canvasCsvFile));
103     return parser.getGrades();
104   }
105 
106   private static void usage(String message) {
107     PrintStream err = System.err;
108 
109     err.println("+++ " + message);
110     err.println();
111     err.println("usage: java GradesFromCanvasImporter canvasGradesCsvFileName gradeBookFileName");
112     err.println("    canvasGradesCsvFileName       Name of the CSV grades file exported from Canvas");
113     err.println("    gradeBookFileName             Gradebook file");
114     err.println();
115     err.println("Imports grades from Canvas into a gradebook");
116     err.println();
117 
118     System.exit(1);
119   }
