3   import com.google.common.io.ByteStreams;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   import java.io.*;
8   import java.util.Date;
9   import java.util.Map;
10  import java.util.jar.Attributes;
11  import java.util.jar.JarFile;
12  import java.util.jar.Manifest;
13  import java.util.zip.ZipEntry;
14  import java.util.zip.ZipOutputStream;
15  
16  public class ZipFileMaker {
17    protected static final Logger logger = LoggerFactory.getLogger("edu.pdx.cs410J.grader");
18  
19    protected final Map<Attributes.Name, String> manifestEntries;
20    private OutputStream zipStream;
21  
22    public ZipFileMaker(File zipFile, Map<Attributes.Name, String> manifestEntries) throws FileNotFoundException {
23      this(new FileOutputStream(zipFile), manifestEntries);
24    }
25  
26    public ZipFileMaker(OutputStream zipStream, Map<Attributes.Name, String> manifestEntries) {
27      this.manifestEntries = manifestEntries;
28      this.zipStream = zipStream;
29    }
30  
31    protected void makeZipFile(Map<ZipEntry, InputStream> zipFileEntries) throws IOException {
32      ZipOutputStream zos = new ZipOutputStream(zipStream);
33      zos.setMethod(ZipOutputStream.DEFLATED);
34  
35      // Create a Manifest for the Zip file containing the name of the
36      // author (userName) and a version that is based on the current
37      // date/time.
38      writeManifestAsEntryInZipFile(zos);
39  
40      // Add the source files to the Zip
41      for (Map.Entry<ZipEntry, InputStream> entry : zipFileEntries.entrySet()) {
42        // Add the entry to the ZIP file
43        zos.putNextEntry(entry.getKey());
44  
45        ByteStreams.copy(entry.getValue(), zos);
46  
47        zos.closeEntry();
48      }
49  
50      zos.close();
51    }
52  
53    private void writeManifestAsEntryInZipFile(ZipOutputStream zos) throws IOException {
54      Manifest manifest = new Manifest();
55      addEntriesToMainManifest(manifest);
56  
57      String entryName = JarFile.MANIFEST_NAME;
58  
59      logger.debug("Adding " + entryName + " to zip");
60      ZipEntry entry = new ZipEntry(entryName);
61      entry.setTime(System.currentTimeMillis());
62      entry.setMethod(ZipEntry.DEFLATED);
63  
64      zos.putNextEntry(entry);
65      manifest.write(new BufferedOutputStream(zos));
66      zos.closeEntry();
67    }
68  
69    private void addEntriesToMainManifest(Manifest manifest) {
70      Attributes attrs = manifest.getMainAttributes();
71  
72      // If a manifest doesn't have a version, the other attributes won't get written out.  Lame.
73      attrs.put(Attributes.Name.MANIFEST_VERSION, new Date().toString());
74  
75      for (Map.Entry<Attributes.Name, String> entry : this.manifestEntries.entrySet()) {
76        attrs.put(entry.getKey(), entry.getValue());
77      }
78    }
79  }
