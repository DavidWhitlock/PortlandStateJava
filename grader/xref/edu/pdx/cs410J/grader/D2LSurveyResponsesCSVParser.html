1   package edu.pdx.cs410J.grader;
2   
3   import au.com.bytecode.opencsv.CSVReader;
4   import com.google.common.annotations.VisibleForTesting;
5   
6   import java.io.IOException;
7   import java.io.Reader;
8   
9   public class D2LSurveyResponsesCSVParser {
10  
11    @VisibleForTesting
12    int questionColumnIndex = -1;
13  
14    @VisibleForTesting
15    int responseColumnIndex = -1;
16  
17    private final SurveyResponsesFromD2L responses;
18  
19    public D2LSurveyResponsesCSVParser(Reader reader) throws IOException {
20      CSVReader csv = new CSVReader( reader );
21      String[] firstLine = csv.readNext();
22      extractColumnNamesFromFirstLineOfCsv(firstLine);
23  
24      responses = new SurveyResponsesFromD2L();
25  
26      String[] surveyLine;
27      while ((surveyLine = csv.readNext()) != null) {
28        addQuestionAndResponseFromLineOfCsv(surveyLine);
29      }
30    }
31  
32    private void addQuestionAndResponseFromLineOfCsv(String[] surveyLine) {
33      String question = getQuestionFromSurveyLine(surveyLine);
34      String response = getResponseFromSurveyLine(surveyLine);
35  
36      this.responses.noteQuestionAndResponse(question, response);
37    }
38  
39    private String getResponseFromSurveyLine(String[] surveyLine) {
40      return surveyLine[responseColumnIndex];
41    }
42  
43    private String getQuestionFromSurveyLine(String[] surveyLine) {
44      return surveyLine[questionColumnIndex];
45    }
46  
47    private void extractColumnNamesFromFirstLineOfCsv(String[] firstLine) {
48      for (int columnIndex = 0; columnIndex < firstLine.length; columnIndex++) {
49        String columnHeading = firstLine[columnIndex];
50        if ("Q Text".equals(columnHeading)) {
51          this.questionColumnIndex = columnIndex;
52  
53        } else if ("Answer".equals(columnHeading)) {
54          this.responseColumnIndex = columnIndex;
55        }
56      }
57    }
58  
59    public SurveyResponsesFromD2L getSurveyResponses() {
60      return this.responses;
61    }
62  }
