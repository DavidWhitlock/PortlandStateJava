28    public static ProjectScore getScoreFrom(Reader reader) throws ScoreNotFoundException {
29      BufferedReader br = new BufferedReader(reader);
30      Optional<String> scoreLine = br.lines().filter(scorePattern.asPredicate()).findFirst();
31  
32      if (scoreLine.isPresent()) {
33        Matcher matcher = scorePattern.matcher(scoreLine.get());
34        if (matcher.find()) {
35          return new ProjectScore(matcher.group(1), matcher.group(2));
36  
37        } else {
38          throw new IllegalStateException("Matcher didn't match \"" + scoreLine.get() + "\"");
39        }
40  
41      } else {
42        throw new ScoreNotFoundException();
43      }
44  
45    }

47    public void recordScoreFromProjectReport(String studentId, Reader report) throws ScoreNotFoundException {
48      ProjectScore score = getScoreFrom(report);
49  
50      if (score.getTotalPoints() != this.assignment.getPoints()) {
51        String message = "Assignment " + this.assignment.getName() + " should be worth " + this.assignment.getPoints() +
52          " points, but the report for " + studentId + " was out of " + score.getTotalPoints();
53        throw new IllegalStateException(message);
54      }
55  
56      Optional<Student> maybeStudent = gradeBook.getStudent(studentId);
57      if (!maybeStudent.isPresent()) {
58        maybeStudent = gradeBook.getStudentWithSsn(studentId);
59      }
60      if (!maybeStudent.isPresent()) {
61        warn("Student \"" + studentId + "\" not found in gradebook");
62        return;
63      }
64  
65      Student student = maybeStudent.get();
66      Grade grade = student.getGrade(this.assignment);
67      if (grade == null) {
68        grade = new Grade(assignment, score.getScore());
69        student.setGrade(assignment.getName(), grade);
70  
71      } else {
72        grade.setScore(score.getScore());
73      }
74  
75      info("Recorded grade of " + score.getScore() + " for " + studentId);
76    }

163   private static String getStudentIdFromFileName(File projectFile) {
164     String fileName = projectFile.getName();
165     int index = fileName.lastIndexOf('.');
166     if (index < 0) {
167       return usage("Project file \"" + fileName + "\" does not have a file extension");
168     }
169 
170     return fileName.substring(0, index);
171   }
