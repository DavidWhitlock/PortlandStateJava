40    @Provides
41    @Singleton
42    public EventBus provideEventBus() {
43      return new EventBusThatPublishesUnhandledExceptionEvents();
44    }
45  
46    @Provides
47    @Singleton
48    @Named("POADownloaderExecutor")
49    public Executor provideDownloaderExecutor(EventBus bus) {
50      return new ThreadPoolExecutor(1, 1, 0L, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<>(), new ThreadFactory() {
51        @Override
52        public Thread newThread(Runnable runnable) {
53          return new Thread(runnable, "POA Downloader");
54        }
55      }) {
56        @Override
57        protected void afterExecute(Runnable r, Throwable t) {
58          super.afterExecute(r, t);
59          if (t == null && r instanceof Future<?> && ((Future<?>) r).isDone()) {
60            try {
61              ((Future<?>) r).get();
62            } catch (CancellationException ce) {
63              t = ce;
64            } catch (ExecutionException ee) {
65              t = ee.getCause();
66            } catch (InterruptedException ie) {
67              // ignore/reset
68              Thread.currentThread().interrupt();
69            }
70          }
71          if (t != null) {
72            bus.post(new UnhandledExceptionEvent(t));
73          }
74        }
75      };
76    }
