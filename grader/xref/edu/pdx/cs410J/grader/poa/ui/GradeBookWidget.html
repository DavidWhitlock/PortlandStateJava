1   package edu.pdx.cs410J.grader.poa.ui;
2   
3   import com.google.inject.Inject;
4   import com.google.inject.Singleton;
5   import edu.pdx.cs410J.grader.poa.GradeBookView;
6   
7   import javax.swing.*;
8   import javax.swing.filechooser.FileNameExtensionFilter;
9   import java.awt.*;
10  
11  @Singleton
12  public class GradeBookWidget extends JPanel implements GradeBookView {
13    private final JLabel gradeBookName;
14    private final JButton loadGradeBookButton;
15    private final JButton saveGradeBookButton;
16    private final Component parentComponent;
17  
18    @Inject
19    public GradeBookWidget(TopLevelJFrame parentFrame) {
20      this.parentComponent = parentFrame;
21      gradeBookName = new JLabel();
22      loadGradeBookButton = new JButton("Load Book");
23      saveGradeBookButton = new JButton("Save Book");
24      saveGradeBookButton.setEnabled(false);
25  
26      this.setLayout(new BoxLayout(this, BoxLayout.LINE_AXIS));
27      this.add(gradeBookName);
28      this.add(Box.createHorizontalStrut(3));
29      this.add(loadGradeBookButton);
30      this.add(Box.createHorizontalStrut(3));
31      this.add(saveGradeBookButton);
32      this.add(Box.createHorizontalGlue());
33    }
34  
35    @Override
36    public void setGradeBookName(String className) {
37      this.gradeBookName.setText(className);
38    }
39  
40    @Override
41    public void addGradeBookFileListener(FileSelectedListener listener) {
42      this.loadGradeBookButton.addActionListener(e -> displayFileChooser(listener));
43    }
44  
45    @Override
46    public void canSaveGradeBook(boolean canSaveGradeBook) {
47      this.saveGradeBookButton.setEnabled(canSaveGradeBook);
48    }
49  
50    @Override
51    public void addSaveGradeBookListener(SaveGradeBookListener listener) {
52      this.saveGradeBookButton.addActionListener(e -> listener.saveGradeBook());
53    }
54  
55    private void displayFileChooser(FileSelectedListener listener) {
56      JFileChooser chooser = new JFileChooser();
57      chooser.setDialogTitle("Select Grade Book file");
58      chooser.setApproveButtonMnemonic('l');
59      chooser.setApproveButtonText("Load");
60      chooser.setApproveButtonToolTipText("Load Grade Book from file");
61      chooser.setFileFilter(new FileNameExtensionFilter("Grade Book XML files", "xml"));
62      chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
63  
64      int result = chooser.showOpenDialog(parentComponent);
65      if (result == JFileChooser.APPROVE_OPTION) {
66        listener.fileSelected(chooser.getSelectedFile());
67      }
68    }
69  }
