3   import com.google.common.eventbus.EventBus;
4   import com.google.common.eventbus.Subscribe;
5   import com.google.inject.Inject;
6   import com.google.inject.Singleton;
7   import edu.pdx.cs410J.grader.gradebook.GradeBook;
8   import edu.pdx.cs410J.grader.gradebook.Student;

10  import java.util.Collections;
11  import java.util.Comparator;
12  import java.util.List;
13  import java.util.stream.Collectors;
14  
15  @Singleton
16  public class StudentsPresenter {
17    private final EventBus bus;
18    private final StudentsView view;
19    private List<Student> students = Collections.emptyList();
20  
21    @Inject
22    public StudentsPresenter(EventBus bus, StudentsView view) {
23      this.bus = bus;
24      this.view = view;
25  
26      this.bus.register(this);
27  
28      this.view.addSelectStudentHandler(this::fireSelectedStudentEventForStudentAtIndex);
29    }
30  
31    private void fireSelectedStudentEventForStudentAtIndex(int index) {
32      Student student;
33      if (index == 0) {
34        student = null;
35  
36      } else {
37        student = this.students.get(index - 1);
38      }
39      fireStudentSelectedEvent(student);
40    }
41  
42    private void fireStudentSelectedEvent(Student student) {
43      this.bus.post(new StudentSelectedEvent(student));
44    }
45  
46    @Subscribe
47    public void populateViewWithStudents(GradeBookLoaded event) {
48      GradeBook book = event.getGradeBook();
49  
50      this.students = book.studentsStream()
51        .sorted(sortStudentsByLastNameFirstNameEmail())
52        .collect(Collectors.toList());
53  
54      List<String> studentsDisplayStrings = this.students.stream()
55        .map(this::getStudentDisplayText)
56        .collect(Collectors.toList());
57      studentsDisplayStrings.add(0, "<unknown student>");
58      this.view.setStudents(studentsDisplayStrings);
59  
60      this.view.setSelectedStudentIndex(0);
61    }
62  
63    private Comparator<Student> sortStudentsByLastNameFirstNameEmail() {
64      return (student1, student2) -> {
65        String lastName1 = getLastNameOfStudent(student1);
66        String lastName2 = getLastNameOfStudent(student2);
67        if (!lastName1.equals(lastName2)) {
68          return lastName1.compareTo(lastName2);
69  
70        } else {
71          String firstName1 = getFirstNameOfStudent(student1);
72          String firstName2 = getFirstNameOfStudent(student2);
73          if (!firstName1.equals(firstName2)) {
74            return firstName1.compareTo(firstName2);
75  
76          } else {
77            return student1.getEmail().compareTo(student2.getEmail());
78          }
79        }
80      };
81    }
82  
83    private String getLastNameOfStudent(Student student) {
84      String lastName = student.getLastName();
85      if (lastName == null) {
86        throw new IllegalStateException("Student " + student.getId() + " doesn't have a last name");
87      }
88      return lastName;
89    }
90  
91    private String getStudentDisplayText(Student student) {
92      return getFirstNameOfStudent(student) + " " + getLastNameOfStudent(student) + " <" + student.getEmail() + ">";
93    }
94  
95    @Subscribe
96    public void attemptToMatchSubmissionWithStudent(POASubmissionSelected selected) {
97      for (int i = 0; i < students.size(); i++) {
98        Student student = students.get(i);
99        if (submitterMatchesStudent(selected.getSubmission(), student)) {
100         this.view.setSelectedStudentIndex(i + 1);
101         fireStudentSelectedEvent(student);
102         return;
103       }
104     }
105 
106     if (this.students.size() > 0) {
107       this.view.setSelectedStudentIndex(0);
108     }
109     fireStudentSelectedEvent(null);
110   }
111 
112   private boolean submitterMatchesStudent(POASubmission submission, Student student) {
113     String submitter = submission.getSubmitter();
114     return submitterContainsStudentName(student, submitter) || submitterContainsStudentEmail(student, submitter);
115   }
116 
117   private boolean submitterContainsStudentEmail(Student student, String submitter) {
118     return submitter.contains(student.getEmail());
119   }
120 
121   private boolean submitterContainsStudentName(Student student, String submitter) {
122     return submitter.contains(getFirstNameOfStudent(student)) && submitter.contains(getLastNameOfStudent(student));
123   }
124 
125   private String getFirstNameOfStudent(Student student) {
126     String firstName = student.getFirstName();
127     if (firstName == null) {
128       throw new IllegalStateException("Student " + student.getId() + " doesn't have a first name");
129     }
130     return firstName;
131   }

133 }
