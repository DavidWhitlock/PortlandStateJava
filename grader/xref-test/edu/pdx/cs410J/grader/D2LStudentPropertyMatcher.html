3   import org.hamcrest.Description;
4   import org.hamcrest.TypeSafeMatcher;
5   
6   import java.util.Iterator;
7   import java.util.List;
8   import java.util.Spliterator;
9   import java.util.function.Consumer;
10  import java.util.function.Function;
11  import java.util.stream.Stream;
12  
13  public class D2LStudentPropertyMatcher extends TypeSafeMatcher<List<GradesFromD2L.D2LStudent>> {
14  
15    private final String expectedValue;
16    private final Function<GradesFromD2L.D2LStudent, String> accessor;
17    private final String propertyName;
18  
19    public D2LStudentPropertyMatcher(Function<GradesFromD2L.D2LStudent, String> accessor, String propertyName, String expectedValue) {
20      this.expectedValue = expectedValue;
21      this.accessor = accessor;
22      this.propertyName = propertyName;
23    }
24  
25    @Override
26    protected boolean matchesSafely(List<GradesFromD2L.D2LStudent> students) {
27      for (GradesFromD2L.D2LStudent student : students) {
28        if (this.accessor.apply(student).equals(expectedValue)) {
29          return true;
30        }
31      }
32  
33      return false;
34    }
35  
36    @Override
37    public void describeTo(Description description) {
38      description.appendText("a student with a " + propertyName + " of ").appendValue(expectedValue);
39    }
40  
41    @Override
42    protected void describeMismatchSafely(List<GradesFromD2L.D2LStudent> students, Description mismatchDescription) {
43      mismatchDescription.appendText("the students' " + propertyName + "s were: ");
44      Stream<String> actualValues = students.stream().map(accessor);
45      mismatchDescription.appendValueList("", ",", "", toIterable(actualValues));
46    }
47  
48    private <T> Iterable<T> toIterable(final Stream<T> stream) {
49      return new Iterable<T>() {
50        @Override
51        public Iterator<T> iterator() {
52          return stream.iterator();
53        }
54  
55        @Override
56        public void forEach(Consumer<? super T> action) {
57          stream.forEach(action);
58        }
59  
60        @Override
61        public Spliterator<T> spliterator() {
62          return stream.spliterator();
63        }
64      };
65    }
66  }
