27    private void noteSubmission(Student student, Assignment project, Double... estimates) {
28      Grade grade = new Grade(project, Grade.NO_GRADE);
29      for (Double estimate : estimates) {
30        Grade.SubmissionInfo submission = new Grade.SubmissionInfo();
31        if (estimate != null) {
32          submission.setEstimatedHours(estimate);
33        }
34        grade.noteSubmission(submission);
35      }
36      student.setGrade(project, grade);
37    }

39    private Student addStudent(GradeBook book, String studentName) {
40      Student student = new Student(studentName);
41      book.addStudent(student);
42      return student;
43    }

45    private Assignment addProject(GradeBook book, ProjectType projectType) {
46      return addProject(book, "project", projectType);
47    }

49    private Assignment addProject(GradeBook book, String projectName, ProjectType projectType) {
50      Assignment project = new Assignment(projectName, 1.0)
51        .setProjectType(projectType);
52      book.addAssignment(project);
53      return project;
54    }

263   @Test
264   void generateMarkdown() {
265     GradeBook book = new GradeBook("test");
266 
267     Assignment appClasses = addProject(book, "AppClasses", APP_CLASSES);
268     Assignment textFile = addProject(book, "TextFile", TEXT_FILE);
269     
270     Student student1 = addStudent(book, "student1");
271     Student student2 = addStudent(book, "student2");
272     Student student3 = addStudent(book, "student3");
273     Student student4 = addStudent(book, "student4");
274     Student student5 = addStudent(book, "student5");
275     
276     noteSubmission(student1, appClasses, 4.0);    
277     noteSubmission(student2, appClasses, 5.0);    
278     noteSubmission(student3, appClasses, 3.0);    
279     noteSubmission(student4, appClasses, 6.0);    
280     noteSubmission(student5, appClasses, 2.0);    
281     noteSubmission(student1, textFile, 7.0);
282     noteSubmission(student2, textFile, 6.0);
283     noteSubmission(student3, textFile, 8.0);
284     noteSubmission(student4, textFile, 6.0);    
285     noteSubmission(student5, textFile, 7.0);
286 
287     ProjectTimeEstimatesSummary summary = new ProjectTimeEstimatesSummary();
288     TimeEstimatesSummaries summaries = summary.getTimeEstimateSummaries(book);
289     TimeEstimatesSummary appClassesEstimates = summaries.getTimeEstimateSummary(appClasses.getProjectType());
290     assertThat(appClassesEstimates.getCount(), equalTo(5));
291 
292     StringWriter sw = new StringWriter();
293     summaries.generateMarkdown(sw, List.of(appClasses.getProjectType(), textFile.getProjectType()));
294     String markdown = sw.toString();
295 
296 //    System.out.println(markdown);
297 
298     List<String> lines = markdown.lines().collect(Collectors.toList());
299 
300     assertThat(lines.get(0), equalTo("|  | App Classes | Text File |"));
301     assertThat(lines.get(1), equalTo("| :--- | ---: | ---: |"));
302     assertThat(lines.get(2), equalTo("| Count | 5 | 5 |"));
303     assertThat(lines.get(3), matchesRegex("\\| Average \\| \\d hours \\| \\d hours \\|"));
304     assertThat(lines.get(4), matchesRegex("\\| Maximum \\| \\d hours \\| \\d hours \\|"));
305     assertThat(lines.get(5), matchesRegex("\\| Top 25% \\| \\d hours \\| \\d hours \\|"));
306     assertThat(lines.get(6), matchesRegex("\\| Median \\| \\d hours \\| \\d hours \\|"));
307     assertThat(lines.get(7), matchesRegex("\\| Bottom 25% \\| \\d hours \\| \\d hours \\|"));
308     assertThat(lines.get(8), matchesRegex("\\| Minimum \\| \\d hours \\| \\d hours \\|"));
309   }
