15    public void defaultLetterGradeRanges() {
16      GradeBook book = new GradeBook("test");
17      GradeBook.LetterGradeRanges undergrad = book.getLetterGradeRanges(UNDERGRADUATE);
18      assertDefaultLetterGradeRanges(undergrad);

20      GradeBook.LetterGradeRanges grad = book.getLetterGradeRanges(GRADUATE);
21      assertDefaultLetterGradeRanges(grad);
22    }

24    private void assertDefaultLetterGradeRanges(GradeBook.LetterGradeRanges ranges) {
25      assertLetterGradeRange(ranges, LetterGrade.A, 94, 100);
26      assertLetterGradeRange(ranges, LetterGrade.A_MINUS, 90, 93);
27      assertLetterGradeRange(ranges, LetterGrade.B_PLUS, 87, 89);
28      assertLetterGradeRange(ranges, LetterGrade.B, 83, 86);
29      assertLetterGradeRange(ranges, LetterGrade.B_MINUS, 80, 82);
30      assertLetterGradeRange(ranges, LetterGrade.C_PLUS, 77, 79);
31      assertLetterGradeRange(ranges, LetterGrade.C, 73, 76);
32      assertLetterGradeRange(ranges, LetterGrade.C_MINUS, 70, 72);
33      assertLetterGradeRange(ranges, LetterGrade.D_PLUS, 67, 69);
34      assertLetterGradeRange(ranges, LetterGrade.D, 63, 66);
35      assertLetterGradeRange(ranges, LetterGrade.D_MINUS, 60, 62);
36      assertLetterGradeRange(ranges, LetterGrade.F, 0, 59);
37    }
38  
39    private void assertLetterGradeRange(GradeBook.LetterGradeRanges ranges, LetterGrade letterGrade, int minimum, int maximum) {
40      LetterGradeRange range = ranges.getRange(letterGrade);
41      assertThat("No range for " + letterGrade, range, not(nullValue()));
42      assertThat("Less than for " + letterGrade, range.maximum(), equalTo(maximum));
43      assertThat(range.minimum(), equalTo(minimum));
44    }

105   @Test
106   public void scoreIsInRange() {
107     GradeBook book = new GradeBook("test");
108     GradeBook.LetterGradeRanges ranges = book.getLetterGradeRanges(UNDERGRADUATE);
109     LetterGradeRange range = ranges.getRange(LetterGrade.A);
110     range.setRange(90, 100);
111 
112     assertThat(range.isScoreInRange(90.0), is(true));
113     assertThat(range.isScoreInRange(100.0), is(true));
114     assertThat(range.isScoreInRange(95.0), is(true));
115     assertThat(range.isScoreInRange(89.5), is(true));
116     assertThat(range.isScoreInRange(101.0), is(false));
117     assertThat(range.isScoreInRange(89.0), is(false));
118     assertThat(range.isScoreInRange(89.4), is(false));
119 
120   }
