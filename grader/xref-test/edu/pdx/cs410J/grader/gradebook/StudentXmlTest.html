31    private Student writeAndReadStudentAsXml(Student student) throws TransformerException, ParserException {
32      Document doc = XmlDumper.toXml(student);
33      byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
34  
35      XmlStudentParser parser = new XmlStudentParser(new InputStreamReader(new ByteArrayInputStream(bytes)));
36      return parser.parseStudent();
37    }

39    @Test
40    public void oneSubmissionIsPersistedInXml() throws TransformerException, ParserException {
41      Student student = new Student("test");
42      String assignmentName = "project";
43      LocalDateTime submissionTime = LocalDateTime.now().minusHours(3).withNano(0);
44      double estimatedHours = 43.5;
45      boolean isLate = true;
46  
47      Grade grade = new Grade(assignmentName, Grade.NO_GRADE);
48      grade.noteSubmission(submissionTime)
49        .setEstimatedHours(estimatedHours)
50        .setIsLate(isLate);
51      student.setGrade(assignmentName, grade);
52  
53      Student student2 = writeAndReadStudentAsXml(student);
54      Grade grade2 = student2.getGrade(assignmentName);
55      assertThat(grade2.getSubmissionInfos(), hasSize(1));
56  
57      Grade.SubmissionInfo info = grade2.getSubmissionInfos().get(0);
58      assertThat(info.getSubmissionTime(), equalTo(submissionTime));
59      assertThat(info.getEstimatedHours(), equalTo(estimatedHours));
60      assertThat(info.isLate(), equalTo(isLate));
61    }

63    @Test
64    public void multipleSubmissionsArePersistedInXml() throws TransformerException, ParserException {
65      Student student = new Student("test");
66      String assignmentName = "project";
67      LocalDateTime submissionTime1 = LocalDateTime.now().minusHours(3).withNano(0);
68      double estimatedHours1 = 24.8;
69      LocalDateTime submissionTime2 = LocalDateTime.now().minusHours(4).withNano(0);
70  
71      Grade grade = new Grade(assignmentName, Grade.NO_GRADE);
72      grade.noteSubmission(submissionTime1).setEstimatedHours(estimatedHours1);
73      grade.noteSubmission(submissionTime2);
74      student.setGrade(assignmentName, grade);
75  
76      Student student2 = writeAndReadStudentAsXml(student);
77      Grade grade2 = student2.getGrade(assignmentName);
78      assertThat(grade2.getSubmissionInfos(), hasSize(2));
79  
80      Grade.SubmissionInfo info1 = grade2.getSubmissionInfos().get(0);
81      assertThat(info1.getSubmissionTime(), equalTo(submissionTime1));
82      assertThat(info1.getEstimatedHours(), equalTo(estimatedHours1));
83  
84      Grade.SubmissionInfo info2 = grade2.getSubmissionInfos().get(1);
85      assertThat(info2.getSubmissionTime(), equalTo(submissionTime2));
86      assertThat(info2.getEstimatedHours(), nullValue());
87    }

111   @Test
112   public void xmlHasNewlinesButNotIndentation() throws TransformerException {
113     Student student = new Student("test");
114     Document doc = XmlDumper.toXml(student);
115     byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
116     String string = new String(bytes);
117     assertThat(string, containsString("\n"));
118     assertThat(string.chars().filter(c -> '\n' == c).count(), greaterThan(4L));
119     assertThat(string, not(containsString(" <")));
120   }

142   private Reader readTextResource(String resourceName) {
143     InputStream resource = getClass().getResourceAsStream(resourceName);
144     if (resource == null) {
145       throw new IllegalArgumentException("Can't find resource " + resourceName);
146     }
147     return new InputStreamReader(resource);
148   }
