023  @Test
024  public void testClickingShowPhoneBillButtonAlertsWithPhoneBillInformation() {
025    final CapturingAlerter alerter = new CapturingAlerter();
026
027    final PhoneBillGwt ui = new PhoneBillGwt(alerter);
028    ui.onModuleLoad();
029
030    // Wait for UI widgets to be created
031    waitBeforeRunning(500, new Runnable() {
032      @Override
033      public void run() {
034        click(ui.showPhoneBillButton);
035      }
036    });
037
038    // Wait for the RPC call to return
039    waitBeforeRunning(1000, new Runnable() {
040      @Override
041      public void run() {
042        String message = alerter.getMessage();
043        assertNotNull("No message was displayed", message);
044        assertTrue(message, message.contains("CS410J's phone bill with 1 phone calls"));
045        finishTest();
046      }
047    });
048
049    delayTestFinish(1000);
050  }

121      @Override
122      public void run() {
123        try {
124          click(ui.showClientSideExceptionButton);
125          fail("Should have thrown an UmbrellaException");
126
127        } catch (UmbrellaException ex) {
128          Throwable cause = ex.getCause();
129          assertTrue(cause instanceof IllegalStateException);
130          IllegalStateException ise = (IllegalStateException) cause;
131          assertTrue(ise.getMessage().contains("Expected exception on the client side"));
132          finishTest();
133        }
134      }
135    });
136
137    // Wait up to 1000 milliseconds for the validation to complete
138    delayTestFinish(1000);
139  }

141  private void waitBeforeRunning(int delayMillis, final Runnable operation) {
142    Timer click = new Timer() {
143      @Override
144      public void run() {
145        operation.run();
146      }
147    };
148    click.schedule(delayMillis);
149  }
150
151  /**
152   * Clicks a <code>Button</code>
153   * <p>
154   * One would think that you could testing clicking a button with Button.click(), but it looks
155   * like you need to fire the native event instead.  Lame.
156   *
157   * @param button The button to click
158   */
159  private void click(final Button button) {
160    assertNotNull("Button is null", button);
161    NativeEvent event = Document.get().createClickEvent(0, 0, 0, 0, 0, false, false, false, false);
162    DomEvent.fireNativeEvent(event, button);
163  }
164
165  private class CapturingAlerter implements PhoneBillGwt.Alerter {
166    private String message;
167
168    @Override
169    public void alert(String message) {
170      this.message = message;
171    }
172
173    public String getMessage() {
174      return message;
175    }
176  }
