051  /**
052   * Tests that the size of the LRU map is bounded
053   *
054   * @see AbstractLRUMap#put(Object, Object)
055   */
056  @Test
057  public void testMapSizeIsBounded() {
058    int maxMappings = 4;
059    AbstractLRUMap<Integer, String> map =
060      createLRUMap(maxMappings);
061    assertEquals(0, map.size());
062
063    for (int i = 0; i < maxMappings; i++) {
064      map.put(i, String.valueOf(i));
065      assertEquals(i + 1, map.size());
066    }
067
068    map.put(maxMappings + 1,
069      String.valueOf(maxMappings + 1));
070
071    assertEquals(maxMappings, map.size());
072  }

074  /**
075   * Makes sure that entrySet() Iterator iterates
076   *
077   * @see AbstractMap#containsKey(Object)
078   */
079  @Test
080  public void testLRUMappingIterates() {
081    int maxMappings = 4;
082    AbstractLRUMap<Integer, String> map =
083      createLRUMap(maxMappings);
084    assertEquals(0, map.size());
085
086    for (int i = 0; i < maxMappings; i++) {
087      map.put(i, String.valueOf(i));
088      assertEquals(i + 1, map.size());
089    }
090
091    try {
092      // get the iterator for this map's mappings 
093      Iterator iter = map.entrySet().iterator();
094
095      // iterate through the mappings 
096      for (int i = 0; i < maxMappings; i++) {
097        iter.next();
098      }
099
100      // there should be no more mappings so this should              
101      // throw a NoSuchElementException error                         
102      iter.next();
103      System.out.println("Bad iterator returned by entrySet()."
104        + "\nFix before continuing unit tests.");
105      System.exit(1);
106
107    } catch (IndexOutOfBoundsException ex) {
108      // pass 
109    } catch (NoSuchElementException ex) {
110      // pass 
111    }
112  }

179  /**
180   * Tests that removing and element doesn't effect the LRU order
181   *
182   * @see AbstractLRUMap#remove(Object)
183   */
184  @Test
185  public void testRemoveFromMap() {
186    int maxMappings = 4;
187    AbstractLRUMap<Integer, String> map =
188      createLRUMap(maxMappings);
189    assertEquals(0, map.size());
190
191    for (int i = 0; i < maxMappings; i++) {
192      map.put(i, String.valueOf(i));
193      assertEquals(i + 1, map.size());
194    }
195
196    assertTrue(map.containsKey(new Integer(0)));
197    assertNotNull(map.remove(new Integer(0)));
198
199    map.put(maxMappings + 1,
200      String.valueOf(maxMappings + 1));
201    assertTrue(map.containsKey(new Integer(1)));
202
203    map.put(maxMappings + 2,
204      String.valueOf(maxMappings + 2));
205
206    assertTrue(!map.containsKey(new Integer(1)));
207  }
