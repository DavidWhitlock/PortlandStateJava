001package edu.pdx.cs410J.apptbookgwt;
002
003import com.gdevelop.gwt.syncrpc.SyncProxy;
004import edu.pdx.cs410J.apptbookgwt.client.AppointmentBook;
005import edu.pdx.cs410J.apptbookgwt.client.AppointmentBookService;
006import edu.pdx.cs410J.web.HttpRequestHelper;
007import org.junit.Test;
008
009import java.io.IOException;
010
011import static junit.framework.TestCase.assertEquals;
012
013public class AppointmentBookServiceSyncProxyIT extends HttpRequestHelper {
014
015  private final int httpPort = Integer.getInteger("http.port", 8080);
016  private String webAppUrl = "http://localhost:" + httpPort + "/apptbook";
017
018  @Test
019  public void gwtWebApplicationIsRunning() throws IOException {
020    Response response = get(this.webAppUrl);
021    assertEquals(200, response.getCode());
022  }
023
024  @Test
025  public void canInvokeAppointmentBookServiceWithGwtSyncProxy() {
026    String moduleName = "apptbook";
027    SyncProxy.setBaseURL(this.webAppUrl + "/" + moduleName + "/");
028
029    AppointmentBookService service = SyncProxy.createSync(AppointmentBookService.class);
030    AppointmentBook apptbook = service.getAppointmentBook();
031    assertEquals("My Owner", apptbook.getOwnerName());
032    assertEquals(1, apptbook.getAppointments().size());
033  }
034
035}
