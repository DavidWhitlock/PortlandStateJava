001package edu.pdx.cs410J.junit;
002
003import static org.junit.Assert.assertNotNull;
004import org.junit.internal.AssumptionViolatedException;
005import org.junit.runner.Description;
006import org.junit.runner.RunWith;
007import org.junit.runner.notification.Failure;
008import org.junit.runner.notification.RunNotifier;
009import org.junit.runners.ParentRunner;
010import org.junit.runners.model.InitializationError;
011import org.junit.BeforeClass;
012import org.junit.AfterClass;
013
014import java.util.ArrayList;
015import java.util.Arrays;
016import java.util.List;
017
018
019/**
020 * The class demonstrates how JUnit 4's {@link ParentRunner} class can be used to generate test cases
021 * dynamically.
022 */
023@RunWith(AutomaticallyGeneratedTest.TestGenerator.class)
024public class AutomaticallyGeneratedTest
025{
026    @BeforeClass
027    public static void setUp() {
028        System.out.println("BeforeClass");
029    }
030
031    @AfterClass
032    public static void tearDown() {
033        System.out.println("AfterClass");
034    }
035
036    public static List<String> getTestNames() {
037        return Arrays.asList( "One", "Two", "Three" );
038    }
039
040    public static class GeneratedTest
041    {
042        private final String name;
043
044        public GeneratedTest(String name) {
045            this.name = name;
046        }
047
048        public String getName() {
049            return name;
050        }
051
052        public void run()
053        {
054            assertNotNull(getName());
055            // assertEquals( 3, getName().length() );
056        }
057    }
058
059
060    public static class TestGenerator extends ParentRunner<GeneratedTest> {
061
062        public TestGenerator( Class<?> testClass )
063            throws InitializationError
064        {
065            super( testClass );
066        }
067
068        @Override
069        protected List<GeneratedTest> getChildren()
070        {
071            List<GeneratedTest> tests = new ArrayList<GeneratedTest>();
072            for (String name : getTestNames() ) {
073                tests.add(new GeneratedTest( name ));                
074            }
075            return tests;
076        }
077
078        @Override
079        protected Description describeChild( GeneratedTest generatedTest )
080        {
081            return Description.createSuiteDescription( generatedTest.getName() );
082        }
083
084        @Override
085        protected void runChild( GeneratedTest generatedTest, RunNotifier notifier )
086        {
087            Description desc = describeChild( generatedTest );
088            notifier.fireTestStarted( desc );
089            try
090            {
091                generatedTest.run();
092            }
093            catch ( AssumptionViolatedException e )
094            {
095                notifier.fireTestAssumptionFailed( new Failure( desc, e) );
096            }
097            catch ( Throwable e )
098            {
099                notifier.fireTestFailure( new Failure( desc, e) );
100            }
101            finally
102            {
103                notifier.fireTestFinished( desc );
104            }
105        }
106    }
107}
