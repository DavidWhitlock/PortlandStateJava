001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import org.junit.Test;
005
006import static org.junit.Assert.fail;
007import static org.mockito.Mockito.*;
008import static edu.pdx.cs410J.gwt.client.mvp.ExceptionPresenter.*;
009
010/**
011 * Tests that an exception triggers the {@link ExceptionPresenter}
012 */
013public class ExceptionPresenterTest {
014
015  /**
016   * Tests that when an exception occurs in an {@link MvpCallback}, the exception view is displayed
017   */
018  @Test
019  public void testExceptionViewDisplayed() {
020    HandlerManager eventBus = new HandlerManager(null);
021
022    MvpCallback callback = new MvpCallback<Void>(eventBus) {
023
024      @Override
025      public void onSuccess(Void aVoid) {
026        fail("onSuccess should not be invoked");
027      }
028    };
029
030    Display view = mock(Display.class);
031    new ExceptionPresenter(view, eventBus);
032
033    String message = "This is a message";
034    Throwable ex = new Throwable(message);
035    ex.fillInStackTrace();
036
037    callback.onFailure(ex);
038
039    verify(view).show();
040    verify(view).setMessage(message);
041    verify(view).setStackTrace(ex.getStackTrace());
042  }
043}
