001package edu.pdx.cs410J.gwt.client.mvp;
002
003import edu.pdx.cs410J.rmi.Movie;
004
005import java.lang.reflect.Constructor;
006import java.lang.reflect.Method;
007
008import static org.junit.Assert.fail;
009
010public abstract class MovieTestCase {
011
012  /**
013   * Uses reflection to create a new movie because the constructor is package-private
014   *
015   * @param title The title of the movie
016   * @param year  The year the movie was released
017   * @return A new movie
018   */
019  Movie newMovie(String title, int year) {
020    try {
021      Constructor<Movie> init = Movie.class.getDeclaredConstructor(String.class, int.class);
022      init.setAccessible(true);
023      return init.newInstance(title, year);
024
025    } catch (Exception e) {
026      fail(e.toString());
027      return null;
028    }
029  }
030
031  /**
032   * Uses reflection to invoke the addCharacter method of Movie
033   * @param movie
034   * @param character
035   * @param actorId
036   */
037  protected void addCharacter(Movie movie, String character, long actorId) {
038    try {
039      Method m = Movie.class.getDeclaredMethod("addCharacter", String.class, long.class);
040      m.setAccessible(true);
041      m.invoke(movie, character, actorId);
042
043    } catch (Exception e) {
044      fail(e.toString());
045    }
046  }
047}
