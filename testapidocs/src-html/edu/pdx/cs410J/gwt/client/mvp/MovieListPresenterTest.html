001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.common.base.Function;
004import com.google.common.collect.Lists;
005import com.google.gwt.event.shared.HandlerManager;
006import com.google.gwt.user.client.rpc.AsyncCallback;
007import edu.pdx.cs410J.rmi.Movie;
008import org.junit.Before;
009import org.junit.Test;
010import org.mockito.ArgumentCaptor;
011import org.mockito.Captor;
012import org.mockito.MockitoAnnotations;
013
014import java.util.ArrayList;
015import java.util.List;
016
017import static edu.pdx.cs410J.gwt.client.mvp.MovieListPresenter.Display;
018import static edu.pdx.cs410J.gwt.client.mvp.MovieListPresenter.Display.MovieSelectedHandler;
019import static org.mockito.Mockito.mock;
020import static org.mockito.Mockito.verify;
021
022/**
023 * Tests the logic of the {@link MovieListPresenter}
024 */
025public class MovieListPresenterTest extends MovieTestCase {
026
027  @Captor
028  private ArgumentCaptor<AsyncCallback<ArrayList<Movie>>> allMoviesCaptor;
029
030  @Before
031  public void initializeMockito() {
032      MockitoAnnotations.initMocks(this);
033  }
034
035  /**
036   * Tests that all movie titles are displayed when the presenter is created
037   */
038  @Test
039  public void testAllMovieTitlesDisplayed() {
040    MovieListPresenter.Display view = mock(MovieListPresenter.Display.class);
041    MovieServiceAsync service = mock(MovieServiceAsync.class);
042    HandlerManager eventBus = mock(HandlerManager.class);
043
044    new MovieListPresenter(view, service, eventBus);
045    verify(service).getAllMovies(allMoviesCaptor.capture());
046
047    List<String> titles = Lists.newArrayList("One", "Two", "Three", "Four");
048
049    List<Movie> movies = Lists.transform(titles, new Function<String, Movie>() {
050      @Override
051      public Movie apply(String title) {
052        return newMovie(title, 2010);
053
054      }
055    });
056    allMoviesCaptor.getValue().onSuccess(Lists.newArrayList(movies));
057
058    verify(view).setTitles(titles);
059  }
060
061  /**
062   * Tests that a {@link DisplayMovieEvent} is fired when a movie is selected
063   */
064  @Test
065  public void testDisplayMovieEventFiredOnSelection() {
066    Display view = mock(Display.class);
067    MovieServiceAsync service = mock(MovieServiceAsync.class);
068    HandlerManager eventBus = new HandlerManager(null);
069
070    DisplayMovieEvent.Handler handler = mock(DisplayMovieEvent.Handler.class);
071    eventBus.addHandler(DisplayMovieEvent.TYPE, handler);
072
073    Movie movie0 = newMovie("Zero", 2010);
074    Movie movie1 = newMovie("One", 2010);
075    newPresenter(view, service, eventBus, movie0, movie1);
076
077    ArgumentCaptor<MovieSelectedHandler> captor = ArgumentCaptor.forClass(MovieSelectedHandler.class);
078    verify(view).addMovieSelectedHandler(captor.capture());
079
080    captor.getValue().onMovieSelected(0);
081    verify(handler).onMovieDisplayed(movie0);
082
083    captor.getValue().onMovieSelected(1);
084    verify(handler).onMovieDisplayed(movie1);
085  }
086
087  /**
088   * Creates a new {@link MovieListPresenter} that displays the given movies
089   * @param view The presenter's view
090   * @param service The remote service
091   * @param eventBus The event bus
092   * @param movies The movies to show
093   * @return The presenter
094   */
095  private MovieListPresenter newPresenter(Display view, MovieServiceAsync service, HandlerManager eventBus, Movie... movies) {
096    MovieListPresenter presenter = new MovieListPresenter(view, service, eventBus);
097    verify(service).getAllMovies(allMoviesCaptor.capture());
098    allMoviesCaptor.getValue().onSuccess(Lists.newArrayList(movies));
099    return presenter;
100  }
101}
