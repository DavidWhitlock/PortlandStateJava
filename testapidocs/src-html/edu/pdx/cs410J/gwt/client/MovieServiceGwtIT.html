001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.core.client.GWT;
004import com.google.gwt.user.client.rpc.AsyncCallback;
005import edu.pdx.cs410J.gwt.client.mvp.MovieService;
006import edu.pdx.cs410J.gwt.client.mvp.MovieServiceAsync;
007import edu.pdx.cs410J.rmi.Movie;
008import org.junit.Test;
009
010import java.util.ArrayList;
011
012/**
013 * Tests that the remote movie service is successfully deployed and invokable
014 */
015public class MovieServiceGwtIT extends IntegrationGwtTestCase {
016
017  /**
018   * Tests that we can get all of the movies that we know about
019   */
020  @Test
021  public void testGetAllMovies() {
022    MovieServiceAsync service = GWT.create(MovieService.class);
023    service.getAllMovies(new AsyncCallback<ArrayList<Movie>>() {
024
025      @Override
026      public void onFailure(Throwable throwable) {
027        fail(throwable.toString());
028        finishTest();
029      }
030
031      @Override
032      public void onSuccess(ArrayList<Movie> movies) {
033        assertFalse(movies.isEmpty());
034        finishTest();
035      }
036    });
037
038    delayTestFinish(1000);
039  }
040}
