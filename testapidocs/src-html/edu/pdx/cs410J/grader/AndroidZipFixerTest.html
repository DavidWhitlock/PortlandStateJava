297  @Test
298  public void contentsOfZipEntriesAreNotModified() throws IOException {
299    String entryName = "build.gradle";
300    String entryContent = "This is a test entry";
301
302    ByteArrayOutputStream zippedBytes = new ByteArrayOutputStream();
303    ZipOutputStream zos = new ZipOutputStream(zippedBytes);
304    zos.setMethod(ZipOutputStream.DEFLATED);
305
306    ZipEntry entry = new ZipEntry(entryName);
307    zos.putNextEntry(entry);
308    zos.write(entryContent.getBytes());
309    zos.closeEntry();
310    zos.close();
311
312    AndroidZipFixer fixer = new AndroidZipFixer(new GradeBook("test"));
313    ByteArrayOutputStream fixedBytes = new ByteArrayOutputStream();
314    fixer.fixZipFile(new ByteArrayInputStream(zippedBytes.toByteArray()), fixedBytes, new HashMap<>());
315
316    ZipInputStream zis = new ZipInputStream(new ByteArrayInputStream(fixedBytes.toByteArray()));
317
318    ZipEntry fixedEntry = zis.getNextEntry();
319    while (fixedEntry != null) {
320      if (fixedEntry.getName().equals(entryName)) {
321        break;
322      }
323      fixedEntry = zis.getNextEntry();
324    }
325
326    assertThat(fixedEntry, notNullValue());
327
328    ByteArrayOutputStream fixedEntryBytes = new ByteArrayOutputStream();
329    ByteStreams.copy(zis, fixedEntryBytes);
330    String fixedEntryContent = new String(fixedEntryBytes.toByteArray());
331
332    assertThat(fixedEntryContent, equalTo(entryContent));
333
334
335  }
