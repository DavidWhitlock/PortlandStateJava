029  @Before
030  public void createTempDirectoryForTest() {

039  @After
040  public void deleteTempDirectoryForTest() throws IOException {
041    Files.walkFileTree(tempDirectory.toPath(), new SimpleFileVisitor<Path>() {
042      @Override
043      public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
044        return deleteFile(file);
045      }
046
047      @Override
048      public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
049        return deleteFile(dir);
050      }
051
052      private FileVisitResult deleteFile(Path file) throws IOException {
053        Files.delete(file);
054        return FileVisitResult.CONTINUE;
055      }
056    });
057
058    assertThat(tempDirectory.exists(), equalTo(false));
059  }

076  @Test
077  public void reportsAreGeneratedInCorrectDirectories() throws IOException {
078    GradeBook gradeBook = new GradeBook("test");
079    Assignment assignment = new Assignment("assignment", 4.0);
080    gradeBook.addAssignment(assignment);
081
082    double undergradScore = 3.0;
083    Student undergrad = addStudentInSectionWithScore(gradeBook, assignment, "undergrad", UNDERGRADUATE, undergradScore);
084
085    double gradScore = 3.5;
086    Student grad = addStudentInSectionWithScore(gradeBook, assignment, "grad", GRADUATE, gradScore);
087
088    SummaryReport.dumpReports(gradeBook.getStudentIds(), gradeBook, tempDirectory, false);
089
090    assertThatStudentHasReportInDirectoryWithScore(undergrad, UNDERGRADUATE_DIRECTORY_NAME, undergradScore);
091    assertThatStudentHasReportInDirectoryWithScore(grad, GRADUATE_DIRECTORY_NAME, gradScore);
092  }
093
094  private Student addStudentInSectionWithScore(GradeBook gradeBook, Assignment assignment, String studentId, Student.Section enrolledSection, double undergradScore) {
095    Student undergrad = new Student(studentId);
096    undergrad.setEnrolledSection(enrolledSection);
097    gradeBook.addStudent(undergrad);
098    undergrad.setGrade(assignment, undergradScore);
099    return undergrad;
100  }
101
102  private void assertThatStudentHasReportInDirectoryWithScore(Student student, String dirFileName, double studentScore) throws IOException {
103    File directory = new File(tempDirectory, dirFileName);
104    assertThat(directory.exists(), equalTo(true));
105
106    File reportFile = new File(directory, SummaryReport.getReportFileName(student.getId()));
107    assertThat(reportFile.exists(), equalTo(true));
108
109    String reportContents = readTextFromFile(reportFile);
110    assertThat(reportContents, containsString(String.valueOf(studentScore)));
111  }
112
113  private String readTextFromFile(File file) throws IOException {
114    StringWriter writer = new StringWriter();
115    CharStreams.copy(new FileReader(file), writer);
116    return writer.getBuffer().toString();
117  }

129    addStudentInSectionWithScore(gradeBook, assignment, betterUndergradStudentName, UNDERGRADUATE, 3.5);

147  private void calculateTotalGradesForStudents(GradeBook gradeBook, boolean assignLetterGrades) {
148    PrintWriter pw = new PrintWriter(new Writer() {
149      @Override
150      public void write(char[] cbuf, int off, int len) throws IOException {
151
152      }
153
154      @Override
155      public void flush() throws IOException {
156
157      }
158
159      @Override
160      public void close() throws IOException {
161
162      }
163    });
164
165    gradeBook.studentsStream().forEach(student -> {
166      SummaryReport.dumpReportTo(gradeBook, student, pw, assignLetterGrades);
167    });
168  }

170  private class CapturingPrintWriter extends PrintWriter {
171    public CapturingPrintWriter() {
172      super(new StringWriter());
173    }
174
175    public String getTextWrittenToWriter() {
176      return ((StringWriter) this.out).getBuffer().toString();
177    }
178  }

180  @Test
181  public void assignLetterGradesAccordingToGradeBookRangesForSection() {
182    GradeBook gradeBook = new GradeBook("test");
183    Assignment assignment = new Assignment("assignment", 10.0);
184    gradeBook.addAssignment(assignment);
185
186    GradeBook.LetterGradeRanges undergradLetterGradeRanges = gradeBook.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
187    undergradLetterGradeRanges.getRange(LetterGrade.A).setRange(95, 100);
188    undergradLetterGradeRanges.getRange(LetterGrade.A_MINUS).setRange(90, 94);
189    undergradLetterGradeRanges.validate();
190
191    GradeBook.LetterGradeRanges gradLetterGradeRanges = gradeBook.getLetterGradeRanges(Student.Section.GRADUATE);
192    gradLetterGradeRanges.getRange(LetterGrade.A).setRange(92, 100);
193    gradLetterGradeRanges.getRange(LetterGrade.A_MINUS).setRange(90, 91);
194    gradLetterGradeRanges.validate();
195
196    String betterUndergradStudentName = "betterUndergrad";
197    Student betterUndergrad = addStudentInSectionWithScore(gradeBook, assignment, betterUndergradStudentName, UNDERGRADUATE, 9.7);
198    assertThat(betterUndergrad.getLetterGrade(), nullValue());
199
200    String worseUndergradStudentName = "worseUndergrad";
201    Student worseUndergrad = addStudentInSectionWithScore(gradeBook, assignment, worseUndergradStudentName, UNDERGRADUATE, 9.2);
202    assertThat(worseUndergrad.getLetterGrade(), nullValue());
203
204    String gradStudentName = "grad";
205    Student grad = addStudentInSectionWithScore(gradeBook, assignment, gradStudentName, GRADUATE, 9.2);
206    assertThat(grad.getLetterGrade(), nullValue());
207
208    calculateTotalGradesForStudents(gradeBook, true);
209
210    assertThat(betterUndergrad.getLetterGrade(), equalTo(LetterGrade.A));
211    assertThat(worseUndergrad.getLetterGrade(), equalTo(LetterGrade.A_MINUS));
212    assertThat(grad.getLetterGrade(), equalTo(LetterGrade.A));
213  }
