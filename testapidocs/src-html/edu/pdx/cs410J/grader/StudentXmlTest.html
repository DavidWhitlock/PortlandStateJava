001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004import org.junit.Test;
005import org.w3c.dom.Document;
006
007import javax.xml.transform.TransformerException;
008import java.io.ByteArrayInputStream;
009import java.io.InputStreamReader;
010import java.time.LocalDateTime;
011
012import static org.hamcrest.MatcherAssert.assertThat;
013import static org.hamcrest.Matchers.*;
014
015public class StudentXmlTest {
016
017  @Test
018  public void letterGradeIsPersistedInXml() throws TransformerException, ParserException {
019    Student student = new Student("test");
020    LetterGrade letterGrade = LetterGrade.A;
021    student.setLetterGrade(letterGrade);
022
023    Student student2 = writeAndReadStudentAsXml(student);
024
025    assertThat(student2.getLetterGrade(), equalTo(letterGrade));
026  }
027
028  private Student writeAndReadStudentAsXml(Student student) throws TransformerException, ParserException {
029    Document doc = XmlDumper.toXml(student);
030    byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
031
032    XmlStudentParser parser = new XmlStudentParser(new InputStreamReader(new ByteArrayInputStream(bytes)));
033    return parser.parseStudent();
034  }
035
036  @Test
037  public void oneSubmissionIsPersistedInXml() throws TransformerException, ParserException {
038    Student student = new Student("test");
039    String assignmentName = "project";
040    LocalDateTime submissionTime = LocalDateTime.now().minusHours(3).withNano(0);
041
042    Grade grade = new Grade(assignmentName, Grade.NO_GRADE);
043    grade.addSubmissionTime(submissionTime);
044    student.setGrade(assignmentName, grade);
045
046    Student student2 = writeAndReadStudentAsXml(student);
047
048    assertThat(student2.getGrade(assignmentName).getSubmissionTimes(), contains(submissionTime));
049  }
050
051  @Test
052  public void multipleSubmissionsArePersistedInXml() throws TransformerException, ParserException {
053    Student student = new Student("test");
054    String assignmentName = "project";
055    LocalDateTime submissionTime1 = LocalDateTime.now().minusHours(3).withNano(0);
056    LocalDateTime submissionTime2 = LocalDateTime.now().minusHours(4).withNano(0);
057
058    Grade grade = new Grade(assignmentName, Grade.NO_GRADE);
059    grade.addSubmissionTime(submissionTime1);
060    grade.addSubmissionTime(submissionTime2);
061    student.setGrade(assignmentName, grade);
062
063    Student student2 = writeAndReadStudentAsXml(student);
064
065    assertThat(student2.getGrade(assignmentName).getSubmissionTimes(), contains(submissionTime1, submissionTime2));
066  }
067
068  @Test
069  public void enrolledUndergraduateSectionIsPersistedInXml() throws TransformerException, ParserException {
070    Student student = new Student("test");
071    Student.Section section = Student.Section.UNDERGRADUATE;
072    student.setEnrolledSection(section);
073
074    Student student2 = writeAndReadStudentAsXml(student);
075
076    assertThat(student2.getEnrolledSection(), equalTo(section));
077  }
078
079  @Test
080  public void enrolledGraduateSectionIsPersistedInXml() throws TransformerException, ParserException {
081    Student student = new Student("test");
082    Student.Section section = Student.Section.GRADUATE;
083    student.setEnrolledSection(section);
084
085    Student student2 = writeAndReadStudentAsXml(student);
086
087    assertThat(student2.getEnrolledSection(), equalTo(section));
088  }
089
090  @Test
091  public void xmlHasNewlinesButNotIndentation() throws TransformerException {
092    Student student = new Student("test");
093    Document doc = XmlDumper.toXml(student);
094    byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
095    String string = new String(bytes);
096    assertThat(string, containsString("\n"));
097    assertThat(string.chars().filter(c -> '\n' == c).count(), greaterThan(4L));
098    assertThat(string, not(containsString(" <")));
099  }
100
101}
