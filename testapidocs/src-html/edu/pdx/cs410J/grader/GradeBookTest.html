001package edu.pdx.cs410J.grader;

003import org.junit.Test;
004
005import static edu.pdx.cs410J.grader.GradeBook.LetterGradeRanges.LetterGradeRange;
006import static edu.pdx.cs410J.grader.GradeBook.LetterGradeRanges.LetterGradeRange.InvalidLetterGradeRange;
007import static edu.pdx.cs410J.grader.Student.Section.GRADUATE;
008import static edu.pdx.cs410J.grader.Student.Section.UNDERGRADUATE;
009import static org.hamcrest.MatcherAssert.assertThat;
010import static org.hamcrest.Matchers.*;
011
012public class GradeBookTest {
013
014  @Test
015  public void defaultLetterGradeRanges() {
016    GradeBook book = new GradeBook("test");
017    GradeBook.LetterGradeRanges undergrad = book.getLetterGradeRanges(UNDERGRADUATE);
018    assertDefaultLetterGradeRanges(undergrad);
019
020    GradeBook.LetterGradeRanges grad = book.getLetterGradeRanges(GRADUATE);
021    assertDefaultLetterGradeRanges(grad);
022  }
023
024  private void assertDefaultLetterGradeRanges(GradeBook.LetterGradeRanges ranges) {
025    assertLetterGradeRange(ranges, LetterGrade.A, 94, 100);
026    assertLetterGradeRange(ranges, LetterGrade.A_MINUS, 90, 93);
027    assertLetterGradeRange(ranges, LetterGrade.B_PLUS, 87, 89);
028    assertLetterGradeRange(ranges, LetterGrade.B, 83, 86);
029    assertLetterGradeRange(ranges, LetterGrade.B_MINUS, 80, 82);
030    assertLetterGradeRange(ranges, LetterGrade.C_PLUS, 77, 79);
031    assertLetterGradeRange(ranges, LetterGrade.C, 73, 76);
032    assertLetterGradeRange(ranges, LetterGrade.C_MINUS, 70, 72);
033    assertLetterGradeRange(ranges, LetterGrade.D_PLUS, 67, 69);
034    assertLetterGradeRange(ranges, LetterGrade.D, 63, 66);
035    assertLetterGradeRange(ranges, LetterGrade.D_MINUS, 60, 62);
036    assertLetterGradeRange(ranges, LetterGrade.F, 0, 59);
037  }
038
039  private void assertLetterGradeRange(GradeBook.LetterGradeRanges ranges, LetterGrade letterGrade, int minimum, int maximum) {
040    LetterGradeRange range = ranges.getRange(letterGrade);
041    assertThat("No range for " + letterGrade, range, not(nullValue()));
042    assertThat("Less than for " + letterGrade, range.maximum(), equalTo(maximum));
043    assertThat(range.minimum(), equalTo(minimum));
044  }

105  @Test
106  public void scoreIsInRange() {
107    GradeBook book = new GradeBook("test");
108    GradeBook.LetterGradeRanges ranges = book.getLetterGradeRanges(UNDERGRADUATE);
109    LetterGradeRange range = ranges.getRange(LetterGrade.A);
110    range.setRange(90, 100);
111
112    assertThat(range.isScoreInRange(90.0), is(true));
113    assertThat(range.isScoreInRange(100.0), is(true));
114    assertThat(range.isScoreInRange(95.0), is(true));
115    assertThat(range.isScoreInRange(89.5), is(true));
116    assertThat(range.isScoreInRange(101.0), is(false));
117    assertThat(range.isScoreInRange(89.0), is(false));
118    assertThat(range.isScoreInRange(89.4), is(false));
119
120  }
121
122  @Test
123  public void scoresAreCorrectlyMappedToLetterGrades() {
124    GradeBook book = new GradeBook("test");
125    assertThat(book.getLetterGradeForScore(UNDERGRADUATE, 99.0), equalTo(LetterGrade.A));
126    assertThat(book.getLetterGradeForScore(UNDERGRADUATE, 93.0), equalTo(LetterGrade.A_MINUS));
127  }
