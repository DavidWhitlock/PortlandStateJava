001package edu.pdx.cs410J.grader;
002
003import org.junit.Test;
004
005import java.io.File;
006import java.time.LocalDateTime;
007
008import static org.hamcrest.MatcherAssert.assertThat;
009import static org.hamcrest.core.IsEqual.equalTo;
010
011public class SubmitTest {
012
013  @Test
014  public void nineDigitStringIsInvalidLoginId() {
015    Submit submit = new Submit();
016    assertThat(submit.isNineDigitStudentId("123456789"), equalTo(true));
017  }
018
019  @Test
020  public void eightDigitStringIsValidLoginId() {
021    Submit submit = new Submit();
022    assertThat(submit.isNineDigitStudentId("12345678"), equalTo(false));
023  }
024
025  @Test
026  public void userNameIsValidLoginId() {
027    Submit submit = new Submit();
028    assertThat(submit.isNineDigitStudentId("whitlock"), equalTo(false));
029  }
030
031  @Test
032  public void emailAddressIsInvalidLoginId() {
033    Submit submit = new Submit();
034    assertThat(submit.looksLikeAnEmailAddress("me@email.com"), equalTo(true));
035  }
036
037  @Test
038  public void canSubmitClassesInStudentPackage() {
039    String userId = "student";
040
041    File file = makeFileWithPath("edu", "pdx", "cs410J", userId, "ProjectClass.java");
042    Submit submit = new Submit();
043    submit.setUserId(userId);
044    assertThat(submit.isInEduPdxCs410JDirectory(file), equalTo(true));
045  }
046
047  private File makeFileWithPath(String... path) {
048    File dir = new File(System.getProperty("user.dir"));
049
050    for (String part : path) {
051      dir = new File(dir, part);
052    }
053
054    return dir;
055  }
056
057  @Test
058  public void canSubmitClassesInSubPackage() {
059    String userId = "student";
060
061    File file = makeFileWithPath("edu", "pdx", "cs410J", userId, "subpackage", "ProjectClass.java");
062    Submit submit = new Submit();
063    submit.setUserId(userId);
064    assertThat(submit.isInEduPdxCs410JDirectory(file), equalTo(true));
065  }
066
067  @Test
068  public void submissionTimeIsInIsoDateTime() {
069    LocalDateTime time = LocalDateTime.of(2018, 7, 8, 5, 59, 32);
070    String formattedTime = Submit.ManifestAttributes.formatSubmissionTime(time);
071    assertThat(formattedTime, equalTo("2018-07-08T05:59:32"));
072  }
073
074  @Test
075  public void legacySubmissionTimeCanBeParsed() {
076    String legacySubmissionTime = "2018-Jul-08 05:59:32";
077    LocalDateTime expected = LocalDateTime.of(2018, 7, 8, 5, 59, 32);
078
079    LocalDateTime actual = Submit.ManifestAttributes.parseSubmissionTime(legacySubmissionTime);
080    assertThat(actual, equalTo(expected));
081  }
082}
