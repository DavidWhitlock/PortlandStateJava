001package edu.pdx.cs410J.grader.poa;
002
003import com.google.common.eventbus.Subscribe;
004import org.junit.Test;
005import org.mockito.ArgumentCaptor;
006
007import static edu.pdx.cs410J.grader.poa.EmailCredentialsView.EmailAddressValueListener;
008import static edu.pdx.cs410J.grader.poa.EmailCredentialsView.PasswordValueListener;
009import static edu.pdx.cs410J.grader.poa.EmailCredentialsView.SubmitCredentialsListener;
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.hamcrest.Matchers.equalTo;
012import static org.mockito.Mockito.mock;
013import static org.mockito.Mockito.verify;
014
015public class EmailCredentialsPresenterTest extends EventBusTestCase {
016
017  private EmailCredentialsView view;
018  private EmailCredentialsPresenter presenter;
019
020  @Override
021  public void setUp() {
022    super.setUp();
023
024    this.view = mock(EmailCredentialsView.class);
025
026    presenter = new EmailCredentialsPresenter(bus, view);
027  }
028
029  @Test
030  public void downloadingSubmissionsDisplaysEmailCredentialsView() {
031    this.bus.post(new DownloadPOASubmissionsRequest());
032
033    verify(this.view).setIsVisible(true);
034  }
035
036  @Test
037  public void emailAddressInViewIsSavedToPresenter() {
038    ArgumentCaptor<EmailAddressValueListener> emailAddress = ArgumentCaptor.forClass(EmailAddressValueListener.class);
039    verify(this.view).addEmailAddressValueListener(emailAddress.capture());
040
041    String address = "me@email.com";
042    emailAddress.getValue().setEmailAddress(address);
043
044    assertThat(presenter.getEmailAddress(), equalTo(address));
045  }
046
047  @Test
048  public void passwordInViewIsSavedToPresenter() {
049    ArgumentCaptor<PasswordValueListener> passwordCaptor = ArgumentCaptor.forClass(PasswordValueListener.class);
050    verify(this.view).addPasswordValueListener(passwordCaptor.capture());
051
052    String password = "password";
053    passwordCaptor.getValue().setPassword(password);
054
055    assertThat(presenter.getPassword(), equalTo(password));
056  }
057
058  @Test
059  public void enteringCredentialsInViewFiresEmailCredentialsEvent() {
060    ArgumentCaptor<SubmitCredentialsListener> submitListener = ArgumentCaptor.forClass(SubmitCredentialsListener.class);
061    verify(this.view).addSubmitCredentialsListener(submitListener.capture());
062
063    EmailCredentialsListener listener = mock(EmailCredentialsListener.class);
064    this.bus.register(listener);
065
066    String emailAddress = "me@email.com";
067    this.presenter.setEmailAddress(emailAddress);
068    String password = "password";
069    this.presenter.setPassword(password);
070
071    submitListener.getValue().submitCredentials();
072
073    ArgumentCaptor<EmailCredentials> credentials = ArgumentCaptor.forClass(EmailCredentials.class);
074    verify(listener).emailCredentials(credentials.capture());
075
076    assertThat(credentials.getValue().getEmailAddress(), equalTo(emailAddress));
077    assertThat(credentials.getValue().getPassword(), equalTo(password));
078  }
079
080  public interface EmailCredentialsListener {
081    @Subscribe
082    void emailCredentials(EmailCredentials event);
083  }
084
085}
