001package edu.pdx.cs410J.grader.poa;
002
003import com.google.common.eventbus.Subscribe;
004import edu.pdx.cs410J.grader.Assignment;
005import edu.pdx.cs410J.grader.GradeBook;
006import org.junit.Test;
007import org.mockito.ArgumentCaptor;
008
009import java.time.LocalDateTime;
010import java.util.Arrays;
011
012import static edu.pdx.cs410J.grader.poa.POAAssignmentsView.AssignmentSelectedHandler;
013import static org.hamcrest.MatcherAssert.assertThat;
014import static org.hamcrest.Matchers.*;
015import static org.mockito.Matchers.any;
016import static org.mockito.Matchers.anyInt;
017import static org.mockito.Mockito.*;
018
019public class POAAssignmentsPresenterTest extends POASubmissionTestCase {
020
021  private POAAssignmentsView view;
022  private GradeBook book;
023  private Assignment assignment0;
024  private Assignment assignment1;
025  private Assignment assignment2;
026
027  @Override
028  public void setUp() {
029    super.setUp();
030
031    this.view = mock(POAAssignmentsView.class);
032
033    new POAAssignmentsPresenter(this.bus, this.view);
034
035    book = new GradeBook("Test Grade Book");
036    assignment0 = new Assignment("POA 0", 1.0).setType(Assignment.AssignmentType.POA);
037    book.addAssignment(assignment0);
038    assignment1 = new Assignment("POA 1", 1.0).setType(Assignment.AssignmentType.POA);
039    book.addAssignment(assignment1);
040    assignment2 = new Assignment("POA 2", 1.0).setType(Assignment.AssignmentType.POA);
041    book.addAssignment(assignment2);
042    book.addAssignment(new Assignment("Quiz 0", 3.0).setType(Assignment.AssignmentType.QUIZ));
043  }
044
045  @Test
046  public void assignmentListPopulatedInViewWhenGradeBookLoaded() {
047    this.bus.post(new GradeBookLoaded(book));
048
049    verify(this.view).setAssignments(Arrays.asList("POA 0", "POA 1", "POA 2"));
050    verify(this.view).setSelectedAssignment(0);
051  }
052
053  @Test
054  public void assignmentSelectedFireWhenAssignmentSelectedInView() {
055    ArgumentCaptor<AssignmentSelectedHandler> viewHandler = ArgumentCaptor.forClass(AssignmentSelectedHandler.class);
056    verify(this.view).addAssignmentSelectedHandler(viewHandler.capture());
057
058    AssignmentSelectedEventHandler eventHandler = mock(AssignmentSelectedEventHandler.class);
059    this.bus.register(eventHandler);
060
061    this.bus.post(new GradeBookLoaded(book));
062
063    viewHandler.getValue().assignmentSelected(1);
064
065    ArgumentCaptor<AssignmentSelectedEvent> event = ArgumentCaptor.forClass(AssignmentSelectedEvent.class);
066    verify(eventHandler).handle(event.capture());
067
068    assertThat(event.getValue().getAssignment(), equalTo(assignment1));
069  }
070
071  private interface AssignmentSelectedEventHandler {
072    @Subscribe
073    void handle(AssignmentSelectedEvent event);
074  }
075
076  @Test
077  public void initiallySelectAssignmentWhoseDueDateHasMostRecentlyPassed() {
078    assignment0.setDueDate(LocalDateTime.now().minusDays(3));
079    assignment1.setDueDate(LocalDateTime.now().minusDays(2));
080    assignment2.setDueDate(LocalDateTime.now().plusDays(2));
081
082    this.bus.post(new GradeBookLoaded(book));
083
084    verify(this.view).setSelectedAssignment(1);
085  }
086
087
088  @Test
089  public void selectAssignmentWhenPOASubmittedWithAssignmentNumberInSubject() {
090    AssignmentSelectedEventHandler eventHandler = mock(AssignmentSelectedEventHandler.class);
091    this.bus.register(eventHandler);
092
093    this.bus.post(new GradeBookLoaded(book));
094
095    POASubmission submission = createPOASubmission("POA for Project 2", "me", LocalDateTime.now());
096    this.bus.post(new POASubmissionSelected(submission));
097
098    verify(this.view).setSelectedAssignment(2);
099
100    ArgumentCaptor<AssignmentSelectedEvent> event = ArgumentCaptor.forClass(AssignmentSelectedEvent.class);
101    verify(eventHandler).handle(event.capture());
102
103    assertThat(event.getValue().getAssignment(), equalTo(assignment2));
104  }
105
106  @Test
107  public void testFindNumbersInString() {
108    assertThat(POAAssignmentsPresenter.findNumbersInString("POA for Project 1"), hasItem("1"));
109    assertThat(POAAssignmentsPresenter.findNumbersInString("Project 5 POA"), hasItem("5"));
110    assertThat(POAAssignmentsPresenter.findNumbersInString("POA for Project 42"), hasItem("42"));
111    assertThat(POAAssignmentsPresenter.findNumbersInString("3 for Project 42"), hasItems("3", "42"));
112  }
113
114  @Test
115  public void populateDueDateInViewWhenGradeBookLoaded() {
116    assignment0.setDueDate(LocalDateTime.now().minusDays(3));
117    assignment1.setDueDate(LocalDateTime.now().minusDays(2));
118    assignment2.setDueDate(LocalDateTime.now().plusDays(2));
119
120    this.bus.post(new GradeBookLoaded(book));
121
122    verify(this.view).setSelectedAssignmentDueDate(POAAssignmentsPresenter.formatDueDate(assignment1));
123  }
124
125  @Test
126  public void populateDueDateInViewWhenAssignmentSelected() {
127    assignment0.setDueDate(LocalDateTime.now().minusDays(2));
128    assignment1.setDueDate(LocalDateTime.now().minusDays(3));
129
130    ArgumentCaptor<AssignmentSelectedHandler> viewHandler = ArgumentCaptor.forClass(AssignmentSelectedHandler.class);
131    verify(this.view).addAssignmentSelectedHandler(viewHandler.capture());
132
133    AssignmentSelectedEventHandler eventHandler = mock(AssignmentSelectedEventHandler.class);
134    this.bus.register(eventHandler);
135
136    this.bus.post(new GradeBookLoaded(book));
137    verify(this.view).setSelectedAssignmentDueDate(POAAssignmentsPresenter.formatDueDate(assignment0));
138
139    viewHandler.getValue().assignmentSelected(1);
140
141    verify(this.view).setSelectedAssignmentDueDate(POAAssignmentsPresenter.formatDueDate(assignment1));
142  }
143
144  @Test
145  public void clearDueDateInViewWhenAssignmentHasNoDueDate() {
146    assignment0.setDueDate(LocalDateTime.now().minusDays(2));
147    assignment1.setDueDate(null);
148
149    ArgumentCaptor<AssignmentSelectedHandler> viewHandler = ArgumentCaptor.forClass(AssignmentSelectedHandler.class);
150    verify(this.view).addAssignmentSelectedHandler(viewHandler.capture());
151
152    AssignmentSelectedEventHandler eventHandler = mock(AssignmentSelectedEventHandler.class);
153    this.bus.register(eventHandler);
154
155    this.bus.post(new GradeBookLoaded(book));
156    verify(this.view).setSelectedAssignmentDueDate(POAAssignmentsPresenter.formatDueDate(assignment0));
157
158    viewHandler.getValue().assignmentSelected(1);
159
160    verify(this.view).setSelectedAssignmentDueDate("");
161  }
162
163  @Test
164  public void noAssignmentSelectedWhenGradeBookHasNoPOAAssignments() {
165    this.bus.post(new GradeBookLoaded(new GradeBook("No POA assignments")));
166
167    verify(this.view, never()).setAssignments(anyList());
168    verify(this.view, never()).setSelectedAssignment(anyInt());
169    verify(this.view, never()).setSelectedAssignmentDueDate(any(String.class));
170  }
