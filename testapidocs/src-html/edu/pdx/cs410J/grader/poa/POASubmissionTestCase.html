001package edu.pdx.cs410J.grader.poa;
002
003import java.time.LocalDateTime;
004
005public abstract class POASubmissionTestCase extends EventBusTestCase {
006
007  protected POASubmission createPOASubmission(String subject, String submitter, LocalDateTime submitTime) {
008    return createPOASubmission(subject, submitter, submitTime, "Content");
009  }
010
011  protected POASubmission createPOASubmission(String subject, String submitter, LocalDateTime submitTime, String content) {
012    POASubmission.Builder builder = POASubmission.builder();
013    builder.setSubject(subject);
014    builder.setSubmitter(submitter);
015    builder.setSubmitTime(submitTime);
016    builder.setContent(content);
017    return builder.create();
018  }
019}
