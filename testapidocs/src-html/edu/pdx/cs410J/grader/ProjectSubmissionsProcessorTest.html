021   * Given: A submission from a student whose name is in the grade book,
022   * but whose id is not

024   * When: The submission is recorded

026   * Then: The submission is noted under the expected id and a new Student

031  @Test
032  public void matchStudentBasedOnFirstAndLastName() throws StudentEmailAttachmentProcessor.SubmissionException {
033    String projectName = "Project";
034
035    GradeBook gradebook = createGradeBookWithAssignment(projectName);
036    Student student = createStudentInGradeBook(gradebook);
037
038    String studentName = student.getFirstName() + " " + student.getLastName();
039    String wrongStudentId = "Not the student id we expect";
040    String wrongEmail = "Not the email that we expect";
041    String submissionComment = "This is only a test";
042
043    Manifest manifest = createManifest(projectName, studentName, wrongStudentId, wrongEmail, submissionComment);
044
045    noteProjectSubmissionInGradeBook(gradebook, manifest);
046
047    assertThat(gradebook.getStudent(wrongStudentId), isNotPresent());
048
049    assertThatProjectSubmissionWasRecordedForStudent(projectName, student);
050  }

165  private Manifest createManifest(String projectName, Student student, LocalDateTime submissionDate, String submissionComment) {
166    return createManifest(projectName, student.getFullName(), student.getId(), student.getEmail(), submissionComment,
167      Submit.ManifestAttributes.formatSubmissionTime(submissionDate));
168  }
169
170  private void noteProjectSubmissionInGradeBook(GradeBook gradebook, Manifest manifest) throws StudentEmailAttachmentProcessor.SubmissionException {
171    ProjectSubmissionsProcessor processor =
172      new ProjectSubmissionsProcessor(new File(System.getProperty("user.dir")), gradebook);
173    processor.noteSubmissionInGradeBook(manifest);
174  }
175
176  private void assertThatProjectSubmissionWasRecordedForStudent(String projectName, Student student) {
177    Grade grade = student.getGrade(projectName);
178    assertThat(grade, not(nullValue()));
179    assertThat(grade.isNotGraded(), equalTo(true));
180  }
181
182  private Manifest createManifest(String projectName, String studentName, String wrongStudentId, String wrongEmail, String submissionComment) {
183    return createManifest(projectName, studentName, wrongStudentId, wrongEmail, submissionComment, Submit.ManifestAttributes.formatSubmissionTime(LocalDateTime.now()));
184  }
185
186  private Manifest createManifest(String projectName, String studentName, String wrongStudentId, String wrongEmail, String submissionComment, String submissionTime) {
187    Manifest manifest = new Manifest();
188    Attributes attributes = manifest.getMainAttributes();
189    attributes.put(USER_ID, wrongStudentId);
190    attributes.put(USER_EMAIL, wrongEmail);
191    attributes.put(USER_NAME, studentName);
192    attributes.put(PROJECT_NAME, projectName);
193    attributes.put(SUBMISSION_TIME, submissionTime);
194    attributes.put(SUBMISSION_COMMENT, submissionComment);
195    return manifest;
196  }
197
198  private Student createStudentInGradeBook(GradeBook gradebook) {
199    Student student = new Student("studentId");
200    student.setFirstName("firstName");
201    student.setLastName("lastName");
202    student.setNickName("nickName");
203    student.setEmail("test@test.com");
204    gradebook.addStudent(student);
205    return student;
206  }
207
208  private GradeBook createGradeBookWithAssignment(String projectName) {
209    return createGradeBookWithAssignment(projectName, null);
210  }
211
212  private GradeBook createGradeBookWithAssignment(String projectName, LocalDateTime dueDate) {
213    GradeBook gradebook = new GradeBook("test");
214    Assignment assignment = new Assignment(projectName, 10.0);
215    assignment.setDueDate(dueDate);
216    gradebook.addAssignment(assignment);
217    return gradebook;
218  }

222  private static Matcher<? super Optional<Student>> isNotPresent() {
223    return new TypeSafeMatcher<Optional<Student>>() {
224      @Override
225      protected boolean matchesSafely(Optional<Student> item) {
226        return !item.isPresent();
227      }
228
229      @Override
230      public void describeTo(Description description) {
231        description.appendText("an Optional<Student> that is not present");
232      }
233    };
234  }
