114  @Test
115  public void scoreIsRecordedInGradeBook() throws ProjectGradesImporter.ScoreNotFoundException {
116    String studentId = "student";
117    Assignment assignment = new Assignment("project", 6.0);
118
119    GradeBook gradeBook = new GradeBook("test");
120    gradeBook.addStudent(new Student(studentId));
121    gradeBook.addAssignment(assignment);
122
123    String score = "5.8";
124    GradedProject project = new GradedProject();
125    project.addLine(score + " out of 6.0");
126    project.addLine("");
127    project.addLine("asdfasd");
128
129    Logger logger = mock(Logger.class);
130    ProjectGradesImporter importer = new ProjectGradesImporter(gradeBook, assignment, logger);
131    importer.recordScoreFromProjectReport(studentId, project.getReader());
132
133    assertThat(gradeBook.getStudent(studentId).get().getGrade(assignment.getName()).getScore(), equalTo(5.8));
134    assertThat(gradeBook.isDirty(), equalTo(true));
135
136    ArgumentCaptor<String> message = ArgumentCaptor.forClass(String.class);
137    verify(logger).info(message.capture());
138    assertThat(message.getValue(), containsString("Recorded grade of " + score + " for " + studentId));
139  }
