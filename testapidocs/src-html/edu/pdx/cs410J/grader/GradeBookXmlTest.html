001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004import org.junit.Test;
005import org.w3c.dom.Document;
006
007import javax.xml.transform.TransformerException;
008import java.io.ByteArrayInputStream;
009import java.io.IOException;
010import java.time.LocalDateTime;
011
012import static edu.pdx.cs410J.grader.GradeBook.LetterGradeRanges;
013import static edu.pdx.cs410J.grader.GradeBook.LetterGradeRanges.LetterGradeRange;
014import static org.hamcrest.MatcherAssert.assertThat;
015import static org.hamcrest.Matchers.equalTo;
016
017public class GradeBookXmlTest {
018
019  @Test
020  public void letterGradeRangesArePersistedToXml() throws IOException, TransformerException, ParserException {
021    GradeBook book = new GradeBook("test");
022
023    LetterGradeRanges undergradRanges = book.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
024    morphLetterGradeRanges(undergradRanges, 1);
025
026    LetterGradeRanges gradRanges = book.getLetterGradeRanges(Student.Section.GRADUATE);
027    morphLetterGradeRanges(gradRanges, 2);
028
029    GradeBook book2 = writeAndReadGradeBookAsXml(book);
030
031    LetterGradeRanges undergradRanges2 = book2.getLetterGradeRanges(Student.Section.UNDERGRADUATE);
032    assertRangesAreEqual(undergradRanges, undergradRanges2);
033
034    LetterGradeRanges gradRanges2 = book2.getLetterGradeRanges(Student.Section.GRADUATE);
035    assertRangesAreEqual(gradRanges, gradRanges2);
036  }
037
038  private void assertRangesAreEqual(LetterGradeRanges ranges, LetterGradeRanges ranges2) {
039    for (LetterGrade letterGrade : LetterGrade.values()) {
040      LetterGradeRange range = ranges.getRange(letterGrade);
041      if (range == null) {
042        continue;
043      }
044      LetterGradeRange range2 = ranges2.getRange(letterGrade);
045      assertThat("Range for " + letterGrade, range2.minimum(), equalTo(range.minimum()));
046    }
047  }
048
049  private void morphLetterGradeRanges(LetterGradeRanges ranges, int morphValue) {
050    for (LetterGrade letterGrade : LetterGrade.values()) {
051      LetterGradeRange range = ranges.getRange(letterGrade);
052      if (range == null) {
053        continue;
054      }
055      range.setRange(range.minimum() + morphValue, range.maximum() + morphValue);
056    }
057    LetterGradeRange fRange = ranges.getRange(LetterGrade.F);
058    fRange.setRange(0, fRange.maximum());
059  }
060
061  private GradeBook writeAndReadGradeBookAsXml(GradeBook book) throws IOException, TransformerException, ParserException {
062    Document doc = XmlDumper.dumpGradeBook(book, new XmlHelper());
063    byte[] bytes = XmlHelper.getBytesForXmlDocument(doc);
064
065    XmlGradeBookParser parser = new XmlGradeBookParser(new ByteArrayInputStream(bytes));
066    return parser.parse();
067  }
068
069  @Test
070  public void dueDatesArePersistedToXml() throws TransformerException, IOException, ParserException {
071    GradeBook book = new GradeBook("test");
072    String assignmentName = "assignment";
073    Assignment assignment = new Assignment(assignmentName, 10.0);
074    book.addAssignment(assignment);
075
076    LocalDateTime dueDate = LocalDateTime.now().minusDays(3).withNano(0);
077    assignment.setDueDate(dueDate);
078
079    GradeBook book2 = writeAndReadGradeBookAsXml(book);
080
081    assertThat(book2.getAssignment(assignmentName).getDueDate(), equalTo(dueDate));
082  }
083
084  @Test
085  public void poaAssignmentsArePersistedToXml() throws TransformerException, IOException, ParserException {
086    GradeBook book = new GradeBook("test");
087    String assignmentName = "assignment";
088    Assignment assignment = new Assignment(assignmentName, 10.0);
089    assignment.setType(Assignment.AssignmentType.POA);
090    book.addAssignment(assignment);
091
092    GradeBook book2 = writeAndReadGradeBookAsXml(book);
093
094    assertThat(book2.getAssignment(assignmentName).getType(), equalTo(Assignment.AssignmentType.POA));
095  }
096}
