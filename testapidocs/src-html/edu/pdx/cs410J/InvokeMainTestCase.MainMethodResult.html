001package edu.pdx.cs410J;
002
003import java.lang.reflect.Method;
004import java.lang.reflect.InvocationTargetException;
005import java.io.PrintStream;
006import java.io.ByteArrayOutputStream;
007import java.security.Permission;
008
009/**
010 * The superclass of test classes that invoke a main method to test a Java program.
011 *
012 * @author David Whitlock
013 * @since Summer 2010
014 */
015public abstract class InvokeMainTestCase
016{
017    /**
018     * Invokes the <code>main</code> method of the given class with the given arguments and returns an object
019     * that represents the result of invoking that method.
020     * @param mainClass The class whose main method is invoked
021     * @param args The arguments passed to the main method
022     * @return The result of the method invocation
023     */
024    protected MainMethodResult invokeMain( Class mainClass, String... args )
025    {
026        return new MainMethodResult( mainClass, args ).invoke();
027    }
028
029    /**
030     * Invokes the <code>main</code> method of a class and captures information about the invocation such as the data
031     * written to standard out and standard error and the exit code.
032     */
033    protected class MainMethodResult
034    {
035        private final Class mainClass;
036        private final String[] args;
037        private Integer exitCode;
038        private String out;
039        private String err;
040
041        MainMethodResult( Class mainClass, String[] args )
042        {
043            this.mainClass = mainClass;
044            this.args = args;
045        }
046
047        /**
048         * Invokes the main method
049         * @return This <code>MainMethodResult</code>
050         */
051        public MainMethodResult invoke()
052        {
053            Method main;
054            try
055            {
056                main = mainClass.getMethod("main", String[].class);
057            }
058            catch ( NoSuchMethodException e )
059            {
060                throw new IllegalArgumentException( "Class " + mainClass.getName() + " does not have a main method" );
061            }
062
063            try
064            {
065                invokeMain( main );
066            }
067            catch ( IllegalAccessException e )
068            {
069                throw new IllegalArgumentException( "Cannot invoke main method of " + mainClass.getName(), e);
070            }
071            catch ( InvocationTargetException e )
072            {
073                throw new IllegalArgumentException( "Error while invoking main method of " + mainClass.getName(), e);
074            }
075            return this;
076        }
077
078        private void invokeMain( Method main )
079            throws IllegalAccessException, InvocationTargetException
080        {
081            SecurityManager oldSecurityManager = System.getSecurityManager();
082            PrintStream oldOut = System.out;
083            PrintStream oldErr = System.err;
084            try {
085                MainMethodResult.ExitStatusSecurityManager essm = new MainMethodResult.ExitStatusSecurityManager( oldSecurityManager );
086                System.setSecurityManager( essm );
087
088                ByteArrayOutputStream newOut = new ByteArrayOutputStream();
089                ByteArrayOutputStream newErr = new ByteArrayOutputStream();
090                System.setOut( new PrintStream(newOut) );
091                System.setErr( new PrintStream(newErr) );
092
093                try {
094                    main.invoke( null, (Object) this.args );
095
096                } catch ( InvocationTargetException ex ) {
097                    if ( ex.getCause() instanceof ExitException ) {
098                        this.exitCode = ((ExitException) ex.getCause()).getExitCode();
099
100                    } else {
101                        throw ex;
102                    }
103                }
104
105                this.out = newOut.toString();
106                this.err = newErr.toString();
107
108            } finally {
109                System.setSecurityManager( oldSecurityManager );
110                System.setOut( oldOut );
111                System.setErr( oldErr );
112            }
113        }
114
115        /**
116         * Returns the exit code of this program (the argument to {@link System#exit(int)}
117         * @return <code>null</code> if {@link System#exit(int)} was not invoked
118         */
119        public Integer getExitCode()
120        {
121            return this.exitCode;
122        }
123
124        /**
125         * Returns the data written to standard out
126         * @return the data written to standard out
127         */
128        public String getTextWrittenToStandardOut()
129        {
130            return out;
131        }
132
133        /**
134         * Returns the data written to standard err
135         * @return the data written to standard err
136         */
137        public String getTextWrittenToStandardError()
138        {
139            return err;
140        }
141
142        /**
143         * A {@link SecurityManager} that delegates security checks to another {@link SecurityManager}, but captures
144         * the exit code called by {@link System#exit(int)}
145         */
146        private class ExitStatusSecurityManager extends SecurityManager
147        {
148            private final SecurityManager delegate;
149
150            public ExitStatusSecurityManager( SecurityManager manager )
151            {
152                this.delegate = manager;
153            }
154
155            @Override
156            public void checkPermission( Permission perm )
157            {
158                if (this.delegate != null) {
159                    this.delegate.checkPermission( perm );
160                }
161            }
162
163            @Override
164            public void checkPermission( Permission perm, Object context )
165            {
166                if (this.delegate != null) {
167                    this.delegate.checkPermission( perm, context );
168                }
169            }
170
171            @Override
172            public void checkExit( int status )
173            {
174                if (this.delegate != null) {
175                    this.delegate.checkExit( status );
176                }
177
178                throw new ExitException( status );
179            }
180
181        }
182
183        /**
184         * An exception that is thrown when the the main method calls {@link System#exit(int)}.  This lets us capture
185         * the exit code without the VM actually exiting.
186         */
187        private class ExitException extends SecurityException
188        {
189            private final int exitCode;
190
191            public ExitException( int exitCode )
192            {
193                this.exitCode = exitCode;
194            }
195
196            public int getExitCode()
197            {
198                return exitCode;
199            }
200        }
201    }
202}
