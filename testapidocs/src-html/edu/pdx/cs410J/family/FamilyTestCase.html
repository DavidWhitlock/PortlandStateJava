001package edu.pdx.cs410J.family;
002
003import org.junit.Assert;
004import static org.junit.Assert.*;
005
006import java.util.Calendar;
007import java.util.Date;
008import java.util.Iterator;
009
010/**
011 * This is the abstract superclass of classes that test family tree
012 * classes.  It contains a number of helper methods.
013 */
014public abstract class FamilyTestCase {
015
016
017  /**
018   * Asserts that two <code>Person</code>s have the same contents
019   */
020  void assertEquals(Person p1, Person p2) {
021    if (p1 == null) {
022      assertNull(p2);
023      return;
024
025    } else if (p2 == null) {
026      assertNull(p1);
027      return;
028    }
029
030    assertNotNull(p1);
031    assertNotNull(p2);
032
033    Assert.assertEquals(p1.getId(), p2.getId());
034    Assert.assertEquals(p1.getGender(), p2.getGender());
035    Assert.assertEquals(p1.getFirstName(), p2.getFirstName());
036    Assert.assertEquals(p1.getMiddleName(), p2.getMiddleName());
037    Assert.assertEquals(p1.getLastName(), p2.getLastName());
038    assertEquals(p1.getMother(), p2.getMother());
039    assertEquals(p1.getFather(), p2.getFather());
040    assertEquals(p1.getDateOfBirth(), p2.getDateOfBirth());
041    assertEquals(p1.getDateOfDeath(), p2.getDateOfDeath());
042
043    Assert.assertEquals(p1.getMarriages().size(), p2.getMarriages().size());
044
045    Iterator<Marriage> iter1 = p1.getMarriages().iterator();
046    Iterator<Marriage> iter2 = p2.getMarriages().iterator();
047    while (iter1.hasNext() && iter2.hasNext()) {
048      assertEquals( iter1.next(), iter2.next() );
049    }
050
051  }
052
053  /**
054   * Asserts that two <code>Date</code> represent the same day
055   * (ignores times)
056   */
057  void assertEquals(Date d1, Date d2) {
058    if (d1 == null) {
059      assertNull(d2);
060      return;
061
062    } else if (d2 == null) {
063      assertNull(d1);
064      return;
065    }
066
067    Calendar c1 = Calendar.getInstance();
068    c1.setTime(d1);
069    Calendar c2 = Calendar.getInstance();
070    c2.setTime(d2);
071
072    Assert.assertEquals(c1.get(Calendar.DAY_OF_YEAR),
073                 c2.get(Calendar.DAY_OF_YEAR));
074    Assert.assertEquals(c1.get(Calendar.YEAR), c2.get(Calendar.YEAR));
075  }
076
077  /**
078   * Asserts that two <code>Marriage</code>s are the same
079   */
080  void assertEquals(Marriage m1, Marriage m2) {
081    if (m1 == null) {
082      assertNotNull(m2);
083      return;
084
085    } else if (m2 == null) {
086      assertNull(m1);
087      return;
088    }
089
090    // To avoid infinite recursion, we compare people using their ids
091    Assert.assertEquals(m1.getHusband().getId(), m2.getHusband().getId());
092    Assert.assertEquals(m1.getWife().getId(), m2.getWife().getId());
093    assertEquals(m1.getDate(), m2.getDate());
094    Assert.assertEquals(m1.getLocation(), m2.getLocation());
095  }
096
097  /**
098   * Asserts that two <code>FamilyTree</code>s has the same contents
099   */
100  void assertEquals(FamilyTree tree1, FamilyTree tree2) {
101    Assert.assertEquals(tree1.getPeople().size(), tree2.getPeople().size());
102
103      for ( Person person : tree1.getPeople() )
104      {
105          Person p2 = tree2.getPerson( person.getId() );
106          assertNotNull( p2 );
107          assertEquals( person, p2 );
108      }
109  }
110
111  /**
112   * Asserts that one String contains another
113   */
114  void assertContains(String container, String containee) {
115    assertTrue(container.indexOf(containee) != -1);
116  }
117
118  /**
119   * Asserts that one String contains another
120   */
121  void assertContains(String message, String container,
122                      String containee) {
123    assertTrue(message, container.indexOf(containee) != -1);
124  }
125
126}
