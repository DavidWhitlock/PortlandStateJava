3   /**
4    * This class demonstrates obtains a lock before access the
5    * <code>BankAccount</code>.  That way, it always sees a consistent
6    * view of the account. 
7    */
8   public class SynchronizedATM extends ATM {
9     public SynchronizedATM(String name, BankAccount account, 
10                           int[] transactions) {
11      super(name, account, transactions);
12    }
13  
14    /**
15     * Perform each transaction on the account
16     */
17    public void run() {
18      for (int i = 0; i < transactions.length; i++) {
19        // Get the lock on the account
20        synchronized(account) {
21          // Get the balance
22          int balance = account.getBalance();
23          out.println(this.name + " got balance " + balance);
24          
25          // Perform the operation
26          out.println(this.name + " perform " + transactions[i]);
27          balance += transactions[i];
28          
29          // Set the balance
30          out.println(this.name + " set balance to " + balance);
31          account.setBalance(balance);
32        }
33        // Give up the lock
34      }
35    }
36  
37    /**
38     * Create a couple of accounts and have them all perform the same
39     * transactions, but in different orders.
40     */
41    public static void main(String[] args) {
42      BankAccount account = new BankAccount();
43      account.setBalance(1000);
44      out.println("Initial balance: " + account.getBalance());
45  
46      // Make some ATMs
47      int[] trans1 = {-200, 400, 100, -300};
48      ATM atm1 = new SynchronizedATM("ATM1", account, trans1);
49      int[] trans2 = {400, 100, -300, -200};
50      ATM atm2 = new SynchronizedATM("ATM2", account, trans2);
51      int[] trans3 = {-300, -200, 100, 400};
52      ATM atm3 = new SynchronizedATM("ATM3", account, trans3);
53  
54      // Make some threads and start them
55      Thread t1 = new Thread(atm1);
56      t1.start();
57      Thread t2 = new Thread(atm2);
58      t2.start();
59      Thread t3 = new Thread(atm3);
60      t3.start();
61  
62      // Wait for all threads to finish
63      try {
64        t1.join();
65        t2.join();
66        t3.join();
67  
68      } catch (InterruptedException ex) {
69        return;
70      }
71  
72      out.println("Final balance: " + account.getBalance());
73    }
