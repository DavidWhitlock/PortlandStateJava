3   import java.io.*;
4   import java.net.*;
5   
6   /**
7    * This program listens to a socket.  It expects strings from a
8    * <code>Speaker</code>.  The conversation terminates with the word
9    * "bye".
10   */
11  public class Listener {
12    private static PrintStream out = System.out;
13    private static PrintStream err = System.err;
14  
15    /**
16     * Main program that reads the port number to listen to from the
17     * command line.
18     */
19    public static void main(String[] args) {
20      int port = 0;
21  
22      try {
23        port = Integer.parseInt(args[0]);
24      } catch (NumberFormatException ex) {
25        err.println("** Bad port number: " + args[0]);
26        System.exit(1);
27      }
28  
29      try {
30        // Backlog of 5 messages
31        ServerSocket server = new ServerSocket(port, 5);
32  
33        // Wait for the Speaker to connect
34        Socket socket = server.accept();
35  
36        // Read from the Socket
37        InputStreamReader isr = 
38  	new InputStreamReader(socket.getInputStream());
39        BufferedReader listener = new BufferedReader(isr);
40  
41        String line = "";
42        while (!line.equals("bye")) {
43  	line = listener.readLine();
44  	out.println(line);
45        }
46  
47        listener.close();
48  
49      } catch (IOException ex) {
50        err.println("** IOException: " + ex);
51        System.exit(1);
52      }
53    }
54  }
