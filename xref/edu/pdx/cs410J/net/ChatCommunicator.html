3   import java.io.*;
4   import java.util.*;
5   import java.net.*;
6   
7   /**
8    * A <code>ChatCommunicator</code> obtains a <code>Socket</code> and
9    * then creates a <code>ChatSpeaker</code> and a
10   * <code>ChatListener</code> that run in their own threads.
11   */
12  public class ChatCommunicator implements Runnable {
13    private static PrintStream err = System.err;
14  
15    private int port;       // Where the socket is
16    private ChatSpeaker speaker;    // Send messsages
17    private ChatListener listener;  // Receives messages
18  
19    /**
20     * Creates a new <code>ChatCommunicator</code> on a given port.
21     *
22     */
23    public ChatCommunicator(int port) {
24      this.port = port;
25    }
26  
27    /**
28     * Starts up this <code>ChatCommunicator</code>.
29     */
30    public void startup() {
31      this.speaker = new ChatSpeaker();
32      this.listener = new ChatListener();
33      (new Thread(this)).start();
34    }
35  
36    /** 
37     * Make the connection to the socket.  If it cannot open a
38     * <code>Socket</code>, is starts a <code>SocketServer</code> and
39     * waits for a connection.  Then, start the speaker and listener.
40     */
41    public void run() {
42      // Attempt to make a socket
43      Socket socket = null;
44      try {
45        socket = new Socket("localhost", port);
46  
47      } catch (IOException ex) {
48        // Nobody listening
49  //        System.out.println("Nobody's listening");
50      }
51  
52      if (socket == null) {
53        // Listen
54        try {
55  	ServerSocket server = new ServerSocket(port, 10);
56  	socket = server.accept();
57  
58        } catch (IOException ex) {
59  	err.println("** IOException: " + ex);
60  	System.exit(1);
61        }
62      }
63  
64      this.speaker.setSocket(socket);
65      this.listener.setSocket(socket);
66  
67      (new Thread(this.speaker)).start();
68      (new Thread(this.listener)).start();
69    }
70  
71    /**
72     * Delegates to the <code>ChatSpeaker</code>
73     */
74    public void sendMessage(ChatMessage message) {
75      this.speaker.sendMessage(message);
76    }
77  
78    /**
79     * Gets messages from the <code>ChatListener</code>
80     */
81    public List getMessages() {
82      return this.listener.getMessages();
83    }
84  }
