3   import java.io.*;
4   import java.util.*;
5   
6   /**
7    * Demonstrates serialization by writing an instance of
8    * <code>Date</code> to a file.
9    */
10  public class WriteDate {
11  
12    /**
13     * Writes the <code>Date</code> object for the current day and time
14     * to a file whose name is specified on the command line.
15     */
16    public static void main(String[] args) {
17      String fileName = args[0];
18  
19      try {
20        FileOutputStream fos = new FileOutputStream(fileName);
21        ObjectOutputStream out = new ObjectOutputStream(fos);
22        Date date = new Date();
23        System.out.println("Writing " + date);
24        out.writeObject(date);
25        out.flush();
26        out.close();
27  
28      } catch (IOException ex) {
29        System.err.println("**IOException: " + ex);
30        System.exit(1);
31      }
32    }
33  
34  }
