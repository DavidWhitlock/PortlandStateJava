5   import javax.swing.*;
6   import javax.swing.event.TableModelEvent;
7   import javax.swing.event.TableModelListener;
8   import java.awt.*;
9   import java.awt.event.ActionEvent;
10  import java.awt.event.ActionListener;
11  import java.util.logging.Logger;
12  
13  public class CheckoutPanel extends JPanel
14  {
15      private final CartTableModel cartTableModel;
16  
17      private final Logger logger;
18  
19      @Inject
20      public CheckoutPanel( BookInventory inventory, CreditCardService cardService, Logger logger ) {
21          setLayout( new BoxLayout( this, BoxLayout.Y_AXIS ) );
22          this.logger = logger;
23  
24          this.cartTableModel = new CartTableModel();
25  
26          final CheckoutInventoryTableModel inventoryModel = new CheckoutInventoryTableModel( inventory );
27          inventoryModel.addTableModelListener( new TableModelListener() {
28  
29              public void tableChanged( TableModelEvent event )
30              {
31                  if (inventoryModel.isLastColumn( event.getColumn() ) ) {
32                      for (int i = event.getFirstRow(); i <= event.getLastRow(); i++) {
33                          addToCart( inventoryModel.decrementInventry(event.getFirstRow()) );
34                      }
35                  }
36              }
37          });
38          this.add( new JScrollPane( new CheckoutInventoryTable( inventoryModel ) ));
39          this.add( Box.createVerticalStrut( 20 ));
40          this.add( new JScrollPane( new CartTable( cartTableModel ) ));
41          this.add( Box.createVerticalStrut( 20 ));
42          this.add( new PaymentPanel( cardService ) );
43  
44          this.add( Box.createVerticalGlue() );
45  
46          this.setPreferredSize( new Dimension( 600, 400) );
47      }
48  
49      private void addToCart(Book book) {
50          logger.fine("Adding " + book + " to cart");
51          this.cartTableModel.addBook(book);
52      }
53  
54      private class PaymentPanel extends JPanel
55      {
56          public PaymentPanel( final CreditCardService cardService ) {
57              setLayout( new BoxLayout( this, BoxLayout.X_AXIS ) );
58  
59              add(new JLabel( "Credit Card" ));
60              add( Box.createHorizontalStrut( 5 ));
61              final JTextField creditCard = new JTextField(30);
62              add(creditCard);
63              add( Box.createHorizontalStrut( 5 ));
64              JButton purchase = new JButton("Purchase");
65              purchase.addActionListener( new ActionListener() {
66  
67                  public void actionPerformed( ActionEvent actionEvent )
68                  {
69                      double total = cartTableModel.getTotal();
70                      String cardNumber = creditCard.getText();
71                      String confirmation = "Do you want to charge " + total + " to your " + cardNumber + " card?";
72                      String title = "Confirm charge";
73                      int confirm =
74                          JOptionPane.showConfirmDialog( PaymentPanel.this, confirmation, title, JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE );
75                      if (confirm == JOptionPane.YES_OPTION) {
76                          CreditCard card = new CreditCard( cardNumber );
77                          cardService.debit( card, total );
78                      }
79                  }
80              });
81              add(Box.createHorizontalStrut( 5 ));
82              add(purchase);
83          }
84      }
85  }
