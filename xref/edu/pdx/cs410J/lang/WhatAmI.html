3   /**
4    * This program uses Java's reflection mechanism to print the name of
5    * a given <code>Object</code>'s class.
6    */
7   public class WhatAmI {
8   
9     /**
10     * Prints out the name of a given <code>Object</code>'s class.
11     */
12    private static void whatAmI(Object o) {
13      Class c = o.getClass();
14      System.out.println("I (" + o + ") am a " + c.getName());
15    }
16  
17    public static void main(String args[]) {
18      whatAmI("Hello");
19      whatAmI(new Integer(4));
20      whatAmI(new Double(2.7));
21      whatAmI(new Cow("Tootie"));
22    }
23  
24  }
