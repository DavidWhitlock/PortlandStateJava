3   /**
4    * This class demonstrate the <code>instanceof</code> operator and the
5    * concept of interfaces using the animal hierarchy.  It uses the
6    * <code>instanceof</code> operator to determine whether or not an
7    * animal implements the <code>Flies</code> interface.
8    */
9   public class DoIFly {
10  
11    /**
12     * Prints out whether or not an animal can fly.  It checks the
13     * run-time type of the <code>Animal</code>.  If the animal
14     * implements the <code>Flies</code> interface, then the animal can
15     * fly.
16     */
17    private static void doIFly(Animal animal) {
18      boolean iFly = (animal instanceof Flies);
19      System.out.print("Does " + animal.getName() + " fly?  ");
20      System.out.println((iFly ? "Yes." : "No."));
21    }
22  
23    /**
24     * This main program creates several animals and then prints out
25     * whether or not they can fly.
26     */
27    public static void main(String[] args) {
28      doIFly(new Cow("Bessy"));
29      doIFly(new Bee("Buzz"));
30      doIFly(new Turkey("Tom"));
31    }
32  
33  }
