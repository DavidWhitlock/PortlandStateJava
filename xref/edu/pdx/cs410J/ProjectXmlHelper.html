1   package edu.pdx.cs410J;
2   
3   import org.xml.sax.*;
4   
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.net.URL;
8   
9   public abstract class ProjectXmlHelper implements ErrorHandler, EntityResolver {
10    private final String publicId;
11    private final String systemId;
12    private final String dtdFileName;
13  
14    protected ProjectXmlHelper(String publicId, String systemId, String dtdFileName) {
15      this.publicId = publicId;
16      this.systemId = systemId;
17      this.dtdFileName = dtdFileName;
18    }
19  
20    @Override
21    public void warning(SAXParseException exception) throws SAXException {
22      throw exception;
23    }
24  
25    @Override
26    public void error(SAXParseException exception) throws SAXException {
27      throw exception;
28    }
29  
30    @Override
31    public void fatalError(SAXParseException exception) throws SAXException {
32      throw exception;
33    }
34  
35    @Override
36    public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
37      if (this.publicId.equals(publicId) || this.systemId.equals(systemId)) {
38        // We're resolving the external entity for the DTD
39        // Check to see if its in the jar file.  This way we don't
40        // need to go all the way to the website to find the DTD.
41        InputStream stream =
42          ProjectXmlHelper.class.getResourceAsStream(this.dtdFileName);
43        if (stream != null) {
44          return new InputSource(stream);
45        }
46      }
47  
48      // Try to access the DTD using the URL
49      try {
50        URL url = new URL(systemId);
51        InputStream stream = url.openStream();
52        return new InputSource(stream);
53  
54      } catch (Exception ex) {
55        return null;
56      }
57    }
58  }
