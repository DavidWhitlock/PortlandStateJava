3   /**
4    * Uses the generic {@link Tuple} class to return a host/port
5    * combination from a single method.
6    *
7    * @author David Whitlock
8    * @since Summer 2005
9    */
10  public class TupleExample {
11  
12    /**
13     * Returns the host and port on which the server runs
14     */
15    public static Tuple<String, Integer> getHostAndPort() {
16      return new Tuple<String, Integer>("pippin", 12345);
17    }
18  
19    /**
20     * Main program that uses a <code>Tuple</code>
21     */
22    public static void main(String[] args) {
23      Tuple<String, Integer> tuple;
24  
25      tuple = getHostAndPort();
26  
27      String host = tuple.getFirst();
28      int port = tuple.getSecond();
29  
30      System.out.println("Running on " + host + ":" + port);
31    }
32  
33  }
