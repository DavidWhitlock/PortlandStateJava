3   import java.util.*;
4   
5   /**
6    * Demonstrates J2SE's "enumerated type" facility.
7    *
8    * @author David Whitlock
9    * @version $Revision: 1.1 $
10   * @since Summer 2004
11   */
12  public class EnumeratedTypes {
13  
14    /**
15     * A enumerated type for the days of the week
16     */
17    private enum Day { SUNDAY, MONDAY, TUESDAY, WEDNESDAY, THURSDAY,
18                       FRIDAY, SATURDAY }
19  
20    /**
21     * Translates a {@link Day} into Spanish
22     */
23    private static String enEspanol(Day day) {
24      switch (day) {
25      case SUNDAY:
26        return "Domingo";
27      case MONDAY:
28        return "Lunes";
29      case TUESDAY:
30        return "Martes";
31      case WEDNESDAY:
32        return "Miercoles";
33      case THURSDAY:
34        return "Jueves";
35      case FRIDAY:
36        return "Viernes";
37      case SATURDAY:
38        return "Sabado";
39      default:
40        String s = "Unknown day: " + day;
41        throw new IllegalArgumentException(s);
42      }
43    }
44  
45    /**
46     * Demonstrates enumerated types by adding several to a {@link
47     * SortedSet} and printing them out using their {@link
48     * Enum#toString()} method and in Spanish.
49     */
50    public static void main(String[] args) {
51      SortedSet<Day> set = new TreeSet<Day>();
52      set.add(Day.WEDNESDAY);
53      set.add(Day.MONDAY);
54      set.add(Day.FRIDAY);
55  
56      System.out.print("Sorted days: ");
57      for (Day day : set) {
58        System.out.print(day + " ");
59      }
60  
61      System.out.print("\nEn espanol: ");
62      for (Day day : set) {
63        System.out.print(enEspanol(day) + " ");
64      }
65      System.out.println("");
66    }
67  
68  }
