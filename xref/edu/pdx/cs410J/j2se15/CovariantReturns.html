3   /**
4    * This class demonstrates <I>covariant returns</I> in JDK 1.5.  The
5    * Java compiler allows an overriding method to change the method's
6    * return type to be a subclass of the overriden method's return
7    * type.  However, you have to be extra careful when using covariant
8    * returns with legacy code.  If someone else overrides a method and
9    * doesn't change the return type, their code will not compile.
10   *
11   * @author David Whitlock
12   * @since Winter 2004
13   */
14  public class CovariantReturns {
15  
16    /**
17     * An animal that can be cloned.  Such pointed political commentary.
18     */
19    static abstract class Animal implements Cloneable {
20  
21      public abstract Object clone();
22  
23    }
24  
25    static class Human extends Animal {
26  
27      public Human clone() {
28        return new Human();
29      }
30    }
31  
32    static class Student extends Human {
33  
34      public Student clone() {
35        return new Student();
36      }
37    }
38  
39    /**
40     * A main class that clones some animals.  Note that we don't need
41     * to cast the result of the <code>clone</code> method.
42     */
43    public static void main(String[] args) {
44      Human human = new Human();
45      @SuppressWarnings("unused")
46      Human human2 = human.clone();
47  
48      Student student = new Student();
49      @SuppressWarnings("unused")
50      Student student2 = student.clone();
51    }
52  
53  }
