20  /**
21   * This class dumps the contents of a <code>GradeBook</code> to an XML
22   * file.  By default, the students in the grade book are dumped to XML
23   * files in the same directory as the grade book's XML file.
24   *
25   * @author David Whitlock
26   * @since Fall 2000
27   */
28  public class XmlDumper extends XmlHelper {
29  
30    private File studentDir = null; // Where to dump student XML files
31    PrintWriter pw = null;          // Where to dump grade book
32  
33    /**
34     * Creates a new <code>XmlDumper</code> that dumps the contents of a
35     * grade book to a given file.
36     */
37    public XmlDumper(File xmlFile) throws IOException {
38      this(new PrintWriter(new FileWriter(xmlFile), true));
39  
40      if (!xmlFile.exists()) {
41        if (!xmlFile.createNewFile()) {
42          throw new IOException("Could not create file " + xmlFile);
43        }
44      }
45  
46      this.setStudentDir(xmlFile.getCanonicalFile().getParentFile());
47    }
48  
49    /**
50     * Creates a new <code>XmlDumper</code> that dumps the contents of a
51     * grade book to a file of the given name.
52     */
53    public XmlDumper(String xmlFileName) throws IOException {
54      this(new File(xmlFileName));
55    }

57    /**
58     * Creates an <code>XmlDumper</code> that dumps the contents of a
59     * grade book to a <code>PrintWriter</code>.  The location of the
60     * student files is unspecified.
61     */
62    private XmlDumper(PrintWriter pw) {
63      this.pw = pw;
64    }

66    /**
67     * Sets the directory in which the XML files for students are
68     * generated.
69     */
70    public void setStudentDir(File dir) {
71      if (dir.exists() && !dir.isDirectory()) {
72        throw new IllegalArgumentException(dir + " is not a directory");
73      }
74      
75      this.studentDir = dir;
76    }

78    /**
79     * Dumps the contents of a <code>GradeBook</code> in XML format.
80     */
81    public void dump(GradeBook book) throws IOException {
82      Document doc = dumpGradeBook(book, this);
83  
84      try {
85        writeXmlToPrintWriter(doc, this.pw);
86  
87      } catch (TransformerException ex) {
88        ex.printStackTrace(System.err);
89        System.exit(1);
90      }
91  
92      dumpDirtyStudents(book);
93  
94      // Mark the grade book as being clean
95      book.makeClean();
96    }

98    static Document dumpGradeBook(GradeBook book, XmlHelper helper) throws IOException {
99      Document doc = createDocumentForGradeBook(helper);
100 
101     Element root = doc.getDocumentElement();
102 
103     appendXmlForClassName(book, doc, root);
104     appendXmlForAssignments(book, doc, root);
105     appendXmlForLetterGradeRanges(book, doc, root);
106     appendXmlForStudents(book, doc, root);
107 
108     return doc;
109   }

111   private static void appendXmlForLetterGradeRanges(GradeBook book, Document doc, Element root) {
112     for (Student.Section section : book.getSections()) {
113       Element lgrNode = appendXmlForLetterGradeRange(book, section, doc);
114       root.appendChild(lgrNode);
115     }
116   }

118   private static Element appendXmlForLetterGradeRange(GradeBook book, Student.Section section, Document doc) {
119     Element lgrNode = doc.createElement("letter-grade-ranges");
120     lgrNode.setAttribute("for-section", getSectionXmlAttributeValue(section));
121     for (LetterGradeRange range : book.getLetterGradeRanges(section)) {
122       appendXmlForLetterGradeRange(range, doc, lgrNode);
123     }
124     return lgrNode;
125   }

127   private static String getSectionXmlAttributeValue(Student.Section section) {
128     return Objects.toString(section, null);
129   }

131   private static void appendXmlForLetterGradeRange(LetterGradeRange range, Document doc, Element parent) {
132     Element node = doc.createElement("letter-grade-range");
133     node.setAttribute("letter-grade", range.letterGrade().asString());
134     node.setAttribute("minimum-score", String.valueOf(range.minimum()));
135     node.setAttribute("maximum-score", String.valueOf(range.maximum()));
136 
137     parent.appendChild(node);
138   }

140   private static Document createDocumentForGradeBook(XmlHelper helper) {
141     Document doc = null;
142 
143     try {
144       DocumentBuilderFactory factory =
145         DocumentBuilderFactory.newInstance();
146       factory.setValidating(true);
147 
148       DocumentBuilder builder = factory.newDocumentBuilder();
149       builder.setErrorHandler(helper);
150       builder.setEntityResolver(helper);
151 
152       DOMImplementation dom =
153         builder.getDOMImplementation();
154       DocumentType docType =
155         dom.createDocumentType("gradebook", publicID, systemID);
156       doc = dom.createDocument(null, "gradebook", docType);
157 
158     } catch (ParserConfigurationException | DOMException ex) {
159       ex.printStackTrace(System.err);
160       System.exit(1);
161 
162     }
163     return doc;
164   }

166   private static void appendXmlForStudents(GradeBook book, Document doc, Element root) {
167     // Students
168     Element studentsNode = doc.createElement("students");
169     for (String id : book.getStudentIds()) {
170       Element studentNode = doc.createElement("id");
171       studentNode.appendChild(doc.createTextNode(id));
172 
173       studentsNode.appendChild(studentNode);
174     }
175 
176     root.appendChild(studentsNode);
177   }

179   private static void appendXmlForClassName(GradeBook book, Document doc, Element root) {
180     // name node
181     Element name = doc.createElement("name");
182     name.appendChild(doc.createTextNode(book.getClassName()));
183     root.appendChild(name);

186   private static void appendXmlForAssignments(GradeBook book, Document doc, Element root) {
187     // assignment nodes
188     Element assignments = doc.createElement("assignments");
189     for (String assignmentName : book.getAssignmentNames()) {
190       Assignment assign = book.getAssignment(assignmentName);
191       Element assignNode = doc.createElement("assignment");
192 
193       setAssignmentTypeAttribute(assign, assignNode);
194       appendTextElementIfValueIsNotNull(assignNode, "name", assign.getName());
195       appendTextElementIfValueIsNotNull(assignNode, "description", assign.getDescription());
196       appendTextElementIfValueIsNotNull(assignNode, "points", String.valueOf(assign.getPoints()));
197       appendTextElementIfValueIsNotNull(assignNode, "due-date", assign.getDueDate());
198 
199       doNotes(doc, assignNode, assign.getNotes());
200 
201       assignments.appendChild(assignNode);
202 
203     }
204     root.appendChild(assignments);
205   }

207   private static void appendTextElementIfValueIsNotNull(Element parent, String elementName, LocalDateTime dateTime) {
208     if (dateTime != null) {
209       appendTextElementIfValueIsNotNull(parent, elementName, dateTime.format(DATE_TIME_FORMAT));
210     }
211   }

213   private static void setAssignmentTypeAttribute(Assignment assign, Element assignNode) {
214     Assignment.AssignmentType type = assign.getType();
215     switch (type) {
216       case PROJECT:
217         assignNode.setAttribute("type", "PROJECT");
218         break;
219 
220       case QUIZ:
221         assignNode.setAttribute("type", "QUIZ");
222         break;
223 
224       case OTHER:
225         assignNode.setAttribute("type", "OTHER");
226         break;
227 
228       case OPTIONAL:
229         assignNode.setAttribute("type", "OPTIONAL");
230         break;
231 
232       case POA:
233         assignNode.setAttribute("type", "POA");
234         break;
235 
236       default:
237         throw new IllegalArgumentException("Can't handle assignment " +
238           "type " + type);
239     }
240   }

242   private void dumpDirtyStudents(GradeBook book) throws IOException {
243     book.forEachStudent(student -> {
244       if (student.isDirty()) {
245         dumpStudent(student);
246       }
247     });
248   }

250   /**
251    * Creates a <code>notes</code> XML element for a given
252    * <code>List</code> of notes.
253    */
254   private static void doNotes(Document doc, Element parent, List<String> notes) {
255     Element notesNode = doc.createElement("notes");
256     for (String note : notes) {
257       Element noteNode = doc.createElement("note");
258       noteNode.appendChild(doc.createTextNode(note));
259       notesNode.appendChild(noteNode);
260     }
261 
262     parent.appendChild(notesNode);
263   }

266    * Dumps a <code>Student</code> out to an XML file whose name is
267    * based on the student's id and resides in the
268    * <code>studentDir</code>.
269    */
270   private void dumpStudent(Student student) {
271     Document doc = toXml(student);
272 
273     // Now dump DOM tree to the file
274     File studentFile = new File(studentDir, student.getId() + ".xml");
275     PrintWriter pw = new PrintWriter(newFileWriter(studentFile), true);
276 
277     try {
278       writeXmlToPrintWriter(doc, pw);
279 
280     } catch (TransformerException ex) {
281       ex.printStackTrace(System.err);
282       System.exit(1);
283     }
284   }

295   /**
296    * Returns a DOM tree that represents a <code>Student</code>
297    */
298   static Document toXml(Student student) {
299     Document doc = createXmlDocument();
300 
301     Element root = doc.getDocumentElement();
302 
303     appendStudentInformation(student, root);
304     appendGradesInformation(student, root);
305     appendLateInformation(student, root);
306     appendResubmittedInformation(student, root);
307     appendNotes(student, root);
308 
309     return doc;
310   }

312   private static void appendNotes(Student student, Element parent) {
313     List<String> notes = student.getNotes();
314     if (!notes.isEmpty()) {
315       doNotes(parent.getOwnerDocument(), parent, notes);
316     }
317   }

319   private static void appendResubmittedInformation(Student student, Element parent) {
320     Document doc = parent.getOwnerDocument();
321     List<String> resubmitted = student.getResubmitted();
322     if (!resubmitted.isEmpty()) {
323       Element resubNode = doc.createElement("resubmitted");
324 
325       for (String assignmentName : resubmitted) {
326         Element nameNode = doc.createElement("name");
327         nameNode.appendChild(doc.createTextNode(assignmentName));
328         resubNode.appendChild(nameNode);
329       }
330 
331       parent.appendChild(resubNode);
332     }

351   private static void appendGradesInformation(Student student, Element parent) {
352     Document doc = parent.getOwnerDocument();
353     Iterator gradeNames = student.getGradeNames().iterator();
354     if (gradeNames.hasNext()) {
355       Element gradesNode = doc.createElement("grades");
356       while (gradeNames.hasNext()) {
357         String gradeName = (String) gradeNames.next();
358         Grade grade = student.getGrade(gradeName);
359 
360         Element gradeNode = doc.createElement("grade");
361 
362         Element nameNode = doc.createElement("name");
363         nameNode.appendChild(doc.createTextNode(grade.getAssignmentName()));
364         gradeNode.appendChild(nameNode);
365 
366         Element scoreNode = doc.createElement("score");
367         scoreNode.appendChild(doc.createTextNode(grade.getScore() + ""));
368         gradeNode.appendChild(scoreNode);
369 
370         appendSubmissionTimesInformation(grade.getSubmissionTimes(), gradeNode);
371 
372         doNotes(doc, gradeNode, grade.getNotes());
373 
374         if (grade.getScore() == Grade.INCOMPLETE) {
375           gradeNode.setAttribute("type", "INCOMPLETE");
376 
377         } else if (grade.getScore() == Grade.NO_GRADE) {
378           gradeNode.setAttribute("type", "NO_GRADE");
379         }
380 
381         gradesNode.appendChild(gradeNode);
382       }
383 
384       parent.appendChild(gradesNode);
385     }
386   }

388   private static void appendSubmissionTimesInformation(List<LocalDateTime> submissionTimes, Element parent) {
389     if (!submissionTimes.isEmpty()) {
390       Document doc = parent.getOwnerDocument();
391       Element submissions = doc.createElement("submissions");
392       parent.appendChild(submissions);
393 
394       submissionTimes.forEach(submissionTime -> {
395         Element submission = doc.createElement("submission");
396         submissions.appendChild(submission);
397         submission.appendChild(doc.createTextNode(submissionTime.format(DATE_TIME_FORMAT)));
398       });
399     }
400 
401   }

403   private static void appendStudentInformation(Student student, Element root) {
404     appendTextElementIfValueIsNotNull(root, "id", student.getId());
405     appendTextElementIfValueIsNotNull(root, "firstName", student.getFirstName());
406     appendTextElementIfValueIsNotNull(root, "lastName", student.getLastName());
407     appendTextElementIfValueIsNotNull(root, "nickName", student.getNickName());
408     appendTextElementIfValueIsNotNull(root, "email", student.getEmail());
409     appendTextElementIfValueIsNotNull(root, "ssn", student.getSsn());
410     appendTextElementIfValueIsNotNull(root, "major", student.getMajor());
411     appendTextElementIfValueIsNotNull(root, "d2l-id", student.getD2LId());
412     appendTextElementIfValueIsNotNull(root, "letter-grade", Objects.toString(student.getLetterGrade(), null));
413 
414     setAttributeIfValueIsNotNull(root, "enrolled-section", getSectionXmlAttributeValue(student.getEnrolledSection()));
415   }

446   private static void appendTextElementIfValueIsNotNull(Element parent, String elementName, String textValue) {
447     if (textValue != null) {
