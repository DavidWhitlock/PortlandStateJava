9   import java.io.File;
10  import java.io.FileNotFoundException;
11  import java.io.IOException;
12  import java.io.PrintStream;
13  
14  /**
15   * This is a little program that combines the grades for the midterm
16   * quiz, the dream job, and the resume into one grade.  This way
17   * people who missed this quiz are not penalized unnecessarily.
18   */
19  public class FixMidterm {
20  
21    public static void main(String[] args) {
22      PrintStream out = System.out;
23      PrintStream err = System.err;
24  
25      if (args.length < 1) {
26        err.println("** usage: java FixMidterm gradebook");
27        System.exit(1);
28      }
29  
30      String xmlFile = args[0];
31  
32      GradeBook book;
33  
34      File file = new File(xmlFile);
35      if (file.exists()) {
36        // Parse a grade book from the XML file
37        try {
38          XmlGradeBookParser parser = new XmlGradeBookParser(file);
39          book = parser.parse();
40  
41        } catch (FileNotFoundException ex) {
42          err.println("** Could not find file: " + ex.getMessage());
43          System.exit(1);
44          return;
45  
46        } catch (IOException ex) {
47          err.println("** IOException during parsing: " + ex.getMessage());
48          System.exit(1);
49          return;
50  
51        } catch (ParserException ex) {
52          err.println("** Error during parsing: " + ex.getMessage());
53          System.exit(1);
54          return;
55        }
56  
57      } else {
58        err.println("** File " + file + " does not exist");
59        System.exit(1);
60        return;
61      }
62  
63      book.forEachStudent(student -> {
64        out.print("Fixing " + student.getId() + ": ");
65        out.flush();
66  
67        double total = 0.0;
68  
69        Grade quiz3 = student.getGrade("quiz3");
70        if (quiz3 != null) {
71          total += quiz3.getScore();
72        } else {
73          quiz3 = new Grade("quiz3", 0.0);
74        }
75  
76        out.print("midterm = " + quiz3.getScore() + ", ");
77        out.flush();
78  
79        Grade job = student.getGrade("job");
80        if (job != null) {
81          total += job.getScore();
82          quiz3.addNote("Dream job score: " + job.getScore());
83          out.print("dream job = " + job.getScore() + ", ");
84  
85        } else {
86          quiz3.addNote("Missing dream job");
87        }
88  
89        Grade resume = student.getGrade("resume");
90        if (resume != null) {
91          total += resume.getScore();
92          quiz3.addNote("Resume score: " + resume.getScore());
93          out.print("resume = " + resume.getScore() + ", ");
94          out.flush();
95  
96        } else {
97          quiz3.addNote("Missing resume");
98        }
99  
100       quiz3.setScore(total);
101       student.setGrade(quiz3.getAssignmentName(), quiz3);
102       out.println("total = " + quiz3.getScore());
103     });
104 
105     // Write the grade book to the XML file
106     try {
107       XmlDumper dumper = new XmlDumper(file);
108       dumper.dump(book);
109 
110     } catch (IOException ex) {
111       err.println("** Error while writing XML file: " + ex);
112       System.exit(1);
113     }    
114   }
115 
116 }
