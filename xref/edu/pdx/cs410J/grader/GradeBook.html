169   /**
170    * Returns <code>true</code> if this <code>GradeBook</code> has been
171    * modified.
172    */
173   public boolean isDirty() {
174     if (this.dirty) {
175       return true;
176     }
177 
178     // Are any of the Assignments dirty?
179     Iterator iter = this.assignments.values().iterator();
180     while (iter.hasNext()) {
181       Assignment assign = (Assignment) iter.next();
182       if (assign.isDirty()) {
183         return true;
184       }
185     }
186 
187     // Are any of the Students dirty?
188     iter = this.students.values().iterator();
189     while (iter.hasNext()) {
190       Student student = (Student) iter.next();
191       if (student.isDirty()) {
192         return true;
193       }
194     }
195 
196     return false;
197   }

222   /**
223    * Main program that is used to create a <code>GradeBook</code>
224    */
225   public static void main(String[] args) {
226     String xmlFile = null;
227     String name = null;
228     String importName = null;
229 
230     // Parse the command line
231     for (int i = 0; i < args.length; i++) {
232       if (args[i].equals("-file")) {
233         if (++i >= args.length) {
234           err.println("** Missing file name");
235           usage();
236         }
237 
238         xmlFile = args[i];
239 
240       } else if (args[i].equals("-name")) {
241         if (++i >= args.length) {
242           err.println("** Missing class name");
243           usage();
244         }
245 
246         name = args[i];
247 
248       } else if (args[i].equals("-import")) {
249         if (++i >= args.length) {
250           err.println("** Missing import file name");
251           usage();
252         }
253 
254         importName = args[i];
255 
256       } else {
257         err.println("** Spurious command line option: " + args[i]);
258         usage();
259       }
260     }
261 
262     if (xmlFile == null) {
263       err.println("** No XML file specified");
264       usage();
265     }
266 
267     GradeBook book = null;
268 
269     File file = new File(xmlFile);
270     if (file.exists()) {
271       // Parse a grade book from the XML file
272       try {
273         XmlGradeBookParser parser = new XmlGradeBookParser(file);
274         book = parser.parse();
275 
276       } catch (FileNotFoundException ex) {
277         err.println("** Could not find file: " + ex.getMessage());
278         System.exit(1);
279 
280       } catch (IOException ex) {
281         err.println("** IOException during parsing: " + ex.getMessage());
282         System.exit(1);
283 
284       } catch (ParserException ex) {
285         err.println("** Error during parsing: " + ex.getMessage());
286         System.exit(1);
287       }
288 
289       // Do we import?
290       if (importName != null) {
291         File importFile = new File(importName);
292         if (!importFile.exists()) {
293           err.println("** Import file " + importFile.getName() + 
294                       " does not exist");
295           System.exit(1);
296         }
297 
298         try {
299           XmlStudentParser sp = new XmlStudentParser(importFile);
300           Student student = sp.parseStudent();
301           book.addStudent(student);
302 
303         } catch (IOException ex) {
304           err.println("** Error during parsing: " + ex.getMessage());
305           System.exit(1);
306 
307         } catch (ParserException ex) {
308           err.println("** Error during parsing: " + ex.getMessage());
309           System.exit(1);
310         }
311       }
312 
313     } else if (name == null) {
314       err.println("** Must specify a class name when creating a " +
315                   "grade book");
316       System.exit(1);
317       
318     } else {
319       // Create an empty GradeBook
320       book = new GradeBook(name);
321     }
322 
323     // Write the grade book to the XML file
324     try {
325       XmlDumper dumper = new XmlDumper(file);
326       dumper.dump(book);
327 
328     } catch (IOException ex) {
329       err.println("** Error while writing XML file: " + ex);
330       System.exit(1);
331     }
332   }

338   public LetterGrade getLetterGradeForScore(Section section, double score) {
339     for (LetterGradeRange range : this.getLetterGradeRanges(section)) {

356   static class LetterGradeRanges implements Iterable<LetterGradeRanges.LetterGradeRange> {
357     private final Map<LetterGrade, LetterGradeRange> ranges = new TreeMap<>();
358 
359     private LetterGradeRanges() {
360       createDefaultLetterGradeRange(LetterGrade.A, 94, 100);
361       createDefaultLetterGradeRange(LetterGrade.A_MINUS, 90, 93);
362       createDefaultLetterGradeRange(LetterGrade.B_PLUS, 87, 89);
363       createDefaultLetterGradeRange(LetterGrade.B, 83, 86);
364       createDefaultLetterGradeRange(LetterGrade.B_MINUS, 80, 82);
365       createDefaultLetterGradeRange(LetterGrade.C_PLUS, 77, 79);
366       createDefaultLetterGradeRange(LetterGrade.C, 73, 76);
367       createDefaultLetterGradeRange(LetterGrade.C_MINUS, 70, 72);
368       createDefaultLetterGradeRange(LetterGrade.D_PLUS, 67, 69);
369       createDefaultLetterGradeRange(LetterGrade.D, 63, 66);
370       createDefaultLetterGradeRange(LetterGrade.D_MINUS, 60, 62);
371       createDefaultLetterGradeRange(LetterGrade.F, 0, 59);
372     }
373 
374     private void createDefaultLetterGradeRange(LetterGrade letterGrade, int minimum, int maximum) {
375       this.ranges.put(letterGrade, new LetterGradeRange(letterGrade, minimum, maximum));
376     }
377 
378     public LetterGradeRange getRange(LetterGrade letterGrade) {
379       return ranges.get(letterGrade);
380     }
381 
382     public void validate() {
383       validateThatFRangeHasAMinimumOf0();
384       validateThatARangeContains100();
385       validateThatRangesAreContiguous();
386     }
387 
388     private void validateThatRangesAreContiguous() {
389       LetterGradeRange previous = null;
390 
391       for (LetterGrade letterGrade : ranges.keySet()) {
392         LetterGradeRange current = ranges.get(letterGrade);
393         if (previous != null) {
394           if (previous.minimum() != current.maximum() + 1) {
395             String s = "There is a gap between the range for " + previous + " and " + current;
396             throw new LetterGradeRange.InvalidLetterGradeRange(s);
397           }
398         }
399 
400         previous = current;
401       }
402 
403     }
404 
405     private void validateThatARangeContains100() {
406       LetterGradeRange range = getRange(LetterGrade.A);
407       if (range.maximum() < 100) {
408         throw new LetterGradeRange.InvalidLetterGradeRange("The A range must contain 100");
409       }
410     }
411 
412     private void validateThatFRangeHasAMinimumOf0() {
413       LetterGradeRange range = getRange(LetterGrade.F);
414       if (range.minimum() > 0) {
415         throw new LetterGradeRange.InvalidLetterGradeRange("The F range must contain zero");
416       }
417     }
418

420     public Iterator<LetterGradeRange> iterator() {
421       return this.ranges.values().iterator();
422     }
423 
424     @Override
425     public void forEach(Consumer<? super LetterGradeRange> action) {
426       this.ranges.values().forEach(action);
427     }
428 
429     @Override
430     public Spliterator<LetterGradeRange> spliterator() {
431       return this.ranges.values().spliterator();
432     }

434     static class LetterGradeRange {
435       private final LetterGrade letterGrade;
436       private int maximum;
437       private int minimum;
438 
439       public LetterGradeRange(LetterGrade letterGrade, int minimum, int maximum) {
440         this.letterGrade = letterGrade;
441         setRange(minimum, maximum);
442       }
443 
444       public void setRange(int minimum, int maximum) {
445         if (minimum >= maximum) {
446           String s = String.format("Minimum value (%d) must be less than maximum value (%d)",
447             minimum, maximum);
448           throw new InvalidLetterGradeRange(s);
449         }
450 
451         this.minimum = minimum;
452         this.maximum = maximum;
453       }
454 
455       public int minimum() {
456         return this.minimum;
457       }
458 
459       public int maximum() {
460         return this.maximum;
461       }
462 
463       @Override
464       public String toString() {
465         return "Range for " + letterGrade() + " is " + minimum() + " to " + maximum();
466       }
467 
468       public LetterGrade letterGrade() {
469         return this.letterGrade;
470       }
471 
472       public boolean isScoreInRange(double score) {
473         int intScore = (int) Math.round(score);
474         return intScore >= minimum() && intScore <= maximum();
475       }
476 
477       public static class InvalidLetterGradeRange extends RuntimeException {
478         public InvalidLetterGradeRange(String message) {
479           super(message);
480         }
481       }
482     }
