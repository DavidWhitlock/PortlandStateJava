29     * Computes the student's final average and makes a pretty report.

31    @VisibleForTesting
32    static void dumpReportTo(GradeBook book, Student student,
33                             PrintWriter pw, boolean assignLetterGrades) {
34      NumberFormat format = NumberFormat.getNumberInstance();
35      format.setMinimumFractionDigits(1);
36      format.setMaximumFractionDigits(1);
37  
38      Assignment lowestQuiz = null;
39      double best = 0.0;
40      double total = 0.0;
41  
42      pw.println("Grade summary for: " + student.getFullName());
43      SimpleDateFormat df = 
44        new SimpleDateFormat("EEEE MMMM d, yyyy 'at' h:mm a");
45      pw.println("Generated on: " + df.format(new Date()));
46      pw.println("");
47  
48      for (String assignmentName : getSortedAssignmentNames(book)) {
49        Assignment assignment = book.getAssignment(assignmentName);
50        if (noStudentHasGradeFor(assignment, book)) {
51          continue;
52        }
53  
54        Grade grade = student.getGrade(assignmentName);
55        double score = getScore(grade);
56  
57  
58  //       System.out.println("Examining " + assign + ", score: " + score);
59  
60        if (assignment.getType() == Assignment.AssignmentType.QUIZ) {
61          if (lowestQuiz == null) {
62            lowestQuiz = assignment;
63  //           System.out.println("Lowest quiz: " + lowestQuiz + 
64  //                              ", score: " + score);
65  
66          } else {
67            Grade lowestGrade = student.getGrade(lowestQuiz.getName());
68            if (lowestGrade != null && score < lowestGrade.getScore()) {
69              lowestQuiz = assignment;
70  //             System.out.println("Lowest quiz: " + lowestQuiz + ", score: "
71  //                                + score + ", lowest grade: " +
72  //                                student.getGrade(lowestQuiz.getName()));
73            }
74          }
75        }
76  
77        StringBuffer line = new StringBuffer();
78        line.append("  ");
79        line.append(assignment.getName());
80        line.append(" (");
81        line.append(assignment.getDescription());
82        line.append(")");
83        if (assignment.getType() == Assignment.AssignmentType.OPTIONAL) {
84          line.append(" (OPTIONAL)");
85        }
86        line.append(": ");
87        line.append(format.format(score));
88        line.append("/");
89        line.append(format.format(assignment.getPoints()));
90  
91  
92        // Skip incompletes and no grades
93        if (grade == null) {
94          if (dueDateHasPassed(assignment)) {
95            line.append(" (MISSING GRADE)");
96  
97          } else {
98            line.append(String.format(" (due %s)", formatDueDate(assignment)));
99          }
100 
101       } else if (grade.isIncomplete()) {
102         line.append(" (INCOMPLETE)");
103 
104       } else if (grade.isNotGraded()) {
105         line.append( " (NOT GRADED)");
106       }
107 
108       pw.println(line);
109 
110       // Don't count optional assignments toward the maximum point
111       // total
112       if (assignment.getType() != Assignment.AssignmentType.OPTIONAL) {
113         best += assignment.getPoints();
114       }
115 
116       total += score;
117     }
118 
119     if (lowestQuiz != null) {
120       pw.println("");
121       pw.println("Lowest Quiz grade dropped: " +
122 		  lowestQuiz.getName());
123       pw.println("");
124 
125       // Subtract lowest quiz grade
126       Grade lowestGrade = student.getGrade(lowestQuiz.getName());
127       total -= (lowestGrade != null ? lowestGrade.getScore() : 0);
128       best -= lowestQuiz.getPoints();
129     }
130 
131     // Print out late and resubmitted assignments
132     pw.println("Late assignments:");
133     for (String late : student.getLate()) {
134       pw.println("  " + late);
135     }
136     pw.println();
137 
138     pw.println("Resubmitted assignments:");
139     for (String resubmitted : student.getResubmitted()) {
140       pw.println("  " + resubmitted);
141     }
142     pw.println("");
143 
144     pw.println("Total grade: " + format.format(total)  + "/" +
145                format.format(best));
146 
147     double overallScore = total / best;
148 
149     if (assignLetterGrades) {
150       LetterGrade letterGrade = book.getLetterGradeForScore(student.getEnrolledSection(), overallScore * 100.0);
151       student.setLetterGrade(letterGrade);
152     }
153 
154     if (student.getLetterGrade() != null) {
155       pw.println("Letter Grade: " + student.getLetterGrade());
156     }
157 
158     allTotals.put(student, overallScore);
159   }

161   private static String formatDueDate(Assignment assignment) {
162     LocalDateTime dueDate = assignment.getDueDate();
163     return dueDate.format(DateTimeFormatter.ofLocalizedDateTime(FormatStyle.SHORT));
164   }

166   @VisibleForTesting
167   static boolean dueDateHasPassed(Assignment assignment) {
168     LocalDateTime dueDate = assignment.getDueDate();
169     LocalDateTime now = LocalDateTime.now();
170     return now.isAfter(dueDate);
171   }

173   static boolean noStudentHasGradeFor(Assignment assignment, GradeBook book) {
174     return book.studentsStream().noneMatch(student -> studentHasGradeFor(student, assignment));
175   }
176 
177   private static boolean studentHasGradeFor(Student student, Assignment assignment) {
178     Grade grade = student.getGrade(assignment.getName());
179     return grade != null && !grade.isNotGraded();
180   }

182   private static double getScore(Grade grade) {
183     // Average non-existent scores as zero
184     double score;
185     if (grade == null) {
186       score = 0.0;
187 
188     } else {
189       score = grade.getScore();
190     }
191     return score;
192   }

215   /**
216    * Main program that creates summary reports for every student in a
217    * grade book located in a given XML file.
218    */
219   public static void main(String[] args) {
220     boolean assignLetterGrades = false;
221     String xmlFileName = null;
222     String outputDirName = null;
223     Collection<String> students = new ArrayList<>();
224 
225     for (String arg : args) {
226       if (arg.equals("-assignLetterGrades")) {
227         assignLetterGrades = true;
228 
229       } else if (xmlFileName == null) {
230         xmlFileName = arg;
231 
232       } else if (outputDirName == null) {
233         outputDirName = arg;
234 
235       } else {
236         students.add(arg);
237       }
238     }
239 
240     if (xmlFileName == null) {
241       usage("Missing XML file name");
242     }
243 
244     if (outputDirName == null) {
245       usage("Missing output dir name");
246       return;
247     }
248 
249     String xmlFile = xmlFileName;
250     File outDir = new File(outputDirName);
251     if (!outDir.exists()) {
252       outDir.mkdirs();
253 
254     } else if (!outDir.isDirectory()) {
255       usage(outDir + " is not a directory");
256     }
257 
258     File file = new File(xmlFile);
259     if (!file.exists()) {
260       usage("Grade book file " + xmlFile + " does not exist");
261     }
262 
263     GradeBook book = parseGradeBook(file);
264 
265     // Create a SummaryReport for every student
266     Iterable<String> studentIds;
267 
268     if (!students.isEmpty()) {
269       studentIds = students;
270 
271     } else {
272       studentIds = book.getStudentIds();
273     }
274 
275     dumpReports(studentIds, book, outDir, assignLetterGrades);
276 
277     // Sort students by totals and print out results:
278     Set<Student> students1 = allTotals.keySet();
279     printOutStudentTotals(students1, out);
280 
281     saveGradeBookIfDirty(xmlFileName, book);
282 
283   }

285   @VisibleForTesting
286   static void printOutStudentTotals(Set<Student> allStudents, PrintWriter out) {
287     SortedSet<Student> sorted = getStudentSortedByTotalPoints(allStudents);
288 
289     out.println("Undergraduates:");
290     Stream<Student> undergrads = sorted.stream().filter(student -> student.getEnrolledSection() == UNDERGRADUATE);
291     printOutStudentTotals(out, undergrads);
292 
293     out.println("Graduate Students:");
294     Stream<Student> grads = sorted.stream().filter(student -> student.getEnrolledSection() == GRADUATE);
295     printOutStudentTotals(out, grads);
296 
297   }

299   private static void printOutStudentTotals(PrintWriter out, Stream<Student> students) {
300     NumberFormat format = NumberFormat.getPercentInstance();
301 
302     students.forEach(student -> {
303       Double d = allTotals.get(student);
304       out.print("  " + student + ": " + format.format(d.doubleValue()));
305 
306       if (student.getLetterGrade() != null) {
307         out.print(" " + student.getLetterGrade());
308       }
309 
310       out.println();
311     });
312   }

326   private static SortedSet<Student> getStudentSortedByTotalPoints(Set<Student> students) {
327     SortedSet<Student> sorted = new TreeSet<>(new Comparator<Student>() {
328         @Override
329         public int compare(Student s1, Student s2) {
330           Double d1 = allTotals.get(s1);
331           Double d2 = allTotals.get(s2);
332           if ( d2.compareTo( d1 ) == 0 ) {
333             return s1.getId().compareTo( s2.getId() );
334           } else {
335             return d2.compareTo(d1);
336           }
337         }
338 
339         @Override
340         public boolean equals(Object o) {
341           return true;
342         }
343       });
344 
345     sorted.addAll(students);
346     return sorted;
347   }

349   @VisibleForTesting
350   static void dumpReports(Iterable<String> studentIds, GradeBook book, File outDir, boolean assignLetterGrades) {
351     for (String id : studentIds) {
352       err.println(id);
353 
354       Student student = book.getStudent(id).orElseThrow(noStudentWithId(id));
355 
356       File outFile = new File(getDirectoryForReportFileForStudent(outDir, student), getReportFileName(id));
357       try {
358         PrintWriter pw =
359           new PrintWriter(new FileWriter(outFile), true);
360         dumpReportTo(book, student, pw, assignLetterGrades);
361 
362 //         dumpReportTo(book, student, out);
363       } catch (IOException ex) {
364         printErrorMessageAndExit("While writing report to " + outFile, ex);
365       }
366     }
367   }

369   private static File getDirectoryForReportFileForStudent(File parentDirectory, Student student) {
370     String directoryName;
371     Student.Section enrolledSection = student.getEnrolledSection();
372     switch (enrolledSection) {
373       case UNDERGRADUATE:
374         directoryName = UNDERGRADUATE_DIRECTORY_NAME;
375         break;
376       case GRADUATE:
377         directoryName = GRADUATE_DIRECTORY_NAME;
378         break;
379       default:
380         throw new IllegalStateException("Don't know directory name for " + enrolledSection);
381     }
382     File directory = new File(parentDirectory, directoryName);
383     directory.mkdirs();
384     return directory;
385   }

392   private static GradeBook parseGradeBook(File gradeBookFile) {
393     GradeBook book = null;
394     try {
395       err.println("Parsing " + gradeBookFile);
396       XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
397       book = parser.parse();
398 
399     } catch (FileNotFoundException ex) {
400       printErrorMessageAndExit("** Could not find grade book file: " + gradeBookFile, ex);
401 
402     } catch (ParserException | IOException ex) {
403       printErrorMessageAndExit("** Exception while parsing " + gradeBookFile, ex);
404     }
405     return book;
406   }
