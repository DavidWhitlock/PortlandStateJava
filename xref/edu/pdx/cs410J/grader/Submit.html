186     validateProjectName();
187

190     }

192     if (userName == null) {
193       throw new IllegalStateException("Missing student name");
194     }
195 
196     if (userId == null) {
197       throw new IllegalStateException("Missing login id");
198     }
199 
200     if (isNineDigitStudentId(userId)) {
201       throw new IllegalStateException(loginIdShouldNotBeStudentId(userId));
202     }
203 
204     if (looksLikeAnEmailAddress(userId)) {
205       throw new IllegalStateException(loginIdShouldNotBeEmailAddress(userId));
206     }
207 
208     if (userEmail == null) {
209       throw new IllegalStateException("Missing email address");
210 
211     } else {
212       // Make sure user's email is okay
213       try {
214         new InternetAddress(userEmail);
215 
216       } catch (AddressException ex) {
217         String s = "Invalid email address: " + userEmail;
218         IllegalStateException ex2 = new IllegalStateException(s);
219         ex2.initCause(ex);
220         throw ex2;
221       }
222     }

255   /**
256    * Submits the project to the grader
257    *
258    * @param verify Should the user be prompted to verify his submission?
259    * @return Whether or not the submission actually occurred
260    * @throws IllegalStateException If no source files were found
261    */
262   public boolean submit(boolean verify) throws IOException, MessagingException {
263     // Recursively search the source directory for .java files
264     Set<File> sourceFiles = searchForSourceFiles(fileNames);
265 
266     db(sourceFiles.size() + " source files found");
267 
268     if (sourceFiles.size() == 0) {
269       String s = "No source files were found.";
270       throw new IllegalStateException(s);
271     } else if (sourceFiles.size() < 3) {
272       String s = "Too few source files were submitted. Each project requires at least 3 files be" + 
273                  " submitted. You only submitted " + sourceFiles.size() + " files";
274       throw new IllegalStateException(s);
275     }
276 
277     // Verify submission with user
278     if (verify && !verifySubmission(sourceFiles)) {
279       // User does not want to submit
280       return false;
281     }
282 
283     // Timestamp
284     this.submitTime = LocalDateTime.now();
285 
286     // Create a temporary zip file to hold the source files
287     File zipFile = makeZipFileWith(sourceFiles);
288 
289     // Send the zip file as an email attachment to the TA
290     mailTA(zipFile, sourceFiles);
291 
292     if (sendReceipt) {
293       mailReceipt(sourceFiles);
294     }
295 
296     return true;
297   }

308   /**
309    * Searches for the files given on the command line.  Ignores files
310    * that do not end in .java, or that appear on the "no submit" list.
311    * Files must reside in a directory named
312    * edu/pdx/cs410J/<studentId>.
313    */
314   private Set<File> searchForSourceFiles(Set<String> fileNames) {
315     // Files should be sorted by name
316     SortedSet<File> files =
317       new TreeSet<>((o1, o2) -> o1.toString().compareTo(o2.toString()));
318     populateWithFilesFromSubdirectories(files, fileNames);
319 
320     files.removeIf((file) -> !canBeSubmitted(file));
321 
322     return files;
323   }

325   private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Set<String> fileNames) {
326     populateWithFilesFromSubdirectories(allFiles, fileNames.stream().map(name -> new File(name).getAbsoluteFile()));
327   }

329   private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Stream<File> files) {
330     files.forEach((file) -> {
331       if (file.isDirectory()) {
332         populateWithFilesFromSubdirectories(allFiles, Arrays.stream(file.listFiles()));
333 
334       } else {
335         allFiles.add(file);
336       }
337     });
338 
339   }

341   private boolean canBeSubmitted(File file) {
342     if (!file.exists()) {
343       err.println("** Not submitting file " + file +
344         " because it does not exist");
345       return false;
346     }
347 
348     // Is the file on the "no submit" list?
349     List<String> noSubmit = fetchListOfFilesThatCanNotBeSubmitted();
350     String name = file.getName();
351     if (noSubmit.contains(name)) {
352       err.println("** Not submitting file " + file +
353         " because it is on the \"no submit\" list");
354       return false;
355     }
356 
357     // Does the file name end in .java?
358     if (!name.endsWith(".java")) {
359       err.println("** Not submitting file " + file +
360         " because does end in \".java\"");
361       return false;
362     }
363 
364     // Verify that file is in the correct directory.
365     if (!isInAKoansDirectory(file) && !isInEduPdxCs410JDirectory(file)) {
366       err.println("** Not submitting file " + file +
367         ": it does not reside in a directory named " +
368         "edu" + File.separator + "pdx" + File.separator +
369         "cs410J" + File.separator + userId + " (or in one of the koans directories)");
370       return false;
371     }
372 
373     return true;
374   }

376   private boolean isInAKoansDirectory(File file) {
377     boolean isInAKoansDirectory = hasParentDirectories(file, "beginner") ||
378       hasParentDirectories(file, "intermediate") ||
379       hasParentDirectories(file, "advanced") ||
380       hasParentDirectories(file, "java7") ||
381       hasParentDirectories(file, "java8");
382     if (isInAKoansDirectory) {
383       this.isSubmittingKoans = true;
384       db(file + " is in a koans directory");
385     }
386     return isInAKoansDirectory;
387   }

398   private boolean hasParentDirectories(File file, String... parentDirectoryNames) {
399     File parent = file.getParentFile();
400 
401     // Skip over subpackages
402     while (parent != null && !parent.getName().equals(parentDirectoryNames[0])) {
403       parent = parent.getParentFile();
404     }
405 
406     for (String parentDirectoryName : parentDirectoryNames) {
407       if (parent == null || !parent.getName().equals(parentDirectoryName)) {
408         return false;
409 
410       } else {
411         parent = parent.getParentFile();
412       }
413     }
414 
415     return true;
416   }

422   private List<String> fetchListOfStringsFromUrl(String listUrl) {
423     List<String> noSubmit = new ArrayList<>();
424 
425     try {
426       URL url = new URL(listUrl);
427       InputStreamReader isr = new InputStreamReader(url.openStream());
428       BufferedReader br = new BufferedReader(isr);
429       while (br.ready()) {
430         noSubmit.add(br.readLine().trim());
431       }
432 
433     } catch (MalformedURLException ex) {
434       err.println("** WARNING: Cannot access " + listUrl + ": " +
435         ex.getMessage());
436 
437     } catch (IOException ex) {
438       err.println("** WARNING: Problems while reading " + listUrl + ": " + ex.getMessage());
439     }
440     return noSubmit;
441   }

471     boolean wasMainProjectClassSubmitted = sourceFiles.stream().anyMatch((f) -> f.getName().contains(this.projName));
472     if (!wasMainProjectClassSubmitted && !this.isSubmittingKoans) {
473       String mainProjectClassName = this.projName + ".java";
474       out.println("*** WARNING: You are submitting " + this.projName +
475         ", but did not include " + mainProjectClassName + ".\n" +
476         "    You might want to check the name of the project or the files you are submitting.\n");

522   /**
523    * Creates a Zip file that contains the source files.  The Zip File

526   private File makeZipFileWith(Set<File> sourceFiles) throws IOException {
527     String zipFileName = userName.replace(' ', '_') + "-TEMP";
528     File zipFile = File.createTempFile(zipFileName, ".zip");
529     if (!saveZip) {
530       zipFile.deleteOnExit();
531 
532     } else {
533       out.println("Saving temporary Zip file: " + zipFile);
534     }
535 
536     db("Created Zip file: " + zipFile);
537 
538     Map<File, String> sourceFilesWithNames =
539       sourceFiles.stream().collect(Collectors.toMap(file -> file, this::getRelativeName));
540 
541     new ZipFileOfFilesMaker(sourceFilesWithNames, zipFile, getManifestEntries()).makeZipFile();
542     return zipFile;
543   }

566     Multipart mp = new MimeMultipart();

568     mp.addBodyPart(filePart);

571 
572     out.println("Submitting project to Grader");
573 
574     Transport.send(message);
575   }

577   private MimeBodyPart createZipAttachment(File zipFile) throws MessagingException {
578     // Now attach the Zip file
579     DataSource ds = new FileDataSource(zipFile) {
580       @Override
581       public String getContentType() {
582         return "application/zip";
583       }
584     };
585     DataHandler dh = new DataHandler(ds);
586     MimeBodyPart filePart = new MimeBodyPart();
587 
588     String zipFileTitle = userName.replace(' ', '_') + ".zip";
589 
590     filePart.setDataHandler(dh);
591     filePart.setFileName(zipFileTitle);
592     filePart.setDescription(userName + "'s " + projName);
593 
594     return filePart;
595   }

597   private MimeBodyPart createTextPartOfTAEmail(Set<File> sourceFiles) throws MessagingException {

605     }
606     text.append("Contents:\n");
607

609       text.append("  ").append(getRelativeName(file)).append("\n");
610     }
611     text.append("\n\n");
612 
613     MimeBodyPart textPart = new MimeBodyPart();
614     textPart.setContent(text.toString(), "text/plain");
615 
616     // Try not to display text as separate attachment
617     textPart.setDisposition("inline");
