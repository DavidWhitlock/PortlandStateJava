3   import com.google.common.collect.Sets;
4   import com.google.common.io.ByteStreams;
5   
6   import java.io.*;
7   import java.util.Date;
8   import java.util.HashMap;
9   import java.util.Map;
10  import java.util.Set;
11  import java.util.jar.Attributes;
12  import java.util.jar.JarFile;
13  import java.util.jar.Manifest;
14  import java.util.stream.Collectors;
15  import java.util.zip.ZipEntry;
16  import java.util.zip.ZipOutputStream;
17  
18  class ZipMaker {
19  
20    private final Map<Attributes.Name, String> manifestEntries;
21    private final Map<File, String> sourceFilesAndNames;
22    private final File zipFile;
23  
24    public ZipMaker(Map<File, String> sourceFilesAndNames, File zipFile, Map<Attributes.Name, String> manifestEntries) {
25      this.sourceFilesAndNames = sourceFilesAndNames;
26      this.zipFile = zipFile;
27      this.manifestEntries = manifestEntries;
28    }
29  
30    public File makeZipFile() throws IOException {
31      ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(zipFile));
32      zos.setMethod(ZipOutputStream.DEFLATED);
33  
34      // Create a Manifest for the Zip file containing the name of the
35      // author (userName) and a version that is based on the current
36      // date/time.
37      writeManifestAsEntryInZipFile(zos);
38  
39      // Add the source files to the Zip
40      for (Map.Entry<File, String> fileEntry : sourceFilesAndNames.entrySet()) {
41        File file = fileEntry.getKey();
42        String fileName = fileEntry.getValue();
43        System.out.println("Adding " + fileName + " to zip");
44        ZipEntry entry = new ZipEntry(fileName);
45        entry.setTime(file.lastModified());
46        entry.setSize(file.length());
47  
48        entry.setMethod(ZipEntry.DEFLATED);
49  
50        // Add the entry to the ZIP file
51        zos.putNextEntry(entry);
52  
53        ByteStreams.copy(new FileInputStream(file), zos);
54  
55        zos.closeEntry();
56      }
57  
58      zos.close();
59  
60      return zipFile;
61    }
62  
63    private void writeManifestAsEntryInZipFile(ZipOutputStream zos) throws IOException {
64      Manifest manifest = new Manifest();
65      addEntriesToMainManifest(manifest);
66  
67      String entryName = JarFile.MANIFEST_NAME;
68  
69      System.out.println("Adding " + entryName + " to zip");
70      ZipEntry entry = new ZipEntry(entryName);
71      entry.setTime(System.currentTimeMillis());
72      entry.setMethod(ZipEntry.DEFLATED);
73  
74      zos.putNextEntry(entry);
75      manifest.write(new BufferedOutputStream(zos));
76      zos.closeEntry();
77    }
78  
79    private void addEntriesToMainManifest(Manifest manifest) {
80      Attributes attrs = manifest.getMainAttributes();
81  
82      // If a manifest doesn't have a version, the other attributes won't get written out.  Lame.
83      attrs.put(Attributes.Name.MANIFEST_VERSION, new Date().toString());
84  
85      for (Map.Entry<Attributes.Name, String> entry : this.manifestEntries.entrySet()) {
86        attrs.put(entry.getKey(), entry.getValue());
87      }
88    }
89  
90    public static void main(String[] args) throws IOException {
91      String zipFileName = null;
92      Set<File> files = Sets.newHashSet();
93  
94      for (String arg : args) {
95        if (zipFileName == null) {
96          zipFileName = arg;
97  
98        } else {
99          File file = new File(arg);
100         if (file.exists()) {
101           files.add(file);
102         }
103       }
104     }
105 
106     if (zipFileName == null) {
107       usage("Missing zip file name");
108     }
109 
110     if (files.isEmpty()) {
111       usage("Missing files");
112     }
113 
114     Map<File, String> sourceFilesAndNames =
115       files.stream().collect(Collectors.toMap(file -> file, File::getPath));
116 
117     assert zipFileName != null;
118     File zipFile = new File(zipFileName);
119 
120     Map<Attributes.Name, String> manifestEntries = new HashMap<>();
121     manifestEntries.put(new Attributes.Name("Created-By"), System.getProperty("user.name"));
122     manifestEntries.put(Attributes.Name.MANIFEST_VERSION, new Date().toString());
123 
124     new ZipMaker(sourceFilesAndNames, zipFile, manifestEntries).makeZipFile();
125   }
126 
127   private static void usage(String message) {
128     PrintStream err = System.err;
129     err.println("** " + message);
130     err.println("java ZipMaker zipFileName files+");
131     err.println("  ziprFileName    The name of the zip file to create");
132     err.println("  files+         One or more files to include in the zip");
133     err.println();
134 
135     System.exit(1);
136   }
