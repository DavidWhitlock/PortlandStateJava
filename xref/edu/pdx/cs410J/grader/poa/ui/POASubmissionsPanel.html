1   package edu.pdx.cs410J.grader.poa.ui;
2   
3   import com.google.inject.Singleton;
4   import edu.pdx.cs410J.grader.poa.POASubmissionsView;
5   
6   import javax.swing.*;
7   import javax.swing.event.ListSelectionEvent;
8   import java.awt.*;
9   import java.util.List;
10  import java.util.Vector;
11  
12  @Singleton
13  public class POASubmissionsPanel extends JPanel implements POASubmissionsView {
14  
15    private final JList<String> submissions;
16    private final JButton downloadSubmissions;
17  
18    public POASubmissionsPanel() {
19      submissions = new JList<>();
20      submissions.setVisibleRowCount(10);
21  
22      this.setLayout(new BorderLayout());
23  
24      this.add(new JScrollPane(submissions), BorderLayout.CENTER);
25  
26      downloadSubmissions = new JButton("Download");
27      this.add(downloadSubmissions, BorderLayout.SOUTH);
28    }
29  
30    @Override
31    public void setPOASubmissionsDescriptions(List<String> strings) {
32      submissions.setListData(new Vector<>(strings));
33    }
34  
35    @Override
36    public void addSubmissionSelectedListener(POASubmissionSelectedListener listener) {
37      submissions.addListSelectionListener(e -> {
38        if (isFinalEventInUserSelection(e)) {
39          int selectedIndex = submissions.getSelectedIndex();
40          if (selectedIndex >= 0) {
41            listener.submissionSelected(selectedIndex);
42          }
43        }
44      });
45    }
46  
47    @Override
48    public void addDownloadSubmissionsListener(DownloadSubmissionsListener listener) {
49      this.downloadSubmissions.addActionListener(e -> listener.downloadSubmissions());
50    }
51  
52    @Override
53    public void selectPOASubmission(int index) {
54      this.submissions.setSelectedIndex(index);
55    }
56  
57    private boolean isFinalEventInUserSelection(ListSelectionEvent e) {
58      return !e.getValueIsAdjusting();
59    }
60  }
