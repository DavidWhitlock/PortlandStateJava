1   package edu.pdx.cs410J.grader.poa;
2   
3   import com.google.common.eventbus.EventBus;
4   import com.google.common.eventbus.Subscribe;
5   import com.google.common.eventbus.SubscriberExceptionContext;
6   import com.google.common.eventbus.SubscriberExceptionHandler;
7   
8   public class EventBusThatPublishesUnhandledExceptionEvents extends EventBus {
9   
10    /** We have to use this hokey second event bus because the
11     * SubscriberExceptionHandler (which posts the UnhandledExceptionEvent)
12     * must be created in the constructor before the bus, itself, is
13     * fully initialized and available for use. */
14    private static final EventBus errorMessageBus = new EventBus("Error Messages");
15  
16    public EventBusThatPublishesUnhandledExceptionEvents() {
17      super(new SubscriberExceptionHandler() {
18        @Override
19        public void handleException(Throwable throwable, SubscriberExceptionContext subscriberExceptionContext) {
20          postUncaughtExceptionEvent(throwable);
21        }
22      });
23  
24      errorMessageBus.register(this);
25    }
26  
27    @Subscribe
28    public void postUncaughtExceptionEventToErrorMessageBus(UnhandledExceptionEvent event) {
29      this.post(event);
30    }
31  
32    private static void postUncaughtExceptionEvent(Throwable throwable) {
33      errorMessageBus.post(new UnhandledExceptionEvent(throwable));
34    }
35  }
