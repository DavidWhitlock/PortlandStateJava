1   package edu.pdx.cs410J.grader.poa;

3   import com.google.common.eventbus.EventBus;
4   import com.google.common.eventbus.Subscribe;
5   import com.google.inject.Inject;
6   import edu.pdx.cs410J.ParserException;
7   import edu.pdx.cs410J.grader.gradebook.GradeBook;
8   import edu.pdx.cs410J.grader.gradebook.XmlDumper;
9   import edu.pdx.cs410J.grader.gradebook.XmlGradeBookParser;
10  
11  import java.io.File;
12  import java.io.IOException;
13  
14  public class GradeBookFileManager {
15    private final EventBus bus;
16    private File file;
17  
18    @Inject
19    public GradeBookFileManager(EventBus bus) {
20      this.bus = bus;
21      this.bus.register(this);
22    }
23  
24    @Subscribe
25    public void loadGradeBookFromFile(LoadGradeBook event) throws IOException, ParserException {
26      file = event.getFile();
27      XmlGradeBookParser parser = new XmlGradeBookParser(file);
28      GradeBook book = parser.parse();
29  
30      this.bus.post(new GradeBookLoaded(book));
31    }
32  
33    @Subscribe
34    public void writeGradeBookToFile(SaveGradeBook event) throws IOException {
35      XmlDumper dumper = new XmlDumper(this.file);
36      dumper.dump(event.getGradeBook());
37    }
38  }
