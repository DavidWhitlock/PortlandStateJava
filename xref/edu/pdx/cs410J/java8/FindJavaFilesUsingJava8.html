3   import java.io.File;
4   import java.util.Arrays;
5   
6   /**
7    * This program demonstrates how Lambdas in Java 8 greatly simplify using
8    * "functional interfaces" like <code>FileFilter</code> and
9    * <code>FileNameFilter</code>.
10   *
11   * @since Summer 2014
12   */
13  public class FindJavaFilesUsingJava8 {
14  
15    /**
16     * Prints out the names all of the Java source files in a directory
17     * and then recurses over subdirectories.
18     */
19    private static void findJavaFiles(File directory) {
20      File[] javaFiles = directory.listFiles((dir, name) -> name.endsWith(".java"));
21      Arrays.asList(javaFiles).forEach(System.out::println);
22  
23      File[] subdirectories = directory.listFiles(File::isDirectory);
24      Arrays.asList(subdirectories).forEach(FindJavaFilesUsingJava8::findJavaFiles);
25    }
26  
27    /**
28     * The one command line parameter is the directory in which to start
29     * the search.
30     */
31    public static void main(String[] args) {
32      File file = new File(args[0]);
33      if (file.isDirectory()) {
34        // Look for Java files
35        findJavaFiles(file);
36  
37      } else {
38        System.err.println(file + " is not a directory");
39      }
40    }
41  
42  }
