5   /**
6    * This class provides an example of anonymous inner classes.  It uses
7    * inner classes to provide implementations of the {@link FileFilter}
8    * and {@link FilenameFilter} interfaces that find all files whose
9    * name ends with <code>.class</code>.
10   *
11   * @see edu.pdx.cs410J.core.FindJavaFiles
12   *
13   * @author David Whitlock
14   * @version $Revision: 1.2 $
15   *
16   * @since Spring 2003
17   */
18  public class FindClassFiles {
19    private static void findFiles(File dir) {
20      final String suffix = ".class";
21  
22      File[] classFiles = 
23        dir.listFiles(new FilenameFilter() {
24          public boolean accept(File dir, String name) {
25            return name.endsWith(suffix);
26          }
27        });
28  
29      for (int i = 0; i < classFiles.length; i++)
30        System.out.println(classFiles[i]);
31  
32      File[] subdirs = 
33        dir.listFiles(new FileFilter() {
34          public boolean accept(File file) {
35            return file.isDirectory();
36          }
37        });
38  
39      for (int i = 0; i < subdirs.length; i++)
40        findFiles(subdirs[i]);
41    }
42  }
