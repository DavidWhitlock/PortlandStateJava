3   import java.io.File;
4   import java.net.*;
5   
6   /**
7    * This <code>ClassLoader</code> prints out the name of every class
8    * that it (or its parent) loads.
9    */
10  public class LoggingClassLoader extends URLClassLoader {
11  
12    /**
13     * Creates a <code>LoggingClassLoader</code> that loads classes from
14     * a given array of URLs.  However, before this class loader
15     * attempts to load the class, it will delegate to its parent class
16     * loader.
17     */
18    public LoggingClassLoader(URL[] urls, ClassLoader parent) {
19      super(urls, parent);
20    }
21  
22    /**
23     * Invoked as this class loader is loading a class
24     */
25    public Class loadClass(String className) 
26      throws ClassNotFoundException {
27  
28      System.out.println("Loading " + className);
29      return super.loadClass(className);
30    }
31    
32    /**
33     * Main program that uses a <code>LoggingClassLoader</code> to load
34     * the class with the given name from a given location.
35     */
36    public static void main(String[] args) {
37      File file = new File(args[0]);
38      String className = args[1];
39  
40      try {
41        URL[] urls = new URL[] { file.toURL() };
42        ClassLoader parent = ClassLoader.getSystemClassLoader();
43        ClassLoader cl = new LoggingClassLoader(urls, parent);
44        cl.loadClass(className);
45  
46      } catch (MalformedURLException ex) {
47        String s = "Bad URL: " + ex;
48        System.err.println(s);
49  
50      } catch (ClassNotFoundException ex) {
51        String s = "Could not find class " + className;
52        System.err.println(s);
53      }
54    }
55  
56  }
