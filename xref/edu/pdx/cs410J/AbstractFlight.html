6   /**
7    * This class represents an airline flight.  Each flight has a unique
8    * number identifying it, an origin airport identified by the
9    * airport's three-letter code, a departure time, a destination
10   * airport identified by the airport's three-letter code, and an
11   * arrival time.
12   */
13  public abstract class AbstractFlight implements Serializable {
14  
15    /**
16     * Returns a number that uniquely identifies this flight.
17     */
18    public abstract int getNumber();
19  
20    /**
21     * Returns the three-letter code of the airport at which this flight
22     * originates.
23     */
24    public abstract String getSource();
25  
26    /**
27     * Returns this flight's departure time as a <code>Date</code>.
28     */
29    public Date getDeparture() {
30      return null;
31    }
32  
33    /**
34     * Returns a textual representation of this flight's departure
35     * time.
36     */
37    public abstract String getDepartureString();
38  
39    /**
40     * Returns the three-letter code of the airport at which this flight
41     * terminates.
42     */
43    public abstract String getDestination();
44  
45    /**
46     * Returns this flight's arrival time as a <code>Date</code>.
47     */
48    public Date getArrival() {
49      return null;
50    }
51  
52    /**
53     * Returns a textual representation of this flight's arrival time.
54     */
55    public abstract String getArrivalString();
56  
57    /**
58     * Returns a brief textual description of this flight.
59     */
60    public final String toString() {
61      return "Flight " + this.getNumber() + " departs " + this.getSource()
62             + " at " + this.getDepartureString() + " arrives " +
63             this.getDestination() + " at " + this.getArrivalString();
64    }
65  }
