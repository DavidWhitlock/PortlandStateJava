3   import jakarta.activation.DataHandler;
4   import jakarta.activation.DataSource;
5   import jakarta.activation.FileDataSource;
6   import jakarta.mail.*;
7   import jakarta.mail.internet.*;
8   
9   import java.io.File;
10  import java.io.PrintWriter;
11  import java.util.Properties;
12  
13  /**
14   * This program uses the JavaMail API to send a file to someone as a
15   * MIME attachment to an email.  More information about the JavaMail
16   * API is available from http://java.sun.com/products/javamail.
17   *
18   * @author David Whitlock
19   */
20  public class MailFile {
21  
22    private static final PrintWriter err = new PrintWriter(System.err, true);
23  
24    /**
25     * Prints out information about how to use this program.
26     */
27    private static void usage() {
28      err.println("usage: MailFile [options] file recipient [subject]");
29      err.println("  Where [options] are:");
30      err.println("  -server name   Name of the SMTP server");
31      err.println("  -verbose       Print out extra info");
32      System.exit(1);
33    }
34  
35    /**
36     * Reads the name of the file to be sent, the recipient of the
37     * email, and the subject of the email (which may contain multiple
38     * words) from the command line.  Then uses the JavaMail API to
39     * construct an email message.
40     */
41    public static void main(String[] args) {
42      StringBuilder subject = null;
43      String fileName = null;
44      String recipient = null;
45      String serverName = "mailhost.pdx.edu";
46      boolean debug = false;
47  
48      // Parse the command line
49      for (int i = 0; i < args.length; i++) {
50        if (args[i].equals("-server")) {
51  	if(++i >= args.length) {
52  	  err.println("** Missing server name");
53  	  usage();
54  	}
55  
56  	serverName = args[i];
57  
58        } if (args[i].equals("-verbose")) {
59  	debug = true;
60  
61        } else if (fileName == null) {
62  	fileName = args[i];
63  
64        } else if (recipient == null) {
65  	recipient = args[i];
66  
67        } else {
68  	// Part of the subject
69  	if(subject == null) {
70  	  subject = new StringBuilder();
71  	}
72  
73  	subject.append(args[i]).append(" ");
74        }
75      }
76  
77      // Check to make sure everything is okay
78      if (fileName == null) {
79        err.println("** No file specified");
80        usage();
81      }
82  
83      if (recipient == null) {
84        err.println("** No recipient specified");
85        usage();
86      }
87  
88      if (subject == null) {
89        // Default subject
90        subject = new StringBuilder("A file for you");
91      }
92  
93      // Obtain a Session for sending email
94      Properties props = new Properties();
95      props.put("mail.smtp.host", serverName);
96      Session session = Session.getDefaultInstance(props, null);
97      session.setDebug(debug);
98  
99      MimeMessage message = new MimeMessage(session);
100 
101     try {
102       InternetAddress[] to = {new InternetAddress(recipient)};
103       message.setRecipients(Message.RecipientType.TO, to);
104 
105     } catch (AddressException ex) {
106       err.println("** Invalid address: " + recipient);
107       System.exit(1);
108 
109     } catch (MessagingException ex) {
110       err.println("** MessagingException: " + ex);
111       System.exit(1);
112     }
113     
114     File file = new File(fileName);
115     if (!file.exists()) {
116       err.println("** File " + file + " does not exist!");
117       System.exit(1);
118     }
119 
120     // This will be a multi-part MIME message
121     MimeBodyPart textPart = new MimeBodyPart();
122     try {
123       textPart.setContent("File " + file, "text/plain");
124 
125     } catch (MessagingException ex) {
126       err.println("** Exception with text part: " + ex);
127       System.exit(1);
128     }
129 
130     // Read in the file and make a MimeBodyPart out of it
131     DataSource ds = new FileDataSource(file);
132     DataHandler dh = new DataHandler(ds);
133     MimeBodyPart filePart = new MimeBodyPart();
134     try {
135       filePart.setDataHandler(dh);
136       filePart.setFileName(file.getName());
137       filePart.setDescription("The file you requested");
138 
139     } catch (MessagingException ex) {
140       err.println("** Exception with file part: " + ex);
141       System.exit(1);
142     }
143 
144     try {
145       Multipart mp = new MimeMultipart();
146       mp.addBodyPart(textPart);
147       mp.addBodyPart(filePart);
148 
149       message.setContent(mp);
150 
151       // Finally, send the message
152       Transport.send(message);
153 
154     } catch (MessagingException ex) {
155       err.println("** Exception while adding parts and sending: " +
156 		  ex);
157       System.exit(1);
158     }
159     
160   }
