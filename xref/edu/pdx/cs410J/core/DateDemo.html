1   package edu.pdx.cs410J.core;

3   import java.text.*;
4   import java.util.*;
5   
6   /**
7    * This class demonstrates how to use the Java's day and time
8    * facilities. 
9    */
10  public class DateDemo {
11  
12    private static DateFormat dfShort;
13    private static DateFormat dfMedium;
14    private static DateFormat dfLong;
15    private static DateFormat dfFull;
16  
17    /**
18     * This main method works with dates.  If there are any arguments on
19     * the command line, they are interpreted as a date to be parsed.
20     * Otherwise, the current day/time are used.
21     */
22    public static void main(String[] args) {
23  
24      // Set up the DateFormats
25      dfShort = DateFormat.getDateTimeInstance(DateFormat.SHORT,
26  					     DateFormat.SHORT );
27      dfMedium = DateFormat.getDateTimeInstance(DateFormat.MEDIUM,
28  					      DateFormat.MEDIUM);
29      dfLong = DateFormat.getDateTimeInstance(DateFormat.LONG,
30  					    DateFormat.LONG);
31      dfFull = DateFormat.getDateTimeInstance(DateFormat.FULL,
32  					    DateFormat.FULL);
33      
34      Date day = null;
35  
36      if (args.length == 0) {
37        // Use the current day/time
38        day = new Date();
39  
40      } else {
41        // Parse the command line as if it were a date
42        StringBuffer sb = new StringBuffer();
43        for (int i = 0; i < args.length; i++) {
44  	sb.append(args[i]);
45  	sb.append(' ');
46        }
47  
48        try {
49  	day = dfShort.parse(sb.toString());
50  
51        } catch (ParseException ex) {
52  	System.err.println("** Malformatted date: " + sb);
53  	System.exit(1);
54        }
55      }
56  
57      // Print out the date in several formats
58      System.out.println("Short: " + dfShort.format(day));
59      System.out.println("Medium: " + dfMedium.format(day));
60      System.out.println("Long: " + dfLong.format(day));
61      System.out.println("Full: " + dfFull.format(day));
62  
63    }
64  
65  }
