11    /**
12     * Print out a couple of the system properties and check to see if
13     * the "edu.pdx.cs410J.Debug" property has been set on the command
14     * line.
15     */
16    public static void main(String[] args) {
17      // Print out some properties
18      Properties props = System.getProperties();
19      props.list(System.out);
20  
21      // Is the "edu.pdx.cs410J.Debug" property set?
22      String name = "edu.pdx.cs410J.Debug";
23      boolean debug = Boolean.getBoolean(name);
24      System.out.print("\nAre we debugging? ");
25      System.out.println((debug ? "Yes." : "No."));
26    }
