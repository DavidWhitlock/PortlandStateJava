5   /**
6    * Compares two boxes of ceral by their price
7    */
8   public class CerealComparator implements Comparator<Cereal> {
9     public int compare(Cereal o1, Cereal o2) {
10      double price1 = o1.getPrice();
11      double price2 = o2.getPrice();
12  
13      if (price1 > price2) {
14        return 1;
15      } else if (price1 < price2) {
16        return -1;
17      } else {
18        return 0;
19      }
20    }
21
