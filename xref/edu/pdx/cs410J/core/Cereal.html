3   import java.util.*;
4   
5   /**
6    * This class represents a box of cereal.  It has a name and a price.
7    */
8   public class Cereal implements Comparable<Cereal> {
9     private String name;
10    private double price;
11  
12    /**
13     * Creates a new box of cereal
14     */
15    public Cereal(String name, double price) {
16      this.name = name;
17      this.price = price;
18    }
19  
20    /**
21     * Returns the name of this cereal
22     */
23    public String getName() {
24      return this.name;
25    }
26  
27    /**
28     * Returns the price of this cereal
29     */
30    public double getPrice() {
31      return this.price;
32    }
33  
34    public String toString() {
35      return this.name + " $" + this.price;
36    }
37  
38    /**
39     * Compares two <code>Cereal</code>s based on their name
40     */
41    public int compareTo(Cereal c2) {
42      return this.getName().compareTo(c2.getName());
43    }
44  
45    /**
46     * Two be consistent with the natural ordering, two
47     * <code>Cereal</code>s that have the same name, are themselves the
48     * same.
49     */
50    public boolean equals(Object o) {
51      if (o instanceof Cereal) {
52        Cereal other = (Cereal) o;
53        return this.getName().equals(other.getName());
54      }
55      return false;
56    }
57  
58    /**
59     * Two cereals that are equal must have the same hash code.
60     */
61    public int hashCode() {
62      return this.getName().hashCode();
63    }
64  
65    /**
66     * Demonstrates the natural ordering of <code>Cereals</code> by
67     * adding a bunch of cereals to a {@link SortedSet}
68     */
69    public static void main(String[] args) {
70      SortedSet<Cereal> set = new TreeSet<Cereal>();
71      set.add(new Cereal("Total", 3.56));
72      set.add(new Cereal("Raisin Bran", 2.65));
73      set.add(new Cereal("Sugar Crisps", 2.38));
74  
75      for (Cereal c : set ) {
76        System.out.println(c);
77      }
78    }
79  
80  }
