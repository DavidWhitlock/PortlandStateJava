3   import java.io.FileInputStream;
4   import java.io.IOException;
5   import java.io.ObjectInputStream;
6   import java.util.Date;
7   
8   /**
9    * This class demonstrates object serialization by reading an instance
10   * of <code>Date</code> from a file.
11   */
12  public class ReadDate {
13  
14    /**
15     * Read a <code>Date</code> instance from th file whose name is
16     * specified on the command line.
17     */
18    public static void main(String[] args) {
19      String fileName = args[0];
20  
21      Date date = null;
22      try {
23        FileInputStream fis = new FileInputStream(fileName);
24        ObjectInputStream in = new ObjectInputStream(fis);
25        date = (Date) in.readObject();
26        in.close();
27        System.out.println("Read " + date);
28  
29      } catch (ClassNotFoundException ex) {
30        System.err.println("** No class " + ex);
31        System.exit(1);
32  
33      } catch (IOException ex) {
34        System.err.println("**IOException: " + ex);
35        System.exit(1);
36      }
37    }
38  
39  }
