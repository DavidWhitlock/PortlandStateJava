3   import java.io.*;
4   
5   /**
6    * This program reads <code>double</code>s from the command line and
7    * writes them to a file.  It demonstrates the
8    * <code>FileOutputStream</code> and <code>DataOutputStream</code>
9    * classes.
10   */
11  public class WriteDoubles {
12  
13    static PrintStream err = System.err;
14  
15    /**
16     * The first command line argument is the name of the file, the
17     * remaining are the doubles
18     */
19    public static void main(String[] args) {
20      // Create a FileOutputStream
21      FileOutputStream fos = null;
22      try {
23        fos = new FileOutputStream(args[0]);
24  
25      } catch (FileNotFoundException ex) {
26        err.println("** No such file: " + args[0]);
27        System.exit(1);
28      }
29  
30      // Wrap a DataOutputStream around the FileOutputStream
31      DataOutputStream dos = new DataOutputStream(fos);
32  
33      // Write the doubles to the DataOutputStream
34      for (int i = 1; i < args.length; i++) {
35        try {
36  	double d = Double.parseDouble(args[i]);
37  	dos.writeDouble(d);
38  
39        } catch (NumberFormatException ex) {
40  	err.println("** Not a double: " + args[i]);
41  
42        } catch (IOException ex) {
43  	err.println("** " + ex);
44  	System.exit(1);
45        }
46      }
47    }
48  }
