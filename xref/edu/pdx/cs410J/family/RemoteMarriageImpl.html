3   import java.rmi.*;
4   import java.util.Date;
5   
6   /**
7    * This is class implements the <code>RemoteMarriage</code> interface.
8    * Basically, it delegates all of its behavior to an underlying {@link
9    * edu.pdx.cs410J.family.Marriage} that lives only on the server.
10   */
11  @SuppressWarnings("serial")
12  class RemoteMarriageImpl extends java.rmi.server.UnicastRemoteObject
13    implements RemoteMarriage {
14  
15    /** The underlying Marriage that is being modeled. */
16    private transient Marriage marriage;
17  
18    //////////////////////  Constructors  //////////////////////
19  
20    /**
21     * Creates a new <code>RemoteMarriageImpl</code> that delegates most
22     * of its behavior to a given <code>Marriage</code>
23     */
24    RemoteMarriageImpl(Marriage marriage) throws RemoteException {
25      this.marriage = marriage;
26    }
27  
28    ////////////////////  Instance Methods  ////////////////////
29  
30    public int getHusbandId() throws RemoteException {
31      return this.marriage.getHusband().getId();
32    }
33  
34    public int getWifeId() throws RemoteException {
35      return this.marriage.getWife().getId();
36    }
37  
38    public Date getDate() throws RemoteException {
39      return this.marriage.getDate();
40    }
41  
42    public void setDate(Date date) throws RemoteException {
43      this.marriage.setDate(date);
44    }
45  
46    public String getLocation() throws RemoteException {
47      return this.marriage.getLocation();
48    }
49  
50    public void setLocation(String location) throws RemoteException {
51      this.marriage.setLocation(location);
52    }
53  
54    public String getDescription() throws RemoteException {
55      return this.marriage.toString();
56    }
57  
58  }
