33    /**
34     * Adds the GUI components to a <code>FamilyTreePanel</code>
35     */
36    void addComponents() {
37      System.out.println("Creating Panel");
38  
39      // Set up the content
40      Dimension minSizeLeft = new Dimension(50, 100);
41      this.treeList = new FamilyTreeList();
42      this.treeList.setToolTipText("Click to select a person in the " +
43                                   "family tree");
44      this.treeList.addListSelectionListener(new ListSelectionListener()
45        {
46          public void valueChanged(ListSelectionEvent e) {
47            Person person =
48              FamilyTreePanel.this.treeList.getSelectedPerson();
49            showPerson(person);
50          }
51        });
52      this.treeList.setMinimumSize(minSizeLeft);
53      JScrollPane scrollPane = new JScrollPane(this.treeList);
54      scrollPane.setMinimumSize(minSizeLeft);
55      
56      JPanel newPersonPanel = null;
57      if (this.canEdit()) {
58        newPersonPanel = new JPanel();
59        newPersonPanel.setLayout(new BoxLayout(newPersonPanel,
60                                               BoxLayout.X_AXIS));
61        newPersonPanel.add(Box.createHorizontalGlue());
62        JButton newPersonButton = new JButton("New Person");
63        newPersonButton.setToolTipText("Creates a new person");
64        newPersonButton.setMnemonic(KeyEvent.VK_N);
65        newPersonButton.addActionListener(new ActionListener() {
66            public void actionPerformed(ActionEvent e) {
67              newPerson();
68            }
69          });
70        newPersonPanel.add(newPersonButton);
71        newPersonPanel.add(Box.createHorizontalGlue());
72      }
73  
74      JPanel treePanel = new JPanel();
75      treePanel.setLayout(new BorderLayout());
76      treePanel.add(scrollPane, BorderLayout.CENTER);
77      if (this.canEdit()) {
78        treePanel.add(newPersonPanel, BorderLayout.SOUTH);
79      }
80  
81      this.personPanel = new PersonPanel(this);
82  
83      JSplitPane splitPane = 
84        new JSplitPane(JSplitPane.HORIZONTAL_SPLIT, treePanel, this.personPanel);
85  
86      this.setLayout(new BorderLayout());
87      this.add(splitPane, BorderLayout.CENTER);
88  
89      JPanel sourcePanel = new JPanel();
90      sourcePanel.setLayout(new BoxLayout(sourcePanel, BoxLayout.X_AXIS));
91      this.sourceLocation = new JLabel();
92      this.sourceLocation.setToolTipText("Location of XML file");
93      sourcePanel.add(this.sourceLocation);
94      sourcePanel.add(Box.createHorizontalGlue());
95      
96      Border sourceBorder = BorderFactory.createEmptyBorder(2, 2, 2, 2);
97      sourcePanel.setBorder(sourceBorder);
98      
99      this.add(sourcePanel, BorderLayout.SOUTH);
100   }

188   /**
189    * Sets the source of the XML file displayed in this GUI
190    */
191   protected void setURLSource(URL url) {
192     try {
193       this.tree = parseSource(url);
194 
195     } catch (IOException ex) {
196       System.out.println("source = " + this.sourceLocation + ": " + ex);
197       this.sourceLocation.setText(ex.toString());
198       return;
199 
200     } catch (FamilyTreeException ex) {
201       this.sourceLocation.setText(ex.toString());
202       return;
203     }
204 
205     this.treeList.fillInList(this.tree);
206     this.sourceLocation.setText(url.toExternalForm());
207   }
208 
209   /**
210    * Parses a <code>URL</code> and tries to extract a family tree in
211    * XML format from it.
212    */ 
213   protected FamilyTree parseSource(URL url) 
214     throws IOException, FamilyTreeException { 
215     
216     InputStream stream = url.openStream();
217 
218     XmlParser parser = new XmlParser(new InputStreamReader(stream));
219     return parser.parse();
220   }

222   /**
223    * Creates a <code>FamilyTreePanel</code> that is displayed inside
224    * a <code>JFrame</code>
225    */
226   public static void main(String[] args) {
227     if (args.length < 1) {
228       System.err.println("** No URL Specified!!");
229       System.exit(1);
230     }
231 
232     URL url = null;
233     try {
234       url = new URL(args[0]);
235 
236     } catch (MalformedURLException ex) {
237       System.err.println(ex.toString());
238       System.exit(1);
239     }
240 
241     FamilyTreePanel viewer = new FamilyTreePanel();
242     viewer.setURLSource(url);
243     
244 
245     JFrame frame = new JFrame("Family Tree Panel");
246     frame.getContentPane().add(viewer);
247 
248     frame.addWindowListener(new WindowAdapter() {
249         public void windowClosing(WindowEvent e) {
250           System.exit(0);
251         }
252       });
253 
254     frame.pack();
255     frame.setVisible(true);
256   }
