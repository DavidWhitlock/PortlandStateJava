3   import java.rmi.*;
4   import java.util.*;
5   
6   /**
7    * <P>This is a concrete class that implements the
8    * <code>RemotePerson</code> interface.  It delegates most of its
9    * functionality to an underlying {@link Person} object.</P>
10   *
11   * <P>Note that because this class is only executed on the
12   * server-side, it does not have to be public.  However, the fact that
13   * it is a <code>UnicastRemoteObject</code> allows the client to
14   * execute its methods remotely.</P>
15   */
16  @SuppressWarnings("serial")
17  class RemotePersonImpl extends java.rmi.server.UnicastRemoteObject
18    implements RemotePerson {
19  
20    /** The underlying Person that is being modeled */
21    private transient Person person;
22  
23    /** The remote family tree that created this remote person.  Note
24     * this field is transient and is therefore not serialized. */
25    private transient RemoteFamilyTree tree;
26  
27    ////////////////////  Constructors  ////////////////////////
28  
29    /**
30     * Creates a new <code>RemotePersonImpl</code> that delegates most
31     * of its behavior to a given <code>Person</code>.
32     */
33    RemotePersonImpl(RemoteFamilyTree tree, Person person) 
34      throws RemoteException {
35  
36      this.tree = tree;
37      this.person = person;
38    }
39  
40    ////////////////////  Instance Methods  ////////////////////
41  
42    public int getId() throws RemoteException {
43      return this.person.getId();
44    }
45  
46    public Person.Gender getGender() throws RemoteException {
47      return this.person.getGender();
48    }
49  
50    public String getFirstName() throws RemoteException {
51      return this.person.getFirstName();
52    }
53  
54    public void setFirstName(String firstName) throws RemoteException {
55      this.person.setFirstName(firstName);
56    }
57  
58    public String getMiddleName() throws RemoteException {
59      return this.person.getMiddleName();
60    }
61  
62    public void setMiddleName(String middleName) 
63      throws RemoteException {
64      this.person.setMiddleName(middleName);
65    }
66  
67    public String getLastName() throws RemoteException {
68      return this.person.getLastName();
69    }
70  
71    public void setLastName(String lastName) throws RemoteException {
72      this.person.setLastName(lastName);
73    }
74  
75    public int getFatherId() throws RemoteException {
76      return this.person.getFatherId();
77    }
78  
79    public void setFatherId(int fatherId) throws RemoteException {
80      RemotePersonImpl rPerson = 
81        (RemotePersonImpl) this.tree.getPerson(fatherId);
82      if (rPerson == null) {
83        String s = "Could not find person with id " + fatherId;
84        throw new FamilyTreeException(s);
85      }
86      this.person.setFather(rPerson.person);
87    }
88  
89    public int getMotherId() throws RemoteException {
90      return this.person.getMotherId();
91    }
92  
93    public void setMotherId(int motherId) throws RemoteException {
94      RemotePersonImpl rPerson = 
95        (RemotePersonImpl) this.tree.getPerson(motherId);
96      if (rPerson == null) {
97        String s = "Could not find person with id " + motherId;
98        throw new FamilyTreeException(s);
99      }
100     this.person.setMother(rPerson.person);
101   }
102 
103   public Date getDateOfBirth() throws RemoteException {
104     return this.person.getDateOfBirth();
105   }
106 
107   public void setDateOfBirth(Date dob) throws RemoteException {
108     this.person.setDateOfBirth(dob);
109   }
110 
111   public Date getDateOfDeath() throws RemoteException {
112     return this.person.getDateOfDeath();
113   }
114 
115   public void setDateOfDeath(Date dod) throws RemoteException {
116     this.person.setDateOfDeath(dod);
117   }
118 
119   public String getDescription() throws RemoteException {
120     return this.person.toString();
121   }
122 
123   ////////////////////  Utility Methods  ////////////////////
124 
125   /**
126    * Two <code>RemotePersonImpl</code>s are considered equal if their
127    * underlying persons have the same id.
128    */
129   public boolean equals(Object o) {
130     if (o instanceof RemotePerson) {
131       RemotePerson other = (RemotePerson) o;
132       try {
133         return this.getId() == other.getId();
134 
135       } catch (RemoteException ex) {
136         return false;
137       }
138  
139     } else {
140       return false;
141     }
142   }
