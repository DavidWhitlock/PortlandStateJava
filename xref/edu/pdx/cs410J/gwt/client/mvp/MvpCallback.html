1   package edu.pdx.cs410J.gwt.client.mvp;
2   
3   import com.google.gwt.event.shared.HandlerManager;
4   import com.google.gwt.user.client.rpc.AsyncCallback;
5   
6   /**
7    * An abstract superclass of {@link AsyncCallback} that handles exceptions in a consistent way
8    */
9   public abstract class MvpCallback<T> implements AsyncCallback<T> {
10    protected final HandlerManager eventBus;
11  
12    public MvpCallback(HandlerManager eventBus) {
13      this.eventBus = eventBus;
14    }
15  
16    @Override
17    public void onFailure(Throwable throwable) {
18      eventBus.fireEvent(new ExceptionEvent(throwable));
19    }
20  }
