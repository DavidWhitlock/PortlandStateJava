1   package edu.pdx.cs410J.gwt.client.mvp;
2   
3   import com.google.gwt.event.dom.client.ClickEvent;
4   import com.google.gwt.event.dom.client.ClickHandler;
5   import com.google.gwt.user.client.ui.*;
6   
7   /**
8    * A dialog that is popped up in response to an exception being thrown
9    */
10  public class ExceptionDialog extends DialogBox implements ExceptionPresenter.Display {
11    private final Label message = new Label();
12    private final TextArea details = new TextArea();
13  
14    public ExceptionDialog() {
15      setText("An error has occurred");
16      setAutoHideEnabled(false);
17      setModal(true);
18  
19      DockPanel panel = new DockPanel();
20      panel.add(message, DockPanel.NORTH);
21  
22      details.setReadOnly(true);
23      details.setCharacterWidth(80);
24      details.setVisibleLines(30);
25      panel.add(details, DockPanel.CENTER);
26  
27      Button ok = new Button("OK");
28      ok.addClickHandler(new ClickHandler() {
29  
30        @Override
31        public void onClick(ClickEvent clickEvent) {
32          ExceptionDialog.this.hide();
33        }
34      });
35      panel.add(ok, DockPanel.SOUTH);
36  
37      setWidget(panel);
38    }
39  
40    @Override
41    public void setMessage(String message) {
42      this.message.setText(message);
43    }
44  
45    @Override
46    public void setStackTrace(StackTraceElement[] trace) {
47      StringBuilder sb = new StringBuilder();
48      for (StackTraceElement line : trace) {
49        sb.append("  at ");
50        sb.append(line.getClassName()).append(".").append(line.getMethodName());
51        sb.append("(").append(line.getFileName()).append(":").append(line.getLineNumber()).append(")");
52        sb.append("\n");
53      }
54  
55      this.details.setText(sb.toString());
56    }
57  
58    private boolean noRecursion;  // Hack?
59  
60    @Override
61    public void show() {
62      if (noRecursion) {
63        noRecursion = false;
64        center();
65        noRecursion = true;
66      }
67      super.show();
68    }
69  }
