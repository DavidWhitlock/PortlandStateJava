3   import com.google.gwt.event.dom.client.ChangeEvent;
4   import com.google.gwt.event.dom.client.ChangeHandler;
5   import com.google.gwt.user.client.ui.Composite;
6   import com.google.gwt.user.client.ui.ListBox;
7   import com.google.gwt.user.client.ui.Widget;
8   
9   import java.util.List;
10  
11  /**
12   * The UI for the list of movies.
13   */
14  public class MovieListView extends Composite implements MovieListPresenter.Display {
15  
16    private final ListBox movies;
17  
18    public MovieListView() {
19      this.movies = new ListBox();
20      this.movies.setVisibleItemCount(10);
21      initWidget(this.movies);
22    }
23  
24    @Override
25    public void setTitles(List<String> titles) {
26      movies.clear();
27      for (String title : titles) {
28        movies.addItem(title);
29      }
30    }
31  
32    @Override
33    public void addMovieSelectedHandler(final MovieSelectedHandler handler) {
34      movies.addChangeHandler(new ChangeHandler() {
35  
36        @Override
37        public void onChange(ChangeEvent changeEvent) {
38          handler.onMovieSelected(movies.getSelectedIndex());
39        }
40      });
41    }
42  
43      @Override
44      public Widget asWidget() {
45          return this;
46      }
47  }
