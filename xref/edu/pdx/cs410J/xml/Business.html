3   import org.w3c.dom.*;

5   /**
6    * This class represents a business whose name, address and phone
7    * number are listed in a phone book.  A <code>Business</code> is
8    * constructed from an XML DOM tree.  If we were doing this for real,
9    * we'd want a way of constructing an empty <code>Business</code> and
10   * filling in its fields.
11   */
12  public class Business extends PhoneBookEntry {
13    protected String name;
14  
15    /**
16     * Create a new <code>Business</code> from an <code>Element</code>
17     * in a DOM tree.
18     */
19    public Business(Element root) {
20      NodeList elements = root.getChildNodes();
21      for (int i = 0; i < elements.getLength(); i++) {
22        Node node = elements.item(i);
23  
24        if (!(node instanceof Element)) {
25  	continue;
26        }
27  
28        Element element = (Element) node;
29        
30        if (element.getNodeName().equals("name")) {
31  	Node text = element.getFirstChild();
32  	this.name = text.getNodeValue();
33  
34        } else if (element.getNodeName().equals("address")) {
35  	fillInAddress(element);
36  
37        } else if (element.getNodeName().equals("phone")) {
38  	fillInPhone(element);
39  
40        } else {
41  	String s = "Unknown element: " + element.getNodeName() + " (" +
42            element.getNodeValue() + ")";
43          throw new IllegalArgumentException(s);
44        }
45      }
46    }
47  
48    public String toString() {
49      StringBuffer sb = new StringBuffer();
50      sb.append(this.name + "\n");
51      sb.append(super.toString());
52  
53      return sb.toString();
54    }
55  }
