001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import com.google.gwt.user.client.rpc.AsyncCallback;
005import edu.pdx.cs410J.rmi.Movie;
006import org.junit.Test;
007import org.mockito.ArgumentCaptor;
008
009import static edu.pdx.cs410J.gwt.client.mvp.MovieEditorPresenter.Display;
010import static edu.pdx.cs410J.gwt.client.mvp.MovieEditorPresenter.Display.NewCharacterHandler;
011import static org.mockito.Matchers.any;
012import static org.mockito.Matchers.eq;
013import static org.mockito.Mockito.mock;
014import static org.mockito.Mockito.verify;
015
016/**
017 * Tests that the movie editor UI is updated by a {@link MovieEditorPresenter}
018 */
019public class MovieEditorPresenterTest extends MovieTestCase {
020
021  /**
022   * Tests that a movie's title and year are displayed appropriately
023   */
024  @Test
025  public void testTitleAndYear() {
026    Display view = mock(Display.class);
027    HandlerManager eventBus = new HandlerManager(null);
028    MovieServiceAsync service = mock(MovieServiceAsync.class);
029    
030    new MovieEditorPresenter(view, eventBus, service);
031
032    String title = "Movie 1";
033    int year = 2010;
034
035    eventBus.fireEvent(new DisplayMovieEvent(newMovie(title, year)));
036
037    verify(view).setTitle(title);
038    verify(view).setYear(year);
039  }
040
041  /**
042   * Tests that a movie's characters are displayed appropriately
043   */
044  @Test
045  public void testCharacters() {
046    Display view = mock(Display.class);
047    HandlerManager eventBus = new HandlerManager(null);
048    MovieServiceAsync service = mock(MovieServiceAsync.class);
049
050    new MovieEditorPresenter(view, eventBus, service);
051
052    Movie movie = newMovie("Movie 1", 2010);
053    addCharacter(movie, "Character 1", 1L);
054    eventBus.fireEvent(new DisplayMovieEvent(movie));
055
056    verify(view).setCharacters(movie.getCharacters());
057  }
058
059  /**
060   * Tests that a new character is created appropriately
061   */
062  @Test
063  public void testNewCharacter() {
064    Display view = mock(Display.class);
065    HandlerManager eventBus = new HandlerManager(null);
066    MovieServiceAsync service = mock(MovieServiceAsync.class);
067
068    new MovieEditorPresenter(view, eventBus, service);
069
070    ArgumentCaptor<NewCharacterHandler> captor = ArgumentCaptor.forClass(NewCharacterHandler.class);
071    verify(view).addNewCharacterHandler(captor.capture());
072
073    Movie movie = newMovie("Movie 1", 2010);
074    eventBus.fireEvent(new DisplayMovieEvent(movie));
075
076    String character = "Character 1";
077    long actorId = 123L;
078    captor.getValue().onNewCharacter(character, actorId);
079
080    verify(service).addCharacter(eq(movie.getId()), eq(character), eq(actorId), any(AsyncCallback.class));
081  }
082}
