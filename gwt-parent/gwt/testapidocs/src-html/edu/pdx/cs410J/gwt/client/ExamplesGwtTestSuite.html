001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.junit.tools.GWTTestSuite;
004import junit.framework.Test;
005import junit.framework.TestCase;
006
007/**
008 * A suite of GWT tests that can all be run inside the same development mode shell instance.
009 * Making this a <code>TestCase</code> makes the GWT Maven plugin happy
010 */
011public class ExamplesGwtTestSuite extends TestCase {
012
013  public static Test suite() {
014    GWTTestSuite suite = new GWTTestSuite("GWT Examples");
015
016    suite.addTestSuite(DivisionServiceGwtIT.class);
017    suite.addTestSuite(DivisionServiceExampleGwtIT.class);
018    suite.addTestSuite(MovieServiceGwtIT.class);
019
020    return suite;
021  }
022
023}
