001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.dom.client.Document;
004import com.google.gwt.dom.client.NativeEvent;
005import com.google.gwt.event.dom.client.DomEvent;
006import com.google.gwt.user.client.Timer;
007import com.google.gwt.user.client.ui.Button;
008import org.junit.Test;
009
010public class DivisionServiceExampleGwtIT extends IntegrationGwtTestCase{
011
012  /**
013   * Tests that the division service UI
014   */
015  @Test
016  public void testDivisionUI() {
017    final DivisionServiceExample ui = new DivisionServiceExample();
018    int dividend = 6;
019    int divisor = 3;
020    final int quotient = 2;
021
022    ui.dividend.setText(String.valueOf(dividend));
023    ui.divisor.setText(String.valueOf(divisor));
024
025    click(ui.equals);
026
027    Timer verify = new Timer() {
028      @Override
029      public void run() {
030        assertEquals(String.valueOf(quotient), ui.quotient.getText());
031        finishTest();
032      }
033    };
034
035    // Wait for the RPC call to return
036    verify.schedule(500);
037
038    delayTestFinish(1000);
039  }
040
041  /**
042   * Clicks a <code>Button</code>
043   *
044   * One would think that you could testing clicking a button with Button.click(), but it looks
045   * like you need to fire the native event instead.  Lame.
046   *
047   * @param button
048   *        The button to click 
049   */
050  private void click(Button button) {
051    NativeEvent event = Document.get().createClickEvent(0, 0, 0, 0, 0, false, false, false, false);
052    DomEvent.fireNativeEvent(event, button);
053  }
054}
