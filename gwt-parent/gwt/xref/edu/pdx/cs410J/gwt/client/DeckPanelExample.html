3   import com.google.gwt.event.dom.client.ChangeHandler;
4   import com.google.gwt.event.dom.client.ChangeEvent;
5   import com.google.gwt.user.client.ui.*;
6   
7   /**
8    * Demonstrates GWT's {@link DeckPanel}
9    */
10  public class DeckPanelExample extends Example {
11    public DeckPanelExample() {
12      super("Deck Panel");
13  
14      DockPanel dock = new DockPanel();
15      final DeckPanel deck = new DeckPanel();
16      final ListBox list = new ListBox();
17      list.addChangeHandler(new ChangeHandler() {
18          public void onChange( ChangeEvent changeEvent )
19          {
20              int index = list.getSelectedIndex();
21              deck.showWidget(index);
22          }
23      });
24  
25      for (String label : new String[] { "One", "Two", "Three", "Four"}) {
26        list.addItem(label);
27        deck.add(new Button(label));
28      }
29      deck.showWidget(0);
30  
31      dock.add(list, DockPanel.NORTH);
32      dock.add(deck, DockPanel.CENTER);
33      add(dock);
34    }
35  }
