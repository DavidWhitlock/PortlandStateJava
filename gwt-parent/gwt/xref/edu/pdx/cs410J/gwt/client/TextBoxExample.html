3   import com.google.gwt.user.client.ui.TextBox;
4   import com.google.gwt.user.client.ui.PasswordTextBox;
5   
6   /**
7    * Demonstrates GWT text boxes
8    */
9   public class TextBoxExample extends Example {
10    public TextBoxExample() {
11      super("Text Boxes");
12  
13      TextBox text = new TextBox();
14      text.setVisibleLength(15);
15      text.setMaxLength(10);
16      add(text);
17  
18      TextBox readOnly = new TextBox();
19      readOnly.setReadOnly(true);
20      readOnly.setVisibleLength(15);
21      readOnly.setText("Read only text");
22      add(readOnly);
23  
24      PasswordTextBox password = new PasswordTextBox();
25      password.setVisibleLength(15);
26      add(password);
27    }
28  }
