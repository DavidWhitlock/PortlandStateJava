4   import com.google.gwt.user.client.rpc.SerializationStreamReader;
5   import com.google.gwt.user.client.rpc.SerializationStreamWriter;

9   /**
10   * A custom field serializer for the {@link edu.pdx.cs410J.rmi.Movie} class.  This allows us to use
11   * <code>Movie</code>s with GWT-RPC services without modifying the <code>Movie</code>
12   * class to have a zero-argument constructor
13   *
14   * http://code.google.com/p/wogwt/wiki/CustomFieldSerializer
15   */
16  public class Movie_CustomFieldSerializer {
17  
18    public static void deserialize(SerializationStreamReader reader, Movie movie)
19      throws SerializationException {
20  
21    }
22  
23    public static Movie instantiate(SerializationStreamReader reader)
24      throws SerializationException {
25  
26  
27      long id = reader.readLong();
28      String title = reader.readString();
29      int year = reader.readInt();
30      Map<String, Long> characters = (Map<String, Long>) reader.readObject();
31  
32      Movie movie = new Movie(title, year);
33      movie.setId(id);
34      for (Map.Entry<String, Long> entry : characters.entrySet()) {
35        movie.addCharacter(entry.getKey(), entry.getValue());
36      }
37      return movie;
38    }
39  
40    public static void serialize(SerializationStreamWriter writer, Movie movie)
41      throws SerializationException {
42  
43      writer.writeLong(movie.getId());
44      writer.writeString(movie.getTitle());
45      writer.writeInt(movie.getYear());
46      writer.writeObject(movie.getCharacters());
47    }
48  }
