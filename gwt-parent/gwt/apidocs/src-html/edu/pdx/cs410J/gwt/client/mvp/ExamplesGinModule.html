001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import com.google.gwt.inject.client.AbstractGinModule;
005import com.google.inject.Provides;
006import com.google.inject.Singleton;
007
008/**
009 * A gin module that binds the well-known MVP example objects
010 */
011public class ExamplesGinModule extends AbstractGinModule {
012  @Override
013  protected void configure() {
014//    bind(DivisionPresenter.Display.class).to(DivisionView.class).in(Singleton.class);
015//    bind(DivisionPresenter.class).asEagerSingleton();
016    
017    bind(ExceptionPresenter.Display.class).to(ExceptionDialog.class).in(Singleton.class);
018    bind(ExceptionPresenter.class).asEagerSingleton();
019
020    bind(MovieListPresenter.Display.class).to(MovieListView.class).in(Singleton.class);
021    bind(MovieListPresenter.class).asEagerSingleton();
022  }
023
024  @Provides
025  @Singleton
026  public HandlerManager provideEventBus() {
027    return new HandlerManager(null);
028  }
029}
