001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import com.google.gwt.user.client.rpc.AsyncCallback;
005
006/**
007 * An abstract superclass of {@link AsyncCallback} that handles exceptions in a consistent way
008 */
009public abstract class MvpCallback<T> implements AsyncCallback<T> {
010  protected final HandlerManager eventBus;
011
012  public MvpCallback(HandlerManager eventBus) {
013    this.eventBus = eventBus;
014  }
015
016  @Override
017  public void onFailure(Throwable throwable) {
018    eventBus.fireEvent(new ExceptionEvent(throwable));
019  }
020}
