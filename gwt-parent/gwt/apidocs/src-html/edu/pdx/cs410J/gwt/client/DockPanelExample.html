001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.DockPanel;
004import com.google.gwt.user.client.ui.Button;
005
006/**
007 * Demonstrates GWT's {@link DockPanel}
008 */
009public class DockPanelExample extends Example {
010  public DockPanelExample() {
011    super("Dock Panel");
012    
013    DockPanel dock = new DockPanel();
014    dock.add(new Button("North"), DockPanel.NORTH);
015    dock.add(new Button("South"), DockPanel.SOUTH);
016    dock.add(new Button("East"), DockPanel.EAST);
017    dock.add(new Button("West"), DockPanel.WEST);
018    dock.add(new Button("Center"), DockPanel.CENTER);
019    add(dock);
020  }
021}
