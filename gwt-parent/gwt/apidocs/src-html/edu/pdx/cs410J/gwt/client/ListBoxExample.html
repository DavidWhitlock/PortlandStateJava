001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.ListBox;
004
005/**
006 * Demonstrates the GWT {@link ListBox}
007 */
008public class ListBoxExample extends Example {
009  public ListBoxExample() {
010    super("List Box");
011
012    ListBox box = new ListBox();
013    box.addItem("Monday");
014    box.addItem("Tuesday");
015    box.addItem("Wednesday");
016    box.addItem("Thursday");
017    box.addItem("Friday");
018
019    add(box);
020
021    ListBox multi = new ListBox(true);
022    multi.addItem("Chocolate");
023    multi.addItem("Vanilla");
024    multi.addItem("Strawberry");
025
026    add(multi);
027  }
028}
