001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.EventHandler;
004import com.google.gwt.event.shared.GwtEvent;
005import edu.pdx.cs410J.rmi.Movie;
006
007/**
008 * An event that is fired when a movie is to be displayed in the UI
009 */
010public class DisplayMovieEvent extends GwtEvent<DisplayMovieEvent.Handler> {
011
012  public interface Handler extends EventHandler {
013
014    /**
015     * Invoked when a movie is to be displayed
016     * @param movie The movie to display
017     */
018    void onMovieDisplayed(Movie movie);
019  }
020
021  public static final Type<Handler> TYPE = new Type<Handler>();
022
023  private final Movie movie;
024
025  public DisplayMovieEvent(Movie movie) {
026    this.movie = movie;
027  }
028
029  @Override
030  public Type<Handler> getAssociatedType() {
031    return TYPE;
032  }
033
034  @Override
035  protected void dispatch(Handler handler) {
036    handler.onMovieDisplayed(this.movie);
037  }
038}
