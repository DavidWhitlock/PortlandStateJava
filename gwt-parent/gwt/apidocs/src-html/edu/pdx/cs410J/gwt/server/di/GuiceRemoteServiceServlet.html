001package edu.pdx.cs410J.gwt.server.di;
002
003import com.google.gwt.user.client.rpc.IncompatibleRemoteServiceException;
004import com.google.gwt.user.client.rpc.RemoteService;
005import com.google.gwt.user.client.rpc.SerializationException;
006import com.google.gwt.user.server.rpc.RPC;
007import com.google.gwt.user.server.rpc.RPCRequest;
008import com.google.gwt.user.server.rpc.RemoteServiceServlet;
009import com.google.inject.Inject;
010import com.google.inject.Injector;
011import com.google.inject.Provider;
012import com.google.inject.Singleton;
013
014/**
015 * A GWT remote service that uses Guice to serve GWT-RPC requests.  This means that remote service
016 * implementations do not need to extend {@link RemoteServiceServlet}
017 */
018@Singleton
019public class GuiceRemoteServiceServlet extends RemoteServiceServlet {
020  private final Provider<Injector> injector;
021
022  /**
023   * Zero-argument constructor is necessary for GWT tests
024   */
025  public GuiceRemoteServiceServlet() {
026    injector = new Provider<Injector>() {
027      @Override
028      public Injector get() {
029        return getInjector();
030      }
031    };
032  }
033
034  /**
035   * Lame hack to get the Injector in an environment (such as a GWT test) in which the
036   * servlet is not created with Guice
037   * @return The Guice injector
038   */
039  protected Injector getInjector() {
040    throw new UnsupportedOperationException("You must override this method");
041  }
042
043  @Inject
044  public GuiceRemoteServiceServlet(Provider<Injector> injector) {
045    this.injector = injector;
046  }
047
048  @Override
049  public String processCall(String payload) throws SerializationException {
050    try {
051      RPCRequest req = RPC.decodeRequest(payload, null, this);
052
053      RemoteService service = getServiceInstance(req.getMethod().getDeclaringClass());
054
055      return RPC.invokeAndEncodeResponse(service, req.getMethod(),
056        req.getParameters(), req.getSerializationPolicy());
057      
058    } catch (IncompatibleRemoteServiceException ex) {
059      log("IncompatibleRemoteServiceException in the processCall(String) method.",
060          ex);
061      return RPC.encodeResponseForFailure(null, ex);
062    }
063  }
064
065  @SuppressWarnings({"unchecked"})
066  private RemoteService getServiceInstance(Class serviceClass) {
067    return (RemoteService) injector.get().getInstance(serviceClass);
068  }
069  @Override
070  protected void doUnexpectedFailure(Throwable unhandled) {
071    unhandled.printStackTrace(System.err);
072    super.doUnexpectedFailure(unhandled);
073  }
074
075}
