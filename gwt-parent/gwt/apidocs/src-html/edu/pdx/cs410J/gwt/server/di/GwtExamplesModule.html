001package edu.pdx.cs410J.gwt.server.di;
002
003import com.google.inject.AbstractModule;
004import com.google.inject.Singleton;
005import com.google.inject.servlet.ServletModule;
006import edu.pdx.cs410J.gwt.client.mvp.MovieService;
007import edu.pdx.cs410J.gwt.server.mvp.MovieServiceImpl;
008import edu.pdx.cs410J.rmi.MovieDatabase;
009import edu.pdx.cs410J.rmi.MovieDatabaseImpl;
010
011/**
012 * A Guice module for the server side of the GWT examples.  It binds remote interfaces to their implementations.
013 */
014public class GwtExamplesModule extends AbstractModule {
015
016  @Override
017  protected void configure() {
018    bind(MovieService.class).to(MovieServiceImpl.class).in(Singleton.class);
019    bind(MovieDatabase.class).to(MovieDatabaseImpl.class).in(Singleton.class);
020    bind(MovieDatabaseInitializer.class).asEagerSingleton();
021  }
022}
