001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.dom.client.ChangeEvent;
004import com.google.gwt.event.dom.client.ChangeHandler;
005import com.google.gwt.user.client.ui.Composite;
006import com.google.gwt.user.client.ui.ListBox;
007import com.google.gwt.user.client.ui.Widget;
008
009import java.util.List;
010
011/**
012 * The UI for the list of movies.
013 */
014public class MovieListView extends Composite implements MovieListPresenter.Display {
015
016  private final ListBox movies;
017
018  public MovieListView() {
019    this.movies = new ListBox();
020    this.movies.setVisibleItemCount(10);
021    initWidget(this.movies);
022  }
023
024  @Override
025  public void setTitles(List<String> titles) {
026    movies.clear();
027    for (String title : titles) {
028      movies.addItem(title);
029    }
030  }
031
032  @Override
033  public void addMovieSelectedHandler(final MovieSelectedHandler handler) {
034    movies.addChangeHandler(new ChangeHandler() {
035
036      @Override
037      public void onChange(ChangeEvent changeEvent) {
038        handler.onMovieSelected(movies.getSelectedIndex());
039      }
040    });
041  }
042
043    @Override
044    public Widget asWidget() {
045        return this;
046    }
047}
