001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.dom.client.ClickEvent;
004import com.google.gwt.event.dom.client.ClickHandler;
005import com.google.gwt.user.client.ui.*;
006
007/**
008 * A dialog that is popped up in response to an exception being thrown
009 */
010public class ExceptionDialog extends DialogBox implements ExceptionPresenter.Display {
011  private final Label message = new Label();
012  private final TextArea details = new TextArea();
013
014  public ExceptionDialog() {
015    setText("An error has occurred");
016    setAutoHideEnabled(false);
017    setModal(true);
018
019    DockPanel panel = new DockPanel();
020    panel.add(message, DockPanel.NORTH);
021
022    details.setReadOnly(true);
023    details.setCharacterWidth(80);
024    details.setVisibleLines(30);
025    panel.add(details, DockPanel.CENTER);
026
027    Button ok = new Button("OK");
028    ok.addClickHandler(new ClickHandler() {
029
030      @Override
031      public void onClick(ClickEvent clickEvent) {
032        ExceptionDialog.this.hide();
033      }
034    });
035    panel.add(ok, DockPanel.SOUTH);
036
037    setWidget(panel);
038  }
039
040  @Override
041  public void setMessage(String message) {
042    this.message.setText(message);
043  }
044
045  @Override
046  public void setStackTrace(StackTraceElement[] trace) {
047    StringBuilder sb = new StringBuilder();
048    for (StackTraceElement line : trace) {
049      sb.append("  at ");
050      sb.append(line.getClassName()).append(".").append(line.getMethodName());
051      sb.append("(").append(line.getFileName()).append(":").append(line.getLineNumber()).append(")");
052      sb.append("\n");
053    }
054
055    this.details.setText(sb.toString());
056  }
057
058  private boolean noRecursion;  // Hack?
059
060  @Override
061  public void show() {
062    if (noRecursion) {
063      noRecursion = false;
064      center();
065      noRecursion = true;
066    }
067    super.show();
068  }
069}
