001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.core.client.GWT;
004import com.google.gwt.event.logical.shared.ValueChangeEvent;
005import com.google.gwt.event.logical.shared.ValueChangeHandler;
006import com.google.gwt.i18n.client.DateTimeFormat;
007import com.google.gwt.user.client.ui.Label;
008import com.google.gwt.user.datepicker.client.CalendarUtil;
009import com.google.gwt.user.datepicker.client.DateBox;
010
011import java.util.Date;
012
013public class DateLocalizationExample extends Example
014{
015    private final Label label;
016
017    public DateLocalizationExample()
018    {
019        super( "Date Localization" );
020
021        DateLocalizationStrings strings = GWT.create( DateLocalizationStrings.class );
022        this.label = new Label();
023        this.label.setText( strings.defaultDaysMessage() );
024        this.add(new Label(strings.selectString()));
025
026        final DateLocalizationMessages messages = GWT.create( DateLocalizationMessages.class );
027
028        DateBox dateBox = new DateBox();
029        dateBox.setFormat( new DateBox.DefaultFormat( DateTimeFormat.getLongDateFormat()) );
030        Date date = new Date();
031        CalendarUtil.addMonthsToDate( date, -29 * 12 );
032        dateBox.setValue( date );
033
034        dateBox.addValueChangeHandler( new ValueChangeHandler<Date>() {
035
036            public void onValueChange( ValueChangeEvent<Date> e )
037            {
038              long days = CalendarUtil.getDaysBetween(e.getValue(), new Date() );
039              label.setText( messages.daysMessage( days ) );
040            }
041        });
042
043        this.add( dateBox );
044        this.add( label );
045    }
046}
