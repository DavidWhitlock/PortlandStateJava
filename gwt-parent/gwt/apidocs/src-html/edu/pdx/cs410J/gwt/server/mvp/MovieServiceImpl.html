001package edu.pdx.cs410J.gwt.server.mvp;
002
003import com.google.common.collect.Lists;
004import com.google.inject.Inject;
005import edu.pdx.cs410J.gwt.client.mvp.MovieService;
006import edu.pdx.cs410J.rmi.Movie;
007import edu.pdx.cs410J.rmi.MovieDatabase;
008
009import java.rmi.RemoteException;
010import java.util.ArrayList;
011
012public class MovieServiceImpl implements MovieService {
013  private final MovieDatabase db;
014
015  @Inject
016  public MovieServiceImpl(MovieDatabase db) {
017    this.db = db;
018  }
019
020  @Override
021  public ArrayList<Movie> getAllMovies() {
022    try {
023      return Lists.newArrayList(db.getMovies());
024
025    } catch (RemoteException e) {
026      throw new RuntimeException(e);
027    }
028  }
029
030  @Override
031  public Movie addCharacter(long movieId, String character, long actorId) {
032    try {
033      db.noteCharacter(movieId, character, actorId);
034      return db.getMovie(movieId);
035
036    } catch (RemoteException e) {
037      throw new RuntimeException(e);
038    }
039  }
040}
