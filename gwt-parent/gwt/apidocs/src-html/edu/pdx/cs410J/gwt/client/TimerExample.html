001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.Timer;
004import com.google.gwt.user.client.ui.Label;
005import com.google.gwt.i18n.client.DateTimeFormat;
006
007import java.util.Date;
008
009/**
010 * Demonstrates GWT's {@link Timer} and date formatting capabilities
011 */
012public class TimerExample extends Example {
013  public TimerExample() {
014    super("Timer Example");
015
016    final Label time = new Label();
017    Timer timer = new Timer() {
018
019      public void run() {
020        Date now = new Date();
021        DateTimeFormat format = DateTimeFormat.getFullDateTimeFormat();
022        time.setText("The time is" + format.format(now));
023      }
024    };
025    timer.scheduleRepeating(1000);
026
027    add(time);
028  }
029}
