001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.*;
004import com.google.gwt.event.dom.client.ClickHandler;
005import com.google.gwt.event.dom.client.ClickEvent;
006
007/**
008 * Pops up a GWT {@link DialogBox}
009 */
010public class DialogBoxExample extends Example {
011  public DialogBoxExample() {
012    super("Dialog Box");
013
014    Button autoHide = new Button("Auto-hide dialog");
015    autoHide.addClickHandler(new ClickHandler() {
016        public void onClick( ClickEvent clickEvent )
017        {
018            createDialogBox(true, false).show();
019        }
020    });
021    add(autoHide);
022
023    Button modal = new Button("Modal dialog");
024    modal.addClickHandler(new ClickHandler() {
025      public void onClick(ClickEvent event) {
026        createDialogBox(false, true).show();
027      }
028    });
029    add(modal);
030  }
031
032  private DialogBox createDialogBox(boolean autoHide, boolean modal) {
033    final DialogBox box = new DialogBox(autoHide, modal);
034    box.setText("Auto-hide dialog");
035
036    DockPanel panel = new DockPanel();
037    panel.add(new Label("Click outside me"), DockPanel.CENTER);
038    panel.add(new Button("Close", new ClickHandler() {
039      public void onClick(ClickEvent event) {
040        box.hide();
041      }
042    }), DockPanel.SOUTH);
043    box.setWidget(panel);
044    box.center();
045    return box;
046  }
047}
