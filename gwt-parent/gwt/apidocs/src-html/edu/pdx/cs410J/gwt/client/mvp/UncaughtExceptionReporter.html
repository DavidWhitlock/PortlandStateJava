001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.core.client.GWT;
004import com.google.gwt.event.shared.HandlerManager;
005import com.google.inject.Inject;
006
007/**
008 * Puts an uncaught exception on the event bus so that the error dialog can be popped up
009 */
010public class UncaughtExceptionReporter implements GWT.UncaughtExceptionHandler {
011  private final HandlerManager eventBus;
012
013  @Inject
014  public UncaughtExceptionReporter(HandlerManager eventBus) {
015    this.eventBus = eventBus;
016  }
017
018  @Override
019  public void onUncaughtException(Throwable throwable) {
020    eventBus.fireEvent(new ExceptionEvent(throwable));
021  }
022}
