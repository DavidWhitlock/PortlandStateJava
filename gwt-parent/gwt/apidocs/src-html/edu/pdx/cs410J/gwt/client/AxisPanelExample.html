001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.HorizontalPanel;
004import com.google.gwt.user.client.ui.Label;
005import com.google.gwt.user.client.ui.VerticalPanel;
006import com.google.gwt.user.client.ui.Button;
007
008/**
009 * Demonstrates GWT's {@link HorizontalPanel} and {@link VerticalPanel}
010 */
011public class AxisPanelExample extends Example {
012  public AxisPanelExample() {
013    super("Axis Panels");
014
015    HorizontalPanel horiz = new HorizontalPanel();
016    horiz.add(new Button("A"));
017    horiz.add(new Button("B"));
018    horiz.add(new Button("C"));
019    add(horiz);
020
021    VerticalPanel vert = new VerticalPanel();
022    vert.add(new Label("1"));
023    vert.add(new Label("2"));
024    vert.add(new Label("3"));
025    add(vert);
026  }
027}
