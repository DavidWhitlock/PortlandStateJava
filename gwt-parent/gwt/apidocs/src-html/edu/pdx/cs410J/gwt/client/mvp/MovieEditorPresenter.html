001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import com.google.inject.Inject;
005import edu.pdx.cs410J.rmi.Movie;
006
007import java.util.Map;               
008
009/**
010 * The presenter that edits information about a {@link Movie}
011 */
012public class MovieEditorPresenter {
013  public interface Display {
014
015    void setTitle(String title);
016
017    void setYear(int year);
018
019    void setCharacters(Map<String, Long> characters);
020
021    public interface NewCharacterHandler {
022      public void onNewCharacter(String character, long actorId);
023    }
024
025    void addNewCharacterHandler(NewCharacterHandler handler);
026  }
027
028  private final Display view;
029
030  private final HandlerManager eventBus;
031
032  private Movie movie;
033
034  @Inject
035  public MovieEditorPresenter(Display view, HandlerManager hm, final MovieServiceAsync service) {
036    this.view = view;
037    this.eventBus = hm;
038
039    eventBus.addHandler(DisplayMovieEvent.TYPE, new DisplayMovieEvent.Handler() {
040
041      @Override
042      public void onMovieDisplayed(Movie movie) {
043        edit(movie);
044      }
045    });
046
047    view.addNewCharacterHandler(new Display.NewCharacterHandler() {
048
049      @Override
050      public void onNewCharacter(String character, long actorId) {
051        final HandlerManager eventBus1 = eventBus;
052        service.addCharacter(movie.getId(), character, actorId, new MvpCallback<Movie>(eventBus1) {
053          @Override
054          public void onSuccess(Movie movie) {
055            edit(movie);
056          }
057        });
058      }
059    });
060  }
061
062  private void edit(Movie movie) {
063    this.movie = movie;
064
065    view.setTitle(movie.getTitle());
066    view.setYear(movie.getYear());
067    view.setCharacters(movie.getCharacters());
068  }
069
070
071}
