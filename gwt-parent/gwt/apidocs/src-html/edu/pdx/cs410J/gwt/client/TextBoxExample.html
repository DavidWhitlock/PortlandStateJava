001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.TextBox;
004import com.google.gwt.user.client.ui.PasswordTextBox;
005
006/**
007 * Demonstrates GWT text boxes
008 */
009public class TextBoxExample extends Example {
010  public TextBoxExample() {
011    super("Text Boxes");
012
013    TextBox text = new TextBox();
014    text.setVisibleLength(15);
015    text.setMaxLength(10);
016    add(text);
017
018    TextBox readOnly = new TextBox();
019    readOnly.setReadOnly(true);
020    readOnly.setVisibleLength(15);
021    readOnly.setText("Read only text");
022    add(readOnly);
023
024    PasswordTextBox password = new PasswordTextBox();
025    password.setVisibleLength(15);
026    add(password);
027  }
028}
