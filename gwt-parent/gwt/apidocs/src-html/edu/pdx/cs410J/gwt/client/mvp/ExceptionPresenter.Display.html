001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.HandlerManager;
004import com.google.inject.Inject;
005
006/**
007 * Handles the logic for displaying an exception to the user
008 */
009public class ExceptionPresenter {
010  public interface Display {
011    void setMessage(String message);
012
013    void setStackTrace(StackTraceElement[] trace);
014
015    void show();
016  }
017
018  private final Display view;
019
020  private final HandlerManager eventBus;
021
022  @Inject
023  public ExceptionPresenter(final Display view, HandlerManager hm) {
024    this.view = view;
025    this.eventBus = hm;
026
027    this.eventBus.addHandler(ExceptionEvent.TYPE, new ExceptionEvent.Handler() {
028
029      @Override
030      public void onException(Throwable ex) {
031        setException(ex);
032      }
033    });
034  }
035
036  private void setException(Throwable ex) {
037    view.setMessage(ex.getMessage());
038    view.setStackTrace(ex.getStackTrace());
039    view.show();
040  }
041}
