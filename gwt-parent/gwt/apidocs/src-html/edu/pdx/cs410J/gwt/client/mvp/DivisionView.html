001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.dom.client.ClickHandler;
004import com.google.gwt.event.logical.shared.ValueChangeHandler;
005import com.google.gwt.resources.client.CssResource;
006import com.google.gwt.uibinder.client.UiBinder;
007import com.google.gwt.uibinder.client.UiField;
008import com.google.gwt.user.client.Window;
009import com.google.gwt.user.client.ui.Button;
010import com.google.gwt.user.client.ui.HasValue;
011import com.google.gwt.user.client.ui.HorizontalPanel;
012import com.google.gwt.user.client.ui.TextBox;
013import com.google.inject.Inject;
014import edu.pdx.cs410J.gwt.client.Example;
015
016/**
017 * The view for the MVP implementation of the division service example.  It uses UI Binder to layout the UI.
018 *
019 * @author David Whitlock
020 * @since Summer 2010
021 */
022public class DivisionView extends Example implements DivisionPresenter.Display
023{
024    /** The weirdo interface required by UI Binder */
025    public interface Binder extends UiBinder<HorizontalPanel, DivisionView> {}
026
027    /** Interface with CSS styles needed by this view */
028    public interface DivisionStyle extends CssResource {
029        String error();
030    }
031
032    @UiField
033    TextBox dividend;
034
035    @UiField
036    TextBox divisor;
037
038    @UiField
039    HasValue<String> quotient;
040
041    @UiField
042    Button equals;
043
044    @UiField
045    DivisionStyle style;
046
047    @Inject
048    public DivisionView(Binder binder) {
049        super("Division with MVP");
050
051        add( binder.createAndBindUi( this ));
052    }
053
054    @Override
055    public void setDividendChangeHandler( ValueChangeHandler<String> handler )
056    {
057        dividend.addValueChangeHandler( handler );
058    }
059
060    @Override
061    public void setDividendValid( boolean valid )
062    {
063        if (valid) {
064            dividend.removeStyleName( style.error() );
065
066        } else {
067            dividend.addStyleName( style.error() );
068        }
069    }
070
071    @Override
072    public void setDivisionClickHandler( ClickHandler handler )
073    {
074       equals.addClickHandler( handler );
075    }
076
077    @Override
078    public void setDivisionEnabled( boolean enabled )
079    {
080        equals.setEnabled( enabled );
081    }
082
083    @Override
084    public void setDivisorChangeHandler( ValueChangeHandler<String> handler )
085    {
086       divisor.addValueChangeHandler( handler );
087    }
088
089    @Override
090    public void setDivisorValid( boolean valid )
091    {
092        if (valid) {
093            divisor.removeStyleName( style.error() );
094
095        } else {
096            divisor.addStyleName( style.error() );
097        }
098    }
099
100    @Override
101    public void setErrorMessage( String message )
102    {
103        Window.alert( message );
104    }
105
106    @Override
107    public void setQuotient( String quotient )
108    {
109        this.quotient.setValue( quotient );
110    }
111}
