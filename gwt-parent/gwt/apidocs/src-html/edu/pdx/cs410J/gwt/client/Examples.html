022/**
023 * The main entry point to GWT examples
024 */
025public class Examples implements EntryPoint {
026
027  public void onModuleLoad() {
028    RootLayoutPanel rootPanel = RootLayoutPanel.get();
029
030    TabLayoutPanel allExamples = new TabLayoutPanel(1.5, Unit.EM);
031    Map<String, List<? extends Example>> examples = getExamples();
032    for (String title : examples.keySet()) {
033      ExamplePanel panel = new ExamplePanel(title);
034      for (Example example : examples.get(title)) {
035        panel.add(example, example.getName());
036      }
037      panel.selectFirst();
038      allExamples.add(panel, panel.getTitle());
039    }
040    allExamples.selectTab(0);
041
042    addHistoryListener(allExamples);
043
044    rootPanel.add(allExamples);
045
046    History.fireCurrentHistoryState();
047  }
048
049    /**
050     * Adds a new GWT history token whenever a new tab is selected
051     */
052  private void addHistoryListener( final TabLayoutPanel tabs ) {
053    tabs.addSelectionHandler( new SelectionHandler<Integer>() {
054        public void onSelection( SelectionEvent<Integer> event )
055        {
056            ExamplePanel panel = (ExamplePanel) tabs.getWidget( event.getSelectedItem() );
057            History.newItem( panel.getTitle(), false /* Don't fire event */);
058        }
059    });
060
061    History.addValueChangeHandler(new ValueChangeHandler<String>() {
062      public void onValueChange(ValueChangeEvent<String> event) {
063        String[] tokens = event.getValue().split("\\|");
064        if (tokens.length >= 1) {
065          String tabName = tokens[0];
066          for (Widget w : tabs) {
067            ExamplePanel panel = (ExamplePanel) w;
068            if (panel.getTitle().equals(tabName)) {
069              // Select the tab again only if it is not already selected
070
071              int index = tabs.getWidgetIndex(panel);
072              if (tabs.getSelectedIndex() != index) {
073                tabs.selectTab(index);
074              }
075            }
076
077            if (tokens.length >= 2) {
078              String exampleName = tokens[1];
079              panel.selectExample(exampleName);
080
081            } else {
082              panel.selectFirstExample();
083            }
084          }
085        }
086      }
087    });
088  }
089

126

129
130      examples.put("Model/View/Presenter",

136  }
137
138  private class ExamplePanel extends DockLayoutPanel {
139    private VerticalPanel buttons = new VerticalPanel();
140    private DeckPanel deck = new DeckPanel();
141    private final String title;
142
143    public ExamplePanel( String title ) {
144      super(Unit.EM);
145      
146      this.title = title;
147      addWest(buttons, 10);
148      add(deck);
149    }
150
151    public void add(final Example example, final String name) {
152      deck.add(example);
153      buttons.add(new ToggleButton(name, new ClickHandler() {
154        public void onClick(ClickEvent event) {
155          for (Widget w : buttons) {
156            if (!w.equals(event.getSource())) {
157              ((ToggleButton) w).setDown(false);
158            }
159          }
160
161          deck.showWidget(deck.getWidgetIndex(example));
162          History.newItem(title + "|" + name, false /* Don't fire event */);
163        }
164      }));
165    }
166
167    public void selectFirst() {
168      ((ToggleButton) buttons.getWidget(0)).setDown(true);
169      deck.showWidget(0);
170    }
171
172    @Override
173    public String getTitle() {
174        return this.title;
175    }
176
177    /**
178     * Selects the example with the given name
179     */
180    public void selectExample(String name) {
181      for (Widget w : buttons) {
182        ToggleButton button = (ToggleButton) w;
183        if (button.getText().equals(name)) {
184          selectButton(button);
185
186        } else {
187          button.setDown(false);
188        }
189
190      }
191    }
192
193    private void selectButton(ToggleButton button) {
194      button.setDown(true);
195      deck.showWidget(buttons.getWidgetIndex(button));
196    }
197
198    public void selectFirstExample() {
199      for (int i = 0; i < buttons.getWidgetCount(); i++) {
200        ToggleButton button = (ToggleButton) buttons.getWidget(i);
201        if (i == 0) {
202          selectButton(button);
203
204        } else {
205          button.setDown(false);
206        }
207      }
208    }
209  }
