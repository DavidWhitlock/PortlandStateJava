001package edu.pdx.cs410J.rmi;
002
003import com.google.gwt.user.client.rpc.SerializationException;
004import com.google.gwt.user.client.rpc.SerializationStreamReader;
005import com.google.gwt.user.client.rpc.SerializationStreamWriter;
006
007import java.util.Map;
008
009/**
010 * A custom field serializer for the {@link edu.pdx.cs410J.rmi.Movie} class.  This allows us to use
011 * <code>Movie</code>s with GWT-RPC services without modifying the <code>Movie</code>
012 * class to have a zero-argument constructor
013 *
014 * http://code.google.com/p/wogwt/wiki/CustomFieldSerializer
015 */
016public class Movie_CustomFieldSerializer {
017
018  public static void deserialize(SerializationStreamReader reader, Movie movie)
019    throws SerializationException {
020
021  }
022
023  public static Movie instantiate(SerializationStreamReader reader)
024    throws SerializationException {
025
026
027    long id = reader.readLong();
028    String title = reader.readString();
029    int year = reader.readInt();
030    Map<String, Long> characters = (Map<String, Long>) reader.readObject();
031
032    Movie movie = new Movie(title, year);
033    movie.setId(id);
034    for (Map.Entry<String, Long> entry : characters.entrySet()) {
035      movie.addCharacter(entry.getKey(), entry.getValue());
036    }
037    return movie;
038  }
039
040  public static void serialize(SerializationStreamWriter writer, Movie movie)
041    throws SerializationException {
042
043    writer.writeLong(movie.getId());
044    writer.writeString(movie.getTitle());
045    writer.writeInt(movie.getYear());
046    writer.writeObject(movie.getCharacters());
047  }
048}
