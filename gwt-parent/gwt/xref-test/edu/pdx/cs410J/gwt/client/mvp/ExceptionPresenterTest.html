8   import static edu.pdx.cs410J.gwt.client.mvp.ExceptionPresenter.*;
9   
10  /**
11   * Tests that an exception triggers the {@link ExceptionPresenter}
12   */
13  public class ExceptionPresenterTest {
14  
15    /**
16     * Tests that when an exception occurs in an {@link MvpCallback}, the exception view is displayed
17     */
18    @Test
19    public void testExceptionViewDisplayed() {
20      HandlerManager eventBus = new HandlerManager(null);
21  
22      MvpCallback callback = new MvpCallback<Void>(eventBus) {
23  
24        @Override
25        public void onSuccess(Void aVoid) {
26          fail("onSuccess should not be invoked");
27        }
28      };
29  
30      Display view = mock(Display.class);
31      new ExceptionPresenter(view, eventBus);
32  
33      String message = "This is a message";
34      Throwable ex = new Throwable(message);
35      ex.fillInStackTrace();
36  
37      callback.onFailure(ex);
38  
39      verify(view).show();
40      verify(view).setMessage(message);
41      verify(view).setStackTrace(ex.getStackTrace());
42    }
43  }
