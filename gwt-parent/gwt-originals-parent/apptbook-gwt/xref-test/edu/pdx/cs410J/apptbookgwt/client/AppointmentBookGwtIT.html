11  /**
12   * An integration test for the airline GWT UI.  Remember that GWTTestCase is JUnit 3 style.
13   * So, test methods names must begin with "test".
14   * And since this test code is compiled into JavaScript, you can't use hamcrest matchers.  :(
15   */
16  public class AppointmentBookGwtIT extends GWTTestCase {
17    @Override
18    public String getModuleName() {
19      return "edu.pdx.cs410J.apptbookgwt.AppointmentBookIntegrationTests";
20    }
21  
22    @Test
23    public void testClickingButtonAlertsWithAppointmentInformation() {
24      final CapturingAlerter alerter = new CapturingAlerter();
25  
26      AppointmentBookGwt ui = new AppointmentBookGwt(alerter);
27      click(ui.button);
28  
29      Timer verify = new Timer() {
30        @Override
31        public void run() {
32          String message = alerter.getMessage();
33          assertNotNull(message);
34          assertTrue(message, message.contains("My Owner's appointment book with 1 appointments"));
35          finishTest();
36        }
37      };
38  
39      // Wait for the RPC call to return
40      verify.schedule(500);
41  
42      delayTestFinish(1000);
43    }
44  
45    /**
46     * Clicks a <code>Button</code>
47     *
48     * One would think that you could testing clicking a button with Button.click(), but it looks
49     * like you need to fire the native event instead.  Lame.
50     *
51     * @param button
52     *        The button to click
53     */
54    private void click(Button button) {
55      NativeEvent event = Document.get().createClickEvent(0, 0, 0, 0, 0, false, false, false, false);
56      DomEvent.fireNativeEvent(event, button);
57    }
58  
59    private class CapturingAlerter implements AppointmentBookGwt.Alerter {
60      private String message;
61  
62      @Override
63      public void alert(String message) {
64        this.message = message;
65      }
66  
67      public String getMessage() {
68        return message;
69      }
70    }
71  }
