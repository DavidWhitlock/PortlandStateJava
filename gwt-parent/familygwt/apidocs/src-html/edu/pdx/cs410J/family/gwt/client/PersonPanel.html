001package edu.pdx.cs410J.family.gwt.client;
002
003import com.google.gwt.user.client.ui.*;
004import com.google.gwt.i18n.client.DateTimeFormat;
005import edu.pdx.cs410J.family.Person;
006import edu.pdx.cs410J.family.Marriage;
007
008import java.util.List;
009import java.util.ArrayList;
010import java.util.Date;
011
012/**
013 * Displays information about a {@link Person}
014 */
015public class PersonPanel extends DockPanel {
016  private final FamilyTreeGWT familyUI;
017
018  private Person person;
019  private List<Marriage> marriages = new ArrayList<Marriage>();
020
021
022  private final Label name;
023  private final Label dob;
024  private final Label dod;
025  private final Label fatherName;
026  private final Label motherName;
027  private final ListBox marriagesList;
028
029  /**
030   * Creates a <code>PersonPanel</code> for displaying
031   * <code>Person</code>s
032   * @param familyUI The parent
033   */
034  public PersonPanel(FamilyTreeGWT familyUI) {
035    this.familyUI = familyUI;
036    VerticalPanel panel = new VerticalPanel();
037    panel.setSpacing(3);
038
039    this.name = new Label();
040    this.name.setHorizontalAlignment(HasHorizontalAlignment.ALIGN_LEFT);
041    panel.add(this.name);
042
043    this.dob = new Label();
044    this.dob.setHorizontalAlignment(HasHorizontalAlignment.ALIGN_LEFT);
045    panel.add(this.dob);
046
047    this.dod = new Label();
048    this.dod.setHorizontalAlignment(HasHorizontalAlignment.ALIGN_LEFT);
049    panel.add(this.dod);
050
051    this.fatherName = new Label();
052    this.fatherName.setHorizontalAlignment(HasHorizontalAlignment.ALIGN_LEFT);
053    this.fatherName.addClickListener(new ClickListener() {
054      public void onClick(Widget widget) {
055        PersonPanel.this.familyUI.displayFather();
056      }
057    });
058    panel.add(this.fatherName);
059
060    this.motherName = new Label();
061    this.motherName.setHorizontalAlignment(HasHorizontalAlignment.ALIGN_LEFT);
062    this.motherName.addClickListener(new ClickListener() {
063      public void onClick(Widget widget) {
064        PersonPanel.this.familyUI.displayMother();
065      }
066    });
067    panel.add(this.motherName);
068
069    fillInLabels();
070
071    panel.add(new Label("Marriages:"));
072    this.marriagesList = new ListBox();
073    this.marriagesList.setVisibleItemCount(10);
074    this.marriagesList.setWidth("100%");
075    ScrollPanel scroll = new ScrollPanel(this.marriagesList);
076    panel.add(scroll);
077    panel.setWidth("100%");
078//    panel.setCellWidth(scroll, "100%");
079 
080    this.add(panel, DockPanel.NORTH);
081  }
082
083  private void fillInLabels() {
084    this.name.setText("Name:");
085    this.dob.setText("Born:");
086    this.dod.setText("Died:");
087    this.fatherName.setText("Father:");
088    this.motherName.setText("Mother:");
089  }
090
091  /**
092   * Dispays information aout a <code>Person</code> in this
093   * <code>PersonPanel</code>
094   */
095  public void showPerson(Person person) {
096    this.person = person;
097
098    fillInLabels();
099
100    if (this.person == null) {
101      // When no person is selected
102      return;
103    }
104
105    DateTimeFormat df = DateTimeFormat.getLongDateFormat();
106
107    this.name.setText("Name: " + person.getFullName());
108
109    Date dob = person.getDateOfBirth();
110    if (dob != null) {
111      this.dob.setText("Born: " + df.format(dob));
112    }
113
114    Date dod = person.getDateOfDeath();
115    if (dod != null) {
116      this.dod.setText("Died: " + df.format(dod));
117    }
118
119    Person father = person.getFather();
120    if (father != null) {
121      this.fatherName.setText("Father: " + father.getFullName());
122    }
123
124    Person mother = person.getMother();
125    if (mother != null) {
126      this.motherName.setText("Mother: " + mother.getFullName());
127    }
128
129    this.marriagesList.clear();
130
131    this.marriages = new ArrayList<Marriage>(person.getMarriages());
132
133    for (Marriage marriage : this.marriages) {
134      StringBuffer sb = new StringBuffer();
135      Person spouse = (marriage.getHusband().getId() == person.getId()
136                       ? marriage.getWife()
137                       : marriage.getHusband());
138      sb.append(spouse.getFullName());
139
140      Date date = marriage.getDate();
141      if (date != null) {
142        sb.append(" on " + df.format(date));
143      }
144
145      String location = marriage.getLocation();
146      if (location != null || !location.equals("")) {
147        sb.append(" in " + location);
148      }
149
150      this.marriagesList.addItem(sb.toString());
151    }
152  }
153
154  /**
155   * Displays and returns the current person's father
156   */
157  public Person showFather() {
158    Person father = this.person.getFather();
159    if (father != null) {
160      this.showPerson(father);
161    }
162    return father;
163  }
164
165  /**
166   * Displays the current person's mother
167   */
168  public Person showMother() {
169    Person mother = this.person.getMother();
170    if (mother != null) {
171      this.showPerson(mother);
172    }
173    return mother;
174  }
175}
