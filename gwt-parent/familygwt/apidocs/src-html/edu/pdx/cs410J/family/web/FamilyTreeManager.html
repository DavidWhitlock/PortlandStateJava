001package edu.pdx.cs410J.family.web;
002
003import edu.pdx.cs410J.family.FamilyTree;
004
005import javax.servlet.http.HttpServletRequest;
006import javax.servlet.http.HttpSession;
007
008/**
009 * Manages the <code>FamilyTree</code> associated with a user's session
010 */
011public class FamilyTreeManager {
012  private static final String ATTRIBUTE = "FamilyTree";
013
014
015  /**
016   * Sets the <code>FamilyTree</code> associated with the current user's session
017   */
018  public static void setFamilyTree(FamilyTree tree, HttpServletRequest request) {
019    HttpSession session = request.getSession(true);
020    session.setAttribute(ATTRIBUTE, tree);
021  }
022
023  /**
024   * Returns the <code>FamilyTree</code> currently associated with the user's session
025   */
026  public static FamilyTree getFamilyTree(HttpServletRequest request) {
027    HttpSession session = request.getSession(true);
028    return (FamilyTree) session.getAttribute(ATTRIBUTE);
029  }
030}
