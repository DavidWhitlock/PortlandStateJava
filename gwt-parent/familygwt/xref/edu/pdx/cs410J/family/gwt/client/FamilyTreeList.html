3   import com.google.gwt.user.client.ui.ListBox;
4   import edu.pdx.cs410J.family.Person;
5   import edu.pdx.cs410J.family.FamilyTree;
6   
7   import java.util.*;
8   
9   /**
10   * Lists the people in a family tree
11   */
12  public class FamilyTreeList extends ListBox {
13    private Map<Integer, Person> indexToPerson = new HashMap<Integer, Person>();
14  
15    public FamilyTreeList() {
16      setVisibleItemCount(5);
17      setSize("200px", "100%");
18    }
19  
20    /**
21     * Fills in the <code>JList</code> with the contents of a
22     * <code>FamilyTree</code>.
23     */
24    public void fillInList(FamilyTree tree) {
25      SortedSet<Person> sortedPeople = new TreeSet<Person>(new Comparator<Person>() {
26          // Sort id's from lowest to highest
27  
28          public int compare(Person p1, Person p2) {
29            return p1.getId() - p2.getId();
30          }
31  
32          public boolean equals(Object o) {
33            return true;
34          }
35        });
36  
37      sortedPeople.addAll(tree.getPeople());
38      this.clear();
39  
40      Iterator<Person> iter = sortedPeople.iterator();
41      for (int i = 0; iter.hasNext(); i++) {
42        Person person = iter.next();
43        this.addItem(person.getFullName() + " (" + person.getId() + ")");
44        indexToPerson.put(new Integer(i), person);
45      }
46  
47    }
48  
49    /**
50     * Returns the person in the list that is currently selected
51     */
52    public Person getSelectedPerson() {
53      return indexToPerson.get(getSelectedIndex());
54    }
55  
56    /**
57     * Sets the selected person.
58     */
59    public void setSelectedPerson(Person person) {
60      Integer index = null;
61      for (Map.Entry<Integer, Person> integerPersonEntry : this.indexToPerson.entrySet()) {
62        Map.Entry<Integer, Person> entry = integerPersonEntry;
63        if (entry.getValue().equals(person)) {
64          index = entry.getKey();
65        }
66      }
67  
68      if (index != null) {
69        this.setSelectedIndex(index.intValue());
70      }
71    }
72  
73  }
