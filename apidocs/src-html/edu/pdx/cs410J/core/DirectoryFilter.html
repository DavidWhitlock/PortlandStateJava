001package edu.pdx.cs410J.core;
002
003import java.io.*;
004
005/**
006 * This class is a <code>FileFilter</code> that only accepts files
007 * that are directories.
008 */
009public class DirectoryFilter implements FileFilter {
010
011  /**
012   * Returns <code>true</code> if a file is a directory
013   */
014  public boolean accept(File file) {
015    if (file.isDirectory()) {
016      return true;
017
018    } else {
019      return false;
020    }
021  }
022
023}
