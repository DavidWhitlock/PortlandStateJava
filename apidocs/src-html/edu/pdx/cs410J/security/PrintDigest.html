001package edu.pdx.cs410J.security;
002
003import java.security.*;
004
005/**
006 * This program computes the MD5 digest of a sentence specified on the
007 * command line.
008 */
009public class PrintDigest {
010
011  public static void main(String[] args) {
012    String message = args[0];
013    MessageDigest algorithm = null;
014    try {
015      algorithm = MessageDigest.getInstance("MD5");
016    } catch (NoSuchAlgorithmException ex) {
017      ex.printStackTrace(System.err);
018      System.exit(1);
019    }
020
021    algorithm.reset();
022    algorithm.update(message.getBytes());
023    byte[] digest = algorithm.digest();
024
025    StringBuffer hexString = new StringBuffer();
026    for (int i = 0; i < digest.length; i++) {
027      String s = Integer.toHexString(0xFF & digest[i]);
028      hexString.append(s);
029    }
030    
031    System.out.println("Message: " + message);
032    System.out.println("Digest: " + hexString.toString());
033  }
034}
