001package edu.pdx.cs410J.phonebillgwt.server;
002
003import com.google.gwt.user.server.rpc.RemoteServiceServlet;
004import edu.pdx.cs410J.phonebillgwt.client.PhoneBill;
005import edu.pdx.cs410J.phonebillgwt.client.PhoneCall;
006import edu.pdx.cs410J.phonebillgwt.client.PhoneBillService;
007
008/**
009 * The server-side implementation of the Phone Bill service
010 */
011public class PhoneBillServiceImpl extends RemoteServiceServlet implements PhoneBillService
012{
013  @Override
014  public PhoneBill getPhoneBill() {
015    PhoneBill phonebill = new PhoneBill();
016    phonebill.addPhoneCall(new PhoneCall());
017    return phonebill;
018  }
019
020  @Override
021  public void throwUndeclaredException() {
022    throw new IllegalStateException("Expected undeclared exception");
023  }
024
025  @Override
026  public void throwDeclaredException() throws IllegalStateException {
027    throw new IllegalStateException("Expected declared exception");
028  }
029
030  /**
031   * Log unhandled exceptions to standard error
032   *
033   * @param unhandled
034   *        The exception that wasn't handled
035   */
036  @Override
037  protected void doUnexpectedFailure(Throwable unhandled) {
038    unhandled.printStackTrace(System.err);
039    super.doUnexpectedFailure(unhandled);
040  }
041
042}
