001package edu.pdx.cs410J.grader;
002
003import com.google.common.annotations.VisibleForTesting;
004import edu.pdx.cs410J.ParserException;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.*;
009import java.util.ArrayList;
010import java.util.List;
011import java.util.Optional;
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015public class ProjectGradesImporter {
016  static final Pattern scorePattern = Pattern.compile("(\\d+\\.?\\d*) out of (\\d+\\.?\\d*)", Pattern.CASE_INSENSITIVE);
017
018  private final GradeBook gradeBook;
019  private final Assignment assignment;
020  private final Logger logger;
021
022  public ProjectGradesImporter(GradeBook gradeBook, Assignment assignment, Logger logger) {
023    this.gradeBook = gradeBook;
024    this.assignment = assignment;
025    this.logger = logger;
026  }
027
028  public static ProjectScore getScoreFrom(Reader reader) throws ScoreNotFoundException {
029    BufferedReader br = new BufferedReader(reader);
030    Optional<String> scoreLine = br.lines().filter(scorePattern.asPredicate()).findFirst();
031
032    if (scoreLine.isPresent()) {
033      Matcher matcher = scorePattern.matcher(scoreLine.get());
034      if (matcher.find()) {
035        return new ProjectScore(matcher.group(1), matcher.group(2));
036
037      } else {
038        throw new IllegalStateException("Matcher didn't match \"" + scoreLine.get() + "\"");
039      }
040
041    } else {
042      throw new ScoreNotFoundException();
043    }
044
045  }
046
047  public void recordScoreFromProjectReport(String studentId, Reader report) throws ScoreNotFoundException {
048    ProjectScore score = getScoreFrom(report);
049
050    if (score.getTotalPoints() != this.assignment.getPoints()) {
051      String message = "Assignment " + this.assignment.getName() + " should be worth " + this.assignment.getPoints() +
052        " points, but the report for " + studentId + " was out of " + score.getTotalPoints();
053      throw new IllegalStateException(message);
054    }
055
056    Optional<Student> maybeStudent = gradeBook.getStudent(studentId);
057    if (!maybeStudent.isPresent()) {
058      maybeStudent = gradeBook.getStudentWithSsn(studentId);
059    }
060    if (!maybeStudent.isPresent()) {
061      warn("Student \"" + studentId + "\" not found in gradebook");
062      return;
063    }
064
065    Student student = maybeStudent.get();
066    Grade grade = student.getGrade(this.assignment);
067    if (grade == null) {
068      grade = new Grade(assignment, score.getScore());
069      student.setGrade(assignment.getName(), grade);
070
071    } else {
072      grade.setScore(score.getScore());
073    }
074
075    info("Recorded grade of " + score.getScore() + " for " + studentId);
076  }
077
078  private void info(String message) {
079    this.logger.info(message);
080  }
081
082  private void warn(String message) {
083    logger.warn(message);
084  }
085
086  static class ProjectScore {
087    private final double score;
088    private final double totalPoints;
089
090    private ProjectScore(String score, String totalPoints) {
091      this.score = Double.parseDouble(score);
092      this.totalPoints = Double.parseDouble(totalPoints);
093    }
094
095    public double getScore() {
096      return this.score;
097    }
098
099    public double getTotalPoints() {
100      return this.totalPoints;
101    }
102  }
103
104  public static void main(String[] args) {
105    String gradeBookFileName = null;
106    String assignmentName = null;
107    List<String> projectFileNames = new ArrayList<>();
108
109    for (String arg : args) {
110      if (gradeBookFileName == null) {
111        gradeBookFileName = arg;
112
113      } else if (assignmentName == null) {
114        assignmentName = arg;
115
116      } else {
117        projectFileNames.add(arg);
118      }
119    }
120
121    usageIfNull(gradeBookFileName, "Missing grade book file");
122    usageIfNull(assignmentName, "Missing assignment name");
123    usageIfEmpty(projectFileNames, "No project file names provided");
124
125    GradeBook gradeBook = getGradeBook(gradeBookFileName);
126    Assignment assignment = getAssignment(assignmentName, gradeBook);
127    Logger logger = LoggerFactory.getLogger(ProjectGradesImporter.class.getPackage().getName());
128
129    ProjectGradesImporter importer = new ProjectGradesImporter(gradeBook, assignment, logger);
130
131    for (String projectFileName : projectFileNames) {
132      File projectFile = getProjectFile(projectFileName);
133      String studentId = getStudentIdFromFileName(projectFile);
134      try {
135        importer.recordScoreFromProjectReport(studentId, new FileReader(projectFile));
136
137      } catch (IllegalStateException ex) {
138        throw new IllegalStateException("While recording score from " + projectFileName, ex);
139
140      } catch (FileNotFoundException ex) {
141        throw new IllegalStateException("Could not find file \"" + projectFile + "\"");
142      } catch (ScoreNotFoundException e) {
143        logger.warn("Could not find score in " + projectFileName);
144      }
145    }
146
147    saveGradeBookIfModified(gradeBook, gradeBookFileName);
148  }
149
150  private static void saveGradeBookIfModified(GradeBook gradeBook, String gradeBookFileName) {
151    if (gradeBook.isDirty()) {
152      File file = new File(gradeBookFileName);
153      try {
154        XmlDumper dumper = new XmlDumper(file);
155        dumper.dump(gradeBook);
156
157      } catch (IOException e) {
158        usage("Can't write grade book in \"" + gradeBookFileName + "\"");
159      }
160    }
161  }
162
163  private static String getStudentIdFromFileName(File projectFile) {
164    String fileName = projectFile.getName();
165    int index = fileName.lastIndexOf('.');
166    if (index < 0) {
167      return usage("Project file \"" + fileName + "\" does not have a file extension");
168    }
169
170    return fileName.substring(0, index);
171  }
172
173  private static File getProjectFile(String projectFileName) {
174    File projectFile = new File(projectFileName);
175    if (!projectFile.exists()) {
176      return usage("Project file \"" + projectFileName + "\" does not exist");
177
178    } else if (!projectFile.isFile()) {
179      return usage("Project file \"" + projectFileName + "\" is not a file");
180    }
181
182    return projectFile;
183  }
184
185  private static Assignment getAssignment(String assignmentName, GradeBook gradeBook) {
186    Assignment assignment = gradeBook.getAssignment(assignmentName);
187    if (assignment == null) {
188      return usage("Could not find assignment \"" + assignmentName + "\" in grade book");
189    }
190    return assignment;
191  }
192
193  private static GradeBook getGradeBook(String gradeBookFileName) {
194    try {
195      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFileName);
196      return parser.parse();
197
198    } catch (IOException ex) {
199      return usage("Couldn't read grade book file \"" + gradeBookFileName + "\"");
200
201    } catch (ParserException ex) {
202      return usage("Couldn't parse grade book file \"" + gradeBookFileName + "\"");
203    }
204  }
205
206  private static void usageIfEmpty(List<String> list, String message) {
207    if (list.isEmpty()) {
208      usage(message);
209    }
210  }
211
212  private static void usageIfNull(String argument, String message) {
213    if (argument == null) {
214      usage(message);
215    }
216  }
217
218  private static <T> T usage(String message) {
219    PrintStream err = System.err;
220
221    err.println("** " + message);
222
223    err.println("java ProjectGradesImporter gradeBookFileName assignmentName projectFileName+");
224    err.println();
225    err.println("Imports grades (of the form \"5.8 out of 6.0\") from project reports into a grade book.");
226    err.println("The name of the project report file begins with the student's id.");
227    err.println();
228    err.println("  gradeBookFileName     Name of file containing grade book");
229    err.println("  assignmentName        Assignment/project whose score is to be recorded");
230    err.println("  projectFileName       Name of file containing graded project");
231    err.println();
232
233    System.exit(1);
234
235    return null;
236  }
