017/**
018 * This panel displays the contents of a grade book.  On the left-hand
019 * side of the panel is a list of the students in the class.  In the
020 * center of the panel is a tabbed pane that lets you view/edit
021 * information about the class, a student, or a student's grades.
022 */
023@SuppressWarnings("serial")
024public class GradeBookPanel extends JPanel {
025  static final int BY_ID = 1;
026  static final int BY_NAME = 2;
027
028  private GradeBook book;  // The GradeBook being displayed
029  private StudentsModel model;
030
031  // GUI components we care about
032  private JList students;
033  private ClassPanel classPanel;
034  private StudentPanel studentPanel;
035  private GradePanel gradePanel;
036
037  /**
038   * Creates a new <code>GradeBookPanel</code> and lays out all of its
039   * components.
040   */
041  public GradeBookPanel(JFrame parent) {
042    this.classPanel = new ClassPanel(parent);
043    this.studentPanel = new StudentPanel();
044    this.gradePanel = new GradePanel();
045
046    JTabbedPane tabs = new JTabbedPane(JTabbedPane.TOP);
047    tabs.addTab("Class", this.classPanel);
048    tabs.addTab("Student", this.studentPanel);
049    tabs.addTab("Grades", this.gradePanel);
050
051    JSplitPane split = new JSplitPane();
052    split.setRightComponent(tabs);
053
054    JPanel leftPanel = new JPanel();
055    leftPanel.setLayout(new BorderLayout());
056    leftPanel.add(new JLabel("Students", JLabel.CENTER),
057                  BorderLayout.NORTH);
058
059    this.students = new JList();
060    this.students.addListSelectionListener(new
061      ListSelectionListener() {
062        public void valueChanged(ListSelectionEvent e) {
063          if (GradeBookPanel.this.model == null) {
064            return;
065          }
066          
067          int index = students.getSelectedIndex();
068
069          if (index >= 0) {
070            String id = model.getIdAt(index);
071            displayStudent(id);
072
073          } else {
074            GradeBookPanel.this.studentPanel.clearContents();
075          }
076        }
077      });
078    this.students.setBorder(BorderFactory.createEmptyBorder(0, 2, 0, 0));
079
080    leftPanel.add(new JScrollPane(this.students), BorderLayout.CENTER);
081    JPanel buttons = new JPanel();
082    buttons.setLayout(new FlowLayout());
083    ButtonGroup group = new ButtonGroup();
084
085    JRadioButton idButton = new JRadioButton("By id");
086    idButton.addActionListener(new ActionListener() {
087        public void actionPerformed(ActionEvent e) {
088          displayStudents(BY_ID);
089        }
090      });
091    group.add(idButton);
092    buttons.add(idButton);
093
094    JRadioButton nameButton = new JRadioButton("By name");
095    nameButton.addActionListener(new ActionListener() {
096        public void actionPerformed(ActionEvent e) {
097          displayStudents(BY_NAME);
098        }
099      });
100    group.add(nameButton);
101    buttons.add(nameButton);
102    leftPanel.add(buttons, BorderLayout.SOUTH);
103
104    split.setLeftComponent(leftPanel);
105
106    this.setLayout(new BorderLayout());
107    this.add(split, BorderLayout.CENTER);
108  }
109
110  /**
111   * Displays information about a given student
112   */
113  private void displayStudent(String id) {
114    Student student =
115      this.book.getStudent(id).orElseThrow(() -> new IllegalStateException("No student with id " + id));
116    this.studentPanel.displayStudent(student);
117    this.gradePanel.displayStudent(student);
118  }
119
120  /**
121   * Displays the names of the students in the class as either their
122   * full names or by the ids.
123   */
124  private void displayStudents(int sortOrder) {
125    if (this.book == null) {
126      // Nothing to do
127      return;
128    }
129
130    this.model = new StudentsModel(this.book, sortOrder);
131    this.students.setModel(model);
132  }
133
134  /**
135   * Displays the contents of a given <code>GradeBook</code>
136   */
137  void displayGradeBook(GradeBook book) {
138    this.book = book;
139    this.displayStudents(BY_ID);
140    this.classPanel.displayGradeBook(book);
141    this.gradePanel.displayAssignmentsFor(book);
142    this.studentPanel.clearContents();
143  }
144
145  /**
146   * Test program
147   */
148  public static void main(String[] args) {
149    final String fileName = args[0];
150
151    GradeBook book = null;
152    try {
153      XmlGradeBookParser parser = new XmlGradeBookParser(fileName);
154      book = parser.parse();
155
156    } catch (FileNotFoundException ex) {
157      System.err.println("** Could not find file: " + ex.getMessage());
158      System.exit(1);
159      
160    } catch (IOException ex) {
161      System.err.println("** IOException during parsing: " + ex.getMessage());
162      System.exit(1);
163
164    } catch (ParserException ex) {
165      System.err.println("** Error during parsing: " + ex);
166      System.exit(1);
167    }
168
169    JFrame frame = new JFrame("GradeBookPanel test");
170    GradeBookPanel bookPanel = new GradeBookPanel(frame);
171    bookPanel.displayGradeBook(book);
172
173    final GradeBook theBook = book;
174    frame.addWindowListener(new WindowAdapter() {
175        public void windowClosing(WindowEvent e) {
176          // Write changes to grade book back to file
177          try {
178            XmlDumper dumper = new XmlDumper(fileName);
179            dumper.dump(theBook);
180
181          } catch (IOException ex) {
182            System.err.println("** Error while writing XML file: " + ex);
183          }
184
185          System.exit(1);
186        }
187      });
188
189    frame.getContentPane().add(bookPanel);
190    
191    frame.pack();
192    frame.setVisible(true);
193  }

197/**
198 * Class used for displaying either student ids or their full names in
199 * a <code>JList</code> 
200 */
201@SuppressWarnings("serial")
202class StudentsModel extends AbstractListModel {
203
204  private ArrayList<String> ids;  // ids in order
205  private ArrayList<String> data; // displayed names in order
206
207  public StudentsModel(final GradeBook book, final int sortedOrder) {
208    this.ids = new ArrayList<String>();
209    this.data = new ArrayList<String>();
210
211    SortedSet<Student> sortedStudents = new TreeSet<Student>(new Comparator<Student>() {
212        public int compare(Student s1, Student s2) {
213          if (sortedOrder == GradeBookPanel.BY_ID) {
214            // Sort by id
215            return s1.getId().compareTo(s2.getId());
216
217          } else if (sortedOrder == GradeBookPanel.BY_NAME) {
218            // Sort by last name
219            String name1 = s1.getLastName();
220            String name2 = s2.getLastName();
221
222            if (name1 == null) {
223              name1 = s1.getId();
224            }
225
226            if (name2 == null) {
227              name2 = s2.getId();
228            }
229
230            if (name1.equalsIgnoreCase(name2)) {
231              name1 = s1.getFirstName();
232              name2 = s2.getFirstName();
233
234              if (name1 == null) {
235                name1 = s1.getId();
236              }
237
238              if (name2 == null) {
239                name2 = s2.getId();
240              }
241
242            } else {
243              return name1.compareTo(name2);
244            }
245
246            if (name1.equalsIgnoreCase(name2)) {
247              return s1.getId().compareTo(s2.getId());
248
249            } else {
250              return name1.compareTo(name2);
251            }
252
253          } else {
254            throw new IllegalArgumentException("Unknown sort order: "
255                                               + sortedOrder);
256          }
257        }
258
259        public boolean equals(Object o) {
260          return o == this;
261        }
262      });
263
264    book.forEachStudent(sortedStudents::add);
265
266    for (Student student : sortedStudents) {
267      this.ids.add(student.getId());
268
269      if (sortedOrder == GradeBookPanel.BY_ID) {
270        // Display id
271        this.data.add(student.getId());
272
273      } else if (sortedOrder == GradeBookPanel.BY_NAME) {
274        // Display full name
275        String name = student.getFullName();
276        if (name == null || name.equals("")) {
277          name = student.getId();
278        }
279        this.data.add(name);
280
281      } else {
282        throw new IllegalArgumentException("Unknown sort order: "
283          + sortedOrder);
284      }
285    }
286
287  }
288
289  /**
290   * Returns the id of the student at the given index
291   */
292  public String getIdAt(int index) {
293    return (String) ids.get(index);
294  }
295
296  public Object getElementAt(int index) {
297    return data.get(index);
298  }
299
300  public int getSize() {
301    return this.data.size();
302  }
