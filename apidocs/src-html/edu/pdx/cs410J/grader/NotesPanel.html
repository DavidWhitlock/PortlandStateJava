001package edu.pdx.cs410J.grader;
002
003import java.awt.*;
004import java.awt.event.*;
005import java.util.*;
006import javax.swing.*;
007import javax.swing.border.*;
008import javax.swing.event.*;
009
010/**
011 * This panel displays and edits notes.
012 *
013 * @author David Whitlock
014 * @version $Revision: 1.6 $
015 * @since Fall 2000
016 */
017@SuppressWarnings("serial")
018public class NotesPanel extends JPanel {
019
020  private Notable notable;
021
022  // GUI components we care about
023  private JList notesList;
024  private JButton add;
025
026  /**
027   * Creates a new <code>NotePanel</code> for displaying the notes of
028   * some <code>Notable</code> object.
029   */
030  public NotesPanel() {
031    this.setLayout(new BorderLayout());
032
033    Border notesBorder = BorderFactory.createTitledBorder("Notes");
034    this.setBorder(notesBorder);
035
036    this.notesList = new JList();
037    this.add(new JScrollPane(notesList), BorderLayout.CENTER);
038
039    JPanel panel = new JPanel();
040    panel.setLayout(new FlowLayout());
041
042    final JTextField field = new JTextField(20);
043    panel.add(field);
044
045    ActionListener action = new ActionListener() {
046        public void actionPerformed(ActionEvent e) {
047          String note = field.getText();
048          if (note != null && !note.equals("") && notable != null) {
049            notable.addNote(note);
050            setNotable(notable);
051          }
052
053          field.setText("");
054        }
055      };
056    field.addActionListener(action);
057
058    this.add = new JButton("Add");
059    this.add.setEnabled(false);
060    this.add.addActionListener(action);
061    panel.add(this.add);
062
063    final JButton delete = new JButton("Delete");
064    delete.setEnabled(false);
065    delete.addActionListener(new ActionListener() {
066        public void actionPerformed(ActionEvent e) {
067          String note = (String) notesList.getSelectedValue();
068          if (note == null) {
069            return;
070          }
071
072          if (notable != null) {
073            notable.removeNote(note);
074            setNotable(notable);
075          }
076        }
077      });
078    notesList.addListSelectionListener(new ListSelectionListener() {
079        public void valueChanged(ListSelectionEvent e) {
080          if (notesList.isSelectionEmpty()) {
081            delete.setEnabled(false);
082
083          } else {
084            delete.setEnabled(true);
085          }
086        }
087      });
088    panel.add(delete);
089
090    this.add(panel, BorderLayout.SOUTH);
091  }
092
093  /**
094   * Sets the <code>Notable</code> that is being displayed/edited by
095   * this <code>NotesPanel</code>.
096   */
097  public void setNotable(Notable notable) {
098    clearNotes();   // Start from scratch
099    this.notable = notable;
100    this.notesList.setListData(notable.getNotes().toArray());
101    this.add.setEnabled(true);
102  }
103
104  /**
105   * Returns the <code>Notable</code> edited by this
106   * <code>NotesPanel</code>. 
107   */
108  public Notable getNotable() {
109    return this.notable;
110  }
111
112  /**
113   * Clears the contents of the notes list
114   */
115  public void clearNotes() {
116    this.notesList.setListData(new Vector());
117    this.notable = null;
118  }
119
120}
