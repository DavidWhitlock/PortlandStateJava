001package edu.pdx.cs410J.grader.poa;
002
003import java.time.LocalDateTime;
004
005public class POASubmission {
006
007  private final String subject;
008  private final String submitter;
009  private final LocalDateTime submitTime;
010  private String content;
011
012  private POASubmission(String subject, String submitter, LocalDateTime submitTime, String content) {
013    this.subject = subject;
014    this.submitter = submitter;
015    this.submitTime = submitTime;
016    this.content = content;
017  }
018
019  public static Builder builder() {
020    return new Builder();
021  }
022
023  public String getSubject() {
024    return subject;
025  }
026
027  public String getSubmitter() {
028    return submitter;
029  }
030
031  public LocalDateTime getSubmitTime() {
032    return submitTime;
033  }
034
035  @Override
036  public String toString() {
037    return String.format("POA from %s with subject %s on %s with %d characters of content",
038      getSubmitter(), getSubject(), getSubmitTime(), getContentLength());
039  }
040
041  private int getContentLength() {
042    String content = getContent();
043    if (content == null) {
044      return 0;
045
046    } else {
047      return content.length();
048    }
049  }
050
051  public String getContent() {
052    return content;
053  }
054
055  public static class Builder {
056    private String subject;
057    private String submitter;
058    private LocalDateTime submitTime;
059    private String content;
060
061    public POASubmission create() {
062      assertNotNull("subject", subject);
063      assertNotNull("submitter", submitter);
064      assertNotNull("submitTime", submitTime);
065      assertNotNull("content", content);
066
067      return new POASubmission(subject, submitter, submitTime, content);
068    }
069
070    private void assertNotNull(String description, Object object) {
071      if (object == null) {
072        throw new IllegalStateException(description + " cannot be null");
073      }
074    }
075
076    public Builder setSubject(String subject) {
077      this.subject = subject;
078      return this;
079    }
080
081    public Builder setSubmitter(String submitter) {
082      this.submitter = submitter;
083      return this;
084    }
085
086    public Builder setSubmitTime(LocalDateTime submitTime) {
087      this.submitTime = submitTime;
088      return this;
089    }
090
091    public Builder setContent(String content) {
092      this.content = content;
093      return this;
094    }
095  }
096}
