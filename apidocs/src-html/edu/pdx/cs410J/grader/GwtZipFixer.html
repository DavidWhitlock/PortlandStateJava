001package edu.pdx.cs410J.grader;
002
003import com.google.common.annotations.VisibleForTesting;
004import com.google.common.io.ByteStreams;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import java.io.*;
009import java.util.ArrayList;
010import java.util.List;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013import java.util.stream.Stream;
014import java.util.zip.ZipEntry;
015import java.util.zip.ZipInputStream;
016import java.util.zip.ZipOutputStream;
017
018public class GwtZipFixer {
019
020  private static final Logger logger = LoggerFactory.getLogger("edu.pdx.cs410J.grader");
021
022  public static void main(String[] args) {
023    String outputDirectoryName = null;
024    List<String> zipFileNames = new ArrayList<>();
025
026    for (String arg : args) {
027      if (outputDirectoryName == null) {
028        outputDirectoryName = arg;
029
030      } else {
031        zipFileNames.add(arg);
032      }
033    }
034
035    if (outputDirectoryName == null) {
036      usage("Missing output directory");
037      return;
038    }
039
040    if (zipFileNames.isEmpty()) {
041      usage("Missing zip file");
042      return;
043    }
044
045    File outputDirectory = new File(outputDirectoryName);
046
047    for (String zipFileName : zipFileNames) {
048      File zipFile = new File(zipFileName);
049      try {
050        fixZipFile(zipFile, outputDirectory);
051
052      } catch (IOException e) {
053        usage("While fixing zip file " + zipFile + ": " + e);
054      }
055
056    }
057  }
058
059  private static void fixZipFile(File zipFile, File outputDirectory) throws IOException {
060    File fixedZipFile = getFixedZipFile(zipFile, outputDirectory);
061
062    try (
063      ZipInputStream input = new ZipInputStream(new FileInputStream(zipFile));
064      ZipOutputStream output = new ZipOutputStream(new FileOutputStream(fixedZipFile))
065    ) {
066      for (ZipEntry entry = input.getNextEntry(); entry != null; entry = input.getNextEntry()) {
067        String entryName = entry.getName();
068        String fixedEntryName = getFixedEntryName(entryName);
069
070        if (fixedEntryName != null) {
071          logger.debug(entryName + " fixed to " + fixedEntryName);
072
073          ZipEntry fixedEntry = new ZipEntry(fixedEntryName);
074          output.putNextEntry(fixedEntry);
075
076          ByteStreams.copy(input, output);
077
078        } else {
079          logger.debug(entryName + " ignored");
080        }
081
082        output.flush();
083      }
084    }
085  }
086
087  @VisibleForTesting
088  static String getFixedEntryName(String entryName) {
089    Stream<String> ignore = Stream.of("__MACOSX", "/test", "/it", "/target/", ".DS_Store");
090    if (ignore.anyMatch(entryName::contains)) {
091      return null;
092    }
093
094    if (entryName.contains("pom.xml")) {
095      return "pom.xml";
096    }
097
098    String fixedName = replaceRegexWithPrefix(entryName, ".*/main/(.*)", "src/main/");
099
100    if (fixedName == null) {
101      fixedName = replaceRegexWithPrefix(entryName, ".*/java/(.*)", "src/main/java/");
102    }
103
104    if (fixedName == null) {
105      fixedName = replaceRegexWithPrefix(entryName, ".*/resources/(.*)", "src/main/resources/");
106    }
107
108    if (fixedName == null) {
109      fixedName = replaceRegexWithPrefix(entryName, ".*/webapp/(.*)", "src/main/webapp/");
110    }
111
112    if (fixedName == null) {
113      fixedName = replaceRegexWithPrefix(entryName, "^edu/(.*)", "src/main/java/edu/");
114    }
115
116    return fixedName;
117  }
118
119  private static String replaceRegexWithPrefix(String entryName, String regex, String replaceWithPrefix) {
120    Pattern pattern = Pattern.compile(regex);
121    Matcher matcher = pattern.matcher(entryName);
122
123    if (matcher.matches()) {
124      String portionUnderMain = matcher.group(1);
125      if ("".equals(portionUnderMain)) {
126        return null;
127
128      } else {
129        return replaceWithPrefix + portionUnderMain;
130      }
131
132    } else {
133      return null;
134    }
135  }
136
137  private static File getFixedZipFile(File zipFile, File outputDirectory) {
138    return new File(outputDirectory, zipFile.getName());
139  }
140
141  private static void usage(String message) {
142    PrintStream err = System.err;
143
144    err.println("+++ " + message);
145    err.println();
146    err.println("usage: java GwtZipFixer outputDirectory zipFile+");
147    err.println("    outputDirectory     Name of direct into which fixed zip files should be written");
148    err.println("    zipFile             Name of zip file to be fixed");
149    err.println();
150    err.println("Fixes the contents of a zip file submitted for the GWT ");
151    err.println("project so that it will work with the grading scripts");
152    err.println();
153
154    System.exit(1);
155  }
156}
