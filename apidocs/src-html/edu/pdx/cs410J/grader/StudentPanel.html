020@SuppressWarnings("serial")
021public class StudentPanel extends JPanel {

023  private Student student;

038  /**
039   * Creates and lays out a new <code>StudentPanel</code>
040   */
041  public StudentPanel() {
042    this.setLayout(new BorderLayout());
043
044    // Center panel that contains information about the student
045    JPanel infoPanel = new JPanel();
046    infoPanel.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 2));
047    infoPanel.setLayout(new BorderLayout());
048
049    JPanel labels = new JPanel();
050    labels.setLayout(new GridLayout(0, 1));
051    labels.setBorder(BorderFactory.createEmptyBorder(0, 2, 0, 2));
052    labels.add(new JLabel("Id:"));
053    labels.add(new JLabel("First name:"));
054    labels.add(new JLabel("Last name:"));
055    labels.add(new JLabel("Nick name:"));
056    labels.add(new JLabel("Email:"));
057    labels.add(new JLabel("SSN:"));
058    labels.add(new JLabel("Major:"));
059    labels.add(new JLabel("D2L Id:"));
060    labels.add(new JLabel("Letter Grade:"));
061    labels.add(new JLabel("Enrolled Section:"));
062
063    JPanel fields = new JPanel();
064    fields.setLayout(new GridLayout(0, 1));
065    this.idField = new JTextField(10);
066    this.idField.setEditable(false);
067    fields.add(this.idField);
068    this.firstNameField = new JTextField(10);
069    fields.add(this.firstNameField);
070    this.lastNameField = new JTextField(10);
071    fields.add(this.lastNameField);
072    this.nickNameField = new JTextField(10);
073    fields.add(this.nickNameField);
074    this.emailField = new JTextField(15);
075    fields.add(this.emailField);
076    this.ssnField = new JTextField(11);
077    fields.add(this.ssnField);
078    this.majorField = new JTextField(15);
079    fields.add(this.majorField);
080    this.d2lIdField = new JTextField(10);
081    fields.add(this.d2lIdField);
082    this.letterGraderComboBox = createLetterGradeComboBox();
083    fields.add(this.letterGraderComboBox);
084    this.enrolledSectionComboBox= createEnrolledSectionComboBox();
085    fields.add(this.enrolledSectionComboBox);
086
087    infoPanel.add(labels, BorderLayout.WEST);
088    infoPanel.add(fields, BorderLayout.CENTER);
089
090    this.add(infoPanel, BorderLayout.NORTH);
091
092    // Add a NotePanel
093    this.notes = new NotesPanel();
094    this.add(notes, BorderLayout.CENTER);
095
096    // Update button
097    JPanel buttons = new JPanel();
098    buttons.setLayout(new FlowLayout());
099    JButton update = new JButton("Update");
100    update.addActionListener(new ActionListener() {
101        public void actionPerformed(ActionEvent e) {
102          if (StudentPanel.this.student != null) {
103            updateStudent(StudentPanel.this.student);
104          }
105        }
106      });
107    buttons.add(update);
108
109    this.add(buttons, BorderLayout.SOUTH);
110  }
111
112  private JComboBox<Student.Section> createEnrolledSectionComboBox() {
113    return createComboBoxWithEnumValues(Student.Section.values());
114  }
115
116  private JComboBox<LetterGrade> createLetterGradeComboBox() {
117    return createComboBoxWithEnumValues(LetterGrade.values());
118  }
119
120  private <E> JComboBox<E> createComboBoxWithEnumValues(E[] values) {
121    Vector<E> vector = new Vector<>();
122    vector.add(null);
123    vector.addAll(Arrays.asList(values));
124    return new JComboBox<>(vector);
125  }

127  /**
128   * Clears the contents of the student fields
129   */
130  void clearContents() {
131    this.idField.setText("");
132    this.firstNameField.setText("");
133    this.lastNameField.setText("");
134    this.nickNameField.setText("");
135    this.emailField.setText("");
136    this.ssnField.setText("");
137    this.majorField.setText("");
138    this.d2lIdField.setText("");
139    this.letterGraderComboBox.setSelectedItem(null);
140    this.enrolledSectionComboBox.setSelectedItem(null);
141    this.notes.clearNotes();
142  }

144  /**
145   * Displays a <code>Student</code> in this
146   * <code>StudentPanel</code>.
147   */
148  public void displayStudent(Student student) {
149    this.student = student;
150    this.clearContents();
151
152//     System.out.println("Displaying student: " +
153//                        student.getDescription());
154
155    // Fill in GUI components
156    this.idField.setText(student.getId());
157
158    String firstName = student.getFirstName();
159    if (firstName != null && !firstName.equals("")) {
160      this.firstNameField.setText(firstName);
161    }
162
163    String lastName = student.getLastName();
164    if (lastName != null && !lastName.equals("")) {
165      this.lastNameField.setText(lastName);
166    }
167
168    String nickName = student.getNickName();
169    if (nickName != null && !nickName.equals("")) {
170      this.nickNameField.setText(nickName);
171    }
172
173    String email = student.getEmail();
174    if (email != null && !email.equals("")) {
175      this.emailField.setText(email);
176    }
177
178    String ssn = student.getSsn();
179    if (ssn != null && !ssn.equals("")) {
180      this.ssnField.setText(ssn);
181    }
182
183    String major = student.getMajor();
184    if (major != null && !major.equals("")) {
185      this.majorField.setText(major);
186    }
187
188    String d2lId = student.getD2LId();
189    if (d2lId != null && !d2lId.equals("")) {
190      this.d2lIdField.setText(d2lId);
191    }
192
193    LetterGrade letterGrade = student.getLetterGrade();
194    if (letterGrade != null) {
195      this.letterGraderComboBox.setSelectedItem(letterGrade);
196    }
197
198    Student.Section section = student.getEnrolledSection();
199    if (section != null) {
200      this.enrolledSectionComboBox.setSelectedItem(section);
201    }
202
203    this.notes.setNotable(student);
204  }
205
206  /**
207   * Updates a <code>Student</code> based on the contents of this
208   * <code>StudentPanel</code>.
209   */
210  private void updateStudent(Student student) {
211    String firstName = this.firstNameField.getText();
212    if (firstName != null) {
213      student.setFirstName(firstName);
214    }
215
216    String lastName = this.lastNameField.getText();
217    if (lastName != null) {
218      if (lastName.equals("")) {
219        student.setLastName(null);
220      } else {
221        student.setLastName(lastName);
222      }
223    }
224
225    String nickName = this.nickNameField.getText();
226    if (nickName != null) {
227      if (nickName.equals("")) {
228        student.setNickName(null);
229      } else {
230        student.setNickName(nickName);
231      }
232    }
233
234    String email = this.emailField.getText();
235    if (email != null) {
236      if (email.equals("")) {
237        student.setEmail(null);
238      } else {
239        student.setEmail(email);
240      }
241    }
242
243    String ssn = this.ssnField.getText();
244    if (ssn != null) {
245      if (ssn.equals("")) {
246        student.setSsn(null);
247      } else {
248        student.setSsn(ssn);
249      }
250    }
251
252    String major = this.majorField.getText();
253    if (major != null) {
254      if (major.equals("")) {
255        student.setMajor(null);
256      } else {
257        student.setMajor(major);
258      }
259    }
260    
261    String d2lId = this.d2lIdField.getText();
262    if (d2lId != null) {
263      if (d2lId.equals("")) {
264        student.setD2LId(null);
265      } else {
266        student.setD2LId(d2lId);
267      }
268    }
269
270    LetterGrade letterGrade = (LetterGrade) this.letterGraderComboBox.getSelectedItem();
271    student.setLetterGrade(letterGrade);
272
273    Student.Section section = (Student.Section) this.enrolledSectionComboBox.getSelectedItem();
274    student.setEnrolledSection(section);
275
276    // The NotesPanel takes care of adding notes
277
278//     System.out.println("Updated student: " +
279//                        student.getDescription());
280  }

282  /**
283   * Test program
284   */
285  public static void main(String[] args) {
286    final String fileName = args[0];
287    String studentId = args[1];
288
289    GradeBook book = null;
290    try {
291      XmlGradeBookParser parser = new XmlGradeBookParser(fileName);
292      book = parser.parse();
293
294    } catch (FileNotFoundException ex) {
295      System.err.println("** Could not find file: " + ex.getMessage());
296      System.exit(1);
297      
298    } catch (IOException ex) {
299      System.err.println("** IOException during parsing: " + ex.getMessage());
300      System.exit(1);
301
302    } catch (ParserException ex) {
303      System.err.println("** Error during parsing: " + ex);
304      System.exit(1);
305    }
306
307    Student student = book.getStudent(studentId).orElseThrow(cannotFindStudent(studentId));
308
309    StudentPanel studentPanel = new StudentPanel();
310    studentPanel.displayStudent(student);
311
312    JFrame frame = new JFrame("StudentPanel test");
313    final GradeBook theBook = book;
314    frame.addWindowListener(new WindowAdapter() {
315        public void windowClosing(WindowEvent e) {
316         // Write changes to grade book back to file
317          try {
318            XmlDumper dumper = new XmlDumper(fileName);
319            dumper.dump(theBook);
320
321          } catch (IOException ex) {
322            System.err.println("** Error while writing XML file: " + ex);
323          }
324
325          System.exit(1);
326        }
327      });
328
329    frame.getContentPane().add(studentPanel);
330    
331    frame.pack();
332    frame.setVisible(true);
333  }
334
335  @SuppressWarnings("ThrowableInstanceNeverThrown")
336  private static Supplier<IllegalStateException> cannotFindStudent(String studentId) {
337    return () -> new IllegalStateException("Cannot find student with id " + studentId);
338  }
