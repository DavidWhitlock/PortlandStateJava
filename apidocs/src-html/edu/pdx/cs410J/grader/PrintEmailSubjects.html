001package edu.pdx.cs410J.grader;
002
003import java.io.*;
004import java.util.*;
005import javax.mail.*;
006
007/**
008 * This program reads your INBOX and prints out the subjects of each
009 * of the emails.
010 */
011public class PrintEmailSubjects {
012  private static final PrintStream out = System.out;
013  private static final PrintStream err = System.err;
014
015  /**
016   * Prints usage information about this program
017   */
018  private static void usage(String s) {
019    err.println("\n** " + s + "\n");
020    err.println("usage: java PrintEmailSubjects mailbox");
021    err.println("");
022    System.exit(1);
023  }
024
025  public static void main(String[] args) throws Throwable {
026    String mailbox = null;
027    boolean DEBUG = false;
028
029    for (int i = 0; i < args.length; i++) {
030      if (args[i].equals("-debug")) {
031        DEBUG = true;
032
033      } else if (mailbox == null) {
034        mailbox = args[i];
035
036      } else {
037        usage("Spurious command line: " + args);
038      }
039    }
040
041    if (mailbox == null) {
042      usage("Missing mailbox");
043    }
044
045    // Get a connection to the mail server
046    Properties props = new Properties();
047    Session session = Session.getDefaultInstance(props, null);
048    session.setDebug(DEBUG);
049
050    Provider[] providers = session.getProviders();
051    for (int i = 0; i < providers.length; i++) {
052      Provider provider = providers[i];
053      out.println(provider);
054    }
055
056    Store store = session.getStore("imap");
057    Folder folder = store.getFolder(mailbox);
058    Message[] messages = folder.getMessages();
059    out.println("Folder " + folder.getName() + " with " +
060                messages.length + " messages");
061    for (int i = 0; i < messages.length; i++) {
062      Message message = messages[i];
063      out.println("  " + i + ": " + message.getSubject());
064    }
065
066    out.println("");
067
068  }
069
070}
