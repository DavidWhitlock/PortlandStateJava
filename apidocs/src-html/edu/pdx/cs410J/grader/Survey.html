015/**
016 * This program presents a survey that all students in CS410J should
017 * answer.  It emails the results of the survey to the TA and emails a

021  private static final PrintWriter out = new PrintWriter(System.out, true);

024    new BufferedReader(new InputStreamReader(System.in));

030  /**
031   * Returns a textual summary of a <code>Student</code>
032   */
033  private static String getSummary(Student student) {
034    StringBuilder sb = new StringBuilder();
035    sb.append("Name: ").append(student.getFullName()).append("\n");
036    sb.append("UNIX login: ").append(student.getId()).append("\n");
037    if (student.getEmail() != null) {
038      sb.append("Email: ").append(student.getEmail()).append("\n");
039    }
040    if (student.getSsn() != null) {
041      sb.append("Student id: ").append(student.getSsn()).append("\n");
042    }
043    if (student.getMajor() != null) {
044      sb.append("Major: ").append(student.getMajor()).append("\n");
045    }
046    sb.append("Enrolled in: ").append(student.getEnrolledSection().asString()).append("\n");
047    return sb.toString();
048  }

050  /**
051   * Ask the student a question and return his response
052   */
053  private static String ask(String question) {
054    out.print(breakUpInto80CharacterLines(question));
055    out.print(" ");
056    out.flush();
057
058    String response = null;
059    try {
060      response = in.readLine();
061
062    } catch (IOException ex) {
063      printErrorMessageAndExit("** IOException while reading response", ex);
064    }
065
066    return response;
067  }

081  public static void main(String[] args) {
082    parseCommandLine(args);
083
084    printIntroduction();
085
086    Student student = gatherStudentInformation();
087
088    String learn = ask("What do you hope to learn in CS410J?");
089    String comments = ask("What else would you like to tell me?");
090
091    addNotesToStudent(student, learn, comments);
092
093    emailSurveyResults(student, learn, comments);
094  }

096  private static void addNotesToStudent(Student student, String learn, String comments) {
097    if (isNotEmpty(learn)) {
098      student.addNote(student.getFullName() + " would like to learn " + learn);
099    }
100
101    if (isNotEmpty(comments)) {
102      student.addNote(student.getFullName() + " has these comments: " + comments);
103    }
104  }

106  private static Student gatherStudentInformation() {
107    String firstName = ask("What is your first name?");
108    String lastName = ask("What is your last name?");
109    String nickName = ask("What is your nickname? (Leave blank if " +
110                          "you don't have one)");
111    String id = ask("MANDATORY: What is your UNIX login id?");
112
113    if (isEmpty(id)) {
114      printErrorMessageAndExit("** You must enter a valid UNIX login id");
115    }
116
117    Student student = new Student(id);
118    setValueIfNotEmpty(firstName, student::setFirstName);
119    setValueIfNotEmpty(lastName, student::setLastName);
120    setValueIfNotEmpty(nickName, student::setNickName);
121
122    askQuestionAndSetValue("What is your email address (doesn't have to be PSU)?", student::setEmail);
123    askQuestionAndSetValue("What is your student id (XXXXXXXXX)?", student::setSsn);
124    askQuestionAndSetValue("What is your major?", student::setMajor);
125
126    askEnrolledSectionQuestion(student);
127
128    return student;
129  }

131  private static void askEnrolledSectionQuestion(Student student) {
132    String answer = ask("MANDATORY: Are you enrolled in the undergraduate or graduate section of this course? [u/g]");
133    if (isEmpty(answer)) {
134      printErrorMessageAndExit("Missing enrolled section. Please enter a \"u\" or \"g\"");
135    }
136
137    if (answer.toLowerCase().startsWith("u")) {
138      student.setEnrolledSection(Student.Section.UNDERGRADUATE);
139
140    } else if (answer.toLowerCase().startsWith("g")) {
141      student.setEnrolledSection(Student.Section.GRADUATE);
142
143    } else {
144      printErrorMessageAndExit("Unknown section \"" + answer + "\".  Please enter a \"u\" or \"g\"");
145    }
146
147  }

149  private static void askQuestionAndSetValue(String question, Consumer<String> setter) {
150    String answer = ask(question);
151    setValueIfNotEmpty(answer, setter);
152  }
153
154  static void setValueIfNotEmpty(String string, Consumer<String> setter) {
155    if (isNotEmpty(string)) {
156      setter.accept(string);
157    }
158  }

160  private static void emailSurveyResults(Student student, String learn, String comments) {
161    String summary = verifyInformation(student);
162
163    // Email the results of the survey to the TA and CC the student
164
165    MimeMessage message = createEmailMessage(student);
166    MimeBodyPart textPart = createEmailText(learn, comments, summary);
167    MimeBodyPart xmlFilePart = createXmlAttachment(student);
168    addAttachmentsAndSendEmail(message, textPart, xmlFilePart);
169  }
170
171  private static void addAttachmentsAndSendEmail(MimeMessage message, MimeBodyPart textPart, MimeBodyPart filePart) {
172    // Finally, add the attachments to the message and send it
173    try {
174      Multipart mp = new MimeMultipart();
175      mp.addBodyPart(textPart);
176      mp.addBodyPart(filePart);
177
178      message.setContent(mp);
179
180      Transport.send(message);
181
182      logSentEmail(message);
183
184    } catch (MessagingException ex) {
185      printErrorMessageAndExit("** Exception while adding parts and sending", ex);
186    }
187  }
188
189  private static void logSentEmail(MimeMessage message) throws MessagingException {
190    StringBuilder sb = new StringBuilder();
191    sb.append("\nAn email with with subject \"");
192    sb.append(message.getSubject());
193    sb.append("\" was sent to ");
194
195    Address[] recipients = message.getAllRecipients();
196    for (int i = 0; i < recipients.length; i++) {
197      Address recipient = recipients[i];
198      sb.append(recipient);
199      if (i < recipients.length - 2) {
200        sb.append(", ");
201
202      } else if (i < recipients.length - 1) {
203        sb.append(" and ");
204      }
205    }
206
207    out.println(breakUpInto80CharacterLines(sb.toString()));
208  }

210  private static MimeBodyPart createXmlAttachment(Student student) {
211    byte[] xmlBytes = getXmlBytes(student);
212
213    if (saveStudentXmlFile) {
214      writeStudentXmlToFile(xmlBytes, student);
215    }
216
217    DataSource ds = new ByteArrayDataSource(xmlBytes, "text/xml");
218    DataHandler dh = new DataHandler(ds);
219    MimeBodyPart filePart = new MimeBodyPart();
220    try {
221      String xmlFileTitle = student.getId() + ".xml";
222
223      filePart.setDataHandler(dh);
224      filePart.setFileName(xmlFileTitle);
225      filePart.setDescription("XML file for " + student.getFullName());
226
227    } catch (MessagingException ex) {
228      printErrorMessageAndExit("** Exception with file part", ex);
229    }
230    return filePart;
231  }
232
233  private static void writeStudentXmlToFile(byte[] xmlBytes, Student student) {
234    File directory = new File(System.getProperty("user.dir"));
235    File file = new File(directory, student.getId() + ".xml");
236    try (FileOutputStream fos = new FileOutputStream(file)) {
237      fos.write(xmlBytes);
238      fos.flush();
239
240    } catch (IOException e) {
241      printErrorMessageAndExit("Could not write student XML file: " + file, e);
242    }
243
244    out.println("\nSaved student XML file to " + file + "\n");
245  }

247  private static MimeBodyPart createEmailText(String learn, String comments, String summary) {
248    // Create the text portion of the message
249    StringBuilder text = new StringBuilder();
250    text.append("Results of CS410J Survey:\n\n");
251    text.append(summary);
252    text.append("\n\nWhat do you hope to learn in CS410J?\n\n");
253    text.append(learn);
254    text.append("\n\nIs there anything else you'd like to tell me?\n\n");
255    text.append(comments);
256    text.append("\n\nThanks for filling out this survey!\n\nDave");
257
258    MimeBodyPart textPart = new MimeBodyPart();
259    try {
260      textPart.setContent(text.toString(), "text/plain");
261
262      // Try not to display text as separate attachment
263      textPart.setDisposition("inline");
264
265    } catch (MessagingException ex) {
266      printErrorMessageAndExit("** Exception with text part", ex);
267    }
268    return textPart;
269  }

271  private static MimeMessage createEmailMessage(Student student) {
272    MimeMessage message = null;
273    try {
274      message = newEmailTo(newEmailSession(false), TA_EMAIL, "CS410J Survey for " + student.getFullName());
275
276      String studentEmail = student.getEmail();
277      if (studentEmail != null) {
278        InternetAddress[] cc = {new InternetAddress(studentEmail)};
279        message.setRecipients(Message.RecipientType.CC, cc);
280      }
281
282    } catch (AddressException ex) {
283      printErrorMessageAndExit("** Exception with email address", ex);
284
285    } catch (MessagingException ex) {
286      printErrorMessageAndExit("** Exception while setting recipients email", ex);
287    }
288    return message;
289  }

291  private static byte[] getXmlBytes(Student student) {
292    // Create a temporary "file" to hold the Student's XML file.  We
293    // use a byte array so that potentially sensitive data (SSN, etc.)
294    // is not written to disk
295    byte[] bytes = null;
296
297    Document xmlDoc = XmlDumper.toXml(student);
298
299
300    try {
301      bytes = XmlHelper.getBytesForXmlDocument(xmlDoc);
302
303    } catch (TransformerException ex) {
304      ex.printStackTrace(System.err);
305      System.exit(1);
306    }
307    return bytes;
308  }

310  private static String verifyInformation(Student student) {
311    String summary = getSummary(student);
312
313    out.println("\nYou entered the following information about " +
314                "yourself:\n");
315    out.println(summary);
316
317    String verify = ask("\nIs this information correct (y/n)?");
318    if (!verify.equals("y")) {
319      printErrorMessageAndExit("** Not sending information.  Exiting.");
320    }
321    return summary;
322  }

344  private static void printIntroduction() {
345    // Ask the student a bunch of questions
346    String welcome =
347      "Welcome to the CS410J Survey Program.  I'd like to ask you a couple of " +
348      "questions about yourself.  Except for your UNIX login id and the section " +
349      "that you are enrolled in, no question" +
350      "is mandatory.  Your answers will be emailed to the TA and a receipt " +
351      "will be emailed to you.";
352
353    out.println("");
354    out.println(breakUpInto80CharacterLines(welcome));
355    out.println("");
356  }

358  @VisibleForTesting
359  static String breakUpInto80CharacterLines(String message) {
360    StringBuilder sb = new StringBuilder();
361    int currentLineLength = 0;
362    String[] words = message.split(" ");
363    for (String word : words) {
364      if (currentLineLength + word.length() > 80) {
365        sb.append('\n');
366        sb.append(word);
367        currentLineLength = word.length();
368
369      } else {
370        if (currentLineLength > 0) {
371          sb.append(' ');
372        }
373        sb.append(word);
374        currentLineLength += word.length() + 1;
375      }
376
377    }
378    return sb.toString();
379  }

381  private static void parseCommandLine(String[] args) {
382    // Parse the command line
383    for (int i = 0; i < args.length; i++) {
384      String arg = args[i];
385      if (arg.equals("-mailServer")) {
386        if (++i >= args.length) {
387          err.println("** Missing mail server name");
388          usage();
389        }
390
391        serverName = arg;
392
393      } else if (arg.equals("-saveStudentXmlFile")) {
394        saveStudentXmlFile = true;
395
396      } else if (arg.startsWith("-")) {
397        err.println("** Unknown command line option: " + arg);
398        usage();
399
400      } else {
401        err.println("** Spurious command line: " + arg);
402        usage();
403      }
404    }
405  }
