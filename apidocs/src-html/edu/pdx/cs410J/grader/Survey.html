015/**
016 * This program presents a survey that all students in CS410J should
017 * answer.  It emails the results of the survey to the TA and emails a

021  private static final PrintWriter out = new PrintWriter(System.out, true);

028  /**
029   * Returns a textual summary of a <code>Student</code>
030   */
031  private static String getSummary(Student student) {
032    StringBuilder sb = new StringBuilder();
033    sb.append("Name: ").append(student.getFullName()).append("\n");
034    sb.append("UNIX login: ").append(student.getId()).append("\n");
035    if (student.getEmail() != null) {
036      sb.append("Email: ").append(student.getEmail()).append("\n");
037    }
038    if (student.getSsn() != null) {
039      sb.append("Student id: ").append(student.getSsn()).append("\n");
040    }
041    if (student.getMajor() != null) {
042      sb.append("Major: ").append(student.getMajor()).append("\n");
043    }
044    sb.append("Enrolled in: ").append(student.getEnrolledSection().asString()).append("\n");
045    return sb.toString();
046  }

048  /**
049   * Ask the student a question and return his response
050   */
051  private static String ask(String question) {
052    out.print(breakUpInto80CharacterLines(question));
053    out.print(" ");
054    out.flush();
055
056    String response = null;
057    try {
058      response = in.readLine();
059
060    } catch (IOException ex) {
061      printErrorMessageAndExit("** IOException while reading response", ex);
062    }
063
064    return response;
065  }

094  private static void addNotesToStudent(Student student, String learn, String comments) {
095    if (isNotEmpty(learn)) {
096      student.addNote(student.getFullName() + " would like to learn " + learn);
097    }
098
099    if (isNotEmpty(comments)) {
100      student.addNote(student.getFullName() + " has these comments: " + comments);
101    }
102  }

104  private static Student gatherStudentInformation() {
105    String firstName = ask("What is your first name?");
106    String lastName = ask("What is your last name?");
107    String nickName = ask("What is your nickname? (Leave blank if " +
108                          "you don't have one)");
109    String id = ask("MANDATORY: What is your UNIX login id?");
110
111    if (isEmpty(id)) {
112      printErrorMessageAndExit("** You must enter a valid UNIX login id");
113
114    } else if (isEmailAddress(id)) {
115      printErrorMessageAndExit("** Your student id cannot be an email address");
116    }
117
118    Student student = new Student(id);
119    setValueIfNotEmpty(firstName, student::setFirstName);
120    setValueIfNotEmpty(lastName, student::setLastName);
121    setValueIfNotEmpty(nickName, student::setNickName);
122
123    askQuestionAndSetValue("What is your email address (doesn't have to be PSU)?", student::setEmail);
124    askQuestionAndSetValue("What is your student id (XXXXXXXXX)?", student::setSsn);
125    askQuestionAndSetValue("What is your major?", student::setMajor);
126
127    askEnrolledSectionQuestion(student);
128
129    return student;
130  }

132  @VisibleForTesting
133  static boolean isEmailAddress(String id) {
134    try {
135      boolean strict = true;
136      new InternetAddress(id, strict);
137      return true;
138
139    } catch (AddressException e) {
140      return false;
141    }
142  }

144  private static void askEnrolledSectionQuestion(Student student) {
145    String answer = ask("MANDATORY: Are you enrolled in the undergraduate or graduate section of this course? [u/g]");
146    if (isEmpty(answer)) {
147      printErrorMessageAndExit("Missing enrolled section. Please enter a \"u\" or \"g\"");
148    }
149
150    if (answer.toLowerCase().startsWith("u")) {
151      student.setEnrolledSection(Student.Section.UNDERGRADUATE);
152
153    } else if (answer.toLowerCase().startsWith("g")) {
154      student.setEnrolledSection(Student.Section.GRADUATE);
155
156    } else {
157      printErrorMessageAndExit("Unknown section \"" + answer + "\".  Please enter a \"u\" or \"g\"");
158    }
159
160  }

162  private static void askQuestionAndSetValue(String question, Consumer<String> setter) {
163    String answer = ask(question);
164    setValueIfNotEmpty(answer, setter);
165  }
166
167  static void setValueIfNotEmpty(String string, Consumer<String> setter) {
168    if (isNotEmpty(string)) {
169      setter.accept(string);
170    }
171  }

173  private static void emailSurveyResults(Student student, String learn, String comments) {
174    String summary = verifyInformation(student);
175
176    // Email the results of the survey to the TA and CC the student
177    out.println("Emailing your information to the Grader");
178
179    MimeMessage message = createEmailMessage(student);
180    MimeBodyPart textPart = createEmailText(learn, comments, summary);
181    MimeBodyPart xmlFilePart = createXmlAttachment(student);
182    addAttachmentsAndSendEmail(message, textPart, xmlFilePart);
183  }
184
185  private static void addAttachmentsAndSendEmail(MimeMessage message, MimeBodyPart textPart, MimeBodyPart filePart) {
186    // Finally, add the attachments to the message and send it
187    try {
188      Multipart mp = new MimeMultipart();
189      mp.addBodyPart(textPart);
190      mp.addBodyPart(filePart);
191
192      message.setContent(mp);
193
194      Transport.send(message);
195
196      logSentEmail(message);
197
198    } catch (MessagingException ex) {
199      printErrorMessageAndExit("** Exception while adding parts and sending", ex);
200    }
201  }
202
203  private static void logSentEmail(MimeMessage message) throws MessagingException {
204    StringBuilder sb = new StringBuilder();
205    sb.append("\nAn email with with subject \"");
206    sb.append(message.getSubject());
207    sb.append("\" was sent to ");
208
209    Address[] recipients = message.getAllRecipients();
210    for (int i = 0; i < recipients.length; i++) {
211      Address recipient = recipients[i];
212      sb.append(recipient);
213      if (i < recipients.length - 2) {
214        sb.append(", ");
215
216      } else if (i < recipients.length - 1) {
217        sb.append(" and ");
218      }
219    }
220
221    out.println(breakUpInto80CharacterLines(sb.toString()));
222  }

224  private static MimeBodyPart createXmlAttachment(Student student) {
225    byte[] xmlBytes = getXmlBytes(student);
226
227    if (saveStudentXmlFile) {
228      writeStudentXmlToFile(xmlBytes, student);
229    }
230
231    DataSource ds = new ByteArrayDataSource(xmlBytes, "text/xml");
232    DataHandler dh = new DataHandler(ds);
233    MimeBodyPart filePart = new MimeBodyPart();
234    try {
235      String xmlFileTitle = student.getId() + ".xml";
236
237      filePart.setDataHandler(dh);
238      filePart.setFileName(xmlFileTitle);
239      filePart.setDescription("XML file for " + student.getFullName());
240
241    } catch (MessagingException ex) {
242      printErrorMessageAndExit("** Exception with file part", ex);
243    }
244    return filePart;
245  }
246
247  private static void writeStudentXmlToFile(byte[] xmlBytes, Student student) {
248    File directory = new File(System.getProperty("user.dir"));
249    File file = new File(directory, student.getId() + ".xml");
250    try (FileOutputStream fos = new FileOutputStream(file)) {
251      fos.write(xmlBytes);
252      fos.flush();
253
254    } catch (IOException e) {
255      printErrorMessageAndExit("Could not write student XML file: " + file, e);
256    }
257
258    out.println("\nSaved student XML file to " + file + "\n");
259  }

261  private static MimeBodyPart createEmailText(String learn, String comments, String summary) {
262    // Create the text portion of the message
263    StringBuilder text = new StringBuilder();
264    text.append("Results of CS410J Survey:\n\n");
265    text.append(summary);
266    text.append("\n\nWhat do you hope to learn in CS410J?\n\n");
267    text.append(learn);
268    text.append("\n\nIs there anything else you'd like to tell me?\n\n");
269    text.append(comments);
270    text.append("\n\nThanks for filling out this survey!\n\nDave");
271
272    MimeBodyPart textPart = new MimeBodyPart();
273    try {
274      textPart.setContent(text.toString(), "text/plain");
275
276      // Try not to display text as separate attachment
277      textPart.setDisposition("inline");
278
279    } catch (MessagingException ex) {
280      printErrorMessageAndExit("** Exception with text part", ex);
281    }
282    return textPart;
283  }

285  private static MimeMessage createEmailMessage(Student student) {
286    MimeMessage message = null;
287    try {
288      InternetAddress studentEmail = newInternetAddress(student.getEmail(), student.getFullName());
289      String subject = "CS410J Survey for " + student.getFullName();
290      message = newEmailTo(newEmailSession(false), TA_EMAIL).from(studentEmail).withSubject(subject).createMessage();
291
292      InternetAddress[] cc = { studentEmail };
293      message.setRecipients(Message.RecipientType.CC, cc);
294
295    } catch (AddressException ex) {
296      printErrorMessageAndExit("** Exception with email address", ex);
297
298    } catch (MessagingException ex) {
299      printErrorMessageAndExit("** Exception while setting recipients email", ex);
300    }
301    return message;
302  }

304  private static byte[] getXmlBytes(Student student) {
305    // Create a temporary "file" to hold the Student's XML file.  We
306    // use a byte array so that potentially sensitive data (SSN, etc.)
307    // is not written to disk
308    byte[] bytes = null;
309
310    Document xmlDoc = XmlDumper.toXml(student);
311
312
313    try {
314      bytes = XmlHelper.getBytesForXmlDocument(xmlDoc);
315
316    } catch (TransformerException ex) {
317      ex.printStackTrace(System.err);
318      System.exit(1);
319    }
320    return bytes;
321  }

323  private static String verifyInformation(Student student) {
324    String summary = getSummary(student);
325
326    out.println("\nYou entered the following information about " +
327                "yourself:\n");
328    out.println(summary);
329
330    String verify = ask("\nIs this information correct (y/n)?");
331    if (!verify.equals("y")) {
332      printErrorMessageAndExit("** Not sending information.  Exiting.");
333    }
334    return summary;
335  }

351  }

357  private static void printIntroduction() {
358    // Ask the student a bunch of questions
359    String welcome =
360      "Welcome to the CS410J Survey Program.  I'd like to ask you a couple of " +
361      "questions about yourself.  Except for your UNIX login id and the section " +
362      "that you are enrolled in, no question" +
363      "is mandatory.  Your answers will be emailed to the TA and a receipt " +
364      "will be emailed to you.";
365
366    out.println("");
367    out.println(breakUpInto80CharacterLines(welcome));
368    out.println("");
369  }

371  @VisibleForTesting
372  static String breakUpInto80CharacterLines(String message) {
373    StringBuilder sb = new StringBuilder();
374    int currentLineLength = 0;
375    String[] words = message.split(" ");
376    for (String word : words) {
377      if (currentLineLength + word.length() > 80) {
378        sb.append('\n');
379        sb.append(word);
380        currentLineLength = word.length();
381
382      } else {
383        if (currentLineLength > 0) {
384          sb.append(' ');
385        }
386        sb.append(word);
387        currentLineLength += word.length() + 1;
388      }
389
390    }
391    return sb.toString();
392  }

394  private static void parseCommandLine(String[] args) {
395    // Parse the command line
396    for (int i = 0; i < args.length; i++) {
397      String arg = args[i];
398      if (arg.equals("-mailServer")) {
399        if (++i >= args.length) {
400          err.println("** Missing mail server name");
401          usage();
402        }
403
404        serverName = arg;
405
406      } else if (arg.equals("-saveStudentXmlFile")) {
407        saveStudentXmlFile = true;
408
409      } else if (arg.startsWith("-")) {
410        err.println("** Unknown command line option: " + arg);
411        usage();
412
413      } else {
414        err.println("** Spurious command line: " + arg);
415        usage();
416      }
417    }
418  }
