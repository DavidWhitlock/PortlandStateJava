001package edu.pdx.cs410J.grader;
002
003import com.google.common.annotations.VisibleForTesting;
004import edu.pdx.cs410J.ParserException;
005
006import java.io.*;
007
008public class FetchAndProcessGraderEmail {
009
010  public static void main(String[] args) {
011    String passwordFile = null;
012    String directoryName = null;
013    String gradeBookFile = null;
014    String whatToFetch = null;
015
016    for (String arg : args) {
017      if (passwordFile == null) {
018        passwordFile = arg;
019
020      } else if (directoryName == null) {
021        directoryName = arg;
022
023      } else if (gradeBookFile == null) {
024        gradeBookFile = arg;
025
026      } else if (whatToFetch == null) {
027        whatToFetch = arg;
028
029      } else {
030        usage("Extraneous argument: " + arg);
031      }
032    }
033
034    if (passwordFile == null) {
035      usage("Missing password file");
036    }
037
038    if (directoryName == null) {
039      usage("Missing directory name");
040    }
041
042    if (gradeBookFile == null) {
043      usage("Missing grade book file");
044    }
045
046    if (whatToFetch == null) {
047      usage("Missing kind of email to fetch");
048    }
049
050    String password = readGraderEmailPasswordFromFile(passwordFile);
051    File directory = getDirectory(directoryName);
052    GradeBook gradeBook = getGradeBook(gradeBookFile);
053
054    GraderEmailAccount account = new GraderEmailAccount(password);
055    fetchAndProcessGraderEmails(whatToFetch, account, directory, gradeBook);
056
057    saveGradeBookIfModified(gradeBook, gradeBookFile);
058  }
059
060  @VisibleForTesting
061  static void fetchAndProcessGraderEmails(String whatToFetch, GraderEmailAccount account, File directory, GradeBook gradeBook) {
062    StudentEmailAttachmentProcessor processor = getStudentEmailAttachmentProcessor(whatToFetch, directory, gradeBook);
063    account.fetchAttachmentsFromUnreadMessagesInFolder(processor.getEmailFolder(), processor);
064  }
065
066  private static void saveGradeBookIfModified(GradeBook gradeBook, String gradeBookFile) {
067    if (gradeBook.isDirty()) {
068      File file = new File(gradeBookFile);
069      try {
070        XmlDumper dumper = new XmlDumper(file);
071        dumper.dump(gradeBook);
072
073      } catch (IOException e) {
074        usage("Can't write grade book in \"" + gradeBookFile + "\"");
075      }
076    }
077  }
078
079  private static StudentEmailAttachmentProcessor getStudentEmailAttachmentProcessor(String whatToFetch, File directory, GradeBook gradeBook) {
080    if (whatToFetch.equalsIgnoreCase("projects")) {
081      return new ProjectSubmissionsProcessor(directory, gradeBook);
082
083    } else if (whatToFetch.equalsIgnoreCase("surveys")) {
084      return new SurveySubmissionsProcessor(directory, gradeBook);
085
086    } else if (whatToFetch.equalsIgnoreCase("gwtProjects")) {
087      return new GwtProjectSubmissionsProcessor(directory, gradeBook);
088
089    } else {
090      return usage("Cannot fetch \"" + whatToFetch + "\"");
091    }
092  }
093
094  private static GradeBook getGradeBook(String gradeBookFile) {
095    try {
096      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
097      return parser.parse();
098
099    } catch (IOException ex) {
100      return usage("Couldn't read grade book file \"" + gradeBookFile + "\"");
101
102    } catch (ParserException ex) {
103      return usage("Couldn't parse grade book file \"" + gradeBookFile + "\"");
104    }
105  }
106
107  private static File getDirectory(String directoryName) {
108    File directory = new File(directoryName);
109    if (!directory.exists()) {
110      return usage("Download directory \"" + directoryName + "\" does not exist");
111    }
112
113    if (!directory.isDirectory()) {
114      return usage("Download directory \"" + directoryName + "\" is not a directory");
115    }
116    return directory;
117  }
118
119  private static String readGraderEmailPasswordFromFile(String passwordFile) {
120    File file = new File(passwordFile);
121    if (!file.exists()) {
122      return usage("Password file \"" + passwordFile + "\" does not exist");
123    }
124
125    try {
126      BufferedReader br = new BufferedReader(new FileReader(file));
127      return br.readLine();
128
129    } catch (IOException e) {
130      return usage("Exception while reading password file");
131    }
132  }
133
134  private static <T> T usage(String message) {
135    PrintStream err = System.err;
136    err.println("** " + message);
137    err.println();
138    err.println("FetchAndProcessGraderEmail passwordFile directory gradeBookFile whatToFetch");
139    err.println("  passwordFile    File containing the password for the Grader's email account");
140    err.println("  directory       The directory in which to download attachments");
141    err.println("  gradeBookFile   Grade Book XML file related to this submission");
142    err.println("  whatToFetch     What kind of student emails should be fetched");
143    err.println("      projects    Project submissions");
144    err.println("      surveys     Student surveys");
145    err.println("      gwtProjects GWT Project submissions");
146    err.println();
147
148    System.exit(1);
149
150    throw new IllegalStateException("Shouldn't reach here");
151  }
152}
