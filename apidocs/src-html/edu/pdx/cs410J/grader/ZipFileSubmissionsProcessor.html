001package edu.pdx.cs410J.grader;
002
003import com.google.common.annotations.VisibleForTesting;
004import com.google.common.io.ByteStreams;
005
006import javax.mail.Address;
007import javax.mail.Message;
008import javax.mail.MessagingException;
009import javax.mail.internet.InternetAddress;
010import java.io.File;
011import java.io.FileOutputStream;
012import java.io.IOException;
013import java.io.InputStream;
014import java.time.LocalDateTime;
015import java.time.ZoneId;
016import java.util.Date;
017import java.util.List;
018import java.util.Optional;
019import java.util.function.Predicate;
020import java.util.stream.Stream;
021
022public abstract class ZipFileSubmissionsProcessor extends StudentEmailAttachmentProcessor {
023  public ZipFileSubmissionsProcessor(File directory, GradeBook gradeBook) {
024    super(directory, gradeBook);
025  }
026
027  @Override
028  public Iterable<? extends String> getSupportedContentTypes() {
029    return List.of("application/zip", "application/x-zip-compressed", "application/x-gzip");
030  }
031
032  @Override
033  public void processAttachment(Message message, String fileName, InputStream inputStream) {
034    String studentId;
035    try {
036      studentId = getIdOfStudentInGradeBookWhoSent(message);
037
038    } catch (SubmissionException ex) {
039      logException("While getting student id", ex);
040      return;
041    }
042
043    try {
044      File file = getLocationToWriteFile(fileName, studentId);
045      info("Writing \"" + fileName + "\" to " + file);
046
047      ByteStreams.copy(inputStream, new FileOutputStream(file));
048    } catch (IOException ex) {
049      logException("While writing \"" + fileName + "\" to \"" + directory + "\"", ex);
050    }
051
052    try {
053      noteSubmissionInGradeBook(message);
054
055    } catch (SubmissionException ex) {
056      logException("While noting submission from \"" + fileName + "\"", ex);
057    }
058  }
059
060  private String getIdOfStudentInGradeBookWhoSent(Message message) throws SubmissionException {
061    Student student = getStudentFromGradeBook(message);
062    return student.getId();
063  }
064
065  private void noteSubmissionInGradeBook(Message message) throws SubmissionException {
066    Student student = getStudentFromGradeBook(message);
067    Assignment project = getKoansProjectFromGradeBook();
068    String note = getSubmissionNote(message);
069
070    Grade grade = student.getGrade(project);
071    if (grade == null) {
072      grade = new Grade(project, Grade.NO_GRADE);
073      student.setGrade(project.getName(), grade);
074    }
075    grade.addNote(note);
076    grade.addSubmissionTime(getSentDate(message));
077  }
078
079  private String getSubmissionNote(Message message) throws SubmissionException {
080    String senderName = getSenderName(message);
081    LocalDateTime sentDate = getSentDate(message);
082
083    return getSubmissionNote(senderName, sentDate);
084  }
085
086  private LocalDateTime getSentDate(Message message) throws SubmissionException {
087    Date sentDate;
088    try {
089      sentDate = message.getSentDate();
090    } catch (MessagingException e) {
091      throw new SubmissionException("While getting the sent date", e);
092    }
093    return LocalDateTime.ofInstant(sentDate.toInstant(), ZoneId.systemDefault());
094  }
095
096  @VisibleForTesting
097  static String getSubmissionNote(String senderName, LocalDateTime submissionTime) {
098    return "Submitted by " + senderName + " on " + Submit.ManifestAttributes.formatSubmissionTime(submissionTime);
099  }
100
101  @VisibleForTesting
102  static String getSubmissionNoteUsingLegacyDateFormat(String senderName, LocalDateTime submissionTime) {
103    return "Submitted by " + senderName + " on " + Submit.ManifestAttributes.formatSubmissionTimeUsingLegacyFormat(submissionTime);
104  }
105
106  private Assignment getKoansProjectFromGradeBook() throws SubmissionException {
107    Assignment assignment = gradeBook.getAssignment(getAssignmentName());
108    if (assignment == null) {
109      throw new SubmissionException("Could not find \"koans\" assignment in grade book");
110    }
111    return assignment;
112  }
113
114  protected abstract String getAssignmentName();
115
116  private Student getStudentFromGradeBook(Message message) throws SubmissionException {
117    String senderName = getSenderName(message);
118    Optional<Student> student = getStudentWithName(senderName);
119    if (student.isPresent()) {
120      return student.get();
121    }
122
123    String senderAddress = getSenderEmailAddress(message);
124    student = getStudentWithEmailAddress(senderAddress);
125    return student.orElseThrow(() -> {
126      String m = "Couldn't find student named \"" + senderName + "\" with email address \"" +
127        senderAddress + "\" in gradebook";
128      return new SubmissionException(m);
129    });
130  }
131
132  private Optional<Student> getStudentWithEmailAddress(String senderAddress) {
133    return getStudentsInGradeBook().filter(doesStudentHaveEmailAddress(senderAddress)).findAny();
134  }
135
136  private Predicate<Student> doesStudentHaveEmailAddress(String address) {
137    return student -> student.getEmail().equals(address);
138  }
139
140  private String getSenderEmailAddress(Message message) throws SubmissionException {
141    Address from = getSender(message);
142
143    if (from instanceof InternetAddress) {
144      return getEmailAddress((InternetAddress) from);
145    } else {
146      return from.toString();
147    }
148  }
149
150  private Optional<Student> getStudentWithName(String studentName) {
151    return getStudentsInGradeBook().filter(doesStudentHaveName(studentName)).findAny();
152  }
153
154  private Stream<Student> getStudentsInGradeBook() {
155    return gradeBook.studentsStream();
156  }
157
158  private Predicate<Student> doesStudentHaveName(String studentName) {
159    return student -> getFullName(student).equals(studentName) || getNickName(student).equals(studentName);
160  }
161
162  private String getNickName(Student student) {
163    return student.getNickName() + " " + student.getLastName();
164  }
165
166  private String getFullName(Student student) {
167    return student.getFirstName() + " " + student.getLastName();
168  }
169
170  private File getLocationToWriteFile(String fileName, String studentId) throws IOException {
171    File dir = new File(directory, getAssignmentName());
172    if (!dir.exists() && !dir.mkdirs()) {
173      throw new IOException("Could not create directory \"" + dir + "\"");
174    }
175    return new File(dir, appendFileSuffixToStudentId(fileName, studentId));
176  }
177
178  private String appendFileSuffixToStudentId(String fileName, String studentId) {
179    return studentId + fileName.substring(fileName.lastIndexOf('.'));
180  }
181
182  private String getSenderName(Message message) throws SubmissionException {
183    Address from = getSender(message);
184
185    if (from instanceof InternetAddress) {
186      return getStudentName((InternetAddress) from);
187    } else {
188      return from.toString();
189    }
190  }
191
192  private Address getSender(Message message) throws SubmissionException {
193    Address from;
194    try {
195      from = message.getFrom()[0];
196
197    } catch (MessagingException ex) {
198      throw new SubmissionException("Could not get from address from " + message, ex);
199    }
200    return from;
201  }
202
203  private String getStudentName(InternetAddress from) {
204    if (from.getPersonal() != null) {
205      return from.getPersonal();
206
207    } else {
208      return getEmailAddress(from);
209    }
210  }
211
212  private String getEmailAddress(InternetAddress from) {
213    return from.getAddress();
214  }
215}
