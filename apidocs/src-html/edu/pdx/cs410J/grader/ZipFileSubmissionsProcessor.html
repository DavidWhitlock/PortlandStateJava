001package edu.pdx.cs410J.grader;
002
003import com.google.common.io.ByteStreams;
004
005import javax.mail.Address;
006import javax.mail.Message;
007import javax.mail.MessagingException;
008import javax.mail.internet.InternetAddress;
009import java.io.File;
010import java.io.FileOutputStream;
011import java.io.IOException;
012import java.io.InputStream;
013import java.text.DateFormat;
014import java.text.SimpleDateFormat;
015import java.util.Collections;
016import java.util.Optional;
017import java.util.function.Predicate;
018import java.util.stream.Stream;
019
020public abstract class ZipFileSubmissionsProcessor extends StudentEmailAttachmentProcessor {
021  public ZipFileSubmissionsProcessor(File directory, GradeBook gradeBook) {
022    super(directory, gradeBook);
023  }
024
025  @Override
026  public Iterable<? extends String> getSupportedContentTypes() {
027    return Collections.singleton("application/zip");
028  }
029
030  @Override
031  public void processAttachment(Message message, String fileName, InputStream inputStream) {
032    String studentId;
033    try {
034      studentId = getIdOfStudentInGradeBookWhoSent(message);
035
036    } catch (SubmissionException ex) {
037      logException("While getting student id", ex);
038      return;
039    }
040
041    try {
042      File file = getLocationToWriteFile(fileName, studentId);
043      info("Writing \"" + fileName + "\" to " + file);
044
045      ByteStreams.copy(inputStream, new FileOutputStream(file));
046    } catch (IOException ex) {
047      logException("While writing \"" + fileName + "\" to \"" + directory + "\"", ex);
048    }
049
050    try {
051      noteSubmissionInGradeBook(message);
052
053    } catch (SubmissionException ex) {
054      logException("While noting submission from \"" + fileName + "\"", ex);
055    }
056  }
057
058  private String getIdOfStudentInGradeBookWhoSent(Message message) throws SubmissionException {
059    Student student = getStudentFromGradeBook(message);
060    return student.getId();
061  }
062
063  private void noteSubmissionInGradeBook(Message message) throws SubmissionException {
064    Student student = getStudentFromGradeBook(message);
065    Assignment project = getKoansProjectFromGradeBook();
066    String note = getSubmissionNote(message);
067
068    Grade grade = student.getGrade(project);
069    if (grade == null) {
070      grade = new Grade(project, Grade.NO_GRADE);
071      student.setGrade(project.getName(), grade);
072    }
073    grade.addNote(note);
074  }
075
076  private String getSubmissionNote(Message message) throws SubmissionException {
077    return "Submitted by " + getSenderName(message) + " on " + getSentDate(message);
078  }
079
080  private String getSentDate(Message message) throws SubmissionException {
081    DateFormat format = new SimpleDateFormat("yyyy-MMM-dd HH:mm:ss");
082    try {
083      return format.format(message.getSentDate());
084    } catch (MessagingException e) {
085      throw new SubmissionException("While getting the sent date", e);
086    }
087  }
088
089  private Assignment getKoansProjectFromGradeBook() throws SubmissionException {
090    Assignment assignment = gradeBook.getAssignment(getAssignmentName());
091    if (assignment == null) {
092      throw new SubmissionException("Could not find \"koans\" assignment in grade book");
093    }
094    return assignment;
095  }
096
097  protected abstract String getAssignmentName();
098
099  private Student getStudentFromGradeBook(Message message) throws SubmissionException {
100    String senderName = getSenderName(message);
101    Optional<Student> student = getStudentWithName(senderName);
102    if (student.isPresent()) {
103      return student.get();
104    }
105
106    String senderAddress = getSenderEmailAddress(message);
107    student = getStudentWithEmailAddress(senderAddress);
108    return student.orElseThrow(() -> {
109      String m = "Couldn't find student named \"" + senderName + "\" with email address \"" +
110        senderAddress + "\" in gradebook";
111      return new SubmissionException(m);
112    });
113  }
114
115  private Optional<Student> getStudentWithEmailAddress(String senderAddress) {
116    return getStudentsInGradeBook().filter(doesStudentHaveEmailAddress(senderAddress)).findAny();
117  }
118
119  private Predicate<Student> doesStudentHaveEmailAddress(String address) {
120    return student -> student.getEmail().equals(address);
121  }
122
123  private String getSenderEmailAddress(Message message) throws SubmissionException {
124    Address from = getSender(message);
125
126    if (from instanceof InternetAddress) {
127      return getEmailAddress((InternetAddress) from);
128    } else {
129      return from.toString();
130    }
131  }
132
133  private Optional<Student> getStudentWithName(String studentName) {
134    return getStudentsInGradeBook().filter(doesStudentHaveName(studentName)).findAny();
135  }
136
137  private Stream<Student> getStudentsInGradeBook() {
138    return gradeBook.studentsStream();
139  }
140
141  private Predicate<Student> doesStudentHaveName(String studentName) {
142    return student -> getFullName(student).equals(studentName) || getNickName(student).equals(studentName);
143  }
144
145  private String getNickName(Student student) {
146    return student.getNickName() + " " + student.getLastName();
147  }
148
149  private String getFullName(Student student) {
150    return student.getFirstName() + " " + student.getLastName();
151  }
152
153  private File getLocationToWriteFile(String fileName, String studentId) throws IOException {
154    File dir = new File(directory, getAssignmentName());
155    if (!dir.exists() && !dir.mkdirs()) {
156      throw new IOException("Could not create directory \"" + dir + "\"");
157    }
158    return new File(dir, appendFileSuffixToStudentId(fileName, studentId));
159  }
160
161  private String appendFileSuffixToStudentId(String fileName, String studentId) {
162    return studentId + fileName.substring(fileName.lastIndexOf('.'));
163  }
164
165  private String getSenderName(Message message) throws SubmissionException {
166    Address from = getSender(message);
167
168    if (from instanceof InternetAddress) {
169      return getStudentName((InternetAddress) from);
170    } else {
171      return from.toString();
172    }
173  }
174
175  private Address getSender(Message message) throws SubmissionException {
176    Address from;
177    try {
178      from = message.getFrom()[0];
179
180    } catch (MessagingException ex) {
181      throw new SubmissionException("Could not get from address from " + message, ex);
182    }
183    return from;
184  }
185
186  private String getStudentName(InternetAddress from) {
187    if (from.getPersonal() != null) {
188      return from.getPersonal();
189
190    } else {
191      return getEmailAddress(from);
192    }
193  }
194
195  private String getEmailAddress(InternetAddress from) {
196    return from.getAddress();
197  }
198}
