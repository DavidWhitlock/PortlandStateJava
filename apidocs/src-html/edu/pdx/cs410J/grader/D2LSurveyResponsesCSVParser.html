001package edu.pdx.cs410J.grader;
002
003import au.com.bytecode.opencsv.CSVReader;
004import com.google.common.annotations.VisibleForTesting;
005
006import java.io.IOException;
007import java.io.Reader;
008
009public class D2LSurveyResponsesCSVParser {
010
011  @VisibleForTesting
012  int questionColumnIndex = -1;
013
014  @VisibleForTesting
015  int responseColumnIndex = -1;
016
017  private final SurveyResponsesFromD2L responses;
018
019  public D2LSurveyResponsesCSVParser(Reader reader) throws IOException {
020    CSVReader csv = new CSVReader( reader );
021    String[] firstLine = csv.readNext();
022    extractColumnNamesFromFirstLineOfCsv(firstLine);
023
024    responses = new SurveyResponsesFromD2L();
025
026    String[] surveyLine;
027    while ((surveyLine = csv.readNext()) != null) {
028      addQuestionAndResponseFromLineOfCsv(surveyLine);
029    }
030  }
031
032  private void addQuestionAndResponseFromLineOfCsv(String[] surveyLine) {
033    String question = getQuestionFromSurveyLine(surveyLine);
034    String response = getResponseFromSurveyLine(surveyLine);
035
036    this.responses.noteQuestionAndResponse(question, response);
037  }
038
039  private String getResponseFromSurveyLine(String[] surveyLine) {
040    return surveyLine[responseColumnIndex];
041  }
042
043  private String getQuestionFromSurveyLine(String[] surveyLine) {
044    return surveyLine[questionColumnIndex];
045  }
046
047  private void extractColumnNamesFromFirstLineOfCsv(String[] firstLine) {
048    for (int columnIndex = 0; columnIndex < firstLine.length; columnIndex++) {
049      String columnHeading = firstLine[columnIndex];
050      if ("Q Text".equals(columnHeading)) {
051        this.questionColumnIndex = columnIndex;
052
053      } else if ("Answer".equals(columnHeading)) {
054        this.responseColumnIndex = columnIndex;
055      }
056    }
057  }
058
059  public SurveyResponsesFromD2L getSurveyResponses() {
060    return this.responses;
061  }
062}
