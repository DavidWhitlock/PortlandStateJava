001package edu.pdx.cs410J.grader;
002
003import com.google.common.io.Files;
004import edu.pdx.cs410J.ParserException;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import javax.mail.MessagingException;
009import javax.mail.Session;
010import javax.mail.Transport;
011import javax.mail.internet.MimeMessage;
012import java.io.File;
013import java.io.IOException;
014import java.io.PrintStream;
015import java.nio.charset.Charset;
016import java.util.*;
017
018public class StudentFileMailer extends EmailSender {
019  private static Logger logger = LoggerFactory.getLogger(StudentFileMailer.class.getPackage().getName());
020
021  private final String subject;
022  private final Session session;
023
024  public StudentFileMailer(String subject, Session session) {
025    this.subject = subject;
026    this.session = session;
027  }
028
029  public static void main(String[] args) {
030    String gradeBookFileName = null;
031    String subject = null;
032    List<String> fileNames = new ArrayList<>();
033
034    for (String arg : args) {
035      if (gradeBookFileName == null) {
036        gradeBookFileName = arg;
037
038      } else if (subject == null) {
039        subject = arg;
040
041      } else {
042        fileNames.add(arg);
043      }
044    }
045
046    usageIfNull(gradeBookFileName, "Missing grade book file name");
047    usageIfNull(subject, "Missing email subject");
048    usageIfEmpty(fileNames, "Missing file names");
049    List<File> files = getFiles(fileNames);
050    GradeBook gradeBook = getGradeBook(gradeBookFileName);
051    Map<Student, File> filesToSendToStudents = getFilesToSendToStudents(files, gradeBook);
052
053    StudentFileMailer mailer = new StudentFileMailer(subject, newEmailSession(false));
054    filesToSendToStudents.forEach((student, file) -> {
055      try {
056        mailer.mailFileToStudent(file, student);
057
058      } catch (MessagingException | IOException ex) {
059        logger.error("While mailing \"" + file + "\" to " + student, ex);
060      }
061    });
062  }
063
064  private void mailFileToStudent(File file, Student student) throws MessagingException, IOException {
065    String studentEmail = getEmailAddress(student);
066
067    logger.info("Mailing \"" + file + "\" to \"" + studentEmail + "\"");
068
069    MimeMessage message = newEmailTo(this.session, studentEmail, this.subject);
070    message.setText(readTextFromFile(file));
071
072    sendEmailMessage(message);
073  }
074
075  private String readTextFromFile(File file) throws IOException {
076    StringBuilder sb = new StringBuilder();
077    Files.copy(file, Charset.defaultCharset(), sb);
078    return sb.toString();
079  }
080
081  private void sendEmailMessage(MimeMessage message) throws MessagingException {
082    Transport.send(message);
083  }
084
085  private String getEmailAddress(Student student) {
086    StringBuilder sb = new StringBuilder();
087    sb.append(student.getFirstName());
088    sb.append(" ");
089    if (student.getNickName() != null) {
090      sb.append("\"");
091      sb.append(student.getNickName());
092      sb.append("\" ");
093    }
094    sb.append(student.getLastName());
095    sb.append(" <");
096    sb.append(student.getEmail());
097    sb.append(">");
098
099    return sb.toString();
100  }
101
102  private static Map<Student, File> getFilesToSendToStudents(List<File> files, GradeBook gradeBook) {
103    Map<Student, File> filesToSendToStudents = new HashMap<>();
104    for (File file : files) {
105      String studentId = getStudentIdFromFileName(file);
106      Optional<Student> maybeStudent = gradeBook.getStudent(studentId);
107      if (!maybeStudent.isPresent()) {
108        maybeStudent = gradeBook.getStudentWithSsn(studentId);
109      }
110      if (!maybeStudent.isPresent()) {
111        cannotFindStudent(studentId);
112      } else {
113        filesToSendToStudents.put(maybeStudent.get(), file);
114      }
115    }
116
117    return filesToSendToStudents;
118  }
119
120  private static void cannotFindStudent(String studentId) {
121    logger.warn("Cannot find student with id \"" + studentId + "\" in grade book");
122  }
123
124  private static String getStudentIdFromFileName(File file) {
125    String studentId;
126    String fileName = file.getName();
127    int index = fileName.lastIndexOf('.');
128    if (index < 0) {
129      studentId = fileName;
130
131    } else {
132      studentId = fileName.substring(0, index);
133    }
134    return studentId;
135  }
136
137  private static List<File> getFiles(List<String> fileNames) {
138    List<File> files = new ArrayList<>(fileNames.size());
139    for (String fileName : fileNames) {
140      File file = new File(fileName);
141      if (!file.exists()) {
142        throw new IllegalStateException("File \"" + fileName + "\" does not exist");
143
144      } else if (!file.isFile()) {
145        throw new IllegalStateException("File \"" + fileName + "\" is not a file");
146      }
147
148      files.add(file);
149    }
150
151    return files;
152  }
153
154  private static GradeBook getGradeBook(String gradeBookFileName) {
155    try {
156      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFileName);
157      return parser.parse();
158
159    } catch (IOException ex) {
160      return usage("Couldn't read grade book file \"" + gradeBookFileName + "\"");
161
162    } catch (ParserException ex) {
163      return usage("Couldn't parse grade book file \"" + gradeBookFileName + "\"");
164    }
165  }
166
167
168  private static void usageIfEmpty(List<String> list, String message) {
169    if (list.isEmpty()) {
170      usage(message);
171    }
172  }
173
174  private static void usageIfNull(String arg, String message) {
175    if (arg == null) {
176      usage(message);
177    }
178  }
179
180  private static <T> T usage(String message) {
181    PrintStream err = System.err;
182
183    err.println("** " + message);
184    err.println();
185    err.println("usage: java StudentFileMailer gradeBookFileName subject fileName+");
186    err.println();
187    err.println("Emails a file to a student in a grade book.  The name of the file must");
188    err.println("begin with the student's id.");
189    err.println();
190    err.println("  gradeBookFileName    Name of the grade book file");
191    err.println("  subject              The subject of the email");
192    err.println("  fileName             A file to send to the student");
193    err.println();
194
195    System.exit(1);
196
197    throw new UnsupportedOperationException("Should never get here");
198  }
199}
