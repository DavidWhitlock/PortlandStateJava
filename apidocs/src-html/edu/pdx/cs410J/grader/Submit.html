023/**
024 * This class is used to submit assignments in CS410J.  The user
025 * specified his or her email address as well as the base directory
026 * for his/her source files on the command line.  The directory is
027 * searched recursively for files ending in .java.  Those files are
028 * placed in a zip file and emailed to the grader.  A confirmation
029 * email is sent to the submitter.
030 *
031 * More information about the JavaMail API can be found at:
032 *
033 * <a href="http://java.sun.com/products/javamail">
034 * http://java.sun.com/products/javamail</a>
035 *
036 * @author David Whitlock
037 * @since Fall 2000 (Refactored to use fewer static methods in Spring 2006)
038 */

185  /**
186   * Validates the state of this submission
187   *
188   * @throws IllegalStateException If any state is incorrect or missing
189   */
190  public void validate() {
191    validateProjectName();
192
193    if (projName == null) {
194      throw new IllegalStateException("Missing project name");
195    }
196
197    if (userName == null) {
198      throw new IllegalStateException("Missing student name");
199    }
200
201    if (userId == null) {
202      throw new IllegalStateException("Missing login id");
203    }
204
205    if (isNineDigitStudentId(userId)) {
206      throw new IllegalStateException(loginIdShouldNotBeStudentId(userId));
207    }
208
209    if (looksLikeAnEmailAddress(userId)) {
210      throw new IllegalStateException(loginIdShouldNotBeEmailAddress(userId));
211    }
212
213    if (userEmail == null) {
214      throw new IllegalStateException("Missing email address");
215
216    } else {
217      // Make sure user's email is okay
218      try {
219        new InternetAddress(userEmail);
220
221      } catch (AddressException ex) {
222        String s = "Invalid email address: " + userEmail;
223        IllegalStateException ex2 = new IllegalStateException(s);
224        ex2.initCause(ex);
225        throw ex2;
226      }
227    }
228  }

260  /**
261   * Submits the project to the grader
262   *
263   * @param verify Should the user be prompted to verify his submission?
264   * @return Whether or not the submission actually occurred
265   * @throws IllegalStateException If no source files were found
266   */
267  public boolean submit(boolean verify) throws IOException, MessagingException {
268    // Recursively search the source directory for .java files
269    Set<File> sourceFiles = searchForSourceFiles(fileNames);
270
271    db(sourceFiles.size() + " source files found");
272
273    if (sourceFiles.size() == 0) {
274      String s = "No source files were found.";
275      throw new IllegalStateException(s);
276    } else if (sourceFiles.size() < 3) {
277      String s = "Too few source files were submitted. Each project requires at least 3 files be" + 
278                 " submitted. You only submitted " + sourceFiles.size() + " files";
279      throw new IllegalStateException(s);
280    }
281
282    // Verify submission with user
283    if (verify && !verifySubmission(sourceFiles)) {
284      // User does not want to submit
285      return false;
286    }
287
288    // Timestamp
289    this.submitTime = LocalDateTime.now();
290
291    // Create a temporary zip file to hold the source files
292    File zipFile = makeZipFileWith(sourceFiles);
293
294    // Send the zip file as an email attachment to the TA
295    mailTA(zipFile, sourceFiles);
296
297    if (sendReceipt) {
298      mailReceipt(sourceFiles);
299    }
300
301    return true;
302  }

313  /**
314   * Searches for the files given on the command line.  Ignores files
315   * that do not end in .java, or that appear on the "no submit" list.
316   * Files must reside in a directory named
317   * edu/pdx/cs410J/<studentId>.
318   */
319  private Set<File> searchForSourceFiles(Set<String> fileNames) {
320    // Files should be sorted by name
321    SortedSet<File> files =
322      new TreeSet<>((o1, o2) -> o1.toString().compareTo(o2.toString()));
323    populateWithFilesFromSubdirectories(files, fileNames);
324
325    files.removeIf((file) -> !canBeSubmitted(file));
326
327    return files;
328  }
329
330  private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Set<String> fileNames) {
331    populateWithFilesFromSubdirectories(allFiles, fileNames.stream().map(name -> new File(name).getAbsoluteFile()));
332  }
333
334  private void populateWithFilesFromSubdirectories(SortedSet<File> allFiles, Stream<File> files) {
335    files.forEach((file) -> {
336      if (file.isDirectory()) {
337        populateWithFilesFromSubdirectories(allFiles, Arrays.stream(file.listFiles()));
338
339      } else {
340        allFiles.add(file);
341      }
342    });
343
344  }

346  private boolean canBeSubmitted(File file) {
347    if (!file.exists()) {
348      err.println("** Not submitting file " + file +
349        " because it does not exist");
350      return false;
351    }
352
353    // Is the file on the "no submit" list?
354    List<String> noSubmit = fetchListOfFilesThatCanNotBeSubmitted();
355    String name = file.getName();
356    if (noSubmit.contains(name)) {
357      err.println("** Not submitting file " + file +
358        " because it is on the \"no submit\" list");
359      return false;
360    }
361
362    // Does the file name end in .java?
363    if (!name.endsWith(".java")) {
364      err.println("** Not submitting file " + file +
365        " because does end in \".java\"");
366      return false;
367    }
368
369    // Verify that file is in the correct directory.
370    if (!isInAKoansDirectory(file) && !isInEduPdxCs410JDirectory(file)) {
371      err.println("** Not submitting file " + file +
372        ": it does not reside in a directory named " +
373        "edu" + File.separator + "pdx" + File.separator +
374        "cs410J" + File.separator + userId + " (or in one of the koans directories)");
375      return false;
376    }
377
378    return true;
379  }
380
381  private boolean isInAKoansDirectory(File file) {
382    boolean isInAKoansDirectory = hasParentDirectories(file, "beginner") ||
383      hasParentDirectories(file, "intermediate") ||
384      hasParentDirectories(file, "advanced") ||
385      hasParentDirectories(file, "java7") ||
386      hasParentDirectories(file, "java8");
387    if (isInAKoansDirectory) {
388      this.isSubmittingKoans = true;
389      db(file + " is in a koans directory");
390    }
391    return isInAKoansDirectory;
392  }
393
394  @VisibleForTesting
395  boolean isInEduPdxCs410JDirectory(File file) {
396    boolean isInEduPdxCs410JDirectory = hasParentDirectories(file, userId, "cs410J", "pdx", "edu");
397    if (isInEduPdxCs410JDirectory) {
398      db(file + " is in the edu/pdx/cs410J directory");
399    }
400    return isInEduPdxCs410JDirectory;
401  }

403  private boolean hasParentDirectories(File file, String... parentDirectoryNames) {
404    File parent = file.getParentFile();
405
406    // Skip over subpackages
407    while (parent != null && !parent.getName().equals(parentDirectoryNames[0])) {
408      parent = parent.getParentFile();
409    }
410
411    for (String parentDirectoryName : parentDirectoryNames) {
412      if (parent == null || !parent.getName().equals(parentDirectoryName)) {
413        return false;
414
415      } else {
416        parent = parent.getParentFile();
417      }
418    }
419
420    return true;
421  }

423  private List<String> fetchListOfFilesThatCanNotBeSubmitted() {
424    return fetchListOfStringsFromUrl(NO_SUBMIT_LIST_URL);
425  }

427  private List<String> fetchListOfStringsFromUrl(String listUrl) {
428    List<String> noSubmit = new ArrayList<>();
429
430    try {
431      URL url = new URL(listUrl);
432      InputStreamReader isr = new InputStreamReader(url.openStream());
433      BufferedReader br = new BufferedReader(isr);
434      while (br.ready()) {
435        noSubmit.add(br.readLine().trim());
436      }
437
438    } catch (MalformedURLException ex) {
439      err.println("** WARNING: Cannot access " + listUrl + ": " +
440        ex.getMessage());
441
442    } catch (IOException ex) {
443      err.println("** WARNING: Problems while reading " + listUrl + ": " + ex.getMessage());
444    }
445    return noSubmit;
446  }

448  /**
449   * Prints a summary of what is about to be submitted and prompts the
450   * user to verify that it is correct.
451   *
452   * @return <code>true</code> if the user wants to submit
453   */
454  private boolean verifySubmission(Set<File> sourceFiles) {
455    // Print out what is going to be submitted
456    out.print("\n" + userName);
457    out.print("'s submission for ");
458    out.println(projName);
459
460    for (File file : sourceFiles) {
461      out.println("  " + file);
462    }
463
464    if (comment != null) {
465      out.println("\nComment: " + comment + "\n\n");
466    }
467
468    out.println("A receipt will be sent to: " + userEmail + "\n");
469
470    warnIfMainProjectClassIsNotSubmitted(sourceFiles);
471
472    return doesUserWantToSubmit();
473  }
474
475  private void warnIfMainProjectClassIsNotSubmitted(Set<File> sourceFiles) {
476    boolean wasMainProjectClassSubmitted = sourceFiles.stream().anyMatch((f) -> f.getName().contains(this.projName));
477    if (!wasMainProjectClassSubmitted && !this.isSubmittingKoans) {
478      String mainProjectClassName = this.projName + ".java";
479      out.println("*** WARNING: You are submitting " + this.projName +
480        ", but did not include " + mainProjectClassName + ".\n" +
481        "    You might want to check the name of the project or the files you are submitting.\n");
482    }
483  }

485  private boolean doesUserWantToSubmit() {
486    InputStreamReader isr = new InputStreamReader(System.in);
487    BufferedReader in = new BufferedReader(isr);
488
489    while (true) {
490      out.print("Do you wish to continue with the submission? (yes/no) ");
491      out.flush();
492
493      try {
494        String line = in.readLine().trim();
495        switch (line) {
496          case "yes":
497            return true;
498
499          case "no":
500            return false;
501
502          default:
503            err.println("** Please enter yes or no");
504            break;
505        }
506
507      } catch (IOException ex) {
508        err.println("** Exception while reading from System.in: " + ex);
509      }
510    }
511  }

527  /**
528   * Creates a Zip file that contains the source files.  The Zip File
529   * is temporary and is deleted when the program exits.
530   */
531  private File makeZipFileWith(Set<File> sourceFiles) throws IOException {
532    String zipFileName = userName.replace(' ', '_') + "-TEMP";
533    File zipFile = File.createTempFile(zipFileName, ".zip");
534    if (!saveZip) {
535      zipFile.deleteOnExit();
536
537    } else {
538      out.println("Saving temporary Zip file: " + zipFile);
539    }
540
541    db("Created Zip file: " + zipFile);
542
543    Map<File, String> sourceFilesWithNames =
544      sourceFiles.stream().collect(Collectors.toMap(file -> file, this::getRelativeName));
545
546    return new ZipMaker(sourceFilesWithNames, zipFile, getManifestEntries()).makeZipFile();
547  }
548
549  private Map<Attributes.Name, String> getManifestEntries() {
550    Map<Attributes.Name, String> manifestEntries = new HashMap<>();
551    manifestEntries.put(ManifestAttributes.USER_NAME, userName);
552    manifestEntries.put(ManifestAttributes.USER_ID, userId);
553    manifestEntries.put(ManifestAttributes.USER_EMAIL, userEmail);
554    manifestEntries.put(ManifestAttributes.PROJECT_NAME, projName);
555    manifestEntries.put(ManifestAttributes.SUBMISSION_COMMENT, comment);
556    manifestEntries.put(ManifestAttributes.SUBMISSION_TIME, ManifestAttributes.formatSubmissionTime(submitTime));
557    return manifestEntries;
558  }

560  /**
561   * Sends the Zip file to the TA as a MIME attachment.  Also includes
562   * a textual summary of the contents of the Zip file.
563   */
564  private void mailTA(File zipFile, Set<File> sourceFiles) throws MessagingException {
565    MimeMessage message = newEmailTo(newEmailSession(debug), TA_EMAIL, "CS410J-SUBMIT " + userName + "'s " + projName);
566
567    MimeBodyPart textPart = createTextPartOfTAEmail(sourceFiles);
568    MimeBodyPart filePart = createZipAttachment(zipFile);
569
570    Multipart mp = new MimeMultipart();
571    mp.addBodyPart(textPart);
572    mp.addBodyPart(filePart);
573
574    message.setContent(mp);
575
576    out.println("Submitting project to Grader");
577
578    Transport.send(message);
579  }
580
581  private MimeBodyPart createZipAttachment(File zipFile) throws MessagingException {
582    // Now attach the Zip file
583    DataSource ds = new FileDataSource(zipFile) {
584      @Override
585      public String getContentType() {
586        return "application/zip";
587      }
588    };
589    DataHandler dh = new DataHandler(ds);
590    MimeBodyPart filePart = new MimeBodyPart();
591
592    String zipFileTitle = userName.replace(' ', '_') + ".zip";
593
594    filePart.setDataHandler(dh);
595    filePart.setFileName(zipFileTitle);
596    filePart.setDescription(userName + "'s " + projName);
597
598    return filePart;
599  }

601  private MimeBodyPart createTextPartOfTAEmail(Set<File> sourceFiles) throws MessagingException {
602    // Create the text portion of the message
603    StringBuilder text = new StringBuilder();
604    text.append("Student name: ").append(userName).append(" (").append(userEmail).append(")\n");
605    text.append("Project name: ").append(projName).append("\n");
606    text.append("Submitted on: ").append(humanReadableSubmitDate()).append("\n");
607    if (comment != null) {
608      text.append("\nComment: ").append(comment).append("\n\n");
609    }
610    text.append("Contents:\n");
611
612    for (File file : sourceFiles) {
613      text.append("  ").append(getRelativeName(file)).append("\n");
614    }
615    text.append("\n\n");
616
617    MimeBodyPart textPart = new MimeBodyPart();
618    textPart.setContent(text.toString(), "text/plain");
619
620    // Try not to display text as separate attachment
621    textPart.setDisposition("inline");
622    return textPart;
623  }

629  /**
630   * Sends a email to the user as a receipt of the submission.
631   */
632  private void mailReceipt(Set<File> sourceFiles) throws MessagingException {
633    MimeMessage message = newEmailTo(newEmailSession(debug), userEmail, "CS410J " + projName + " submission");
634
635    // Create the contents of the message
636    StringBuilder text = new StringBuilder();
637    text.append("On ").append(humanReadableSubmitDate()).append("\n");
638    text.append(userName).append(" (").append(userEmail).append(")\n");
639    text.append("submitted the following files for ").append(projName).append(":\n");
640
641    for (File file : sourceFiles) {
642      text.append("  ").append(file.getAbsolutePath()).append("\n");
643    }
644
645    if (comment != null) {
646      text.append("\nComment: ").append(comment).append("\n\n");
647    }
648
649    text.append("\n\n");
650    text.append("Have a nice day.");
651
652    // Add the text to the message and send it
653    message.setText(text.toString());
654    message.setDisposition("inline");
655
656    out.println("Sending receipt to you at " + userEmail);
657
658    Transport.send(message);
659  }

685  /**
686   * Parses the command line, finds the source files, prompts the user
687   * to verify whether or not the settings are correct, and then sends
688   * an email to the Grader.
689   */
690  public static void main(String[] args) throws IOException, MessagingException {
691    Submit submit = new Submit();
692
693    // Parse the command line
694    for (int i = 0; i < args.length; i++) {
695      // Check for options first
696      if (args[i].equals("-smtp")) {
697        if (++i >= args.length) {
698          usage("No SMTP server specified");
699        }
700
701        submit.setEmailServerHostName(args[i]);
702
703      } else if (args[i].equals("-verbose")) {
704        submit.setDebug(true);
705
706      } else if (args[i].equals("-savezip")) {
707        submit.setSaveZip(true);
708
709      } else if (args[i].equals("-comment")) {
710        if (++i >= args.length) {
711          usage("No comment specified");
712        }
713
714        submit.setComment(args[i]);
715
716      } else if (submit.projName == null) {
717        submit.setProjectName(args[i]);
718
719      } else if (submit.userName == null) {
720        submit.setUserName(args[i]);
721
722      } else if (submit.userId == null) {
723        submit.setUserId(args[i]);
724
725      } else if (submit.userEmail == null) {
726        submit.setUserEmail(args[i]);
727
728      } else {
729        // The name of a source file
730        submit.addFile(args[i]);
731      }
732    }
733
734    boolean submitted;
735
736    try {
737      // Make sure that user entered enough information
738      submit.validate();
739
740      submit.db("Command line successfully parsed.");
741
742      submitted = submit.submit(true);
743
744    } catch (IllegalStateException ex) {
745      usage(ex.getMessage());
746      return;
747    }
748
749    // All done.
750    if (submitted) {
751      out.println(submit.projName + " submitted successfully.  Thank you.");
752
753    } else {
754      out.println(submit.projName + " not submitted.");
755    }
