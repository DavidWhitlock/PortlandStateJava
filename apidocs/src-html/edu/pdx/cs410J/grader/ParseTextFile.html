001package edu.pdx.cs410J.grader;
002
003import java.io.*;
004import java.util.*;
005
006/**
007 * Quicky program that parses a text file containing student's grades.
008 */
009public class ParseTextFile {
010  private static PrintWriter err = new PrintWriter(System.err, true);
011
012  /**
013   * Prints usage information for this program
014   */
015  private static void usage() {
016    err.println("\njava ParseTextFile textFile courseName xmlFile");
017    err.println("\n");
018    System.exit(1);
019  }
020
021  public static void main(String[] args) {
022    if (args.length < 3) {
023      err.println("** Missing command line arguments");
024      usage();
025    }
026    
027    String textFile = args[0];
028    String courseName = args[1];
029    String xmlFile = args[2];
030
031    // Make a grade book
032    GradeBook book = new GradeBook(courseName);
033
034    // Parse the text file
035    BufferedReader in = null;
036    try {
037      in = new BufferedReader(new FileReader(textFile));
038
039      List<String> assignments = new ArrayList<String>();
040
041      // First line of file contains names of assignments
042      String line = in.readLine();
043      StringTokenizer st = new StringTokenizer(line);
044      st.nextToken();    // Skip "id"
045      while (st.hasMoreTokens()) {
046        String assignment = st.nextToken();
047        assignments.add(assignment);
048      }
049
050      line = in.readLine();
051      while (line != null) {
052        st = new StringTokenizer(line);
053        String studentName = st.nextToken();
054        Student student =
055          book.getStudent(studentName).orElseThrow(() -> new IllegalStateException("No student with id " + studentName));
056
057        for (int i = 0; st.hasMoreTokens(); i++) {
058          String s = st.nextToken();
059          String assignment = (String) assignments.get(i);
060          double score;
061
062          if (s.equals("I")) {
063            score = Grade.INCOMPLETE;
064
065          } else {
066            try {
067              score = Double.parseDouble(s);
068            } catch (NumberFormatException ex) {
069              err.println("** Bad number: " + s);
070              System.exit(1);
071              score = -4.2;
072            }
073          }
074
075          Grade grade = new Grade(assignment, score);
076          student.setGrade(assignment, grade);
077        }
078
079        line = in.readLine();
080      }
081    } catch (IOException ex) {
082      ex.printStackTrace();
083      System.exit(1);
084    }
085
086    // Save grade book to XML file
087    try {
088      XmlDumper dumper = new XmlDumper(xmlFile);
089      dumper.dump(book);
090
091    } catch (IOException ex) {
092      err.println("** Error while writing XML file: " + ex);
093      System.exit(1);
094    }
095  }
096}
