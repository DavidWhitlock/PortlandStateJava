089  /**
090   * Create and lay out a new <code>GradeBookGUI</code>
091   */
092  public GradeBookGUI(String title) {
093    super(title);
094
095    // Read preference information
096    this.prefs = Preferences.userNodeForPackage(this.getClass());
097    this.recentFiles = new ArrayList<String>();
098    for (int i = 0; i < MAX_RECENT_FILES; i++) {
099      String recent = this.prefs.get(RECENT_FILE + i, null);
100      if (recent != null) {
101        File file = new File(recent);
102        if (file.exists() && file.isFile()) {
103          this.recentFiles.add(0, recent);
104        }
105      }
106    }
107    
108    // Set up the actions
109    this.setupActions();
110
111    // Add the menus
112    JMenuBar menuBar = new JMenuBar();
113    this.setJMenuBar(menuBar);
114    this.addFileMenu(menuBar);
115    this.updateRecentFilesMenu();
116
117    this.getContentPane().setLayout(new BorderLayout());
118
119    // Add the tool bar
120    this.getContentPane().add(this.createToolBar(),
121                              BorderLayout.NORTH);
122
123    // Add the GradeBookPanel
124    this.bookPanel = new GradeBookPanel(this);
125    this.getContentPane().add(this.bookPanel, BorderLayout.CENTER);
126
127    // Add the status label.  Status messages are displaed for five
128    // seconds
129    this.status = new JLabel(" ") {
130        private java.util.Timer timer =
131          new java.util.Timer(true /* isDaemon */);
132        private TimerTask lastTask = null;
133
134        public void setText(String text) {
135          super.setText(text);
136          if ("".equals(text.trim())) {
137            // Avoid recusion
138            return;
139          }
140
141          if (lastTask != null) {
142            lastTask.cancel();
143          }
144          lastTask = new TimerTask() {
145              public void run() {
146                SwingUtilities.invokeLater(new Runnable() {
147                    public void run() {
148                      // Empty label looks funny with border
149                      GradeBookGUI.this.status.setText(" ");
150                    }
151                  });
152              }
153            };
154          timer.schedule(lastTask, 5 * 1000);
155        }
156      };
157    Border inside = BorderFactory.createEmptyBorder(0, 2, 0, 0);
158    Border outside = BorderFactory.createLoweredBevelBorder();
159    Border border =
160      BorderFactory.createCompoundBorder(outside, inside); 
161    this.status.setBorder(border);
162    this.getContentPane().add(this.status, BorderLayout.SOUTH);
163
164    // Handle exit events
165    this.addWindowListener(new WindowAdapter() {
166        public void windowClosing(WindowEvent e) {
167          exit();
168        }
169      });
170  }

357  /**
358   * Makes note that the current <code>file</code> has been changed.
359   * Updates the preferences accordingly.
360   *
361   * @since Fall 2004
362   */
363  private void noteRecentFile() {
364    assert this.file != null;
365    String name = this.file.getAbsolutePath();
366    this.recentFiles.remove(name);
367    this.recentFiles.add(0, name);
368    while (this.recentFiles.size() > MAX_RECENT_FILES) {
369      this.recentFiles.remove(this.recentFiles.size() - 1);
370    }
371    for (int i = 0;
372         i < this.recentFiles.size() && i < MAX_RECENT_FILES;
373         i++) {
374      this.prefs.put(RECENT_FILE + i,
375                     (String) this.recentFiles.get(i));
376    }
377    updateRecentFilesMenu();
378  }
379
380  /**
381   * Updates the contents of the "recent files" menu
382   *
383   * @since Fall 2004
384   */
385  private void updateRecentFilesMenu() {
386    this.recentFilesMenu.removeAll();
387    JMenuItem clear = new JMenuItem("Clear recent files");
388    clear.addActionListener(new ActionListener() {
389        public void actionPerformed(ActionEvent e) {
390          GradeBookGUI.this.recentFiles.clear();
391          updateRecentFilesMenu();
392        }
393      });
394    this.recentFilesMenu.add(clear);
395    this.recentFilesMenu.addSeparator();
396
397    for (int i = 0; i < this.recentFiles.size(); i++) {
398      String fileName = (String) this.recentFiles.get(i);
399      final File file = new File(fileName);
400      if (file.exists()) {
401        JMenuItem item = new JMenuItem((i+1) + ": " + fileName);
402        item.addActionListener(new ActionListener() {
403            public void actionPerformed(ActionEvent e) {
404              open(file);
405            }
406          });
407        this.recentFilesMenu.add(item);
408      }
409    }
410  }

412  /**
413   * Saves the grade book being edited to a file
414   */
415  private void save() {
416    if (this.book == null) {
417      // Nothing to do
418      return;
419    }
420
421    if (this.file == null) {
422      saveAs();
423      // saveAs() will recursively invoke save() if the user selects a
424      // file
425      return;
426    }
427
428    // Dump grade to file as XML
429    try {
430      XmlDumper dumper = new XmlDumper(this.file);
431      dumper.dump(this.book);
432      this.status.setText("Saved " + this.file);
433
434    } catch (IOException ex) {
435      error("While writing grade book", ex);
436      return;
437    }
438  }

440  /**
441   * Creates a <code>JFileChooser</code> suitable for dealing with
442   * xml files.
443   */
444  private JFileChooser getFileChooser() {
445    JFileChooser chooser = new JFileChooser();
446    if (this.file != null) {
447      chooser.setCurrentDirectory(file.getAbsoluteFile().getParentFile());
448    } else {
449      String cwd = System.getProperty("user.dir");
450      chooser.setCurrentDirectory(new File(cwd));
451    }
452
453    chooser.setFileFilter(new javax.swing.filechooser.FileFilter() {
454        public boolean accept(File file) {
455          if (file.isDirectory()) {
456            return true;
457          }
458
459          String fileName = file.getName();
460          return fileName.endsWith(".xml");
461        }
462
463        public String getDescription() {
464          return "XML files";
465        }
466      });
467    chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
468    chooser.setMultiSelectionEnabled(false);
469
470    return chooser;
471  }

473  /**
474   * Prompts the user for a file to save the grade book in.
475   */
476  private void saveAs() {
477    JFileChooser chooser = this.getFileChooser();
478    chooser.setDialogTitle("Save as...");
479    chooser.setDialogType(JFileChooser.SAVE_DIALOG);
480    int response = chooser.showSaveDialog(this);
481
482    if (response == JFileChooser.APPROVE_OPTION) {
483      this.file = chooser.getSelectedFile();
484      save();
485      noteRecentFile();
486    }
487  }

489  /**
490   * Prompts the user for a file to display
491   */
492  private void open() {
493    JFileChooser chooser = this.getFileChooser();
494    chooser.setDialogTitle("Open grade book file");
495    chooser.setDialogType(JFileChooser.OPEN_DIALOG);
496    int response = chooser.showOpenDialog(this);
497
498    if (response == JFileChooser.APPROVE_OPTION) {
499      // Read the grade book from the file and display it in the GUI
500      File file = chooser.getSelectedFile();
501      open(file);
502    }
503  }

505  /**
506   * Opens the grade book stored in the given file
507   */ 
508  private void open(File file) {
509    this.checkDirtyGradeBook();
510
511    try {
512      XmlGradeBookParser parser = new XmlGradeBookParser(file);
513      GradeBook book = parser.parse();
514      this.displayGradeBook(book);
515      this.file = file;
516      noteRecentFile();
517
518    } catch (IOException ex) {
519      error("While parsing file " + file.getName(), ex);
520      return;
521
522    } catch (ParserException ex) {
523      error("While parsing file " + file.getName(), ex);
524    }
525
526    this.status.setText("Opened " + file);
527  }

529  /**
530   * Prompts the user for a file from which a student is read and
531   * added to the grade book
532   */
533  private void importStudent() {
534    if (this.book == null) {
535      error("You must open a grade book before importing a student");
536      return;
537    }
538    
539    JFileChooser chooser = this.getFileChooser();
540    chooser.setDialogTitle("Import student XML file");
541    chooser.setDialogType(JFileChooser.OPEN_DIALOG);
542    chooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
543    chooser.setMultiSelectionEnabled(true);
544    int response = chooser.showOpenDialog(this);
545    
546    if (response == JFileChooser.APPROVE_OPTION) {
547      // Read the student from each of the selected files and add it
548      // to the grade book
549      File[] files = chooser.getSelectedFiles();
550      for (int i = 0; i < files.length; i++) {
551        File file = files[i];
552        try {
553          XmlStudentParser sp = new XmlStudentParser(file);
554          Student student = sp.parseStudent();
555          this.book.addStudent(student);
556          this.status.setText("Imported " + student.getFullName());
557        
558        } catch (IOException ex) {
559          error("Could not access " + file.getName(), ex);
560          return;
561
562        } catch (ParserException ex) {
563          error("While parsing file " + file.getName(), ex);
564          return;
565        }
566      }
567    }
568  }

570  /**
571   * Creates a new grade book to be edited
572   */
573  private void newGradeBook() {
574    // Prompt user for name of grade book
575    String className = 
576      JOptionPane.showInputDialog(this, 
577                                  "Enter the name of the class",
578                                  "Enter class name",
579                                  JOptionPane.INFORMATION_MESSAGE);
580
581    if (className == null || className.equals("")) {
582      return;
583    }
584
585    this.displayGradeBook(new GradeBook(className));
586    this.file = null;
587    this.status.setText("Created new graded book for \"" + className +
588                        "\"");
589  }

591  /**

593   */
594  private void exit() {
595    checkDirtyGradeBook();
596    System.exit(0);
597  }

599  /**
600   * If the grade book has been modified and not saved, ask the user
601   * if he wants to save it.
602   */
603  private void checkDirtyGradeBook() {
604    if (this.book != null && this.book.isDirty()) {
605      int response = JOptionPane.showConfirmDialog(this, new String[] {
606        "You have made changes to the grade book.",
607        "Do you want to save them?"},
608        "Save Changes?",
609        JOptionPane.YES_NO_CANCEL_OPTION,
610        JOptionPane.QUESTION_MESSAGE);
611      
612      if (response == JOptionPane.YES_OPTION) {
613        save();
614
615      } else if (response == JOptionPane.CANCEL_OPTION) {
616        // Don't exit
617        return;
618      }
619    }
620  }

634  /**

636   */

638    GradeBookGUI gui = new GradeBookGUI("CS410J Grade Book Program");

641      PrintStream err = System.err;
642      File file = new File(args[0]);
643      if (!file.exists()) {
644        err.println("** " + file + " does not exist");
645
646      } else if (!file.isFile()) {
647        err.println("** " + file + " is not a file");
648
649      } else {
650        gui.open(file);
651      }
652    }
653
654    gui.pack();
655    gui.setVisible(true);
656  }
