087  private JPanel createAssignmentListPanel() {
088    JPanel listPanel = new JPanel();

092    listPanel.setLayout(new BorderLayout());

167  private Component createSubmissionsList() {
168    JPanel submissionsPanel = new JPanel();
169    submissionsPanel.setLayout(new BorderLayout());
170    submissionsPanel.setBorder(BorderFactory.createTitledBorder("Submissions"));
171
172    this.submissionTimesList = new JList<>();
173    this.submissionTimesList.setCellRenderer(new DefaultListCellRenderer() {
174      @Override
175      public Component getListCellRendererComponent(JList<?> list, Object dateTime, int index, boolean isSelected, boolean cellHasFocus) {
176        LocalDateTime submissionTime = (LocalDateTime) dateTime;
177        String value = submissionTime.format(DateTimeFormatter.ofLocalizedDateTime(FormatStyle.SHORT));
178        return super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
179      }
180    });
181
182    submissionsPanel.add(new JScrollPane(this.submissionTimesList), BorderLayout.CENTER);
183
184    return submissionsPanel;
185  }

196  private JPanel createGradeInfoPanel() {
197    JPanel p = new JPanel();
198    p.setLayout(new BorderLayout());
199

209    JPanel fields = new JPanel();
210    fields.setLayout(new GridLayout(0, 1));

212    this.gradeField.addActionListener(e -> createOrUpdateGrade());

214
215    p.add(labels, BorderLayout.WEST);
216    p.add(fields, BorderLayout.CENTER);

219
220  private void createOrUpdateGrade() {
221    Assignment assign = getSelectedAssignment();
222    if (assign != null && student != null) {
223      Grade grade = student.getGrade(assign.getName());
224      if (grade == null) {
225        createGrade();
226
227      } else {
228        updateGrade(grade);
229      }
230    }
231  }

233  /**
234   * Displays the assignments for a given <code>GradeBook</code>
235   */
236  public void displayAssignmentsFor(GradeBook book) {
237    this.book = book;
238
239    // Clear grade fields
240    Vector<String> names = new Vector<>(book.getAssignmentNames());
241    this.assignmentsList.setListData(names);
242
243    this.assignmentLabel.setText("Assignment");
244    this.gradeLabel.setText("Grade (out of ):");
245    this.gradeField.setText("");
246  }
247
248  /**
249   * Displays the grades for a given <code>Student</code>
250   */
251  public void displayStudent(Student student) {
252    // Clear any grade info
253    this.assignmentLabel.setText("Assignment");
254    this.gradeLabel.setText("Grade (out of ):");
255    this.gradeField.setText("");
256    this.notes.clearNotes();
257
258    this.student = student;
259    String name = student.getFullName();
260    if (name.equals("")) {
261      name = student.getId();
262    }
263    this.studentNameLabel.setText(name + " (" + student.getId() + ")");
264    this.lateList.setListData(new Vector<>(student.getLate()));
265    this.resubmitList.setListData(new Vector<>(student.getResubmitted()));
266
267    // Might as well refresh assignments list
268    this.displayAssignmentsFor(this.book);
269
270    // Display the grade for the currently selected
271    Assignment assign = getSelectedAssignment();
272    if (assign != null) {
273      displayGradeFor(assign);
274    }
275  }
276
277  /**
278   * Displays the current students grade for a given assignment
279   */
280  private void displayGradeFor(Assignment assign) {
281    this.assignmentLabel.setText("Assignment " + assign.getName() +
282                                 " (" + assign.getDescription() + ")");
283    this.gradeLabel.setText("Grade (out of " + assign.getPoints() +
284                            "):");
285    
286    if (this.student != null) {
287      Grade grade = student.getGrade(assign.getName());
288      if (grade != null) {
289        this.gradeField.setText(String.valueOf(grade.getScore()));
290        this.notes.setNotable(grade);
291        this.submissionTimesList.setModel(createListModel(grade.getSubmissionTimes()));
292        return;
293      }
294    }
295
296    this.gradeField.setText("");
297    this.notes.clearNotes();
298  }
299
300  private ListModel<LocalDateTime> createListModel(List<LocalDateTime> submissionTimes) {
301    return new AbstractListModel<LocalDateTime>() {
302      @Override
303      public int getSize() {
304        return submissionTimes.size();
305      }
306
307      @Override
308      public LocalDateTime getElementAt(int index) {
309        return submissionTimes.get(index);
310      }
311    };
312  }
313
314  /**
315   * Returns the <code>Assignment</code> that is currently selected in
316   * the assignmentsList.
317   */
318  private Assignment getSelectedAssignment() {
319    String name = this.assignmentsList.getSelectedValue();
320    if (name != null) {
321      this.assignmentIndex = this.assignmentsList.getSelectedIndex();
322      return this.book.getAssignment(name);
323
324    } else if (this.assignmentIndex == -1) {
325      return null;
326
327    } else if (this.assignmentIndex >=
328               this.assignmentsList.getModel().getSize()) {
329      // Changed grade book?
330      this.assignmentIndex = -1;
331      return null;
332
333    } else {
334      this.assignmentsList.setSelectedIndex(this.assignmentIndex);
335      return getSelectedAssignment();
336    }
337  }
338
339  /**
340   * Creates a new <code>Grade</code> based on the contents of this
341   * <code>GradePanel</code>
342   */
343  private Grade createGrade() {
344    Assignment assign = getSelectedAssignment();
345
346    if (this.student == null) {
347      return null;
348
349    } else if (assign == null) {
350      String s = "Please select an assignment";
351      return error(s);
352    }
353
354    String score = this.gradeField.getText();
355
356    try {
357      double d = Double.parseDouble(score);
358      Grade grade = new Grade(assign.getName(), d);
359      this.notes.setNotable(grade);
360      this.student.setGrade(assign.getName(), grade);
361      return grade;
362
363    } catch (NumberFormatException ex) {
364      return error(score + " is not a number");
365    }
366  }
367
368  /**
369   * Updates the current <code>Grade</code> based on the contents of
370   * this <code>GradePanel</code>
371   */
372  private void updateGrade(Grade grade) {
373    String score = this.gradeField.getText();
374    try {
375      // All we can change is the score
376      double d = Double.parseDouble(score);
377      grade.setScore(d);
378
379    } catch (NumberFormatException ex) {
380      error(score + " is not a number");
381    }
382
383    // Remember that the NotesPanel automatically updates the grade
384  }
385
386  private <T> T error(String message) {
387    JOptionPane.showMessageDialog(this,
388      new String[]{message},
389      "Error",
390      JOptionPane.ERROR_MESSAGE);
391    return null;
392  }

397  public static void main(String[] args) {
398    final String fileName = args[0];
399    String studentName = args[1];
400
401    GradeBook book = null;
402    try {
403      XmlGradeBookParser parser = new XmlGradeBookParser(fileName);
404      book = parser.parse();
405
406    } catch (FileNotFoundException ex) {
407      System.err.println("** Could not find file: " + ex.getMessage());
408      System.exit(1);
409      
410    } catch (IOException ex) {
411      System.err.println("** IOException during parsing: " + ex.getMessage());
412      System.exit(1);
413
414    } catch (ParserException ex) {
415      System.err.println("** Error during parsing: " + ex);
416      System.exit(1);
417    }
418
419    JFrame frame = new JFrame("GradePanel test");
420    GradePanel gradePanel = new GradePanel();
421    gradePanel.displayAssignmentsFor(book);
422
423    Student student =
424      book.getStudent(studentName).orElseThrow(() -> new IllegalStateException("No student with id " + studentName));
425    gradePanel.displayStudent(student);
426
427    final GradeBook theBook = book;
428    frame.addWindowListener(new WindowAdapter() {
429        @Override
430        public void windowClosing(WindowEvent e) {
431          // Write changes to grade book back to file
432          try {
433            XmlDumper dumper = new XmlDumper(fileName);
434            dumper.dump(theBook);
435
436          } catch (IOException ex) {
437            System.err.println("** Error while writing XML file: " + ex);
438          }
439
440          System.exit(1);
441        }
442      });
443
444    frame.getContentPane().add(gradePanel);
445    
446    frame.pack();
447    frame.setVisible(true);
448  }
