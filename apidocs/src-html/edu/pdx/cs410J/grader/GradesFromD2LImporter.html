001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.io.*;
008import java.util.List;
009import java.util.Optional;
010
011public class GradesFromD2LImporter {
012
013  private static final Logger logger = LoggerFactory.getLogger("edu.pdx.cs410J.grader");
014
015  public static void importGradesFromD2L(GradesFromD2L d2lGrades, GradeBook gradebook) {
016    List<GradesFromD2L.D2LStudent> students = d2lGrades.getStudents();
017    logger.debug("Importing grades for " + students.size() + " students");
018
019    for (GradesFromD2L.D2LStudent d2LStudent : students) {
020      Optional<Student> student = d2lGrades.findStudentInGradebookForD2LStudent(d2LStudent, gradebook);
021      if (!student.isPresent()) {
022        String message = "Could not find student " + d2LStudent + " in gradebook";
023        System.err.println(message);
024
025      } else {
026        for (String quizName : d2LStudent.getQuizNames()) {
027          Optional<Assignment> optional = d2lGrades.findAssignmentInGradebookForD2lQuiz(quizName, gradebook);
028          Assignment quiz = optional.orElseThrow(() -> new IllegalStateException("No quiz named \"" + quizName + "\" in gradebook"));
029
030          Double score = d2LStudent.getScore(quizName);
031          logger.debug("Recording grade of " + score + " for " + student.get() + " for " + quiz.getName());
032          student.get().setGrade(quiz.getName(), new Grade(quiz, score));
033        }
034      }
035    }
036  }
037
038  public static void main(String[] args) throws IOException, ParserException {
039    String d2lCsvFileName = null;
040    String gradeBookFileName = null;
041
042    for (String arg : args) {
043      if (d2lCsvFileName == null) {
044        d2lCsvFileName = arg;
045
046      } else if (gradeBookFileName == null) {
047        gradeBookFileName = arg;
048
049      } else {
050        usage("Extraneous command line argument: " + arg);
051      }
052    }
053
054    if (d2lCsvFileName == null) {
055      usage("Missing D2L CSV file name");
056    }
057
058    if (gradeBookFileName == null) {
059      usage("Missing grade book file name");
060    }
061
062    GradesFromD2L d2lGrades = parseD2lCsvFile(d2lCsvFileName);
063    GradeBook gradebook = parseGradeBook(gradeBookFileName);
064    importGradesFromD2L(d2lGrades, gradebook);
065
066    saveGradeBookIfModified(gradebook, gradeBookFileName);
067  }
068
069  private static void saveGradeBookIfModified(GradeBook gradeBook, String gradeBookFileName) {
070    if (gradeBook.isDirty()) {
071      File file = new File(gradeBookFileName);
072      try {
073        XmlDumper dumper = new XmlDumper(file);
074        dumper.dump(gradeBook);
075
076      } catch (IOException e) {
077        usage("Can't write grade book in \"" + gradeBookFileName + "\"");
078      }
079    }
080  }
081
082  private static GradeBook parseGradeBook(String gradeBookFileName) throws IOException, ParserException {
083    File gradeBookFile = new File(gradeBookFileName);
084    if (!gradeBookFile.exists()) {
085      usage("Grade Book file \"" + gradeBookFile + "\" does not exist");
086    }
087
088    XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
089    return parser.parse();
090  }
091
092  private static GradesFromD2L parseD2lCsvFile(String d2lCsvFileName) throws IOException {
093    File d2lCsvFile = new File(d2lCsvFileName);
094    if (!d2lCsvFile.exists()) {
095      usage("D2L CSV file \"" + d2lCsvFile + "\" does not exist");
096    }
097
098    D2LGradesCSVParser parser = new D2LGradesCSVParser(new FileReader(d2lCsvFile));
099    return parser.getGrades();
100  }
101
102  private static void usage(String message) {
103    PrintStream err = System.err;
104
105    err.println("+++ " + message);
106    err.println();
107    err.println("usage: java GradesFromD2LImporter d2lGradesCsvFileName gradeBookFileName");
108    err.println("    d2lGradesCsvFileName       Name of the CSV grades file exported from D2L");
109    err.println("    gradeBookFileName    Gradebook file");
110    err.println();
111    err.println("Imports grades from D2L into a gradebook");
112    err.println();
113
114    System.exit(1);
115  }
116}
