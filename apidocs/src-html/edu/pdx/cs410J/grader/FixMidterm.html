001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004
005import java.io.File;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.PrintStream;
009
010/**
011 * This is a little program that combines the grades for the midterm
012 * quiz, the dream job, and the resume into one grade.  This way
013 * people who missed this quiz are not penalized unnecessarily.
014 */
015public class FixMidterm {
016
017  public static void main(String[] args) {
018    PrintStream out = System.out;
019    PrintStream err = System.err;
020
021    if (args.length < 1) {
022      err.println("** usage: java FixMidterm gradebook");
023      System.exit(1);
024    }
025
026    String xmlFile = args[0];
027
028    GradeBook book;
029
030    File file = new File(xmlFile);
031    if (file.exists()) {
032      // Parse a grade book from the XML file
033      try {
034        XmlGradeBookParser parser = new XmlGradeBookParser(file);
035        book = parser.parse();
036
037      } catch (FileNotFoundException ex) {
038        err.println("** Could not find file: " + ex.getMessage());
039        System.exit(1);
040        return;
041
042      } catch (IOException ex) {
043        err.println("** IOException during parsing: " + ex.getMessage());
044        System.exit(1);
045        return;
046
047      } catch (ParserException ex) {
048        err.println("** Error during parsing: " + ex.getMessage());
049        System.exit(1);
050        return;
051      }
052
053    } else {
054      err.println("** File " + file + " does not exist");
055      System.exit(1);
056      return;
057    }
058
059    book.forEachStudent(student -> {
060      out.print("Fixing " + student.getId() + ": ");
061      out.flush();
062
063      double total = 0.0;
064
065      Grade quiz3 = student.getGrade("quiz3");
066      if (quiz3 != null) {
067        total += quiz3.getScore();
068      } else {
069        quiz3 = new Grade("quiz3", 0.0);
070      }
071
072      out.print("midterm = " + quiz3.getScore() + ", ");
073      out.flush();
074
075      Grade job = student.getGrade("job");
076      if (job != null) {
077        total += job.getScore();
078        quiz3.addNote("Dream job score: " + job.getScore());
079        out.print("dream job = " + job.getScore() + ", ");
080
081      } else {
082        quiz3.addNote("Missing dream job");
083      }
084
085      Grade resume = student.getGrade("resume");
086      if (resume != null) {
087        total += resume.getScore();
088        quiz3.addNote("Resume score: " + resume.getScore());
089        out.print("resume = " + resume.getScore() + ", ");
090        out.flush();
091
092      } else {
093        quiz3.addNote("Missing resume");
094      }
095
096      quiz3.setScore(total);
097      student.setGrade(quiz3.getAssignmentName(), quiz3);
098      out.println("total = " + quiz3.getScore());
099    });
100
101    // Write the grade book to the XML file
102    try {
103      XmlDumper dumper = new XmlDumper(file);
104      dumper.dump(book);
105
106    } catch (IOException ex) {
107      err.println("** Error while writing XML file: " + ex);
108      System.exit(1);
109    }    
110  }
111
112}
