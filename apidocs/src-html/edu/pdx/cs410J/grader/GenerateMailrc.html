001package edu.pdx.cs410J.grader;
002
003import edu.pdx.cs410J.ParserException;
004
005import java.io.*;
006import java.util.Comparator;
007import java.util.Iterator;
008import java.util.SortedSet;
009import java.util.TreeSet;
010
011/**
012 * This program generates a .mailrc file containing a mail alias for
013 * every student in a given grade book.
014 */
015public class GenerateMailrc {
016
017  /**
018   * Generates a mailrc file for the students in a given grade book.
019   */
020  static void generateMailrc(GradeBook book, PrintWriter mailrc) {
021    mailrc.println("# Mail aliases for " + book.getClassName());
022
023    // Sort all of the students by their id
024    SortedSet<Student> sorted = new TreeSet<Student>(new Comparator<Student>() {
025      public int compare(Student o1, Student o2) {
026        String id1 = o1.getId();
027        String id2 = o2.getId();
028        return id1.compareTo(id2);
029      }
030    });
031
032    book.forEachStudent(sorted::add);
033
034    Iterator iter = sorted.iterator();
035    while (iter.hasNext()) {
036      // alias id "email (firstName LastName)"
037      Student student = (Student) iter.next();
038      String email = student.getEmail();
039      if (email != null) {
040        mailrc.print("alias " + student.getId() + " \"");
041        StringBuffer name = new StringBuffer();
042
043        if (student.getNickName() != null) {
044          name.append(student.getNickName());
045          name.append(" ");
046
047        } else if (student.getFirstName() != null) {
048          name.append(student.getFirstName());
049          name.append(" ");
050        }
051        if(student.getLastName() != null) {
052          name.append(student.getLastName());
053        }
054        if(!name.toString().equals("")) {
055          mailrc.print(name.toString().trim() + " <" + email + ">");
056        }
057        mailrc.println("\"");
058      }
059    }
060  }
061
062  /**
063   * Displays usage information for this program
064   */
065  private static void usage() {
066    PrintStream err = System.err;
067    err.println("usage: java GenerateMailrc gradeBookFile mailrcFile");
068    System.exit(1);
069  }
070
071  /**
072   * Main program.
073   */
074  public static void main(String[] args) {
075    if (args.length < 2) {
076      System.err.println("** Missing arguments");
077      usage();
078    }
079
080    String bookName = args[0];
081    String mailrcName = args[1];
082
083    GradeBook book = null;
084    PrintWriter mailrc = null;
085    try {
086      XmlGradeBookParser parser = new XmlGradeBookParser(bookName);
087      book = parser.parse();
088      mailrc = new PrintWriter(new FileWriter(mailrcName), true);
089
090    } catch (FileNotFoundException ex) {
091      System.err.println("** Could not find file: " + ex.getMessage());
092      System.exit(1);
093      
094    } catch (IOException ex) {
095      System.err.println("** IOException during parsing: " + ex.getMessage());
096      System.exit(1);
097
098    } catch (ParserException ex) {
099      System.err.println("** Error during parsing: " + ex);
100      System.exit(1);
101    }
102
103    generateMailrc(book, mailrc);
104  }
105
106}
