003import java.lang.reflect.*;
004import java.io.*;
005import java.util.*;

007/**
008 * This program uses Java reflection to invoke a static method of a
009 * given class.
010 */
011public class InvokeStatic {
012  private static PrintStream out = System.out;
013  private static PrintStream err = System.err;
014  private static final boolean VERBOSE = 
015    Boolean.getBoolean("InvokeStatic.VERBOSE");
016  
017  /**
018   * Turns a string into the an instance of the given type.  Array
019   * elements are separated by a space.
020   */
021  private static Object parseParam(String param, Class type) 
022    throws Throwable {
023
024    if (type.isArray()) {
025      Class elementType = type.getComponentType();
026      StringTokenizer st = new StringTokenizer(param, " ");
027      List<Object> elements = new ArrayList<Object>();
028      while (st.hasMoreTokens()) {
029        elements.add(parseParam(st.nextToken(), elementType));
030      }
031
032      Object array = Array.newInstance(elementType, elements.size());
033      for (int i = 0; i < elements.size(); i++) {
034        Array.set(array, i, elements.get(i));
035      }
036      return array;
037
038    } else {
039      // The type must have a constructor that takes a string argument
040      if (type.equals(Character.TYPE)) {
041        type = Character.class;
042
043      } else if (type.equals(Boolean.TYPE)) {
044        type = Boolean.class;
045
046      } else if (type.equals(Byte.TYPE)) {
047        type = Byte.class;
048
049      } else if (type.equals(Short.TYPE)) {
050        type = Short.class;
051
052      } else if (type.equals(Integer.TYPE)) {
053        type = Integer.class;
054
055      } else if (type.equals(Long.TYPE)) {
056        type = Long.class;
057
058      } else if (type.equals(Float.TYPE)) {
059        type = Float.class;
060
061      } else if (type.equals(Double.TYPE)) {
062        type = Double.class;
063      }
064
065      Constructor init = 
066        type.getConstructor(new Class[] { String.class });
067      return init.newInstance(new Object[] { param });
068    }
069  }
070
071  /**
072   * This method converts a string into a Class type.  Array types are
073   * denoted with a preceeding "[".
074   */
075  private static Class parseType(String name)
076    throws ClassNotFoundException {
077    
078    int lastBracket = name.lastIndexOf('[');
079    if (lastBracket == -1) {
080      // Not an array type
081
082      if (name.equals("char")) {
083        return Character.TYPE;
084
085      } else if (name.equals("boolean")) {
086        return Boolean.TYPE;
087
088      } else if (name.equals("byte")) {
089        return Byte.TYPE;
090
091      } else if (name.equals("short")) {
092        return Short.TYPE;
093
094      } else if (name.equals("int")) {
095        return Integer.TYPE;
096
097      } else if (name.equals("long")) {
098        return Long.TYPE;
099
100      } else if (name.equals("float")) {
101        return Float.TYPE;
102
103      } else if (name.equals("double")) {
104        return Double.TYPE;
105
106      } else {
107        return Class.forName(name);
108      }
109
110    } else {
111      // Array type
112      String className = name.substring(lastBracket + 1);
113      Class elementType = parseType(className);
114      int[] dimensions = new int[lastBracket + 1];
115      for (int i = 0; i < dimensions.length; i++) {
116        dimensions[i] = 0;
117      }
118      return Array.newInstance(elementType, dimensions).getClass();
119    }
120  }
121
122  /**
123   * Returns a default value for a given type.  If the type is an
124   * array type, then it returns an empty array of that type.
125   * Otherwise, it returns <code>null</code>.
126   */
127  private static Object getDefaultValueFor(Class type) {
128    if (type.isArray()) {
129      return Array.newInstance(type.getComponentType(), 0);
130
131    } else {
132      return null;
133    }
134  }
135
136  /**
137   * Prints usage information for this program
138   */
139  private static void usage(String s) {
140    err.println("\n** " + s + "\n");
141    err.println("usage java InvokeStatic className methodName " +
142                "[paramTypes] [(param)*]");
143    err.println("  className   The name of the class containing the "
144                + "static method");
145    err.println("  methodName  The name of the static method");
146    err.println("  paramTypes  The types of the method's arguments " +
147                "(defaults to an array of");
148    err.println("  param       An argument to the method (defaults "
149                + "to an empty array of");
150    err.println("              Strings");
151    err.println("              Strings");
152    err.println("");
153    System.exit(1);
154  }
155
156  public static void main(String[] args) throws Throwable {
157    String className = null;
158    String methodName = null;
159    String paramTypes = null;
160    List<String> params = new ArrayList<String>();
161
162    // Parse the command line
163    for (int i = 0; i < args.length; i++) {
164      if (className == null) {
165        className = args[i];
166
167      } else if (methodName == null) {
168        methodName = args[i];
169
170      } else if (paramTypes == null) {
171        paramTypes = args[i];
172
173      } else {
174        // It's an arguement
175        params.add(args[i]);
176      }
177    }
178
179    if (className == null) {
180      usage("Missing class name");
181    }
182
183    if (methodName == null) {
184      usage("Missing method name");
185    }
186
187    if (paramTypes == null) {
188      paramTypes = ""; // "[java.lang.String";
189    }
190
191//      if (params.isEmpty()) {
192//        params.add("");
193//      }
194
195    // Translate the methodArgsTypes into types
196    List<Object> argsTypes = new ArrayList<Object>();
197    StringTokenizer st = new StringTokenizer(paramTypes, ";");
198    while (st.hasMoreTokens()) {
199      argsTypes.add(parseType(st.nextToken()));
200    }
201
202    Class[] types = (Class[]) argsTypes.toArray(new Class[0]);
203    List<Object> actuals = new ArrayList<Object>();
204
205    // Translate the parameters into objects
206    for (int i = 0; i < types.length; i++) {
207      Class type = types[i];
208      if (params.isEmpty()) {
209        actuals.add(getDefaultValueFor(type));
210
211      } else if (type.isArray()) {
212        // The params are elements of the array
213        Class elementType = type.getComponentType();
214        Object array = Array.newInstance(elementType, params.size());
215
216        for (int j = 0; j < params.size(); j++) {
217          String param = (String) params.get(j);
218          Array.set(array, j, parseParam(param, elementType));
219        }
220
221        actuals.add(array);
222
223      } else {
224        String param = (String) params.get(i);
225        actuals.add(parseParam(param, type));
226      }
227    }
228
229    // Make sure that the number of parameter types equals the number
230    // of parameters
231    if (!actuals.isEmpty() && types.length != actuals.size()) {
232      String s = "Param mismatch: " + types.length + 
233        " param types and " + actuals.size() + " params";
234      throw new IllegalArgumentException(s);
235    }
236
237    if (VERBOSE) {
238      StringBuffer sb = new StringBuffer();
239      sb.append("Invoking ");
240      sb.append(className);
241      sb.append(".");
242      sb.append(methodName);
243      sb.append("(");
244      for (int i = 0; i < types.length; i++) {
245        sb.append(types[i].getName());
246        if (i != types.length - 1) {
247          sb.append(", ");
248        }
249      }
250      sb.append(")");
251      out.println(sb.toString());
252      out.println("params are " + actuals);
253    }
254
255    Class c = Class.forName(className);
256    Method m = c.getDeclaredMethod(methodName, types);
257    m.setAccessible(true);
258
259    try {
260      Object value = m.invoke(null, actuals.toArray());
261
262      if (!m.getReturnType().equals(Void.TYPE)) {
263        System.out.println("\nMethod returned: " + value);
264      }
265
266    } catch (InvocationTargetException ex) {
267      throw ex.getTargetException();
268    }
269  }
