001package edu.pdx.cs410J.grader;
002
003import java.awt.BorderLayout;
004import java.awt.Container;
005import java.awt.event.ActionEvent;
006import java.awt.event.ActionListener;
007
008import javax.swing.Box;
009import javax.swing.BoxLayout;
010import javax.swing.JButton;
011import javax.swing.JDialog;
012import javax.swing.JFrame;
013import javax.swing.JPanel;
014
015/**
016 * This dialog is used to create a new <code>Assignment</code>
017 */
018@SuppressWarnings("serial")
019public class NewAssignmentDialog extends JDialog {
020
021  private Assignment assignment;
022
023  // GUI components we care about
024  private AssignmentPanel assignmentPanel;
025
026  /**
027   * Creates a new <code>NewAssignmentDialog</code> and populates with
028   * GUI components.
029   */
030  public NewAssignmentDialog(JFrame owner) {
031    super(owner, "New Assignment", true /* modal */);
032
033    Container pane = this.getContentPane();
034    pane.setLayout(new BorderLayout());
035
036    this.assignmentPanel = new AssignmentPanel(true);
037    pane.add(this.assignmentPanel, BorderLayout.CENTER);
038
039    JPanel buttons = new JPanel();
040    buttons.setLayout(new BoxLayout(buttons, BoxLayout.X_AXIS));
041    buttons.add(Box.createHorizontalGlue());
042    
043    JButton ok = new JButton("OK");
044    ok.addActionListener(new ActionListener() {
045        public void actionPerformed(ActionEvent e) {
046          // Create a new Assignment from the contents of assignment
047          // panel
048          assignment = assignmentPanel.createAssignment();
049          if (assignment != null) {
050            // Assignment is okay
051            NewAssignmentDialog.this.dispose();
052          }
053        }
054      });
055    buttons.add(ok);
056
057    buttons.add(Box.createHorizontalGlue());
058
059    JButton cancel = new JButton("Cancel");
060    cancel.addActionListener(new ActionListener() {
061        public void actionPerformed(ActionEvent e) {
062          // Do not create a new assignment
063          assignment = null;
064          NewAssignmentDialog.this.dispose();
065        }
066      });
067    buttons.add(cancel);
068
069    buttons.add(Box.createHorizontalGlue());
070
071    pane.add(buttons, BorderLayout.SOUTH);
072  }
073
074  /**
075   * Returns the <code>Assignment</code> created by this dialog.  If
076   * no assignment was created, <code>null</code> is returned.
077   */
078  public Assignment getAssignment() {
079    return this.assignment;
080  }
081
082}
