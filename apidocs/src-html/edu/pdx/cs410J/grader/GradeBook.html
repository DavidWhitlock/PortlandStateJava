016/**
017 * This class represents a grade book that contains information about
018 * a CS410J class: the assignments, the students and their grades.

020 * @author David Whitlock
021 * @since Fall 2000
022 */
023public class GradeBook {
024
025  /** The name of the class */
026  private String className;
027

029  private Map<String, Assignment> assignments = new TreeMap<>();

034  /** Has the grade book been modified? */
035  private boolean dirty = true;

134    if (this.students.remove(student.getId()) != null) {
135      this.setDirty(true);

147  /**
148   * Sets the dirtiness of this <code>GradeBook</code>
149   */
150  public void setDirty(boolean dirty) {
151    this.dirty = dirty;
152  }
153
154  /**
155   * Marks this <code>GradeBook</code> as being clean
156   */
157  public void makeClean() {
158    this.setDirty(false);
159
160    for (Assignment assignment : this.assignments.values()) {
161      assignment.makeClean();
162    }
163
164    for (Student student : this.students.values()) {
165      student.makeClean();
166    }
167  }
168
169  /**
170   * Returns <code>true</code> if this <code>GradeBook</code> has been
171   * modified.
172   */
173  public boolean isDirty() {
174    if (this.dirty) {
175      return true;
176    }
177
178    // Are any of the Assignments dirty?
179    Iterator iter = this.assignments.values().iterator();
180    while (iter.hasNext()) {
181      Assignment assign = (Assignment) iter.next();
182      if (assign.isDirty()) {
183        return true;
184      }
185    }
186
187    // Are any of the Students dirty?
188    iter = this.students.values().iterator();
189    while (iter.hasNext()) {
190      Student student = (Student) iter.next();
191      if (student.isDirty()) {
192        return true;
193      }
194    }
195
196    return false;
197  }

202   */
203  public String toString() {
204    return "Grade book for " + this.getClassName() + " with " +
205           this.getStudentIds().size() + " students";
206  }

215    err.println("Where [options] are:");

222  /**
223   * Main program that is used to create a <code>GradeBook</code>
224   */
225  public static void main(String[] args) {
226    String xmlFile = null;
227    String name = null;
228    String importName = null;
229
230    // Parse the command line
231    for (int i = 0; i < args.length; i++) {
232      if (args[i].equals("-file")) {
233        if (++i >= args.length) {
234          err.println("** Missing file name");
235          usage();
236        }
237
238        xmlFile = args[i];
239
240      } else if (args[i].equals("-name")) {
241        if (++i >= args.length) {
242          err.println("** Missing class name");
243          usage();
244        }
245
246        name = args[i];
247
248      } else if (args[i].equals("-import")) {
249        if (++i >= args.length) {
250          err.println("** Missing import file name");
251          usage();
252        }
253
254        importName = args[i];
255
256      } else {
257        err.println("** Spurious command line option: " + args[i]);
258        usage();
259      }
260    }
261
262    if (xmlFile == null) {
263      err.println("** No XML file specified");
264      usage();
265    }
266
267    GradeBook book = null;
268
269    File file = new File(xmlFile);
270    if (file.exists()) {
271      // Parse a grade book from the XML file
272      try {
273        XmlGradeBookParser parser = new XmlGradeBookParser(file);
274        book = parser.parse();
275
276      } catch (FileNotFoundException ex) {
277        err.println("** Could not find file: " + ex.getMessage());
278        System.exit(1);
279
280      } catch (IOException ex) {
281        err.println("** IOException during parsing: " + ex.getMessage());
282        System.exit(1);
283
284      } catch (ParserException ex) {
285        err.println("** Error during parsing: " + ex.getMessage());
286        System.exit(1);
287      }
288
289      // Do we import?
290      if (importName != null) {
291        File importFile = new File(importName);
292        if (!importFile.exists()) {
293          err.println("** Import file " + importFile.getName() + 
294                      " does not exist");
295          System.exit(1);
296        }
297
298        try {
299          XmlStudentParser sp = new XmlStudentParser(importFile);
300          Student student = sp.parseStudent();
301          book.addStudent(student);
302
303        } catch (IOException ex) {
304          err.println("** Error during parsing: " + ex.getMessage());
305          System.exit(1);
306
307        } catch (ParserException ex) {
308          err.println("** Error during parsing: " + ex.getMessage());
309          System.exit(1);
310        }
311      }
312
313    } else if (name == null) {
314      err.println("** Must specify a class name when creating a " +
315                  "grade book");
316      System.exit(1);
317      
318    } else {
319      // Create an empty GradeBook
320      book = new GradeBook(name);
321    }
322
323    // Write the grade book to the XML file
324    try {
325      XmlDumper dumper = new XmlDumper(file);
326      dumper.dump(book);
327
328    } catch (IOException ex) {
329      err.println("** Error while writing XML file: " + ex);
330      System.exit(1);
331    }
332  }

338  public LetterGrade getLetterGradeForScore(Section section, double score) {
339    for (LetterGradeRange range : this.getLetterGradeRanges(section)) {
340      if (range.isScoreInRange(score)) {
341        return range.letterGrade();
342      }
343    }
344
345    throw new IllegalStateException("Could not find a letter grade range for " + score);
346  }

360  static class LetterGradeRanges implements Iterable<LetterGradeRanges.LetterGradeRange> {
361    private final Map<LetterGrade, LetterGradeRange> ranges = new TreeMap<>();
362
363    private LetterGradeRanges() {
364      createDefaultLetterGradeRange(LetterGrade.A, 94, 100);
365      createDefaultLetterGradeRange(LetterGrade.A_MINUS, 90, 93);
366      createDefaultLetterGradeRange(LetterGrade.B_PLUS, 87, 89);
367      createDefaultLetterGradeRange(LetterGrade.B, 83, 86);
368      createDefaultLetterGradeRange(LetterGrade.B_MINUS, 80, 82);
369      createDefaultLetterGradeRange(LetterGrade.C_PLUS, 77, 79);
370      createDefaultLetterGradeRange(LetterGrade.C, 73, 76);
371      createDefaultLetterGradeRange(LetterGrade.C_MINUS, 70, 72);
372      createDefaultLetterGradeRange(LetterGrade.D_PLUS, 67, 69);
373      createDefaultLetterGradeRange(LetterGrade.D, 63, 66);
374      createDefaultLetterGradeRange(LetterGrade.D_MINUS, 60, 62);
375      createDefaultLetterGradeRange(LetterGrade.F, 0, 59);
376    }
377
378    private void createDefaultLetterGradeRange(LetterGrade letterGrade, int minimum, int maximum) {
379      this.ranges.put(letterGrade, new LetterGradeRange(letterGrade, minimum, maximum));
380    }
381
382    public LetterGradeRange getRange(LetterGrade letterGrade) {
383      return ranges.get(letterGrade);
384    }
385
386    public void validate() {
387      validateThatFRangeHasAMinimumOf0();
388      validateThatARangeContains100();
389      validateThatRangesAreContiguous();
390    }
391
392    private void validateThatRangesAreContiguous() {
393      LetterGradeRange previous = null;
394
395      for (LetterGrade letterGrade : ranges.keySet()) {
396        LetterGradeRange current = ranges.get(letterGrade);
397        if (previous != null) {
398          if (previous.minimum() != current.maximum() + 1) {
399            String s = "There is a gap between the range for " + previous + " and " + current;
400            throw new LetterGradeRange.InvalidLetterGradeRange(s);
401          }
402        }
403
404        previous = current;
405      }
406
407    }
408
409    private void validateThatARangeContains100() {
410      LetterGradeRange range = getRange(LetterGrade.A);
411      if (range.maximum() < 100) {
412        throw new LetterGradeRange.InvalidLetterGradeRange("The A range must contain 100");
413      }
414    }
415
416    private void validateThatFRangeHasAMinimumOf0() {
417      LetterGradeRange range = getRange(LetterGrade.F);
418      if (range.minimum() > 0) {
419        throw new LetterGradeRange.InvalidLetterGradeRange("The F range must contain zero");
420      }
421    }
422
423    @Override
424    public Iterator<LetterGradeRange> iterator() {
425      return this.ranges.values().iterator();
426    }
427
428    @Override
429    public void forEach(Consumer<? super LetterGradeRange> action) {
430      this.ranges.values().forEach(action);
431    }
432
433    @Override
434    public Spliterator<LetterGradeRange> spliterator() {
435      return this.ranges.values().spliterator();
436    }
437
438    static class LetterGradeRange {
439      private final LetterGrade letterGrade;
440      private int maximum;
441      private int minimum;
442
443      public LetterGradeRange(LetterGrade letterGrade, int minimum, int maximum) {
444        this.letterGrade = letterGrade;
445        setRange(minimum, maximum);
446      }
447
448      public void setRange(int minimum, int maximum) {
449        if (minimum >= maximum) {
450          String s = String.format("Minimum value (%d) must be less than maximum value (%d)",
451            minimum, maximum);
452          throw new InvalidLetterGradeRange(s);
453        }
454
455        this.minimum = minimum;
456        this.maximum = maximum;
457      }
458
459      public int minimum() {
460        return this.minimum;
461      }
462
463      public int maximum() {
464        return this.maximum;
465      }
466
467      @Override
468      public String toString() {
469        return "Range for " + letterGrade() + " is " + minimum() + " to " + maximum();
470      }
471
472      public LetterGrade letterGrade() {
473        return this.letterGrade;
474      }
475
476      public boolean isScoreInRange(double score) {
477        int intScore = (int) Math.round(score);
478        return intScore >= minimum() && intScore <= maximum();
479      }
480
481      public static class InvalidLetterGradeRange extends RuntimeException {
482        public InvalidLetterGradeRange(String message) {
483          super(message);
484        }
485      }
486    }
487  }
