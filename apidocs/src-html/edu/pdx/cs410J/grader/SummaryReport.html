024  /**
025   * Computes the student's final average and makes a pretty report.
026   */
027  @VisibleForTesting
028  static void dumpReportTo(GradeBook book, Student student,
029                                   PrintWriter pw, boolean assignLetterGrades) {
030    NumberFormat format = NumberFormat.getNumberInstance();
031    format.setMinimumFractionDigits(1);
032    format.setMaximumFractionDigits(1);
033
034    Assignment lowestQuiz = null;
035    double best = 0.0;
036    double total = 0.0;
037
038    pw.println("Grade summary for: " + student.getFullName());
039    SimpleDateFormat df = 
040      new SimpleDateFormat("EEEE MMMM d, yyyy 'at' h:mm a");
041    pw.println("Generated on: " + df.format(new Date()));
042    pw.println("");
043
044    for (String assignmentName : getSortedAssignmentNames(book)) {
045      Assignment assignment = book.getAssignment(assignmentName);
046      if (noStudentHasGradeFor(assignment, book)) {
047        continue;
048      }
049
050      Grade grade = student.getGrade(assignmentName);
051      double score = getScore(grade);
052
053
054//       System.out.println("Examining " + assign + ", score: " + score);
055
056      if (assignment.getType() == Assignment.AssignmentType.QUIZ) {
057        if (lowestQuiz == null) {
058          lowestQuiz = assignment;
059//           System.out.println("Lowest quiz: " + lowestQuiz + 
060//                              ", score: " + score);
061
062        } else {
063          Grade lowestGrade = student.getGrade(lowestQuiz.getName());
064          if (lowestGrade != null && score < lowestGrade.getScore()) {
065            lowestQuiz = assignment;
066//             System.out.println("Lowest quiz: " + lowestQuiz + ", score: "
067//                                + score + ", lowest grade: " +
068//                                student.getGrade(lowestQuiz.getName()));
069          }
070        }
071      }
072
073      StringBuffer line = new StringBuffer();
074      line.append("  ");
075      line.append(assignment.getName());
076      line.append(" (");
077      line.append(assignment.getDescription());
078      line.append(")");
079      if (assignment.getType() == Assignment.AssignmentType.OPTIONAL) {
080        line.append(" (OPTIONAL)");
081      }
082      line.append(": ");
083      line.append(format.format(score));
084      line.append("/");
085      line.append(format.format(assignment.getPoints()));
086
087
088      // Skip incompletes and no grades
089      if (grade == null) {
090        line.append(" (MISSING GRADE)");
091
092      } else if (grade.isIncomplete()) {
093        line.append(" (INCOMPLETE)");
094
095      } else if (grade.isNotGraded()) {
096        line.append( "(NOT GRADED)");
097      }
098
099      pw.println(line);
100
101      // Don't count optional assignments toward the maximum point
102      // total
103      if (assignment.getType() != Assignment.AssignmentType.OPTIONAL) {
104        best += assignment.getPoints();
105      }
106
107      total += score;
108    }
109
110    if (lowestQuiz != null) {
111      pw.println("");
112      pw.println("Lowest Quiz grade dropped: " +
113                  lowestQuiz.getName());
114      pw.println("");
115
116      // Subtract lowest quiz grade
117      Grade lowestGrade = student.getGrade(lowestQuiz.getName());
118      total -= (lowestGrade != null ? lowestGrade.getScore() : 0);
119      best -= lowestQuiz.getPoints();
120    }
121
122    // Print out late and resubmitted assignments
123    pw.println("Late assignments:");
124    for (String late : student.getLate()) {
125      pw.println("  " + late);
126    }
127    pw.println();
128
129    pw.println("Resubmitted assignments:");
130    for (String resubmitted : student.getResubmitted()) {
131      pw.println("  " + resubmitted);
132    }
133    pw.println("");
134
135    pw.println("Total grade: " + format.format(total)  + "/" +
136               format.format(best));
137
138    double overallScore = total / best;
139
140    if (assignLetterGrades) {
141      LetterGrade letterGrade = book.getLetterGradeForScore(student.getEnrolledSection(), overallScore * 100.0);
142      student.setLetterGrade(letterGrade);
143    }
144
145    if (student.getLetterGrade() != null) {
146      pw.println("Letter Grade: " + student.getLetterGrade());
147    }
148
149    allTotals.put(student, overallScore);
150  }
151
152  static boolean noStudentHasGradeFor(Assignment assignment, GradeBook book) {
153    return book.studentsStream().noneMatch(student -> studentHasGradeFor(student, assignment));
154  }
155
156  private static boolean studentHasGradeFor(Student student, Assignment assignment) {
157    Grade grade = student.getGrade(assignment.getName());
158    return grade != null && !grade.isNotGraded();
159  }
160
161  private static double getScore(Grade grade) {
162    // Average non-existent scores as zero
163    double score;
164    if (grade == null) {
165      score = 0.0;
166
167    } else {
168      score = grade.getScore();
169    }
170    return score;
171  }
172
173  private static SortedSet<String> getSortedAssignmentNames(GradeBook book) {
174    return new TreeSet<>(book.getAssignmentNames());
175  }

183  private static void usage(String s) {
184    err.println("\n** " + s + "\n");
185    err.println("\njava SummaryReport -assignLetterGrades xmlFile outDir (student)*");
186    err.println("\n");

194  /**
195   * Main program that creates summary reports for every student in a
196   * grade book located in a given XML file.
197   */
198  public static void main(String[] args) {
199    boolean assignLetterGrades = false;
200    String xmlFileName = null;
201    String outputDirName = null;
202    Collection<String> students = new ArrayList<>();
203
204    for (String arg : args) {
205      if (arg.equals("-assignLetterGrades")) {
206        assignLetterGrades = true;
207
208      } else if (xmlFileName == null) {
209        xmlFileName = arg;
210
211      } else if (outputDirName == null) {
212        outputDirName = arg;
213
214      } else {
215        students.add(arg);
216      }
217    }
218
219    if (xmlFileName == null) {
220      usage("Missing XML file name");
221    }
222
223    if (outputDirName == null) {
224      usage("Missing output dir name");
225      return;
226    }
227
228    String xmlFile = xmlFileName;
229    File outDir = new File(outputDirName);
230    if (!outDir.exists()) {
231      outDir.mkdirs();
232
233    } else if (!outDir.isDirectory()) {
234      usage(outDir + " is not a directory");
235    }
236
237    File file = new File(xmlFile);
238    if (!file.exists()) {
239      usage("Grade book file " + xmlFile + " does not exist");
240    }
241
242    GradeBook book = parseGradeBook(file);
243
244    // Create a SummaryReport for every student
245    Iterable<String> studentIds;
246
247    if (!students.isEmpty()) {
248      studentIds = students;
249
250    } else {
251      studentIds = book.getStudentIds();
252    }
253
254    dumpReports(studentIds, book, outDir, assignLetterGrades);
255
256    // Sort students by totals and print out results:
257    Set<Student> students1 = allTotals.keySet();
258    printOutStudentTotals(students1, out);
259
260    saveGradeBookIfDirty(xmlFileName, book);
261
262  }

264  @VisibleForTesting
265  static void printOutStudentTotals(Set<Student> allStudents, PrintWriter out) {
266    SortedSet<Student> sorted = getStudentSortedByTotalPoints(allStudents);
267
268    out.println("Undergraduates:");
269    Stream<Student> undergrads = sorted.stream().filter(student -> student.getEnrolledSection() == UNDERGRADUATE);
270    printOutStudentTotals(out, undergrads);
271
272    out.println("Graduate Students:");
273    Stream<Student> grads = sorted.stream().filter(student -> student.getEnrolledSection() == GRADUATE);
274    printOutStudentTotals(out, grads);
275
276  }
277
278  private static void printOutStudentTotals(PrintWriter out, Stream<Student> students) {
279    NumberFormat format = NumberFormat.getPercentInstance();
280
281    students.forEach(student -> {
282      Double d = allTotals.get(student);
283      out.print("  " + student + ": " + format.format(d.doubleValue()));
284
285      if (student.getLetterGrade() != null) {
286        out.print(" " + student.getLetterGrade());
287      }
288
289      out.println();
290    });
291  }

293  private static void saveGradeBookIfDirty(String xmlFileName, GradeBook book) {
294    if (book.isDirty()) {
295      try {
296        XmlDumper dumper = new XmlDumper(xmlFileName);
297        dumper.dump(book);
298
299      } catch (IOException ex) {
300        printErrorMessageAndExit("While saving gradebook to " + xmlFileName, ex);
301      }
302    }
303  }

305  private static SortedSet<Student> getStudentSortedByTotalPoints(Set<Student> students) {
306    SortedSet<Student> sorted = new TreeSet<>(new Comparator<Student>() {
307        @Override
308        public int compare(Student s1, Student s2) {
309          Double d1 = allTotals.get(s1);
310          Double d2 = allTotals.get(s2);
311          if ( d2.compareTo( d1 ) == 0 ) {
312            return s1.getId().compareTo( s2.getId() );
313          } else {
314            return d2.compareTo(d1);
315          }
316        }
317
318        @Override
319        public boolean equals(Object o) {
320          return true;
321        }
322      });
323
324    sorted.addAll(students);
325    return sorted;
326  }

328  @VisibleForTesting
329  static void dumpReports(Iterable<String> studentIds, GradeBook book, File outDir, boolean assignLetterGrades) {
330    for (String id : studentIds) {
331      err.println(id);
332
333      Student student = book.getStudent(id).orElseThrow(noStudentWithId(id));
334
335      File outFile = new File(getDirectoryForReportFileForStudent(outDir, student), getReportFileName(id));
336      try {
337        PrintWriter pw =
338          new PrintWriter(new FileWriter(outFile), true);
339        dumpReportTo(book, student, pw, assignLetterGrades);
340
341//         dumpReportTo(book, student, out);
342      } catch (IOException ex) {
343        printErrorMessageAndExit("While writing report to " + outFile, ex);
344      }
345    }
346  }
347
348  private static File getDirectoryForReportFileForStudent(File parentDirectory, Student student) {
349    String directoryName;
350    Student.Section enrolledSection = student.getEnrolledSection();
351    switch (enrolledSection) {
352      case UNDERGRADUATE:
353        directoryName = UNDERGRADUATE_DIRECTORY_NAME;
354        break;
355      case GRADUATE:
356        directoryName = GRADUATE_DIRECTORY_NAME;
357        break;
358      default:
359        throw new IllegalStateException("Don't know directory name for " + enrolledSection);
360    }
361    File directory = new File(parentDirectory, directoryName);
362    directory.mkdirs();
363    return directory;
364  }
365
366  @VisibleForTesting
367  static String getReportFileName(String studentId) {
368    return studentId + ".report";
369  }

371  private static GradeBook parseGradeBook(File gradeBookFile) {
372    GradeBook book = null;
373    try {
374      err.println("Parsing " + gradeBookFile);
375      XmlGradeBookParser parser = new XmlGradeBookParser(gradeBookFile);
376      book = parser.parse();
377
378    } catch (FileNotFoundException ex) {
379      printErrorMessageAndExit("** Could not find grade book file: " + gradeBookFile, ex);
380
381    } catch (ParserException | IOException ex) {
382      printErrorMessageAndExit("** Exception while parsing " + gradeBookFile, ex);
383    }
384    return book;
385  }
386
387  private static void printErrorMessageAndExit(String message, Throwable ex) {
388    err.println(message);
389    System.exit(1);
390  }

392  @SuppressWarnings("ThrowableInstanceNeverThrown")
393  private static Supplier<? extends IllegalStateException> noStudentWithId(String id) {
394    return () -> new IllegalStateException("No student with id " + id);
395  }
