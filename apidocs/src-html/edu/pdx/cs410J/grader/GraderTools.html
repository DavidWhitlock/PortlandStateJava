001package edu.pdx.cs410J.grader;
002
003import ch.qos.logback.classic.Level;
004import ch.qos.logback.classic.Logger;
005import com.google.common.annotations.VisibleForTesting;
006import edu.pdx.cs410J.grader.poa.ui.PlanOfAttackGrader;
007import org.slf4j.LoggerFactory;
008
009import java.io.PrintStream;
010import java.lang.reflect.InvocationTargetException;
011import java.lang.reflect.Method;
012import java.util.ArrayList;
013import java.util.List;
014
015public class GraderTools {
016
017  public static void main(String[] args) throws NoSuchMethodException, IllegalAccessException, InvocationTargetException {
018    String tool = null;
019    List<String> toolArgs = new ArrayList<>();
020    for (String arg : args) {
021      if (arg.equals("-debug")) {
022        setLoggingLevelToDebug();
023
024      } else if (tool == null) {
025        tool = arg;
026
027      } else {
028        toolArgs.add(arg);
029      }
030    }
031
032    if (tool == null) {
033      usage("Missing tool");
034    }
035
036
037    invokeMainMethod(getToolClass(tool), toolArgs.toArray(new String[toolArgs.size()]));
038  }
039
040  @VisibleForTesting
041  static void setLoggingLevelToDebug() {
042    Logger logger = (Logger) LoggerFactory.getLogger("edu.pdx.cs410J.grader");
043    logger.setLevel(Level.DEBUG);
044  }
045
046  private static Class getToolClass(String tool) {
047    switch (tool) {
048      case "gradebook":
049        return GradeBookGUI.class;
050
051      case "fetch":
052        return FetchAndProcessGraderEmail.class;
053
054      case "importFromD2L" :
055        return GradesFromD2LImporter.class;
056
057      case "importFromProjectReports" :
058        return ProjectGradesImporter.class;
059
060      case "mailFileToStudent" :
061        return StudentFileMailer.class;
062
063      case "gradePOAs":
064        return PlanOfAttackGrader.class;
065
066      case "generateGradeSummary":
067        return SummaryReport.class;
068
069      case "htmlForSurveyResults":
070        return SurveyResponsesFromD2LGenerator.class;
071
072      case "fixAndroidZips":
073              return AndroidZipFixer.class;
074
075      default:
076        usage("Unknown tool: " + tool);
077        return null;
078    }
079  }
080
081  @SuppressWarnings("unchecked")
082  private static void invokeMainMethod(Class aClass, String[] args) throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
083    Method main = aClass.getMethod("main", args.getClass());
084    main.invoke(aClass, new Object[] { args });
085
086  }
087
088  private static void usage(String message) {
089    PrintStream err = System.err;
090    err.println("** " + message);
091    err.println();
092    err.println("Executes one of the Grader tools");
093    err.println();
094    err.println("usage: GraderTools tool toolArg*");
095    err.println("  tool                        The tool to execute");
096    err.println("    gradebook                 The Grade Book GUI");
097    err.println("    fetch                     Fetch student surveys or projects from the Grader's");
098    err.println("                              emails account");
099    err.println("    importFromD2L             Import grades from a D2L CSV");
100    err.println("    importFromProjectReports  Import grades from graded project reports");
101    err.println("    mailFileToStudent         Email text files to students");
102    err.println("    gradePOAs                 Tool for downloading and grading POAs");
103    err.println("    generateGradeSummary      Generate grade summary report for one or more students");
104    err.println("    htmlForSurveyResults      Generate an html file for the responses to a D2L survey");
105    err.println("    fixAndroidZips            Fix zip files for the Android project to work with grading script");
106    err.println("  toolArg                     A command line argument to send to the tool");
107    err.println();
108
109    System.exit(1);
110  }
111}
