001package edu.pdx.cs410J.apptbookgwt.client;
002
003import com.google.gwt.user.client.rpc.AsyncCallback;
004
005/**
006 * The client-side interface to the Appointment Book service
007 */
008public interface AppointmentBookServiceAsync {
009
010  /**
011   * Return the current date/time on the server
012   */
013  void getAppointmentBook(AsyncCallback<AppointmentBook> async);
014
015  /**
016   * Always throws an exception so that we can see how to handle uncaught
017   * exceptions in GWT.
018   */
019  void throwUndeclaredException(AsyncCallback<Void> async);
020
021  /**
022   * Always throws a declared exception so that we can see GWT handles it.
023   */
024  void throwDeclaredException(AsyncCallback<Void> async);
025
026}
