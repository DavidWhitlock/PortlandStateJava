001package edu.pdx.cs410J.family;
002
003import edu.pdx.cs410J.family.web.FamilyTreeManager;
004import org.apache.commons.fileupload.FileItem;
005import org.apache.commons.fileupload.FileItemFactory;
006import org.apache.commons.fileupload.FileUploadException;
007import org.apache.commons.fileupload.disk.DiskFileItemFactory;
008import org.apache.commons.fileupload.servlet.ServletFileUpload;
009
010import javax.servlet.ServletException;
011import javax.servlet.http.HttpServlet;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014import java.io.IOException;
015import java.io.InputStreamReader;
016import java.io.PrintWriter;
017import java.util.List;
018
019/**
020 * A servlet that uploads a family tree XML file and stores it in the session.
021 */
022public class FamilyTreeUploadServlet extends HttpServlet {
023  @Override
024  protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
025    response.setContentType("text/plain");
026
027    PrintWriter pw = response.getWriter();
028
029    if (ServletFileUpload.isMultipartContent(request)) {
030      FileItemFactory factory = new DiskFileItemFactory();
031      ServletFileUpload upload = new ServletFileUpload(factory);
032      List<FileItem> items;
033      try {
034        items = upload.parseRequest(request);
035
036      } catch (FileUploadException ex) {
037        throw new ServletException("Could not upload file", ex);
038      }
039
040      FileItem file = null;
041      for (FileItem item : items) {
042        if (!item.isFormField()) {
043          if (file != null) {
044            response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Only one file can be uploaded");
045            return;
046
047          } else {
048            file = item;
049          }
050        }
051      }
052
053      if (file == null) {
054        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No file specified");
055        return;
056      }
057
058      XmlParser parser = new XmlParser(new InputStreamReader(file.getInputStream()));
059
060      pw.println("Uploading " + file.getName());
061
062      FamilyTree tree;
063      try {
064        tree = parser.parse();
065
066      } catch (FamilyTreeException ex) {
067        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Couldn't parse " + file.getName());
068        return;
069      }
070
071      FamilyTreeManager.setFamilyTree(tree, request);
072      pw.println("Uploaded " + file.getName());
073
074    } else {
075      response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No files specified");
076    }
077  }
078
079  @Override
080  protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
081    response.setContentType("text/plain");
082
083    PrintWriter pw = response.getWriter();
084
085    FamilyTree tree = FamilyTreeManager.getFamilyTree(request);
086    if (tree == null) {
087      pw.println("No tree has been uploaded");
088
089    } else {
090      TextDumper dumper = new TextDumper(pw);
091      dumper.dump(tree);
092    }
093  }
094}
