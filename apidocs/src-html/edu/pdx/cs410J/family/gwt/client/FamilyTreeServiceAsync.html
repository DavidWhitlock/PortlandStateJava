001package edu.pdx.cs410J.family.gwt.client;
002
003import com.google.gwt.user.client.rpc.AsyncCallback;
004import com.google.gwt.user.client.rpc.ServiceDefTarget;
005import com.google.gwt.core.client.GWT;
006
007/**
008 * The client-side interface to the <code>FamilyTreeService</code>
009 */
010public interface FamilyTreeServiceAsync {
011  /**
012   * Returns the family tree being edited by the user
013   */
014  void getFamilyTree(AsyncCallback async);
015
016  public static class Factory {
017
018    public static FamilyTreeServiceAsync create() {
019      FamilyTreeServiceAsync service = GWT.create(FamilyTreeService.class);
020
021      ServiceDefTarget endpoint = (ServiceDefTarget) service;
022      String moduleRelativeURL = GWT.getModuleBaseURL() + "tree";
023      endpoint.setServiceEntryPoint(moduleRelativeURL);
024
025      return service;
026    }
027  }
028
029}
