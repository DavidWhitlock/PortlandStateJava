001package edu.pdx.cs410J.family.gwt.client;
002
003import com.google.gwt.core.client.EntryPoint;
004import com.google.gwt.user.client.ui.*;
005import com.google.gwt.user.client.Command;
006import com.google.gwt.user.client.Window;
007import com.google.gwt.user.client.rpc.AsyncCallback;
008import edu.pdx.cs410J.family.Person;
009import edu.pdx.cs410J.family.FamilyTree;
010
011/**
012 * The entry point of the Family Tree GWT Web Application
013 */
014public class FamilyTreeGWT extends SimplePanel implements EntryPoint {
015  private FamilyTreeList treeList;
016  private FamilyTree tree = new FamilyTree();
017  private boolean isDirty = false;
018  private final FamilyTreeServiceAsync service;
019  private final PersonPanel personPanel;
020
021  public FamilyTreeGWT() {
022    service = FamilyTreeServiceAsync.Factory.create();
023
024    treeList = new FamilyTreeList();
025    treeList.addChangeListener(new ChangeListener() {
026
027      public void onChange(Widget widget) {
028        Person person = treeList.getSelectedPerson();
029        showPerson(person);
030      }
031    });
032
033    HorizontalPanel buttons = new HorizontalPanel();
034    Button newPerson = new Button("New Person");
035    newPerson.addClickListener(new ClickListener() {
036      public void onClick(Widget widget) {
037        newPerson();
038      }
039    });
040    buttons.add(newPerson);
041
042    DockPanel left = new DockPanel();
043    left.setSize("100%", "100%");
044    left.add(treeList, DockPanel.CENTER);
045    left.setCellHeight(treeList, "100%");
046    left.add(buttons, DockPanel.SOUTH);
047    left.setCellHorizontalAlignment(buttons, DockPanel.ALIGN_CENTER);
048
049    personPanel = new PersonPanel(this);
050    personPanel.setSize("100%", "100%");
051
052    HorizontalSplitPanel split = new HorizontalSplitPanel();
053    split.setLeftWidget(left);
054    split.setRightWidget(personPanel);
055    split.setSplitPosition("200px");
056    split.setSize("100%", "100%");
057
058    DockPanel panel = new DockPanel();
059    panel.setSize("100%", "100%");
060
061    MenuBar menuBar = new MenuBar();
062    addFileMenu(menuBar);
063    addPersonMenu(menuBar);
064    panel.add(menuBar, DockPanel.NORTH);
065
066    panel.add(split, DockPanel.CENTER);
067    panel.setCellHeight(split, "100%");
068
069
070    setWidget(panel);
071
072    getFamilyTreeService().getFamilyTree(new AsyncCallback<FamilyTree>() {
073
074      public void onFailure(Throwable throwable) {
075        Window.alert("While invoking remote method: " + throwable.getMessage());
076      }
077
078      public void onSuccess(FamilyTree tree) {
079        if (tree != null) {
080          FamilyTreeGWT.this.tree = tree;
081          treeList.fillInList(tree);
082        }
083      }
084    });
085
086  }
087
088  private void addPersonMenu(MenuBar menuBar) {
089
090  }
091
092  private void addFileMenu(MenuBar menuBar) {
093    MenuBar file = new MenuBar(true);
094    menuBar.addItem("File", file);
095
096    file.addItem("Open", new Command() {
097      public void execute() {
098        open();
099      }
100    });
101
102    file.addItem("Save", new Command() {
103      public void execute() {
104        save();
105      }
106    });
107  }
108
109  private void save() {
110    Window.alert("Saving family tree");
111  }
112
113  private void open() {
114    final DialogBox open = new DialogBox(false, true);
115    open.setText("Open Family Tree XML File");
116
117    final FormPanel form = new FormPanel();
118    form.setAction("upload");
119    form.setMethod(FormPanel.METHOD_POST);
120    form.setEncoding(FormPanel.ENCODING_MULTIPART);
121
122    VerticalPanel panel = new VerticalPanel();
123    panel.add(new Label("Choose a file to upload"));
124
125    final FileUpload upload = new FileUpload();
126    upload.setName("xmlFile");
127    panel.add(upload);
128
129    HorizontalPanel buttons = new HorizontalPanel();
130
131    buttons.add(new Button("Upload", new ClickListener() {
132      public void onClick(Widget sender) {
133        form.submit();
134      }
135    }));
136    buttons.add(new Button("Cancel", new ClickListener() {
137      public void onClick(Widget widget) {
138        open.hide();
139      }
140    }));
141
142    panel.add(buttons);
143
144    form.addFormHandler(new FormHandler() {
145
146      public void onSubmit(FormSubmitEvent event) {
147        if (upload.getFilename().length() == 0) {
148          Window.alert("No file name specified");
149          event.setCancelled(true);
150        }
151      }
152
153      public void onSubmitComplete(FormSubmitCompleteEvent event) {
154        getFamilyTreeService().getFamilyTree(new AsyncCallback<FamilyTree>() {
155
156          public void onFailure(Throwable throwable) {
157            Window.alert("While invoking remote method: " + throwable.getMessage());
158          }
159
160          public void onSuccess(FamilyTree tree) {
161            FamilyTreeGWT.this.tree = tree;
162            treeList.fillInList(tree);
163            open.hide();
164          }
165        }); 
166      }
167    });
168
169    form.add(panel);
170
171    open.add(form);
172
173    open.center();
174    open.show();
175  }
176
177  private FamilyTreeServiceAsync getFamilyTreeService() {
178    return service;
179  }
180
181  private void newPerson() {
182    EditPersonDialog dialog = new EditPersonDialog(this);
183    dialog.center();
184  }
185
186  private void showPerson(Person person) {
187    personPanel.showPerson(person); 
188  }
189
190  public void onModuleLoad() {
191    RootPanel.get().add(new FamilyTreeGWT());
192  }
193
194  public void displayFather() {
195    Person person = this.treeList.getSelectedPerson();
196    if (person != null) {
197      Person father = person.getFather();
198      if (father != null) {
199        this.treeList.setSelectedPerson(father);
200        showPerson(father);
201      }
202    }
203  }
204
205  public void displayMother() {
206    Person person = this.treeList.getSelectedPerson();
207    if (person != null) {
208      Person mother = person.getMother();
209      if (mother != null) {
210        this.treeList.setSelectedPerson(mother);
211        showPerson(mother);
212      }
213    }
214  }
215
216  public void update(Person person) {
217    this.setDirty(true);
218    this.showPerson(person);
219    this.treeList.fillInList(this.tree);
220  }
221
222  private void setDirty(boolean isDirty) {
223    this.isDirty = isDirty;
224  }
225}
