001package edu.pdx.cs410J.family.gwt.client;
002
003import com.google.gwt.user.client.ui.ListBox;
004import edu.pdx.cs410J.family.Person;
005import edu.pdx.cs410J.family.FamilyTree;
006
007import java.util.*;
008
009/**
010 * Lists the people in a family tree
011 */
012public class FamilyTreeList extends ListBox {
013  private Map<Integer, Person> indexToPerson = new HashMap<Integer, Person>();
014
015  public FamilyTreeList() {
016    setVisibleItemCount(5);
017    setSize("200px", "100%");
018  }
019
020  /**
021   * Fills in the <code>JList</code> with the contents of a
022   * <code>FamilyTree</code>.
023   */
024  public void fillInList(FamilyTree tree) {
025    SortedSet<Person> sortedPeople = new TreeSet<Person>(new Comparator<Person>() {
026        // Sort id's from lowest to highest
027
028        public int compare(Person p1, Person p2) {
029          return p1.getId() - p2.getId();
030        }
031
032        public boolean equals(Object o) {
033          return true;
034        }
035      });
036
037    sortedPeople.addAll(tree.getPeople());
038    this.clear();
039
040    Iterator<Person> iter = sortedPeople.iterator();
041    for (int i = 0; iter.hasNext(); i++) {
042      Person person = iter.next();
043      this.addItem(person.getFullName() + " (" + person.getId() + ")");
044      indexToPerson.put(new Integer(i), person);
045    }
046
047  }
048
049  /**
050   * Returns the person in the list that is currently selected
051   */
052  public Person getSelectedPerson() {
053    return indexToPerson.get(getSelectedIndex());
054  }
055
056  /**
057   * Sets the selected person.
058   */
059  public void setSelectedPerson(Person person) {
060    Integer index = null;
061    for (Map.Entry<Integer, Person> integerPersonEntry : this.indexToPerson.entrySet()) {
062      Map.Entry<Integer, Person> entry = integerPersonEntry;
063      if (entry.getValue().equals(person)) {
064        index = entry.getKey();
065      }
066    }
067
068    if (index != null) {
069      this.setSelectedIndex(index.intValue());
070    }
071  }
072
073}
