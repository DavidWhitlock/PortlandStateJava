001package edu.pdx.cs410J.phonebillweb;
002
003import com.google.common.annotations.VisibleForTesting;
004import edu.pdx.cs410J.web.HttpRequestHelper;
005
006import java.io.IOException;
007import java.util.Map;
008
009import static java.net.HttpURLConnection.HTTP_OK;
010
011/**
012 * A helper class for accessing the rest client.  Note that this class provides
013 * an example of how to make gets and posts to a URL.  You'll need to change it
014 * to do something other than just send dictionary entries.
015 */
016public class PhoneBillRestClient extends HttpRequestHelper
017{
018    private static final String WEB_APP = "phonebill";
019    private static final String SERVLET = "calls";
020
021    private final String url;
022
023
024    /**
025     * Creates a client to the Phone Bil REST service running on the given host and port
026     * @param hostName The name of the host
027     * @param port The port
028     */
029    public PhoneBillRestClient( String hostName, int port )
030    {
031        this.url = String.format( "http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET );
032    }
033
034    /**
035     * Returns all dictionary entries from the server
036     */
037    public Map<String, String> getAllDictionaryEntries() throws IOException {
038      Response response = get(this.url, Map.of());
039      return Messages.parseDictionary(response.getContent());
040    }
041
042    /**
043     * Returns the definition for the given word
044     */
045    public String getDefinition(String word) throws IOException {
046      Response response = get(this.url, Map.of("word", word));
047      throwExceptionIfNotOkayHttpStatus(response);
048      String content = response.getContent();
049      return Messages.parseDictionaryEntry(content).getValue();
050    }
051
052    public void addDictionaryEntry(String word, String definition) throws IOException {
053      Response response = postToMyURL(Map.of("word", word, "definition", definition));
054      throwExceptionIfNotOkayHttpStatus(response);
055    }
056
057    @VisibleForTesting
058    Response postToMyURL(Map<String, String> dictionaryEntries) throws IOException {
059      return post(this.url, dictionaryEntries);
060    }
061
062    public void removeAllDictionaryEntries() throws IOException {
063      Response response = delete(this.url, Map.of());
064      throwExceptionIfNotOkayHttpStatus(response);
065    }
066
067    private Response throwExceptionIfNotOkayHttpStatus(Response response) {
068      int code = response.getCode();
069      if (code != HTTP_OK) {
070        throw new PhoneBillRestException(code);
071      }
072      return response;
073    }
074
075    @VisibleForTesting
076    class PhoneBillRestException extends RuntimeException {
077      PhoneBillRestException(int httpStatusCode) {
078        super("Got an HTTP Status Code of " + httpStatusCode);
079      }
080    }
081
082}
