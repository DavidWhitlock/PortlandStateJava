001package edu.pdx.cs410J;
002
003import java.io.Serializable;
004import java.util.Date;
005
006/**
007 * This abstract class represents a phone call between a caller (the
008 * phone number of the person who originates the call) and callee (the
009 * phone number of the person whose receives the phone call).  Phone
010 * calls begin and end at given times.
011 *
012 * @author David Whitlock
013 */
014public abstract class AbstractPhoneCall implements Serializable {
015
016  /**
017   * Returns the phone number of the person who originated this phone
018   * call.
019   */
020  public abstract String getCaller();
021
022  /**
023   * Returns the phone number of the person who received this phone
024   * call.
025   */
026  public abstract String getCallee();
027
028  /**
029   * Returns the time that this phone call was originated as a
030   * {@link Date}. 
031   */
032  public Date getStartTime() {
033    return null;
034  }
035
036  /**
037   * Returns a textual representation of the time that this phone call
038   * was originated.
039   */
040  public abstract String getStartTimeString();
041
042  /**
043   * Returns the time that this phone call was completed as a
044   * {@link Date}.
045   */
046  public Date getEndTime() {
047    return null;
048  }
049
050  /**
051   * Returns a textual representation of the time that this phone call
052   * was completed.
053   */
054  public abstract String getEndTimeString();
055
056  /**
057   * Returns a brief textual description of this phone call.
058   */
059  public String toString() {
060    return "Phone call from " + this.getCaller() + " to " +
061      this.getCallee() + " from " + this.getStartTimeString() + 
062      " to " + this.getEndTimeString();
063  }
064
065}
