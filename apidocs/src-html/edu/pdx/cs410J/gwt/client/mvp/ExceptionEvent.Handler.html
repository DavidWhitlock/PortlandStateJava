001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.EventHandler;
004import com.google.gwt.event.shared.GwtEvent;
005
006/**
007 * Event triggered when an exception is thrown, most often from an {@link MvpCallback} 
008 */
009public class ExceptionEvent extends GwtEvent<ExceptionEvent.Handler> {
010
011  public static Type<Handler> TYPE = new Type<Handler>();
012
013  private final Throwable exception;
014
015  public interface Handler extends EventHandler {
016    void onException(Throwable ex);
017  }
018
019  public ExceptionEvent(Throwable ex) {
020    this.exception = ex;
021  }
022
023  @Override
024  public Type<Handler> getAssociatedType() {
025    return TYPE;
026  }
027
028  @Override
029  protected void dispatch(Handler handler) {
030    handler.onException(this.exception);
031  }
032}
