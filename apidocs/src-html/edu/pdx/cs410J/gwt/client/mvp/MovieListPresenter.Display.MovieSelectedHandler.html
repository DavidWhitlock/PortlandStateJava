001package edu.pdx.cs410J.gwt.client.mvp;
002
003import com.google.gwt.event.shared.EventHandler;
004import com.google.gwt.event.shared.HandlerManager;
005import com.google.gwt.user.client.ui.Widget;
006import com.google.inject.Inject;
007import edu.pdx.cs410J.rmi.Movie;
008
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * Implements the logic for displaying a list of {@link Movie}s
014 */
015public class MovieListPresenter {
016
017  public interface Display {
018
019    void setTitles(List<String> titles);
020
021    void addMovieSelectedHandler(MovieSelectedHandler handler);
022
023      Widget asWidget();
024
025      public interface MovieSelectedHandler extends EventHandler {
026      /**
027       * Invoked when a movie is selected
028       * @param index The index of the selected movie
029       */
030      public void onMovieSelected(int index);
031    }
032  }
033
034  /** The movies being presented */
035  private List<Movie> movies;
036
037  /** The view of this presenter */
038  private final Display view;
039
040  private final HandlerManager eventBus;
041
042  @Inject
043  public MovieListPresenter(Display view, MovieServiceAsync service, final HandlerManager hm) {
044    this.view = view;
045    this.eventBus = hm;
046
047    service.getAllMovies(new MvpCallback<ArrayList<Movie>>(eventBus) {
048      @Override
049      public void onSuccess(ArrayList<Movie> movies) {
050        setMovies(movies);
051      }
052    });
053
054    this.view.addMovieSelectedHandler(new Display.MovieSelectedHandler() {
055
056      @Override
057      public void onMovieSelected(int index) {
058        eventBus.fireEvent(new DisplayMovieEvent(movies.get(index)));
059      }
060    });
061  }
062
063  private void setMovies(List<Movie> movies) {
064    this.movies = movies;
065
066    List<String> titles = new ArrayList<String>(movies.size());
067    for (Movie movie : movies) {
068      titles.add(movie.getTitle());
069    }
070
071    this.view.setTitles(titles);
072  }
073
074
075}
