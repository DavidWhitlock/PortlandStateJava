001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.TabPanel;
004import com.google.gwt.user.client.ui.Button;
005
006/**
007 * Demonstrates GWT's {@link TabPanel}
008 */
009public class TabPanelExample extends Example {
010  public TabPanelExample() {
011    super("Tab Panel");
012
013    TabPanel tabs = new TabPanel();
014    tabs.add(new Button("One"), "One");
015    tabs.add(new Button("Two"), "Two");
016    tabs.add(new Button("Three"), "Three");
017    tabs.add(new Button("Four"), "Four");
018    tabs.selectTab(0);
019    add(tabs);
020  }
021}
