001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.core.client.GWT;
004import com.google.gwt.user.client.rpc.RemoteService;
005import com.google.gwt.user.client.rpc.RemoteServiceRelativePath;
006
007/**
008 * A GWT remote service that divides two <code>double</code>s
009 */
010@RemoteServiceRelativePath("division")
011public interface DivisionService extends RemoteService {
012
013  /**
014   * Divides two numbers and returns their quotient
015   */
016  public int divide(int dividend, int divisor);
017
018  /**
019   * Inner class for accessing the async service
020   */
021  public static class Helper {
022    private static DivisionServiceAsync instance;
023
024    /**
025     * Returns the async interface for the <code>DivisionService</code>
026     */
027    public static DivisionServiceAsync getAsync() {
028      if (instance == null) {
029        instance = GWT.create(DivisionService.class);
030      }
031      return instance;
032    }
033  }
034}
