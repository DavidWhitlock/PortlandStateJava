001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.user.client.ui.Button;
004import com.google.gwt.user.client.ui.CheckBox;
005import com.google.gwt.user.client.ui.PushButton;
006import com.google.gwt.user.client.ui.ToggleButton;
007
008/**
009 * Demonstrates the various kinds of GWT buttons
010 */
011public class ButtonsExample extends Example {
012  public ButtonsExample() {
013    super("Buttons");
014
015    Button button = new Button("Disabled button");
016    button.setEnabled(false);
017    add(button);
018
019    CheckBox check = new CheckBox("Check box");
020    check.setChecked(true);
021    add(check);
022
023    PushButton push = new PushButton("Click me", "Clicked");
024    add(push);
025
026    ToggleButton toggle = new ToggleButton("Up", "Down");
027    add(toggle);
028
029  }
030}
