001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.event.dom.client.*;
004import com.google.gwt.user.client.ui.*;
005
006/**
007 * Demonstrates several of GWT's low-level event Handlers
008 */
009public class FocusPanelExample extends Example {
010  public FocusPanelExample() {
011    super("Lots of Handlers");
012
013    DockPanel dock = new DockPanel();
014
015    final Label label = new Label("Move the mouse below");
016    dock.add(label, DockPanel.NORTH);
017    dock.setCellHorizontalAlignment(label, DockPanel.ALIGN_CENTER);
018
019    final Label status = new Label("Status");
020    dock.add(status, DockPanel.SOUTH);
021    dock.setCellHorizontalAlignment(status, DockPanel.ALIGN_CENTER);
022
023    FocusPanel panel = new FocusPanel(new Label("Mouse me!"));
024    panel.addFocusHandler(new FocusHandler() {
025
026      public void onFocus( FocusEvent event) {
027        status.setText("Got focus");
028      }
029
030      public void onLostFocus(FocusEvent event) {
031        status.setText("Lost focus");
032      }
033    });
034      KeyEventHandler handler = new KeyEventHandler( status );
035      panel.addKeyDownHandler( handler );
036      panel.addKeyPressHandler( handler );
037      panel.addKeyUpHandler( handler );
038
039    panel.addMouseWheelHandler(new MouseWheelHandler() {
040      public void onMouseWheel(MouseWheelEvent event) {
041        String dir = event.isNorth() ? "North" : "South";
042        status.setText("Mouse wheel " + dir + " at " + event.getDeltaY());
043      }
044    });
045
046    dock.add(panel, DockPanel.CENTER);
047    dock.setCellHeight(panel, "200px");
048    dock.setCellWidth(panel, "200px");
049    dock.setCellHorizontalAlignment(panel, DockPanel.ALIGN_CENTER);
050    dock.setCellVerticalAlignment(panel, DockPanel.ALIGN_MIDDLE);
051
052    add(dock);
053  }
054
055    /**
056     * Handles all kinds of keyboard events
057     */
058    static class KeyEventHandler implements KeyDownHandler, KeyPressHandler, KeyUpHandler {
059        private Label status;
060
061        public KeyEventHandler( Label status )
062        {
063            this.status = status;
064        }
065
066        public void onKeyDown(KeyDownEvent event) {
067          status.setText("Key down " + getModifierString(event) + describe(event.getNativeKeyCode()));
068        }
069
070        public void onKeyPress(KeyPressEvent event) {
071          status.setText("Key press " + getModifierString(event) + event.getCharCode());
072        }
073
074        public void onKeyUp(KeyUpEvent event) {
075          status.setText("Key up " + getModifierString(event) + describe( event.getNativeKeyCode() ));
076        }
077
078        /**
079         * Returns a description of the given native key code associated with a {@link KeyEvent}
080         */
081        private String describe( int nativeKeyCode )
082        {
083            switch ( nativeKeyCode ) {
084                case KeyCodes.KEY_ALT:
085                    return "ALT";
086                case KeyCodes.KEY_BACKSPACE:
087                    return "BACKSPACE";
088                case KeyCodes.KEY_CTRL:
089                    return "CRTL";
090                case KeyCodes.KEY_DELETE:
091                    return "DELETE";
092                case KeyCodes.KEY_DOWN:
093                    return "DOWN";
094                case KeyCodes.KEY_END:
095                    return "END";
096                case KeyCodes.KEY_ENTER:
097                    return "ENTER";
098                case KeyCodes.KEY_ESCAPE:
099                    return "ESCAPE";
100                case KeyCodes.KEY_HOME:
101                    return "HOME";
102                case KeyCodes.KEY_LEFT:
103                    return "LEFT";
104                case KeyCodes.KEY_PAGEDOWN:
105                    return "PAGEDOWN";
106                case KeyCodes.KEY_PAGEUP:
107                    return "PAGEUP";
108                case KeyCodes.KEY_RIGHT:
109                    return "RIGHT";
110                case KeyCodes.KEY_SHIFT:
111                    return "SHIFT";
112                case KeyCodes.KEY_TAB:
113                    return "TAB";
114                case KeyCodes.KEY_UP:
115                    return "UP";
116                default:
117                    return Character.toString( (char) nativeKeyCode );
118            }
119        }
120
121        /**
122         * Returns a description of the given {@link KeyEvent} modifier
123         */
124        private String getModifierString(KeyEvent event) {
125          StringBuilder sb = new StringBuilder();
126
127          if (event.isAltKeyDown()) {
128            sb.append("ALT ");
129          }
130
131          if (event.isControlKeyDown()) {
132            sb.append("CTRL ");
133          }
134
135          if (event.isMetaKeyDown()) {
136            sb.append("META ");
137          }
138
139          if (event.isShiftKeyDown()) {
140            sb.append("SHIFT ");
141          }
142
143          return sb.toString();
144        }
145
146    }
147
