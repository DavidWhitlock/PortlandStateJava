001package edu.pdx.cs410J.gwt.client;
002
003import com.google.gwt.event.dom.client.ClickEvent;
004import com.google.gwt.event.dom.client.ClickHandler;
005import com.google.gwt.user.client.Window;
006import com.google.gwt.user.client.ui.Button;
007
008/**
009 * A simple class that pops up a dialog when a button is clicked
010 */
011public class ClickHandlerExample extends Example {
012  public ClickHandlerExample() {
013    super("Click Handler");
014
015    Button button = new Button("Click me");
016    button.addClickHandler(new ClickHandler() {
017        public void onClick( ClickEvent clickEvent )
018        {
019            Window.alert("I was clicked");
020        }
021    });
022    add(button);
023  }
024}
