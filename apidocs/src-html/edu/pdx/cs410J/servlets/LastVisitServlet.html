001package edu.pdx.cs410J.servlets;
002
003import javax.servlet.http.HttpServlet;
004import javax.servlet.http.HttpServletRequest;
005import javax.servlet.http.HttpServletResponse;
006import javax.servlet.http.Cookie;
007import javax.servlet.ServletException;
008import java.io.IOException;
009import java.io.PrintWriter;
010import java.text.DateFormat;
011import java.util.Date;
012
013/**
014 * This servlet sets cookies on the client's browser to determine when the last
015 * time the page was visited.
016 */
017public class LastVisitServlet extends HttpServlet {
018
019  private static final String FIRST_VISIT_COOKIE = "firstVisited";
020
021  private static final String LAST_VISIT_COOKIE = "lastVisited";
022
023  private static final DateFormat DATE_FORMAT = DateFormat.getDateTimeInstance();
024
025  protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
026    PrintWriter pw = response.getWriter();
027    String now = DATE_FORMAT.format(new Date());
028    Cookie firstVisitCookie = getCookie(request, FIRST_VISIT_COOKIE);
029    Cookie lastVisitCookie;
030    if (firstVisitCookie == null) {
031      pw.println("<h1>Welcome to the Last Visit servlet</h1>");
032      pw.println("<p>Check your browser cookies</p>");
033      firstVisitCookie = new Cookie(FIRST_VISIT_COOKIE, now);
034      firstVisitCookie.setComment("This first time you viewed this page");
035      firstVisitCookie.setMaxAge(3600);
036      response.addCookie(firstVisitCookie);
037
038    } else {
039      pw.println("<h1>Welcome back to the Last Visit servlet</h1>");
040      pw.println("<p>You first visited on " + firstVisitCookie.getValue() + "</p>");
041
042      lastVisitCookie = getCookie(request, LAST_VISIT_COOKIE);
043      pw.println("<p>You were last here on " + lastVisitCookie.getValue() + "</p>");
044    }
045
046    lastVisitCookie = new Cookie(LAST_VISIT_COOKIE, now);
047    lastVisitCookie.setMaxAge(3600);
048    response.addCookie(lastVisitCookie);
049  }
050
051  /**
052   * Returns the cookie from the given request with the given name
053   *
054   * @return <code>null</code> if no cookie with that name exists
055   */
056  private Cookie getCookie(HttpServletRequest request, String name) {
057    Cookie[] cookies = request.getCookies();
058    if (cookies == null) {
059      return null;
060    }
061    for (Cookie cookie : cookies) {
062      if (cookie.getName().equals(name)) {
063        return cookie;
064      }
065    }
066
067    return null;
068  }
069}
