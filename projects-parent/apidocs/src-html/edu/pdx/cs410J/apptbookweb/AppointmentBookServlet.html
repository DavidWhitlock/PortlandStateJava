001package edu.pdx.cs410J.apptbookweb;
002
003import com.google.common.annotations.VisibleForTesting;
004
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010import java.io.PrintWriter;
011import java.util.HashMap;
012import java.util.Map;
013
014/**
015 * This servlet ultimately provides a REST API for working with an
016 * <code>AppointmentBook</code>.  However, in its current state, it is an example
017 * of how to use HTTP and Java servlets to store simple dictionary of words
018 * and their definitions.
019 */
020public class AppointmentBookServlet extends HttpServlet
021{
022    static final String WORD_PARAMETER = "word";
023    static final String DEFINITION_PARAMETER = "definition";
024
025    private final Map<String, String> dictionary = new HashMap<>();
026
027    /**
028     * Handles an HTTP GET request from a client by writing the definition of the
029     * word specified in the "word" HTTP parameter to the HTTP response.  If the
030     * "word" parameter is not specified, all of the entries in the dictionary
031     * are written to the HTTP response.
032     */
033    @Override
034    protected void doGet( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
035    {
036        response.setContentType( "text/plain" );
037
038        String word = getParameter( WORD_PARAMETER, request );
039        if (word != null) {
040            writeDefinition(word, response);
041
042        } else {
043            writeAllDictionaryEntries(response);
044        }
045    }
046
047    /**
048     * Handles an HTTP POST request by storing the dictionary entry for the
049     * "word" and "definition" request parameters.  It writes the dictionary
050     * entry to the HTTP response.
051     */
052    @Override
053    protected void doPost( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
054    {
055        response.setContentType( "text/plain" );
056
057        String word = getParameter(WORD_PARAMETER, request );
058        if (word == null) {
059            missingRequiredParameter(response, WORD_PARAMETER);
060            return;
061        }
062
063        String definition = getParameter(DEFINITION_PARAMETER, request );
064        if ( definition == null) {
065            missingRequiredParameter( response, DEFINITION_PARAMETER );
066            return;
067        }
068
069        this.dictionary.put(word, definition);
070
071        PrintWriter pw = response.getWriter();
072        pw.println(Messages.definedWordAs(word, definition));
073        pw.flush();
074
075        response.setStatus( HttpServletResponse.SC_OK);
076    }
077
078    /**
079     * Handles an HTTP DELETE request by removing all dictionary entries.  This
080     * behavior is exposed for testing purposes only.  It's probably not
081     * something that you'd want a real application to expose.
082     */
083    @Override
084    protected void doDelete(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
085        response.setContentType("text/plain");
086
087        this.dictionary.clear();
088
089        PrintWriter pw = response.getWriter();
090        pw.println(Messages.allDictionaryEntriesDeleted());
091        pw.flush();
092
093        response.setStatus(HttpServletResponse.SC_OK);
094
095    }
096
097    /**
098     * Writes an error message about a missing parameter to the HTTP response.
099     *
100     * The text of the error message is created by {@link Messages#missingRequiredParameter(String)}
101     */
102    private void missingRequiredParameter( HttpServletResponse response, String parameterName )
103        throws IOException
104    {
105        String message = Messages.missingRequiredParameter(parameterName);
106        response.sendError(HttpServletResponse.SC_PRECONDITION_FAILED, message);
107    }
108
109    /**
110     * Writes the definition of the given word to the HTTP response.
111     *
112     * The text of the message is formatted with
113     * {@link Messages#formatDictionaryEntry(String, String)}
114     */
115    private void writeDefinition(String word, HttpServletResponse response) throws IOException {
116        String definition = this.dictionary.get(word);
117
118        if (definition == null) {
119            response.setStatus(HttpServletResponse.SC_NOT_FOUND);
120
121        } else {
122            PrintWriter pw = response.getWriter();
123            pw.println(Messages.formatDictionaryEntry(word, definition));
124
125            pw.flush();
126
127            response.setStatus(HttpServletResponse.SC_OK);
128        }
129    }
130
131    /**
132     * Writes all of the dictionary entries to the HTTP response.
133     *
134     * The text of the message is formatted with
135     * {@link Messages#formatDictionaryEntry(String, String)}
136     */
137    private void writeAllDictionaryEntries(HttpServletResponse response ) throws IOException
138    {
139        PrintWriter pw = response.getWriter();
140        Messages.formatDictionaryEntries(pw, dictionary);
141
142        pw.flush();
143
144        response.setStatus( HttpServletResponse.SC_OK );
145    }
146
147    /**
148     * Returns the value of the HTTP request parameter with the given name.
149     *
150     * @return <code>null</code> if the value of the parameter is
151     *         <code>null</code> or is the empty string
152     */
153    private String getParameter(String name, HttpServletRequest request) {
154      String value = request.getParameter(name);
155      if (value == null || "".equals(value)) {
156        return null;
157
158      } else {
159        return value;
160      }
161    }
162
163    @VisibleForTesting
164    String getDefinition(String word) {
165        return this.dictionary.get(word);
166    }
167}
