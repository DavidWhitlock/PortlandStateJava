001package edu.pdx.cs410J.phonebillweb;
002
003import edu.pdx.cs410J.web.HttpRequestHelper;
004
005import java.io.IOException;
006import java.io.PrintStream;
007import java.io.PrintWriter;
008import java.io.StringWriter;
009import java.util.Map;
010
011/**
012 * The main class that parses the command line and communicates with the
013 * Phone Bill server using REST.
014 */
015public class Project4 {
016
017    public static final String MISSING_ARGS = "Missing command line arguments";
018
019    public static void main(String... args) {
020        String hostName = null;
021        String portString = null;
022        String word = null;
023        String definition = null;
024
025        for (String arg : args) {
026            if (hostName == null) {
027                hostName = arg;
028
029            } else if ( portString == null) {
030                portString = arg;
031
032            } else if (word == null) {
033                word = arg;
034
035            } else if (definition == null) {
036                definition = arg;
037
038            } else {
039                usage("Extraneous command line argument: " + arg);
040            }
041        }
042
043        if (hostName == null) {
044            usage( MISSING_ARGS );
045
046        } else if ( portString == null) {
047            usage( "Missing port" );
048        }
049
050        int port;
051        try {
052            port = Integer.parseInt( portString );
053            
054        } catch (NumberFormatException ex) {
055            usage("Port \"" + portString + "\" must be an integer");
056            return;
057        }
058
059        PhoneBillRestClient client = new PhoneBillRestClient(hostName, port);
060
061        String message;
062        try {
063            if (word == null) {
064                // Print all word/definition pairs
065                Map<String, String> dictionary = client.getAllDictionaryEntries();
066                StringWriter sw = new StringWriter();
067                Messages.formatDictionaryEntries(new PrintWriter(sw, true), dictionary);
068                message = sw.toString();
069
070            } else if (definition == null) {
071                // Print all dictionary entries
072                message = Messages.formatDictionaryEntry(word, client.getDefinition(word));
073
074            } else {
075                // Post the word/definition pair
076                client.addDictionaryEntry(word, definition);
077                message = Messages.definedWordAs(word, definition);
078            }
079
080        } catch ( IOException ex ) {
081            error("While contacting server: " + ex);
082            return;
083        }
084
085        System.out.println(message);
086
087        System.exit(0);
088    }
089
090    /**
091     * Makes sure that the give response has the expected HTTP status code
092     * @param code The expected status code
093     * @param response The response from the server
094     */
095    private static void checkResponseCode( int code, HttpRequestHelper.Response response )
096    {
097        if (response.getCode() != code) {
098            error(String.format("Expected HTTP code %d, got code %d.\n\n%s", code,
099                                response.getCode(), response.getContent()));
100        }
101    }
102
103    private static void error( String message )
104    {
105        PrintStream err = System.err;
106        err.println("** " + message);
107
108        System.exit(1);
109    }
110
111    /**
112     * Prints usage information for this program and exits
113     * @param message An error message to print
114     */
115    private static void usage( String message )
116    {
117        PrintStream err = System.err;
118        err.println("** " + message);
119        err.println();
120        err.println("usage: java Project4 host port [word] [definition]");
121        err.println("  host         Host of web server");
122        err.println("  port         Port of web server");
123        err.println("  word         Word in dictionary");
124        err.println("  definition   Definition of word");
125        err.println();
126        err.println("This simple program posts words and their definitions");
127        err.println("to the server.");
128        err.println("If no definition is specified, then the word's definition");
129        err.println("is printed.");
130        err.println("If no word is specified, all dictionary entries are printed");
131        err.println();
132
133        System.exit(1);
134    }
135}
