001package edu.pdx.cs410J.phonebillweb;
002
003import com.google.common.annotations.VisibleForTesting;
004
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010import java.io.PrintWriter;
011import java.util.HashMap;
012import java.util.Map;
013
014/**
015 * This servlet ultimately provides a REST API for working with an
016 * <code>PhoneBill</code>.  However, in its current state, it is an example
017 * of how to use HTTP and Java servlets to store simple key/value pairs.
018 */
019public class PhoneBillServlet extends HttpServlet
020{
021    private final Map<String, String> data = new HashMap<>();
022
023    /**
024     * Handles an HTTP GET request from a client by writing the value of the key
025     * specified in the "key" HTTP parameter to the HTTP response.  If the "key"
026     * parameter is not specified, all of the key/value pairs are written to the
027     * HTTP response.
028     */
029    @Override
030    protected void doGet( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
031    {
032        response.setContentType( "text/plain" );
033
034        String key = getParameter( "key", request );
035        if (key != null) {
036            writeValue(key, response);
037
038        } else {
039            writeAllMappings(response);
040        }
041    }
042
043    /**
044     * Handles an HTTP POST request by storing the key/value pair specified by the
045     * "key" and "value" request parameters.  It writes the key/value pair to the
046     * HTTP response.
047     */
048    @Override
049    protected void doPost( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
050    {
051        response.setContentType( "text/plain" );
052
053        String key = getParameter( "key", request );
054        if (key == null) {
055            missingRequiredParameter(response, "key");
056            return;
057        }
058
059        String value = getParameter( "value", request );
060        if ( value == null) {
061            missingRequiredParameter( response, "value" );
062            return;
063        }
064
065        this.data.put(key, value);
066
067        PrintWriter pw = response.getWriter();
068        pw.println(Messages.mappedKeyValue(key, value));
069        pw.flush();
070
071        response.setStatus( HttpServletResponse.SC_OK);
072    }
073
074    /**
075     * Handles an HTTP DELETE request by removing all key/value pairs.  This
076     * behavior is exposed for testing purposes only.  It's probably not
077     * something that you'd want a real application to expose.
078     */
079    @Override
080    protected void doDelete(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
081        response.setContentType("text/plain");
082
083        this.data.clear();
084
085        PrintWriter pw = response.getWriter();
086        pw.println(Messages.allMappingsDeleted());
087        pw.flush();
088
089        response.setStatus(HttpServletResponse.SC_OK);
090
091    }
092
093    /**
094     * Writes an error message about a missing parameter to the HTTP response.
095     *
096     * The text of the error message is created by {@link Messages#missingRequiredParameter(String)}
097     */
098    private void missingRequiredParameter( HttpServletResponse response, String parameterName )
099        throws IOException
100    {
101        String message = Messages.missingRequiredParameter(parameterName);
102        response.sendError(HttpServletResponse.SC_PRECONDITION_FAILED, message);
103    }
104
105    /**
106     * Writes the value of the given key to the HTTP response.
107     *
108     * The text of the message is formatted with
109     * {@link Messages#formatKeyValuePair(String, String)}
110     */
111    private void writeValue( String key, HttpServletResponse response ) throws IOException
112    {
113        String value = this.data.get(key);
114
115        PrintWriter pw = response.getWriter();
116        pw.println(Messages.formatKeyValuePair(key, value));
117
118        pw.flush();
119
120        response.setStatus( HttpServletResponse.SC_OK );
121    }
122
123    /**
124     * Writes all of the key/value pairs to the HTTP response.
125     *
126     * The text of the message is formatted with
127     * {@link Messages#formatKeyValuePair(String, String)}
128     */
129    private void writeAllMappings( HttpServletResponse response ) throws IOException
130    {
131        PrintWriter pw = response.getWriter();
132        Messages.formatKeyValueMap(pw, data);
133
134        pw.flush();
135
136        response.setStatus( HttpServletResponse.SC_OK );
137    }
138
139    /**
140     * Returns the value of the HTTP request parameter with the given name.
141     *
142     * @return <code>null</code> if the value of the parameter is
143     *         <code>null</code> or is the empty string
144     */
145    private String getParameter(String name, HttpServletRequest request) {
146      String value = request.getParameter(name);
147      if (value == null || "".equals(value)) {
148        return null;
149
150      } else {
151        return value;
152      }
153    }
154
155    @VisibleForTesting
156    void setValueForKey(String key, String value) {
157        this.data.put(key, value);
158    }
159
160    @VisibleForTesting
161    String getValueForKey(String key) {
162        return this.data.get(key);
163    }
164
165}
