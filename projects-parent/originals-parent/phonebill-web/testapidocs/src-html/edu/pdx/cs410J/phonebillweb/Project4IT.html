001package edu.pdx.cs410J.phonebillweb;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.FixMethodOrder;
005import org.junit.Test;
006import org.junit.runners.MethodSorters;
007
008import java.io.IOException;
009
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.hamcrest.Matchers.containsString;
012import static org.hamcrest.Matchers.equalTo;
013
014/**
015 * Tests the {@link Project4} class by invoking its main method with various arguments
016 */
017@FixMethodOrder(MethodSorters.NAME_ASCENDING)
018public class Project4IT extends InvokeMainTestCase {
019    private static final String HOSTNAME = "localhost";
020    private static final String PORT = System.getProperty("http.port", "8080");
021
022    @Test
023    public void test0RemoveAllMappings() throws IOException {
024      PhoneBillRestClient client = new PhoneBillRestClient(HOSTNAME, Integer.parseInt(PORT));
025      client.removeAllDictionaryEntries();
026    }
027
028    @Test
029    public void test1NoCommandLineArguments() {
030        MainMethodResult result = invokeMain( Project4.class );
031        assertThat(result.getExitCode(), equalTo(1));
032        assertThat(result.getTextWrittenToStandardError(), containsString(Project4.MISSING_ARGS));
033    }
034
035    @Test
036    public void test2EmptyServer() {
037        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT );
038        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
039        String out = result.getTextWrittenToStandardOut();
040        assertThat(out, out, containsString(Messages.formatWordCount(0)));
041    }
042
043    @Test
044    public void test3NoDefinitions() {
045        String word = "WORD";
046        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, word );
047        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
048        String out = result.getTextWrittenToStandardOut();
049        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, null)));
050    }
051
052    @Test
053    public void test4AddDefinition() {
054        String word = "WORD";
055        String definition = "DEFINITION";
056
057        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, word, definition );
058        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
059        String out = result.getTextWrittenToStandardOut();
060        assertThat(out, out, containsString(Messages.definedWordAs(word, definition)));
061
062        result = invokeMain( Project4.class, HOSTNAME, PORT, word );
063        out = result.getTextWrittenToStandardOut();
064        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, definition)));
065
066        result = invokeMain( Project4.class, HOSTNAME, PORT );
067        out = result.getTextWrittenToStandardOut();
068        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, definition)));
069    }
070}
