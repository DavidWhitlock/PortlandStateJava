001package edu.pdx.cs410J.airlineweb;
002
003import com.google.common.annotations.VisibleForTesting;
004
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServlet;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import java.io.IOException;
010import java.io.PrintWriter;
011import java.util.HashMap;
012import java.util.Map;
013
014/**
015 * This servlet ultimately provides a REST API for working with an
016 * <code>Airline</code>.  However, in its current state, it is an example
017 * of how to use HTTP and Java servlets to store simple key/value pairs.
018 */
019public class AirlineServlet extends HttpServlet {
020  private final Map<String, String> data = new HashMap<>();
021
022  /**
023   * Handles an HTTP GET request from a client by writing the value of the key
024   * specified in the "key" HTTP parameter to the HTTP response.  If the "key"
025   * parameter is not specified, all of the key/value pairs are written to the
026   * HTTP response.
027   */
028  @Override
029  protected void doGet( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
030  {
031      response.setContentType( "text/plain" );
032
033      String key = getParameter( "key", request );
034      if (key != null) {
035          writeValue(key, response);
036
037      } else {
038          writeAllMappings(response);
039      }
040  }
041
042  /**
043   * Handles an HTTP POST request by storing the key/value pair specified by the
044   * "key" and "value" request parameters.  It writes the key/value pair to the
045   * HTTP response.
046   */
047  @Override
048  protected void doPost( HttpServletRequest request, HttpServletResponse response ) throws ServletException, IOException
049  {
050      response.setContentType( "text/plain" );
051
052      String key = getParameter( "key", request );
053      if (key == null) {
054          missingRequiredParameter(response, "key");
055          return;
056      }
057
058      String value = getParameter( "value", request );
059      if ( value == null) {
060          missingRequiredParameter( response, "value" );
061          return;
062      }
063
064      this.data.put(key, value);
065
066      PrintWriter pw = response.getWriter();
067      pw.println(Messages.mappedKeyValue(key, value));
068      pw.flush();
069
070      response.setStatus( HttpServletResponse.SC_OK);
071  }
072
073  /**
074   * Handles an HTTP DELETE request by removing all key/value pairs.  This
075   * behavior is exposed for testing purposes only.  It's probably not
076   * something that you'd want a real application to expose.
077   */
078  @Override
079  protected void doDelete(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
080      response.setContentType("text/plain");
081
082      this.data.clear();
083
084      PrintWriter pw = response.getWriter();
085      pw.println(Messages.allMappingsDeleted());
086      pw.flush();
087
088      response.setStatus(HttpServletResponse.SC_OK);
089
090  }
091
092  /**
093   * Writes an error message about a missing parameter to the HTTP response.
094   *
095   * The text of the error message is created by {@link Messages#missingRequiredParameter(String)}
096   */
097  private void missingRequiredParameter( HttpServletResponse response, String parameterName )
098      throws IOException
099  {
100      String message = Messages.missingRequiredParameter(parameterName);
101      response.sendError(HttpServletResponse.SC_PRECONDITION_FAILED, message);
102  }
103
104  /**
105   * Writes the value of the given key to the HTTP response.
106   *
107   * The text of the message is formatted with
108   * {@link Messages#formatKeyValuePair(String, String)}
109   */
110  private void writeValue( String key, HttpServletResponse response ) throws IOException {
111      String value = this.data.get(key);
112
113      PrintWriter pw = response.getWriter();
114      pw.println(Messages.formatKeyValuePair(key, value));
115
116      pw.flush();
117
118      response.setStatus( HttpServletResponse.SC_OK );
119  }
120
121  /**
122   * Writes all of the key/value pairs to the HTTP response.
123   *
124   * The text of the message is formatted with
125   * {@link Messages#formatKeyValuePair(String, String)}
126   */
127  private void writeAllMappings( HttpServletResponse response ) throws IOException {
128      PrintWriter pw = response.getWriter();
129      Messages.formatKeyValueMap(pw, data);
130
131      pw.flush();
132
133      response.setStatus( HttpServletResponse.SC_OK );
134  }
135
136  /**
137   * Returns the value of the HTTP request parameter with the given name.
138   *
139   * @return <code>null</code> if the value of the parameter is
140   *         <code>null</code> or is the empty string
141   */
142  private String getParameter(String name, HttpServletRequest request) {
143    String value = request.getParameter(name);
144    if (value == null || "".equals(value)) {
145      return null;
146
147    } else {
148      return value;
149    }
150  }
151
152  @VisibleForTesting
153  void setValueForKey(String key, String value) {
154      this.data.put(key, value);
155  }
156
157  @VisibleForTesting
158  String getValueForKey(String key) {
159      return this.data.get(key);
160  }
161}
