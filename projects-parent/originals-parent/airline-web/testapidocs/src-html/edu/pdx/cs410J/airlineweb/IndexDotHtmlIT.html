001package edu.pdx.cs410J.airlineweb;
002
003import edu.pdx.cs410J.web.HttpRequestHelper;
004import edu.pdx.cs410J.web.HttpRequestHelper.Response;
005import org.junit.Test;
006
007import java.io.IOException;
008
009import static org.hamcrest.Matchers.containsString;
010import static org.hamcrest.Matchers.equalTo;
011import static org.junit.Assert.assertThat;
012
013public class IndexDotHtmlIT {
014  private static final String HOSTNAME = "localhost";
015  private static final String PORT = System.getProperty("http.port", "8080");
016
017  @Test
018  public void indexDotHtmlExists() throws IOException {
019    Response indexDotHtml = fetchIndexDotHtml();
020    assertThat(indexDotHtml.getCode(), equalTo(200));
021  }
022
023  @Test
024  public void indexDotHtmlHasResonableContent() throws IOException {
025    Response indexDotHtml = fetchIndexDotHtml();
026    assertThat(indexDotHtml.getContent(), containsString("<form"));
027  }
028
029  private Response fetchIndexDotHtml() throws IOException {
030    int port = Integer.parseInt(PORT);
031    return new IndexDotHtmlHelper(HOSTNAME, port).getIndexDotHtml();
032  }
033
034  static class IndexDotHtmlHelper extends HttpRequestHelper {
035    private static final String WEB_APP = "airline";
036    private final String url;
037
038    IndexDotHtmlHelper(String hostName, int port) {
039      this.url = String.format( "http://%s:%d/%s/%s", hostName, port, WEB_APP, "index.html" );
040    }
041
042    public Response getIndexDotHtml() throws IOException {
043      return get(this.url);
044    }
045  }
046}
