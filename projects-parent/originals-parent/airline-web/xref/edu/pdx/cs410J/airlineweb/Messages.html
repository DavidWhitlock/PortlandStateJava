3   import java.io.PrintWriter;

9   /**
10   * Class for formatting messages on the server side.  This is mainly to enable
11   * test methods that validate that the server returned expected strings.
12   */
13  public class Messages
14  {
15      public static String formatMappingCount(int count )
16      {
17          return String.format( "Server contains %d key/value pairs", count );
18      }
19  
20      public static String formatKeyValuePair( String key, String value )
21      {
22          return String.format("  %s -> %s", key, value);
23      }
24  
25      public static String missingRequiredParameter( String parameterName )
26      {
27          return String.format("The required parameter \"%s\" is missing", parameterName);
28      }
29  
30      public static String mappedKeyValue( String key, String value )
31      {
32          return String.format( "Mapped %s to %s", key, value );
33      }
34  
35      public static String allMappingsDeleted() {
36          return "All mappings have been deleted";
37      }
38  
39      public static Map.Entry<String, String> parseKeyValuePair(String content) {
40          Pattern pattern = Pattern.compile("\\s*(.*) -> (.*)");
41          Matcher matcher = pattern.matcher(content);
42  
43          if (!matcher.find()) {
44              return null;
45          }
46  
47          return new Map.Entry<String, String>() {
48              @Override
49              public String getKey() {
50                  return matcher.group(1);
51              }
52  
53              @Override
54              public String getValue() {
55                  String value = matcher.group(2);
56                  if ("null".equals(value)) {
57                      value = null;
58                  }
59                  return value;
60              }
61  
62              @Override
63              public String setValue(String value) {
64                  throw new UnsupportedOperationException("This method is not implemented yet");
65              }
66          };
67      }
68  
69      public static void formatKeyValueMap(PrintWriter pw, Map<String, String> map) {
70          pw.println(Messages.formatMappingCount(map.size()));
71  
72          for (Map.Entry<String, String> entry : map.entrySet()) {
73              pw.println(Messages.formatKeyValuePair(entry.getKey(), entry.getValue()));
74          }
75      }
76  
77      public static Map<String, String> parseKeyValueMap(String content) {
78          Map<String, String> map = new HashMap<>();
79  
80          String[] lines = content.split("\n");
81          for (int i = 1; i < lines.length; i++) {
82              String line = lines[i];
83              Map.Entry<String, String> entry = parseKeyValuePair(line);
84              map.put(entry.getKey(), entry.getValue());
85          }
86  
87          return map;
88      }
