3   import com.google.common.annotations.VisibleForTesting;
4   import edu.pdx.cs410J.ParserException;
5   import edu.pdx.cs410J.web.HttpRequestHelper;

7   import java.io.IOException;
8   import java.io.StringReader;
9   import java.util.Map;
10  
11  import static edu.pdx.cs410J.web.HttpRequestHelper.Response;
12  import static edu.pdx.cs410J.web.HttpRequestHelper.RestException;
13  import static java.net.HttpURLConnection.HTTP_OK;
14  
15  /**
16   * A helper class for accessing the rest client.  Note that this class provides
17   * an example of how to make gets and posts to a URL.  You'll need to change it
18   * to do something other than just send dictionary entries.
19   */
20  public class AirlineRestClient
21  {
22      private static final String WEB_APP = "airline";
23      private static final String SERVLET = "flights";
24  
25      private final HttpRequestHelper http;
26  
27  
28      /**
29       * Creates a client to the airline REST service running on the given host and port
30       * @param hostName The name of the host
31       * @param port The port
32       */
33      public AirlineRestClient( String hostName, int port )
34      {
35          this(new HttpRequestHelper(String.format("http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET)));
36      }
37  
38      @VisibleForTesting
39      AirlineRestClient(HttpRequestHelper http) {
40        this.http = http;
41      }
42  
43    /**
44     * Returns all dictionary entries from the server
45     */
46    public Map<String, String> getAllDictionaryEntries() throws IOException, ParserException {
47      Response response = http.get(Map.of());
48      throwExceptionIfNotOkayHttpStatus(response);
49  
50      TextParser parser = new TextParser(new StringReader(response.getContent()));
51      return parser.parse();
52    }
53  
54    /**
55     * Returns the definition for the given word
56     */
57    public String getDefinition(String word) throws IOException, ParserException {
58      Response response = http.get(Map.of(AirlineServlet.WORD_PARAMETER, word));
59      throwExceptionIfNotOkayHttpStatus(response);
60      String content = response.getContent();
61  
62      TextParser parser = new TextParser(new StringReader(content));
63      return parser.parse().get(word);
64    }
65  
66    public void addDictionaryEntry(String word, String definition) throws IOException {
67      Response response = http.post(Map.of(AirlineServlet.WORD_PARAMETER, word, AirlineServlet.DEFINITION_PARAMETER, definition));
68      throwExceptionIfNotOkayHttpStatus(response);
69    }
70  
71    public void removeAllDictionaryEntries() throws IOException {
72      Response response = http.delete(Map.of());
73      throwExceptionIfNotOkayHttpStatus(response);
74    }
75  
76    private void throwExceptionIfNotOkayHttpStatus(Response response) {
77      int code = response.getHttpStatusCode();
78      if (code != HTTP_OK) {
79        String message = response.getContent();
80        throw new RestException(code, message);
81      }
82    }
83  
84  }
