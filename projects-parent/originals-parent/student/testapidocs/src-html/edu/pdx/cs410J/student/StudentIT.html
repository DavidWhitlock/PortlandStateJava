001package edu.pdx.cs410J.student;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.Test;
005
006import static org.hamcrest.MatcherAssert.assertThat;
007import static org.hamcrest.core.IsEqual.equalTo;
008import static org.hamcrest.core.StringContains.containsString;
009
010/**
011 * Integration tests for the <code>Student</code> class's main method.
012 * These tests extend <code>InvokeMainTestCase</code> which allows them
013 * to easily invoke the <code>main</code> method of <code>Student</code>.
014 */
015public class StudentIT extends InvokeMainTestCase {
016  @Test
017  public void invokingMainWithNoArgumentsHasExitCodeOf1() {
018    InvokeMainTestCase.MainMethodResult result = invokeMain(Student.class);
019    assertThat(result.getExitCode(), equalTo(1));
020  }
021
022  @Test
023  public void invokingMainWithNoArgumentsPrintsMissingArgumentsToStandardError() {
024    InvokeMainTestCase.MainMethodResult result = invokeMain(Student.class);
025    assertThat(result.getTextWrittenToStandardError(), containsString("Missing command line arguments"));
026  }
027
028
029}
