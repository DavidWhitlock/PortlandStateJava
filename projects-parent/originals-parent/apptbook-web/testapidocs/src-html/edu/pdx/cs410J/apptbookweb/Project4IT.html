001package edu.pdx.cs410J.apptbookweb;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.FixMethodOrder;
005import org.junit.Test;
006import org.junit.runners.MethodSorters;
007
008import java.io.IOException;
009
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.hamcrest.Matchers.containsString;
012import static org.hamcrest.Matchers.equalTo;
013
014/**
015 * An integration test for {@link Project4} that invokes its main method with
016 * various arguments
017 */
018@FixMethodOrder(MethodSorters.NAME_ASCENDING)
019public class Project4IT extends InvokeMainTestCase {
020    private static final String HOSTNAME = "localhost";
021    private static final String PORT = System.getProperty("http.port", "8080");
022
023    @Test
024    public void test0RemoveAllMappings() throws IOException {
025      AppointmentBookRestClient client = new AppointmentBookRestClient(HOSTNAME, Integer.parseInt(PORT));
026      client.removeAllDictionaryEntries();
027    }
028
029    @Test
030    public void test1NoCommandLineArguments() {
031        MainMethodResult result = invokeMain( Project4.class );
032        assertThat(result.getExitCode(), equalTo(1));
033        assertThat(result.getTextWrittenToStandardError(), containsString(Project4.MISSING_ARGS));
034    }
035
036    @Test
037    public void test2EmptyServer() {
038        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT );
039        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
040        String out = result.getTextWrittenToStandardOut();
041        assertThat(out, out, containsString(Messages.formatWordCount(0)));
042    }
043
044    @Test
045    public void test3NoDefinitions() {
046        String word = "WORD";
047        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, word );
048        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
049        String out = result.getTextWrittenToStandardOut();
050        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, null)));
051    }
052
053    @Test
054    public void test4AddDefinition() {
055        String word = "WORD";
056        String definition = "DEFINITION";
057
058        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, word, definition );
059        assertThat(result.getTextWrittenToStandardError(), result.getExitCode(), equalTo(0));
060        String out = result.getTextWrittenToStandardOut();
061        assertThat(out, out, containsString(Messages.definedWordAs(word, definition)));
062
063        result = invokeMain( Project4.class, HOSTNAME, PORT, word );
064        out = result.getTextWrittenToStandardOut();
065        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, definition)));
066
067        result = invokeMain( Project4.class, HOSTNAME, PORT );
068        out = result.getTextWrittenToStandardOut();
069        assertThat(out, out, containsString(Messages.formatDictionaryEntry(word, definition)));
070    }
071}
