001package edu.pdx.cs410J.apptbookweb;
002
003import org.junit.Test;
004
005import java.io.PrintWriter;
006import java.io.StringWriter;
007import java.util.HashMap;
008import java.util.Map;
009
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.hamcrest.Matchers.equalTo;
012import static org.hamcrest.Matchers.nullValue;
013
014public class MessagesTest {
015
016  @Test
017  public void malformedKeyAndPairReturnsNull() {
018    assertThat(Messages.parseKeyValuePair("blah"), nullValue());
019  }
020
021  @Test
022  public void canParseFormattedKeyAndValuePair() {
023    String key = "testKey";
024    String value = "testValue";
025    String formatted = Messages.formatKeyValuePair(key, value);
026    Map.Entry<String, String> parsed = Messages.parseKeyValuePair(formatted);
027    assertThat(parsed.getKey(), equalTo(key));
028    assertThat(parsed.getValue(), equalTo(value));
029  }
030
031  @Test
032  public void canParseFormattedKeyValueWithoutLeadingSpaces() {
033    String key = "testKey";
034    String value = "testValue";
035    String formatted = Messages.formatKeyValuePair(key, value);
036    String trimmed = formatted.trim();
037    Map.Entry<String, String> parsed = Messages.parseKeyValuePair(trimmed);
038    assertThat(parsed.getKey(), equalTo(key));
039    assertThat(parsed.getValue(), equalTo(value));
040
041  }
042
043  @Test
044  public void nullValueIsParsedAsNull() {
045    String key = "testKey";
046    String value = null;
047    String formatted = Messages.formatKeyValuePair(key, value);
048    Map.Entry<String, String> parsed = Messages.parseKeyValuePair(formatted);
049    assertThat(parsed.getKey(), equalTo(key));
050    assertThat(parsed.getValue(), equalTo(value));
051  }
052
053  @Test
054  public void canParseFormattedKeyValueMap() {
055    Map<String, String> map = new HashMap<>();
056
057    for (int i = 0; i < 5; i++) {
058      String key = String.valueOf(i);
059      String value = "QQ" + key;
060      map.put(key, value);
061    }
062
063    StringWriter sw = new StringWriter();
064    Messages.formatKeyValueMap(new PrintWriter(sw, true), map);
065
066    String formatted = sw.toString();
067
068    Map<String, String> actual = Messages.parseKeyValueMap(formatted);
069    assertThat(actual, equalTo(map));
070  }
071}
