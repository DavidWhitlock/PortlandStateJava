001package edu.pdx.cs410J.apptbookweb;
002
003import com.google.common.annotations.VisibleForTesting;
004import edu.pdx.cs410J.web.HttpRequestHelper;
005
006import java.io.IOException;
007import java.util.Map;
008
009import static java.net.HttpURLConnection.HTTP_OK;
010
011/**
012 * A helper class for accessing the rest client
013 */
014public class AppointmentBookRestClient extends HttpRequestHelper {
015  private static final String WEB_APP = "apptbook";
016  private static final String SERVLET = "appointments";
017
018  private final String url;
019
020
021  /**
022   * Creates a client to the appointment book REST service running on the given host and port
023   *
024   * @param hostName The name of the host
025   * @param port     The port
026   */
027  public AppointmentBookRestClient(String hostName, int port) {
028    this.url = String.format("http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET);
029  }
030
031  /**
032   * Returns all keys and values from the server
033   */
034  public Map<String, String> getAllKeysAndValues() throws IOException {
035    Response response = get(this.url);
036    return Messages.parseKeyValueMap(response.getContent());
037  }
038
039  /**
040   * Returns the value for the given key
041   */
042  public String getValue(String key) throws IOException {
043    Response response = get(this.url, "key", key);
044    throwExceptionIfNotOkayHttpStatus(response);
045    String content = response.getContent();
046    return Messages.parseKeyValuePair(content).getValue();
047  }
048
049  public void addKeyValuePair(String key, String value) throws IOException {
050    Response response = postToMyURL("key", key, "value", value);
051    throwExceptionIfNotOkayHttpStatus(response);
052  }
053
054  @VisibleForTesting
055  Response postToMyURL(String... keysAndValues) throws IOException {
056    return post(this.url, keysAndValues);
057  }
058
059  public void removeAllMappings() throws IOException {
060    Response response = delete(this.url);
061    throwExceptionIfNotOkayHttpStatus(response);
062  }
063
064  private Response throwExceptionIfNotOkayHttpStatus(Response response) {
065    int code = response.getCode();
066    if (code != HTTP_OK) {
067      throw new AppointmentBookRestException(code);
068    }
069    return response;
070  }
071
072  private class AppointmentBookRestException extends RuntimeException {
073    public AppointmentBookRestException(int httpStatusCode) {
074      super("Got an HTTP Status Code of " + httpStatusCode);
075    }
076  }
077}
