001package edu.pdx.cs410J;
002
003import org.junit.Test;
004
005import static org.hamcrest.MatcherAssert.assertThat;
006import static org.hamcrest.Matchers.containsString;
007import static org.hamcrest.Matchers.equalTo;
008
009public class InvokeMainTest extends InvokeMainTestCase {
010
011  @Test
012  public void commandLineArgsWrittenToStandardOutAreCaptured() {
013    String[] args = { "One", "Two", "Three"};
014    MainMethodResult result = invokeMain(InvokeMainTest.class, args);
015    assertThat(result.getExitCode(), equalTo(0));
016
017    String out = result.getTextWrittenToStandardOut();
018    for(String arg : args) {
019      assertThat(out, containsString(arg));
020    }
021  }
022
023  public static void main(String... args) {
024    for (String arg : args) {
025      System.out.println(arg);
026    }
027
028    System.exit(0);
029  }
030}
