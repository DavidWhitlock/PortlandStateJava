001package edu.pdx.cs410J;
002
003import org.junit.Test;
004
005import static org.hamcrest.CoreMatchers.equalTo;
006import static org.hamcrest.MatcherAssert.assertThat;
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNull;
009
010/**
011 * Tests the {@link #invokeMain(Class, String[])} method of {@link InvokeMainTestCase}
012 */
013public class InvokeMainIT extends InvokeMainTestCase {
014
015    @Test
016    public void testNoExitCode() {
017        MainMethodResult result = invokeMain( NoExitCode.class );
018        assertNull( result.getExitCode() );
019    }
020
021    @Test
022    public void testZeroExitCode() {
023        MainMethodResult result = invokeMain( ZeroExitCode.class );
024        assertEquals(new Integer(0), result.getExitCode());
025    }
026
027    @Test
028    public void testExitCodeOf1() {
029        MainMethodResult result = invokeMain( ExitCodeOf1.class );
030        assertEquals(new Integer(1), result.getExitCode());
031    }
032
033    @Test
034    public void testCommandLineArguments() {
035        Integer exitCode = 27;
036        MainMethodResult result = invokeMain( ExitCodeFromCommandLine.class, String.valueOf(exitCode) );
037        assertEquals( exitCode, result.getExitCode() );
038    }
039
040    @Test
041    public void testStandardOutput() {
042        MainMethodResult result = invokeMain( WriteArgsToStandardOut.class, "1", "2", "3", "4", "5" );
043        assertEquals( "12345", result.getTextWrittenToStandardOut() );
044    }
045
046    @Test
047    public void testStandardError() {
048        MainMethodResult result = invokeMain( WriteArgsToStandardErr.class, "1", "2", "3", "4", "5" );
049        assertEquals( "12345", result.getTextWrittenToStandardError() );
050    }
051
052  @Test
053  public void codeAfterSystemExitIsNotExecuted() {
054    MainMethodResult result = invokeMain(CodeAfterSystemExit.class);
055    assertThat(result.getExitCode(), equalTo(1));
056    assertThat(result.getTextWrittenToStandardOut(), equalTo(""));
057  }
058
059    private static class NoExitCode
060    {
061        public static void main(String... args) {
062            // Doesn't invoke System.exit()
063        }
064    }
065
066    private static class ZeroExitCode
067    {
068        public static void main(String... args) {
069            System.exit(0);
070        }
071    }
072
073    private static class ExitCodeOf1
074    {
075        public static void main(String... args) {
076            System.exit(1);
077        }
078    }
079
080    private static class ExitCodeFromCommandLine
081    {
082        public static void main(String... args)
083        {
084            System.exit(Integer.parseInt(args[0]));
085        }
086    }
087
088    private static class WriteArgsToStandardOut
089    {
090        public static void main(String... args) {
091            for (String arg : args) {
092                System.out.print(arg);
093            }
094            System.out.flush();
095        }
096    }
097
098    private static class WriteArgsToStandardErr
099    {
100        public static void main(String... args) {
101            for (String arg : args) {
102                System.err.print(arg);
103            }
104            System.err.flush();
105        }
106    }
107
108  private static class CodeAfterSystemExit {
109    public static void main(String... args) {
110      callSystemExit1();
111      System.out.println("Should not get here");
112      throw new RuntimeException("Should not get here");
113    }
114
115    private static void callSystemExit1() {
116      System.exit(1);
117    }
118  }
119}
