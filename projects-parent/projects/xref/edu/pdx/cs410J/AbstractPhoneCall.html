6   /**
7    * This abstract class represents a phone call between a caller (the
8    * phone number of the person who originates the call) and callee (the
9    * phone number of the person whose receives the phone call).  Phone
10   * calls begin and end at given times.
11   *
12   * @author David Whitlock
13   */
14  public abstract class AbstractPhoneCall implements Serializable {
15  
16    /**
17     * Returns the phone number of the person who originated this phone
18     * call.
19     */
20    public abstract String getCaller();
21  
22    /**
23     * Returns the phone number of the person who received this phone
24     * call.
25     */
26    public abstract String getCallee();
27  
28    /**
29     * Returns the time that this phone call was originated as a
30     * {@link Date}. 
31     */
32    public Date getStartTime() {
33      return null;
34    }
35  
36    /**
37     * Returns a textual representation of the time that this phone call
38     * was originated.
39     */
40    public abstract String getStartTimeString();
41  
42    /**
43     * Returns the time that this phone call was completed as a
44     * {@link Date}.
45     */
46    public Date getEndTime() {
47      return null;
48    }
49  
50    /**
51     * Returns a textual representation of the time that this phone call
52     * was completed.
53     */
54    public abstract String getEndTimeString();
55  
56    /**
57     * Returns a brief textual description of this phone call.
58     */
59    public String toString() {
60      return "Phone call from " + this.getCaller() + " to " +
61        this.getCallee() + " from " + this.getStartTimeString() + 
62        " to " + this.getEndTimeString();
63    }
64  
65  }
