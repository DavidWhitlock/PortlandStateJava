025  /**
026   * Invokes the main method of AddPerson with the given command line
027   * string 
028   */
029  private void addPerson(String cmdLine) {
030    if (DEBUG) {
031      System.out.println("\n" + cmdLine + "\n");
032    }
033
034    // Create an array of Strings to send to main
035    StringTokenizer st = new StringTokenizer(cmdLine);
036    String[] args = new String[st.countTokens()];
037    for (int i = 0; st.hasMoreTokens(); i++) {
038      args[i] = st.nextToken();
039    }
040
041    AddPerson.main(args);
042    reset(AddPerson.class);
043  }
044
045  /**
046   * "Resets" all of the static fields in a given class to their
047   * default value
048   */
049  private static void reset(Class c) {
050    Field[] fields = c.getDeclaredFields();
051    for (int i = 0; i < fields.length; i++) {
052      Field field = fields[i];
053      if (!Modifier.isFinal(field.getModifiers())) {
054        field.setAccessible(true);
055        try {
056          reset(field);
057
058        } catch (IllegalAccessException ex) {
059          StringWriter sw = new StringWriter();
060          sw.write("While accessing field " + field + ": ");
061          ex.printStackTrace(new PrintWriter(sw, true));
062          fail(sw.toString());
063        }
064      }
065    }
066  }
067
068  /**
069   * "Resets" the value of the given static field to its default value
070   */
071  private static void reset(Field field)
072    throws IllegalAccessException {
073
074    Class type = field.getType();
075    if (type.equals(boolean.class)) {
076      field.set(null, Boolean.FALSE);
077
078    } else if (type.equals(char.class)) {
079      field.set(null, '\0' );
080
081    } else if (type.equals(byte.class)) {
082      field.set(null, (byte) 0 );
083
084    } else if (type.equals(short.class)) {
085      field.set(null, (short) 0 );
086
087    } else if (type.equals(int.class)) {
088      field.set(null, 0 );
089
090    } else if (type.equals(long.class)) {
091      field.set(null, 0L );
092
093    } else if (type.equals(float.class)) {
094      field.set(null, 0.0f );
095
096    } else if (type.equals(double.class)) {
097      field.set(null, 0.0 );
098
099    } else {
100      assert Object.class.isAssignableFrom(type) : type;
101      field.set(null, null);
102    }
103  }

125  /**
126   * Create a family tree that describes Dave's family.
127   *
128   * @param fileOption
129   *        The kind of file to use
130   */
131  private void createDavesFamily(String fileOption) {
132    // Create a temp file to write the family tree to
133    File file;
134    try {
135      file = File.createTempFile("DavesFamily", fileOption);
136 
137      // XML parser complains about empty file, so delete it before we
138      // start
139      file.delete();
140
141    } catch (IOException ex) {
142      fail("Couldn't create temp file");
143      return;
144    }
145
146    String fileName = file.getAbsolutePath();
147
148    
149    addPerson(fileOption + fileName +
150              " 1 male David Michael Whitlock");
151    
152    addPerson(fileOption + "-parentOf 1 " + fileName + 
153              " 2 male Stanley Jay Whitlock Feb 27, 1948");
154
155    addPerson(fileOption + "-parentOf 1 " + fileName + 
156              " 3 female Carolyn Joyce Granger May 17, 1945");
157
158    noteMarriage(fileOption + "-date Jul 12, 1969 " + fileName +
159                 " 2 3 ");
160
161    // If we get this far, delete the file
162    file.delete();
163  }
