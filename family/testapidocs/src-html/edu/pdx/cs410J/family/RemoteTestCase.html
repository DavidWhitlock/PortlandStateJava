001package edu.pdx.cs410J.family;
002
003import org.junit.After;
004import org.junit.Assert;
005import static org.junit.Assert.assertNotNull;
006import org.junit.Before;
007
008import java.io.File;
009import java.io.PrintWriter;
010import java.io.StringWriter;
011import java.rmi.RemoteException;
012import java.rmi.registry.LocateRegistry;
013import java.rmi.registry.Registry;
014import java.util.Calendar;
015import java.util.Date;
016
017
018/**
019 * This is the abstract superclass for all of the remote family
020 * tree tests.
021 */
022public abstract class RemoteTestCase {
023
024  /** The port on which the RMI registry runs */
025  private static final int RMI_PORT =
026    Integer.getInteger("RMI_PORT", 1999).intValue();
027
028  /** An RMI Registry that is co-located in the VM in which the test
029   * runs */
030  private static Registry registry;
031
032  /** The name of this test */
033  private String name = "RemoteTestCase" + System.currentTimeMillis();
034
035    /**
036   * Creates an empty <code>RemoteFamilyTree</code> and binds it into
037   * the RMI namespace.
038   */
039  @Before
040  public void setUp() {
041    try {
042      File file = File.createTempFile("familyTree", "xml");
043      file.delete();
044      file.deleteOnExit();
045      RemoteFamilyTree tree = new XmlRemoteFamilyTree(file);
046      this.bind(tree);
047
048    } catch (Exception ex) {
049      fail("While getting creating remote family tree", ex);
050    }
051  }
052
053  /**
054   * Unbinds the remote family tree from the RMI namespace
055   */
056  @After
057  public void tearDown() {
058    this.unbind();
059  }
060
061  /**
062   * Returns the name that the <code>RemoteFamilyTree</code> is bound
063   * into in the RMI namespace.
064   */
065  protected String getFamilyName() {
066    return "RemoteTestCase";
067  }
068
069  /**
070   * Returns an RMI registry that is co-located in this VM.  If one
071   * doesn't exist, it is started on {@link #RMI_PORT}
072   */
073  private Registry getRegistry() {
074    if (registry != null) {
075      return registry;
076    }
077
078    synchronized (RemoteTestCase.class) {
079      if (registry != null) {
080        return registry;
081      }
082
083      try {
084        registry = LocateRegistry.createRegistry(RMI_PORT);
085
086      } catch (RemoteException ex) {
087        try {
088          registry = LocateRegistry.createRegistry(RMI_PORT);
089
090        } catch (RemoteException ex1) {
091          fail("Couldn't create local registry", ex);
092        }
093      }
094
095      assertNotNull(registry);
096      return registry;
097    }
098  }
099
100  /**
101   * Returns the <code>RemoteFamilyTree</code> used by this test
102   */
103  protected RemoteFamilyTree getTree() {
104//     if (System.getSecurityManager() == null) {
105//       System.setSecurityManager(new RMISecurityManager());
106//     }
107
108    Registry registry = getRegistry();
109    try {
110      return (RemoteFamilyTree) registry.lookup(this.getName());
111
112    } catch (Exception ex) {
113      fail("While getting remote family tree on " + this.getName(), ex);
114      return null;
115    }
116  }
117
118  /**
119   * Convenience method that binds a given
120   * <code>RemoteFamilyTree</code> into the RMI namespace.
121   */
122  protected void bind(RemoteFamilyTree tree) {
123//     if (System.getSecurityManager() == null) {
124//       System.setSecurityManager(new RMISecurityManager());
125//     }
126
127    Registry registry = getRegistry();
128    try {
129      registry.rebind(this.getName(), tree);
130      System.out.println("Successfully bound RemoteFamilyTree");
131
132    } catch (Exception ex) {
133      fail("While getting remote family tree on " + this.getName(), ex);
134    }
135  }
136
137  /**
138   * Convenience method that unbinds and shuts down the remote family
139   * tree from the RMI namespace.
140   */
141  protected void unbind() {
142//     if (System.getSecurityManager() == null) {
143//       System.setSecurityManager(new RMISecurityManager());
144//     }
145
146    Registry registry = getRegistry();
147    try {
148      RemoteFamilyTree tree = 
149        (RemoteFamilyTree) registry.lookup(this.getName());
150      tree.shutdown();
151      registry.unbind(this.getName());
152      System.out.println("Successfully unbound RemoteFamilyTree");
153
154    } catch (Exception ex) {
155      fail("While getting remote family tree on " + this.getName(), ex);
156    }
157  }
158
159  private String getName()
160  {
161    return name;
162  }
163
164    /**
165   * Asserts the equality of two dates.  Only takes the month, day,
166   * and year into account.
167   */
168  protected void assertEquals(Date d1, Date d2) {
169    Calendar cal1 = Calendar.getInstance();
170    cal1.setTime(d1);
171
172    Calendar cal2 = Calendar.getInstance();
173    cal1.setTime(d2);
174
175    Assert.assertEquals(cal1.get(Calendar.DAY_OF_YEAR),
176                 cal2.get(Calendar.DAY_OF_YEAR));
177    Assert.assertEquals(cal1.get(Calendar.YEAR), cal2.get(Calendar.YEAR));
178  }
179
180  /**
181   * A JUnit failure that is caused by an exception.  This method
182   * provides us with a nice strack trace for the failure.
183   *
184   * @since Winter 2004
185   */
186  public static void fail(String message, Throwable cause) {
187    StringWriter sw = new StringWriter();
188    sw.write(message);
189    sw.write("\nCaused by: ");
190
191    PrintWriter pw = new PrintWriter(sw, true);
192    cause.printStackTrace(pw);
193    Assert.fail(sw.toString());
194  }
195
196}
