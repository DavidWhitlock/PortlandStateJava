33      @Test
34    public void testConformingData() {
35      Person father = new Person(1, Person.MALE);
36      father.setDateOfBirth(new Date());
37      father.setDateOfDeath(new Date());
38  
39      Person mother = new Person(2, Person.FEMALE);
40      mother.setDateOfBirth(new Date());
41      mother.setDateOfDeath(new Date());
42  
43      Person child = new Person(3, Person.FEMALE);
44      child.setFather(father);
45      child.setMother(mother);
46  
47      Marriage m = new Marriage(father, mother);
48      m.setLocation("Here");
49      m.setDate(new Date());
50  
51      father.addMarriage(m);
52      mother.addMarriage(m);
53  
54      FamilyTree tree = new FamilyTree();
55      tree.addPerson(father);
56      tree.addPerson(mother);
57      tree.addPerson(child);
58  
59      String string = getStringFor(tree);
60  
61      FamilyTree tree2 = null;
62  
63      try {
64        tree2 = getFamilyTreeFor(string);
65  
66      } catch (FamilyTreeException ex) {
67        StringWriter sw = new StringWriter();
68        ex.printStackTrace(new PrintWriter(sw, true));
69        fail(sw.toString());
70      }
71  
72      assertEquals(tree, tree2);
73    }
