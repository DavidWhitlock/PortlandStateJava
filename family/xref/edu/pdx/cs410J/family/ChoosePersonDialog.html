8   /**
9    * A <code>ChoosePersonDialog</code> is used to select a person from a
10   * family tree.
11   */
12  @SuppressWarnings("serial")
13  public class ChoosePersonDialog extends JDialog {
14  
15    private Person person;  // The chosen person
16  
17    /**
18     * Creates a new <code>ChoosePersonDialog</code> that displays the
19     * people in a given <code>FamilyTree</code>.
20     */
21    public ChoosePersonDialog(FamilyTree tree, JDialog owner) {
22      super(owner, "Select a Person", true /* modal */);
23  
24      // A ChoosePersonDialog consists of a FamilyTreeList and a couple
25      // of buttons
26      Container pane = this.getContentPane();
27      pane.setLayout(new BorderLayout());
28  
29      final FamilyTreeListtml#FamilyTreeList">FamilyTreeList treeList = new FamilyTreeList();
30      treeList.fillInList(tree);
31  
32      JScrollPane treePane = new JScrollPane(treeList);
33      Border border = BorderFactory.createEmptyBorder(5, 5, 5, 5);
34      treePane.setBorder(border);
35  
36      pane.add(treePane, BorderLayout.CENTER);
37  
38      JPanel buttons = new JPanel();
39      buttons.setLayout(new BoxLayout(buttons, BoxLayout.X_AXIS));
40      buttons.add(Box.createHorizontalGlue());
41  
42      JButton okButton = new JButton("OK");
43      okButton.addActionListener(new ActionListener() {
44          public void actionPerformed(ActionEvent e) {
45            // Get selected person and dispose
46            ChoosePersonDialog.this.person =
47              treeList.getSelectedPerson();
48            ChoosePersonDialog.this.dispose();
49          }
50        });
51      buttons.add(okButton);
52  
53      buttons.add(Box.createHorizontalGlue());
54  
55      JButton cancelButton = new JButton("Cancel");
56      cancelButton.addActionListener(new ActionListener() {
57          public void actionPerformed(ActionEvent e) {
58            // No person selected
59            ChoosePersonDialog.this.person = null;
60            ChoosePersonDialog.this.dispose();
61          }
62        });
63      buttons.add(cancelButton);
64  
65      buttons.add(Box.createHorizontalGlue());
66  
67      pane.add(buttons, BorderLayout.SOUTH);
68    }
69  
70    /**
71     * Returns the choosen person
72     */
73    public Person getPerson() {
74      return this.person;
75    }
76  
77  }
